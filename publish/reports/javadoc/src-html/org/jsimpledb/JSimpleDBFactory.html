006package org.jsimpledb;
007
008import com.google.common.base.Preconditions;
009
010import java.util.Arrays;
011
012import javax.validation.ValidatorFactory;
013
014import org.jsimpledb.core.Database;
015import org.jsimpledb.kv.simple.SimpleKVDatabase;
016
017/**
018 * Factory for {@link JSimpleDB} instances.
019 *
020 * <p>
021 * If no {@link Database} is configured, newly created {@link JSimpleDB} instances will use an initially empty,
022 * in-memory {@link SimpleKVDatabase}.
023 * </p>
024 *
025 * @see JSimpleDB
026 */
027public class JSimpleDBFactory {
028
029    private Database database;
030    private int schemaVersion;
031    private StorageIdGenerator storageIdGenerator = new DefaultStorageIdGenerator();
032    private Iterable<? extends Class<?>> modelClasses;
033    private ValidatorFactory validatorFactory;
034
035    /**
036     * Configure the Java model classes.
037     *
038     * <p>
039     * Note: {@link org.jsimpledb.annotation.JSimpleClass &#64;JSimpleClass}-annotated super-types of any
040     * class in {@code modelClasses} will be included, even if the super-type is not explicitly specified in {@code modelClasses}.
041     * </p>
042     *
043     * @param modelClasses classes annotated with {@link org.jsimpledb.annotation.JSimpleClass &#64;JSimpleClass} annotations
044     * @return this instance
045     */
046    public JSimpleDBFactory setModelClasses(Iterable<? extends Class<?>> modelClasses) {
047        this.modelClasses = modelClasses;
048        return this;
049    }
050
051    /**
052     * Configure the Java model classes.
053     *
054     * <p>
055     * Equivalent to {@link #setModelClasses(Iterable) setModelClasses}{@code (Arrays.asList(modelClasses))}.
056     * </p>
057     *
058     * @param modelClasses classes annotated with {@link org.jsimpledb.annotation.JSimpleClass &#64;JSimpleClass} annotations
059     * @return this instance
060     * @see #setModelClasses(Iterable)
061     */
062    public JSimpleDBFactory setModelClasses(Class<?>... modelClasses) {
063        return this.setModelClasses(Arrays.asList(modelClasses));
064    }
065
066    /**
067     * Configure the underlying {@link Database} for this instance.
068     *
069     * <p>
070     * By default this instance will use an initially empty, in-memory {@link SimpleKVDatabase}.
071     * </p>
072     *
073     * @param database core API database to use
074     * @return this instance
075     */
076    public JSimpleDBFactory setDatabase(Database database) {
077        this.database = database;
078        return this;
079    }
080
081    /**
082     * Configure the schema version number associated with the configured Java model classes.
083     *
084     * <p>
085     * A value of zero means to use whatever is the highest version already recorded in the database.
086     * However, if this instance has no {@link Database} configured, then an empty
087     * {@link SimpleKVDatabase} is used and therefore a schema version of {@code 1} is assumed.
088     * </p>
089     *
090     * @param schemaVersion the schema version number of the schema derived from the configured Java model classes,
091     *  or zero to default to the highest version already recorded in the database
092     * @return this instance
093     */
094    public JSimpleDBFactory setSchemaVersion(int schemaVersion) {
095        this.schemaVersion = schemaVersion;
096        return this;
097    }
098
099    /**
100     * Configure the {@link StorageIdGenerator} for auto-generating storage ID's when not explicitly
101     * specified in {@link org.jsimpledb.annotation.JSimpleClass &#64;JSimpleClass},
102     * {@link org.jsimpledb.annotation.JField &#64;JField}, etc., annotations.
103     *
104     * <p>
105     * This instance will initially be configured with a {@link DefaultStorageIdGenerator}.
106     * To disable auto-generation of storage ID's altogether, configure a null value here.
107     * </p>
108     *
109     * @param storageIdGenerator storage ID generator, or null to disable auto-generation of storage ID's
110     * @return this instance
111     */
112    public JSimpleDBFactory setStorageIdGenerator(StorageIdGenerator storageIdGenerator) {
113        this.storageIdGenerator = storageIdGenerator;
114        return this;
115    }
116
117    /**
118     * Configure a custom {@link ValidatorFactory} used to create {@link javax.validation.Validator}s
119     * for validation within transactions.
120     *
121     * @param validatorFactory factory for validators
122     * @return this instance
123     * @throws IllegalArgumentException if {@code validatorFactory} is null
124     */
125    public JSimpleDBFactory setValidatorFactory(ValidatorFactory validatorFactory) {
126        Preconditions.checkArgument(validatorFactory != null, "null validatorFactory");
127        this.validatorFactory = validatorFactory;
128        return this;
129    }
130
131    /**
132     * Construct a {@link JSimpleDB} instance using this instance's configuration.
133     *
134     * @return newly created {@link JSimpleDB} database
135     * @throws IllegalArgumentException if this instance has an incomplete or invalid configuration
136     * @throws IllegalArgumentException if any Java model class has an invalid annotation
137     */
138    public JSimpleDB newJSimpleDB() {
139        Database database1 = this.database;
140        int schemaVersion1 = this.schemaVersion;
141        if (database1 == null) {
142            database1 = new Database(new SimpleKVDatabase());
143            if (schemaVersion1 == 0)
144                schemaVersion1 = 1;
145        }
146        final JSimpleDB jdb = new JSimpleDB(database1, schemaVersion1, this.storageIdGenerator, this.modelClasses);
147        if (this.validatorFactory != null)
148            jdb.setValidatorFactory(this.validatorFactory);
149        return jdb;
150    }
151}
