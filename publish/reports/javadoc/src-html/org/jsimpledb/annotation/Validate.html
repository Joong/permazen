006package org.jsimpledb.annotation;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.ElementType;
010import java.lang.annotation.Inherited;
011import java.lang.annotation.Retention;
012import java.lang.annotation.RetentionPolicy;
013import java.lang.annotation.Target;
014
015/**
016 * <b>Deprecated.</b> This annotation has been renamed to {@link OnValidate &#64;OnValidate}.
017 *
018 * @deprecated Replaced by {@link OnValidate &#64;OnValidate}
019 */
020@Documented
021@Inherited
022@Retention(RetentionPolicy.RUNTIME)
023@Target(ElementType.METHOD)
024@Deprecated
025public @interface Validate {
026
027    /**
028     * Specify the validation group(s) for which the annotated method should be invoked.
029     *
030     * @return validation group(s) to use for validation; if empty, {@link javax.validation.groups.Default} is assumed
031     */
032    Class<?>[] groups() default {};
033}
