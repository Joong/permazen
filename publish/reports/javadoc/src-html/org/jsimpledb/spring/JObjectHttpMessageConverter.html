006package org.jsimpledb.spring;
007
008import com.google.common.base.Preconditions;
009
010import java.io.IOException;
011import java.util.ArrayList;
012import java.util.Iterator;
013import java.util.List;
014import java.util.concurrent.ConcurrentHashMap;
015
016import org.jsimpledb.JClass;
017import org.jsimpledb.JObject;
018import org.jsimpledb.JSimpleDB;
019import org.jsimpledb.SnapshotJTransaction;
020import org.jsimpledb.core.UnknownTypeException;
021import org.jsimpledb.schema.SchemaObjectType;
022import org.springframework.http.HttpInputMessage;
023import org.springframework.http.HttpOutputMessage;
024import org.springframework.http.MediaType;
025import org.springframework.http.converter.AbstractHttpMessageConverter;
026import org.springframework.http.converter.HttpMessageNotReadableException;
027
028/**
029 * Spring {@link org.springframework.http.converter.HttpMessageConverter HttpMessageConverter} capable of
030 * encoding and decoding one or more {@link JObject}s contained in a {@link SnapshotJTransaction} that is
031 * backed by a {@link org.jsimpledb.kv.util.NavigableMapKVStore}.
032 *
033 * <p>
034 * The payload MIME type is set to {@code application/x-jsimpledb-FOO} where {@code FOO} is the
035 * {@linkplain org.jsimpledb.JClass#getName name} of the JSimpleDB object type.
036 * There must be exactly one instance of this object type in the encoded {@link SnapshotJTransaction},
037 * but there may be arbitrarily many other "supporting" {@link JObject}s riding along with it.
038 *
039 * <p>
040 * Validation of all incoming objects is supported; see {@link #setValidationGroups setValidationGroups()}.
041 *
042 * @see SnapshotJTransactionHttpMessageConverter
043 * @see KVStoreHttpMessageConverter
044 */
045public class JObjectHttpMessageConverter extends AbstractHttpMessageConverter<JObject> {
046
047    private static final String MEDIA_SUBTYPE_PREFIX = "x-jsimpledb-";
048
049    private final JSimpleDB jdb;
050
051    private final ConcurrentHashMap<MediaType, JClass<?>> mediaTypeMap = new ConcurrentHashMap<>();
052    private Class<?>[] validationGroups;
053
054    /**
055     * Constructor.
056     *
057     * @param mediaTypes custom {@link MediaType}s supported by overriding subclasses
058     * @param jdb {@link JSimpleDB} instance
059     * @throws IllegalArgumentException if {@code jdb} is null
060     */
061    public JObjectHttpMessageConverter(JSimpleDB jdb, MediaType... mediaTypes) {
062        super(mediaTypes);
063        Preconditions.checkArgument(jdb != null, "null jdb");
064        this.jdb = jdb;
065    }
066
067    /**
068     * Constructor.
069     *
070     * <p>
071     * Configures this instance for the MIME types {@code application/x-jsimpledb-foo}, {@code application/x-jsimpledb-bar}, etc.
072     * where {@code foo}, {@code bar}, etc. are {@code jdb}'s object type names.
073     *
074     * @param jdb {@link JSimpleDB} instance
075     * @throws IllegalArgumentException if {@code jdb} is null
076     */
077    public JObjectHttpMessageConverter(JSimpleDB jdb) {
078        this(jdb, JObjectHttpMessageConverter.generateMediaTypes(jdb));
079    }
080
081    /**
082     * Set validation groups used to validate all incoming objects.
083     *
084     * <p>
085     * If set to null, no validation is performed. Otherwise, validation of <b>all</b> incoming objects (not just
086     * the root) is performed using the specified validation groups, or {@link javax.validation.groups.Default} if
087     * an empty is specified.
088     *
089     * <p>
090     * By default, this is null.
091     *
092     * @param groups validation group(s) to use for validation; if empty, {@link javax.validation.groups.Default} is assumed;
093     *  if null, no validation is performed
094     */
095    public void setValidationGroups(Class<?>... groups) {
096        this.validationGroups = groups;
097    }
098
099// AbstractHttpMessageConverter
100
101    @Override
102    protected Long getContentLength(JObject jobj, MediaType contentType) {
103        return KVStoreHttpMessageConverter.getKVStoreContentLength(jobj.getTransaction().getTransaction().getKVTransaction());
104    }
105
106    @Override
107    public boolean canRead(Class<?> target, MediaType mediaType) {
108        if (!this.canRead(mediaType))
109            return false;
110        final JClass<?> jclass = this.findJClass(mediaType);
111        if (jclass == null)
112            return false;
113        return target.isAssignableFrom(jclass.getType());
114    }
115
116    @Override
117    public boolean canWrite(Class<?> target, MediaType mediaType) {
118        if (!this.canWrite(mediaType))
119            return false;
120        final JClass<?> jclass = this.findJClass(mediaType);
121        if (jclass == null)
122            return false;
123        return jclass.getType().isAssignableFrom(target);
124    }
125
126    @Override
127    protected boolean supports(Class<?> target) {
128        return this.jdb.getJClassesByType().containsKey(target);
129    }
130
131    @Override
132    protected MediaType getDefaultContentType(JObject jobj) {
133        Preconditions.checkArgument(jobj != null, "null jobj");
134        return JObjectHttpMessageConverter.createMediaType(this.jdb.getJClass(jobj.getObjId()));
135    }
136
137    @Override
138    protected JObject readInternal(Class<? extends JObject> type, HttpInputMessage input) throws IOException {
139
140        // Get snapshot transaction
141        final SnapshotJTransaction jtx = SnapshotJTransactionHttpMessageConverter.readSnapshotTransaction(
142          this.jdb, input, this.validationGroups);
143
144        // Get root object and verify there is exactly one of them
145        final Iterator<? extends JObject> i = jtx.getAll(type).iterator();
146        if (!i.hasNext())
147            throw new HttpMessageNotReadableException("no object of type " + type.getName() + " found in object graph");
148        final JObject jobj = i.next();
149        if (i.hasNext())
150            throw new HttpMessageNotReadableException("more than one object of type " + type.getName() + " found in object graph");
151
152        // Done
153        return jobj;
154    }
155
156    @Override
157    protected void writeInternal(JObject jobj, HttpOutputMessage output) throws IOException {
158        KVStoreHttpMessageConverter.writeKVStore(jobj.getTransaction().getTransaction().getKVTransaction(), output);
159    }
160
161// Internal methods
162
163    private JClass<?> findJClass(MediaType mediaType) {
164
165        // Sanity check
166        Preconditions.checkArgument(mediaType != null, "null mediaType");
167
168        // Check cache
169        final JClass<?> cached = this.mediaTypeMap.get(mediaType);
170        if (cached != null)
171            return cached;
172
173        // Decode MIME type into JClass
174        final String subtype = mediaType.getSubtype();
175        final int prefixLength = MEDIA_SUBTYPE_PREFIX.length();
176        if (subtype == null || subtype.length() < prefixLength || !subtype.substring(0, prefixLength).equals(MEDIA_SUBTYPE_PREFIX))
177            return null;
178        final String typeName = subtype.substring(prefixLength);
179        final SchemaObjectType objectType = this.jdb.getNameIndex().getSchemaObjectType(typeName);
180        if (objectType == null)
181            return null;
182        final JClass<?> jclass;
183        try {
184            jclass = this.jdb.getJClass(objectType.getStorageId());
185        } catch (UnknownTypeException e) {
186            return null;
187        }
188
189        // Cache and return result
190        this.mediaTypeMap.put(mediaType, jclass);
191        return jclass;
192    }
193
194    private static MediaType[] generateMediaTypes(JSimpleDB jdb) {
195        Preconditions.checkArgument(jdb != null, "null jdb");
196        final List<JClass<? extends Object>> jclasses = jdb.getJClasses(Object.class);
197        final ArrayList<MediaType> mediaTypes = new ArrayList<>(jclasses.size());
198        for (JClass<? extends Object> jclass : jclasses)
199            mediaTypes.add(JObjectHttpMessageConverter.createMediaType(jclass));
200        return mediaTypes.toArray(new MediaType[mediaTypes.size()]);
201    }
202
203    private static MediaType createMediaType(JClass<?> jclass) {
204        return new MediaType("application", MEDIA_SUBTYPE_PREFIX + jclass.getName());
205    }
206}
