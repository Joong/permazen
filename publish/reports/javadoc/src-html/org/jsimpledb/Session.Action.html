085
086    private Session(JSimpleDB jdb, Database db, KVDatabase kvdb) {
087        Preconditions.checkArgument(kvdb != null, "null kvdb");
088        if (jdb != null) {
089            Preconditions.checkArgument(db != null, "null db");
090            this.mode = SessionMode.JSIMPLEDB;
091        } else if (db != null)
092            this.mode = SessionMode.CORE_API;
093        else
094            this.mode = SessionMode.KEY_VALUE;
095        this.jdb = jdb;
096        this.db = db;
097        this.kvdb = kvdb;
098    }

296    /**
297     * Associate the current {@link JTransaction} with this instance while performing the given action.
298     *
299     * <p>
300     * If {@code action} throws an {@link Exception}, it will be caught and handled by {@link #reportException reportException()}
301     * and then false returned.
302     *
303     * <p>
304     * There must be a {@link JTransaction} open and {@linkplain JTransaction#getCurrent associated with the current thread}.
305     * It will be left open when this method returns.
306     *
307     * <p>
308     * This method safely handles re-entrant invocation.
309     *
310     * @param action action to perform
311     * @return true if {@code action} completed successfully, false if {@code action} threw an exception
312     * @throws IllegalArgumentException if {@code action} is null
313     * @throws IllegalStateException if there is already an open transaction associated with this instance
314     * @throws IllegalStateException if this instance is not in mode {@link SessionMode#JSIMPLEDB}
315     */
316    public boolean performSessionActionWithCurrentTransaction(Action action) {
317
318        // Sanity check
319        Preconditions.checkArgument(action != null, "null action");
320        Preconditions.checkArgument(SessionMode.JSIMPLEDB.equals(this.mode), "session is not in JSimpleDB mode");
321
322        // Check for re-entrant invocation, otherwise verify no other transaction is associated
323        final Transaction currentTx = JTransaction.getCurrent().getTransaction();
324        final KVTransaction currentKVT = currentTx.getKVTransaction();
325        final boolean associate;
326        if (this.tx == null && this.kvt == null)
327            associate = true;
328        else if (this.tx == currentTx && this.kvt == currentKVT)
329            associate = false;
330        else
331            throw new IllegalStateException("a different transaction is already open in this session");
332
333        // Perform action
334        if (associate) {
335            this.tx = currentTx;
336            this.kvt = currentKVT;
337        }
338        try {
339            action.run(this);
340            return true;
341        } catch (Exception e) {
342            this.reportException(e);
343            return false;
344        } finally {
345            if (associate) {
346                this.tx = null;
347                this.kvt = null;
348            }
349        }
350    }

352    /**
353     * Perform the given action within a new transaction associated with this instance.
354     *
355     * <p>
356     * If {@code action} throws an {@link Exception}, it will be caught and handled by {@link #reportException reportException()}
357     * and then false returned.
358     *
359     * @param action action to perform
360     * @return true if {@code action} completed successfully, false if the transaction could not be created
361     *  or {@code action} threw an exception
362     * @throws IllegalArgumentException if {@code action} is null
363     * @throws IllegalStateException if there is already an open transaction associated with this instance
364     */
365    public boolean performSessionAction(Action action) {
366
367        // Sanity check
368        Preconditions.checkArgument(action != null, "null action");
369
370        // Perform action within new transaction
371        try {
372            if (!this.openTransaction())
373                return false;
374            boolean success = false;
375            this.rollbackOnly = false;
376            try {
377                action.run(this);
378                success = true;
379            } finally {
380                success &= this.closeTransaction(success && !this.rollbackOnly);
381            }
382            return success;
383        } catch (Exception e) {
384            this.reportException(e);
385            return false;
386        } finally {
387            this.tx = null;
388        }
389    }

391    /**
392     * Mark the current transaction to be rolled back.
393     *
394     * @throws IllegalStateException if {@link #performSessionAction performSessionAction()} is not currently being invoked

396    public void setRollbackOnly() {
397        Preconditions.checkState(this.kvt != null, "no transaction is open in this session");
398        this.rollbackOnly = true;
399    }

401    private boolean openTransaction() {
402        try {
403
404            // Sanity check
405            Preconditions.checkState(this.tx == null && this.kvt == null, "a transaction is already open in this session");
406
407            // Open transaction at the appropriate level
408            switch (this.mode) {
409            case KEY_VALUE:
410                this.kvt = this.kvdb.createTransaction();
411                break;
412            case CORE_API:
413                this.tx = this.db.createTransaction(this.schemaModel, this.schemaVersion, this.allowNewSchema);
414                this.kvt = this.tx.getKVTransaction();
415                break;
416            case JSIMPLEDB:
417                Preconditions.checkState(!Session.isCurrentJTransaction(),
418                  "a JSimpleDB transaction is already open in the current thread");
419                final JTransaction jtx = this.jdb.createTransaction(this.allowNewSchema,
420                  validationMode != null ? validationMode : ValidationMode.AUTOMATIC);
421                JTransaction.setCurrent(jtx);
422                this.tx = jtx.getTransaction();
423                this.kvt = this.tx.getKVTransaction();
424                break;
425            default:
426                assert false;
427                break;
428            }
429
430            // Update schema model and version (if appropriate)
431            if (this.tx != null) {
432                final Schema schema = this.tx.getSchema();
433                this.setSchemaModel(schema.getSchemaModel());
434                this.setSchemaVersion(schema.getVersionNumber());
435                this.tx.setReadOnly(this.readOnly);
436            }
437
438            // Done
439            return true;
440        } catch (Exception e) {
441            this.tx = null;
442            this.kvt = null;
443            this.reportException(e);
444            return false;
445        }
446    }
447
448    @SuppressWarnings("fallthrough")
449    private boolean closeTransaction(boolean commit) {
450        try {
451            Preconditions.checkState(this.tx != null || this.kvt != null, "no transaction");
452            switch (this.mode) {
453            case JSIMPLEDB:
454                if (commit && !this.tx.isRollbackOnly())
455                    JTransaction.getCurrent().commit();
456                else
457                    JTransaction.getCurrent().rollback();
458                break;
459            case CORE_API:
460                if (commit && !this.tx.isRollbackOnly())
461                    this.tx.commit();
462                else
463                    this.tx.rollback();
464                break;
465            case KEY_VALUE:
466                if (commit)
467                    this.kvt.commit();
468                else
469                    this.kvt.rollback();
470                break;
471            default:
472                assert false;
473                break;
474            }
475            return true;
476        } catch (Exception e) {
477            this.reportException(e);
478            return false;
479        } finally {
480            switch (this.mode) {
481            case JSIMPLEDB:
482                JTransaction.setCurrent(null);
483                // FALLTHROUGH
484            case CORE_API:
485                this.tx = null;
486                // FALLTHROUGH
487            case KEY_VALUE:
488                this.kvt = null;
489                // FALLTHROUGH
490            default:
491                break;
492            }
493        }
494    }
495
496    private static boolean isCurrentJTransaction() {
497        try {
498            JTransaction.getCurrent();
499            return true;
500        } catch (IllegalStateException e) {
501            return false;
502        }
503    }
