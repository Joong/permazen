030/**
031 * A fixed set of {@link KeyRange} instances that can be treated as a unified whole, in particular as a {@link KeyFilter}.
032 *
033 * <p>
034 * Instances are immutable.
035 * </p>
036 *
037 * @see KeyRange
038 */
039public class KeyRanges implements Iterable<KeyRange>, KeyFilter, SizeEstimating {
040
041    /**
042     * The empty instance containing zero ranges.
043     */
044    public static final KeyRanges EMPTY = new KeyRanges(Collections.<KeyRange>emptyList());
045
046    /**
047     * The "full" instance containing a single {@link KeyRange} that contains all keys.
048     */
049    public static final KeyRanges FULL = new KeyRanges(Arrays.asList(KeyRange.FULL));

057    /**
058     * Constructor.
059     *
060     * <p>
061     * Creates an instance that contains all keys contained by any of the {@link KeyRange}s in {@code ranges}.
062     * The given {@code ranges} may be adjacent, overlap, and/or be listed in any order; this constructor
063     * will normalize them.
064     * </p>
065     *
066     * @param ranges individual key ranges
067     * @throws IllegalArgumentException if {@code ranges} or any {@link KeyRange} therein is null

069    public KeyRanges(Iterable<? extends KeyRange> ranges) {
070        Preconditions.checkArgument(ranges != null, "null ranges");
071        this.ranges = KeyRanges.minimize(Lists.<KeyRange>newArrayList(ranges));

209    /**
210     * Create the inverse of this instance. The inverse contains all keys not contained by this instance.
211     *
212     * @return the inverse of this instance
213     */
214    public KeyRanges inverse() {
215        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + 1);
216        if (this.ranges.isEmpty())
217            return KeyRanges.FULL;
218        final KeyRange first = this.ranges.get(0);
219        int i = 0;
220        byte[] lastMax;
221        if (first.getMin().length == 0) {
222            if ((lastMax = first.getMax()) == null) {
223                assert this.ranges.size() == 1;
224                return KeyRanges.EMPTY;
225            }
226            i++;
227        } else
228            lastMax = ByteUtil.EMPTY;
229        while (true) {
230            if (i == this.ranges.size()) {
231                list.add(new KeyRange(lastMax, null));
232                break;
233            }
234            final KeyRange next = this.ranges.get(i++);
235            list.add(new KeyRange(lastMax, next.getMin()));
236            if ((lastMax = next.getMax()) == null) {
237                assert i == this.ranges.size();
238                break;
239            }
240        }
241        return new KeyRanges(list);
242    }

244    /**
245     * Determine whether this instance contains the given {@link KeyRanges}, i.e., all keys contained by
246     * the given {@link KeyRanges} are also contained by this instance.
247     *
248     * @param ranges other instance to test
249     * @return true if this instance contains {@code ranges}, otherwise false
250     * @throws IllegalArgumentException if {@code ranges} is null
251     */
252    public boolean contains(KeyRanges ranges) {
253        Preconditions.checkArgument(ranges != null, "null ranges");
254        return ranges.equals(this.intersection(ranges));
255    }

257    /**
258     * Determine whether this instance contains the given {@link KeyRange}, i.e., all keys contained by
259     * the given {@link KeyRange} are also contained by this instance.
260     *
261     * @param range key range to test
262     * @return true if this instance contains {@code range}, otherwise false
263     * @throws IllegalArgumentException if {@code range} is null
264     */
265    public boolean contains(KeyRange range) {
266        Preconditions.checkArgument(range != null, "null range");
267        final int[] pair = this.findKeyIndex(range.getMin());
268        if (pair[0] != pair[1] || pair[0] == -1)
269            return false;
270        return this.ranges.get(pair[0]).contains(range);
271    }

273    /**
274     * Find the contiguous {@link KeyRange}(s) within this instance containing, or adjacent to, the given key.
275     *
276     * <p>
277     * This method returns an array of length two: if this instance contains {@code key} then both elements are the
278     * same Java object, namely, the {@link KeyRange} that contains {@code key}; otherwise, the first element is
279     * the nearest {@link KeyRange} to the left of {@code key}, or null if none exists, and the second element is
280     * the {@link KeyRange} to the right of {@code key}, or null if none exists. Note if this instance is empty
281     * then <code>{ null, null }</code> is returned.
282     * </p>
283     *
284     * @param key key to find
285     * @return array with the containing {@link KeyRange} or nearest neighbor to the left (or null) and
286     *  the containing {@link KeyRange} or nearest neighbor to the right (or null)
287     * @throws IllegalArgumentException if {@code key} is null
288     */
289    public KeyRange[] findKey(byte[] key) {
290        final int[] indexes = this.findKeyIndex(key);
291        return new KeyRange[] {
292          indexes[0] != -1 ? this.ranges.get(indexes[0]) : null,
293          indexes[1] != -1 ? this.ranges.get(indexes[1]) : null
294        };
295    }

297    private int[] findKeyIndex(byte[] key) {
298
299        // Sanity check
300        Preconditions.checkArgument(key != null, "null key");
301
302        // Optimization: assume previous success is likely to repeat
303        final int temp = this.lastContainingKeyRangeIndex;
304        if (temp != -1) {
305            if (this.ranges.get(temp).contains(key))
306                return new int[] { temp, temp };
307            this.lastContainingKeyRangeIndex = -1;
308        }
309
310        // Search for matching range
311        final int i = ~Collections.binarySearch(this.ranges, new KeyRange(key, key), KeyRange.SORT_BY_MIN);
312        assert i >= 0;                                                  // this.ranges should never contain new KeyRange(key, key)
313        int leftIndex = -1;
314        if (i > 0) {
315            leftIndex = i - 1;
316            if (this.ranges.get(leftIndex).contains(key)) {
317                this.lastContainingKeyRangeIndex = leftIndex;
318                return new int[] { leftIndex, leftIndex };
319            }
320        }
321        int rightIndex = -1;
322        if (i < this.ranges.size()) {
323            rightIndex = i;
324            if (this.ranges.get(rightIndex).contains(key)) {
325                this.lastContainingKeyRangeIndex = rightIndex;
326                return new int[] { rightIndex, rightIndex };
327            }
328        }
329
330        // Not contained
331        return new int[] { leftIndex, rightIndex };
332    }

334    /**
335     * Return a new {@link KeyRanges} instance equal to this instance with all keys in the given {@link KeyRange} added.
336     *
337     * @param range range to add
338     * @return this instance with {@code range} added
339     * @throws IllegalArgumentException if {@code range} is null
340     */
341    @SuppressWarnings("unchecked")
342    public KeyRanges add(KeyRange range) {
343
344        // Sanity checks
345        Preconditions.checkArgument(range != null, "null range");
346        if (range.isEmpty())
347            return this;
348        if (this.isEmpty())
349            return new KeyRanges(range);
350
351        // Start with same range list
352        final ArrayList<KeyRange> rangeList = (ArrayList<KeyRange>)this.ranges.clone();
353        assert !rangeList.isEmpty();
354
355        // Find where to start replacing ranges with new consolidated interval
356        byte[] minKey = range.getMin();
357        final int[] minFind = this.findKeyIndex(minKey);
358        int replaceIndexMin;
359        if (minFind[0] == minFind[1]
360          || (minFind[0] != -1 && Arrays.equals(minKey, rangeList.get(minFind[0]).getMax()))) {
361            assert minFind[0] != -1;
362            replaceIndexMin = minFind[0];
363            minKey = rangeList.get(replaceIndexMin).getMin();
364        } else
365            replaceIndexMin = minFind[0] + 1;               // works even if minFind[0] == -1
366
367        // Find where to stop replacing ranges with new consolidated interval
368        byte[] maxKey = range.getMax();
369        final int lastIndex = this.ranges.size() - 1;
370        final int[] maxFind = maxKey != null ? this.findKeyIndex(maxKey) :
371          new int[] { lastIndex, this.ranges.get(lastIndex).getMax() != null ? -1 : lastIndex };
372        int replaceIndexMax;
373        if (maxFind[0] == maxFind[1]) {
374            assert maxFind[1] != -1;
375            replaceIndexMax = maxFind[1] + 1;
376            maxKey = rangeList.get(maxFind[1]).getMax();
377        } else
378            replaceIndexMax = maxFind[1] != -1 ? maxFind[1] : lastIndex + 1;
379
380        // Replace all overlapping ranges with a single new range
381        if (replaceIndexMax == replaceIndexMin + 1)
382            rangeList.set(replaceIndexMin, new KeyRange(minKey, maxKey));
383        else {
384            if (replaceIndexMax > replaceIndexMin)
385                rangeList.subList(replaceIndexMin, replaceIndexMax).clear();
386            rangeList.add(replaceIndexMin, new KeyRange(minKey, maxKey));
387        }
388
389        // Done
390        return new KeyRanges(rangeList);
391    }

393    /**
394     * Return a new {@link KeyRanges} instance equal to this instance with all keys in the given {@link KeyRange} removed.
395     *
396     * @param range range to remove
397     * @return this instance with {@code range} removed
398     * @throws IllegalArgumentException if {@code range} is null
399     */
400    @SuppressWarnings("unchecked")
401    public KeyRanges remove(KeyRange range) {
402
403        // Sanity checks
404        Preconditions.checkArgument(range != null, "null range");
405        if (range.isEmpty() || this.ranges.isEmpty())
406            return this;
407
408        // Start with same range list
409        final ArrayList<KeyRange> rangeList = (ArrayList<KeyRange>)this.ranges.clone();
410        assert !rangeList.isEmpty();
411
412        // Find where remove range's endpoints intersect our range list
413        final byte[] minKey = range.getMin();
414        final int[] minFind = this.findKeyIndex(minKey);
415        int imin = minFind[1];                                  // index of our range containing, or to the right of, minKey
416        if (imin == -1)                                         // all our ranges are to the left of minKey, nothing to do
417            return this;
418        final byte[] maxKey = range.getMax();
419        final int lastIndex = this.ranges.size() - 1;
420        final int[] maxFind = maxKey != null ? this.findKeyIndex(maxKey) :
421          new int[] { lastIndex, this.ranges.get(lastIndex).getMax() != null ? -1 : lastIndex };
422        int imax = maxFind[0];                                  // index of our range containing, or to the left of, maxKey
423        if (imax == -1)                                         // all our ranges are to the right of maxKey, nothing to do
424            return this;
425        if (imax < imin)                                        // the remove range is between and not touching two of our ranges
426            return this;
427
428        // Split rmin in two if it contains minKey
429        final KeyRange rmin = rangeList.get(imin);
430        if (ByteUtil.compare(rmin.getMin(), minKey) < 0) {
431            rangeList.set(imin, new KeyRange(rmin.getMin(), minKey));
432
433            // Handle the case were rmin and rmax are the same range
434            if (imax == imin) {
435                if (maxKey != null && (rmin.getMax() == null || ByteUtil.compare(maxKey, rmin.getMax()) < 0))
436                    rangeList.add(imin + 1, new KeyRange(maxKey, rmin.getMax()));
437                return new KeyRanges(rangeList);
438            }
439            imin++;
440        }
441
442        // Split rmax in two if it contains maxKey
443        final KeyRange rmax = rangeList.get(imax);
444        if (maxKey != null && (rmax.getMax() == null || ByteUtil.compare(maxKey, rmax.getMax()) < 0))
445            rangeList.set(imax--, new KeyRange(maxKey, rmax.getMax()));
446
447        // Remove all ranges from rmin through rmax
448        rangeList.subList(imin, imax + 1).clear();
449
450        // Done
451        return new KeyRanges(rangeList);
452    }

454    /**
455     * Create an instance that represents the union of this and the provided instance(s).
456     *
457     * @param others other instances
458     * @return the union of this instance and {@code others}
459     * @throws IllegalArgumentException if {@code others} or any element in {@code others} is null
460     */
461    public KeyRanges union(KeyRanges... others) {
462        Preconditions.checkArgument(others != null, "null others");
463        if (others.length == 0)
464            return this;
465        final ArrayList<ArrayList<KeyRange>> rangeLists = new ArrayList<>(1 + others.length);
466        rangeLists.add(this.ranges);
467        for (KeyRanges other : others) {
468            Preconditions.checkArgument(other != null, "null other");
469            rangeLists.add(other.ranges);
470        }
471        return new KeyRanges(KeyRanges.minimizeSortedByMinNoEmpty(
472          Lists.newArrayList(Iterables.mergeSorted(rangeLists, KeyRange.SORT_BY_MIN))));
473    }
474
475    /**
476     * Create an instance that represents the intersection of this and the provided instance(s).
477     *
478     * @param others other instances
479     * @return the intersection of this instance and {@code others}
480     * @throws IllegalArgumentException if {@code others} or any element in {@code others} is null
481     */
482    public KeyRanges intersection(KeyRanges... others) {
483        Preconditions.checkArgument(others != null, "null others");
484        if (others.length == 0)
485            return this;
486        final KeyRanges[] inverses = new KeyRanges[others.length];
487        for (int i = 0; i < others.length; i++) {
488            Preconditions.checkArgument(others[i] != null, "null other");
489            inverses[i] = others[i].inverse();
490        }
491        return this.inverse().union(inverses).inverse();
492    }

515    /**
516     * Serialize this instance.
517     *
518     * @param out output
519     * @throws IOException if an error occurs
520     * @throws IllegalArgumentException if {@code out} is null
521     */
522    public void serialize(OutputStream out) throws IOException {
523        UnsignedIntEncoder.write(out, this.ranges.size());
524        byte[] prev = null;
525        for (KeyRange range : this.ranges) {
526            final byte[] min = range.getMin();
527            final byte[] max = range.getMax();
528            assert max != null || range == this.ranges.get(this.ranges.size() - 1);
529            KeyListEncoder.write(out, min, prev);
530            KeyListEncoder.write(out, max != null ? max : min, min);            // map final [min, null) to [min, min]
531            prev = max;
532        }
533    }
534
535    /**
536     * Calculate the number of bytes required to serialize this instance via {@link #serialize serialize()}.
537     *
538     * @return number of serialized bytes
539     */
540    public long serializedLength() {
541        long total = UnsignedIntEncoder.encodeLength(this.ranges.size());
542        byte[] prev = null;
543        for (KeyRange range : this.ranges) {
544            final byte[] min = range.getMin();
545            final byte[] max = range.getMax();
546            total += KeyListEncoder.writeLength(min, prev);
547            total += KeyListEncoder.writeLength(max != null ? max : min, min);
548            prev = max;
549        }
550        return total;
551    }

553    /**
554     * Deserialize an instance created by {@link #serialize serialize()}.
555     *
556     * @param input input stream containing data from {@link #serialize serialize()}
557     * @return deserialized instance
558     * @throws IOException if an I/O error occurs
559     * @throws java.io.EOFException if the input ends unexpectedly
560     * @throws IllegalArgumentException if {@code input} is null
561     * @throws IllegalArgumentException if {@code input} is invalid
562     */
563    public static KeyRanges deserialize(InputStream input) throws IOException {
564        Preconditions.checkArgument(input != null, "null input");
565        final int count = UnsignedIntEncoder.read(input);
566        final ArrayList<KeyRange> rangeList = new ArrayList<>(count);
567        byte[] prev = null;
568        for (int i = 0; i < count; i++) {
569            final byte[] min = KeyListEncoder.read(input, prev);
570            final byte[] max = KeyListEncoder.read(input, min);
571            Preconditions.checkArgument(prev == null || ByteUtil.compare(min, prev) > 0, "invalid input");
572            rangeList.add(new KeyRange(min, Arrays.equals(min, max) ? null : max));
573            prev = max;
574        }
575        return new KeyRanges(rangeList);
576    }

578    /**
579     * Deserialize an instance created by {@link #serialize serialize()} in the form of an
580     * iterator of the individual {@link KeyRange}s.
581     *
582     * <p>
583     * If an {@link IOException} is thrown while reading, the returned {@link Iterator}
584     * will throw a {@link RuntimeException} wrapping it. If invalid data is encountered,
585     * the returned {@link Iterator} will throw an {@link IllegalArgumentException}.
586     *
587     * @param input input stream containing data from {@link #serialize serialize()}
588     * @return deserialized iteration of {@link KeyRange}s
589     * @throws IllegalArgumentException if {@code input} is null
590     */
591    public static Iterator<KeyRange> deserializeIterator(final InputStream input) {
592        Preconditions.checkArgument(input != null, "null input");
593        return new UnmodifiableIterator<KeyRange>() {
594
595            private int remain = -1;
596            private byte[] prev;
597
598            @Override
599            public boolean hasNext() {
600                try {
601                    this.init();
602                } catch (IOException e) {
603                    throw new RuntimeException(e);
604                }
605                return this.remain > 0;
606            }
607
608            @Override
609            public KeyRange next() {
610                if (this.remain == 0)
611                    throw new NoSuchElementException();
612                final byte[] min;
613                final byte[] max;
614                try {
615                    this.init();
616                    min = KeyListEncoder.read(input, this.prev);
617                    max = KeyListEncoder.read(input, min);
618                } catch (IOException e) {
619                    throw new RuntimeException(e);
620                }
621                final KeyRange range = new KeyRange(min, Arrays.equals(min, max) ? null : max);
622                this.prev = max;
623                this.remain--;
624                return range;
625            }
626
627            private void init() throws IOException {
628                if (this.remain == -1)
629                    this.remain = UnsignedIntEncoder.read(input);
630            }
631        };
632    }

636    @Override
637    public boolean contains(byte[] key) {
638        final int[] pair = this.findKeyIndex(key);
639        return pair[0] == pair[1] && pair[0] != -1;
640    }

642    @Override
643    public byte[] seekHigher(byte[] key) {
644        final int[] pair = this.findKeyIndex(key);
645        if (pair[0] == pair[1])
646            return pair[0] != -1 ? key : null;
647        return pair[1] != -1 ? this.ranges.get(pair[1]).getMin() : null;
648    }
649
650    @Override
651    public byte[] seekLower(byte[] key) {
652        Preconditions.checkArgument(key != null, "null key");
653        if (key.length == 0) {
654            if (this.ranges.isEmpty())
655                return null;
656            final byte[] lastMax = this.ranges.get(this.ranges.size() - 1).getMax();
657            return lastMax != null ? lastMax : ByteUtil.EMPTY;
658        }
659        final int[] pair = this.findKeyIndex(key);
660        if (pair[0] == pair[1])
661            return pair[0] != -1 ? key : null;
662        return pair[0] != -1 ? this.ranges.get(pair[0]).getMax() : null;
663    }

700    // Return a "minimal" list with these properties:
701    //  - Sorted according to KeyRange.SORT_BY_MIN
702    //  - No overlapping ranges
703    //  - Adjacent ranges consolidated into a single range
704    private static ArrayList<KeyRange> minimize(ArrayList<KeyRange> ranges) {
705
706        // Create array
707        final ArrayList<KeyRange> sortedRanges = new ArrayList<>(ranges);
708
709        // Remove empty ranges
710        for (Iterator<KeyRange> i = sortedRanges.iterator(); i.hasNext(); ) {
711            if (i.next().isEmpty())
712                i.remove();
713        }
714
715        // Sort remaining ranges by min, then max
716        Collections.sort(sortedRanges, KeyRange.SORT_BY_MIN);
717
718        // Proceed
719        return KeyRanges.minimizeSortedByMinNoEmpty(sortedRanges);
720    }
721
722    private static ArrayList<KeyRange> minimizeSortedByMinNoEmpty(ArrayList<KeyRange> ranges) {
723
724        // Sanity check
725        assert KeyRanges.isSortedByMin(ranges);
726        assert !KeyRanges.containsEmpty(ranges);
727
728        // Consolidate
729        final ArrayList<KeyRange> list = new ArrayList<>(ranges.size());
730        KeyRange prev = null;
731        for (KeyRange range : ranges) {
732
733            // Sanity check range
734            Preconditions.checkArgument(range != null, "null range");
735
736            // Handle first in list
737            if (prev == null) {
738                prev = range;
739                continue;
740            }
741
742            // Compare to previous range
743            final int diff1 = KeyRange.compare(range.getMin(), prev.getMin());
744            assert diff1 >= 0;
745            if (diff1 == 0) {                           // range contains prev -> discard prev
746                assert range.contains(prev);
747                prev = range;
748                continue;
749            }
750            final int diff2 = KeyRange.compare(range.getMin(), prev.getMax());
751            if (diff2 <= 0) {                           // prev and range overlap -> take their union
752                final byte[] max = KeyRange.compare(range.getMax(), prev.getMax()) > 0 ? range.getMax() : prev.getMax();
753                prev = new KeyRange(prev.getMin(), max);
754                continue;
755            }
756
757            // OK add it
758            list.add(prev);                             // prev and range don't overlap -> accept prev
759            prev = range;
760            continue;
761        }
762        if (prev != null)
763            list.add(prev);
764        list.trimToSize();
765        assert KeyRanges.isMinimal(list);
766        return list;
767    }
768
769    private static boolean isMinimal(List<KeyRange> ranges) {
770        KeyRange prev = null;
771        for (KeyRange range : ranges) {
772            if (range.isEmpty())
773                return false;
774            if (prev != null && KeyRange.compare(prev.getMax(), range.getMin()) >= 0)
775                return false;
776            prev = range;
777        }
778        return true;
779    }
780
781    private static boolean isSortedByMin(List<KeyRange> ranges) {
782        KeyRange prev = null;
783        for (KeyRange range : ranges) {
784            if (prev != null && KeyRange.SORT_BY_MIN.compare(prev, range) > 0)
785                return false;
786            prev = range;
787        }
788        return true;
789    }
790
791    private static boolean containsEmpty(List<KeyRange> ranges) {
792        for (KeyRange range : ranges) {
793            if (range.isEmpty())
794                return true;
795        }
796        return false;
797    }
