006package org.jsimpledb.cli.cmd;
007
008import java.io.PrintWriter;
009import java.util.HashSet;
010import java.util.Map;
011
012import org.jsimpledb.SessionMode;
013import org.jsimpledb.cli.CliSession;
014import org.jsimpledb.parse.ParseContext;
015import org.jsimpledb.parse.Parser;
016import org.jsimpledb.parse.WordParser;
017import org.jsimpledb.parse.func.AbstractFunction;
018
019@Command(modes = { SessionMode.KEY_VALUE, SessionMode.CORE_API, SessionMode.JSIMPLEDB })
020public class HelpCommand extends AbstractCommand {
021
022    private final CliSession session;
023
024    public HelpCommand(CliSession session) {
025        super("help -a:all command-or-function:cmdfunc?");
026        this.session = session;
027    }
028
029    @Override
030    public String getHelpSummary() {
031        return "Display help information";
032    }
033
034    @Override
035    public String getHelpDetail() {
036        return "Displays the list of known commands and functions, or help information about a specific command or function."
037          + "\nNormally only those appropriate for the current session mode are listed; use `-a' to show all.";
038    }
039
040    @Override
041    protected Parser<?> getParser(String typeName) {
042        if ("cmdfunc".equals(typeName)) {
043            return new WordParser("command/function") {
044                @Override
045                protected HashSet<String> getWords() {
046                    final HashSet<String> names = new HashSet<>();
047                    names.addAll(HelpCommand.this.session.getCommands().keySet());
048                    names.addAll(HelpCommand.this.session.getFunctions().keySet());
049                    return names;
050                }
051            };
052        }
053        return super.getParser(typeName);
054    }
055
056    @Override
057    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
058        final boolean all = params.containsKey("all");
059        final String name = (String)params.get("command-or-function");
060        final SessionMode sessionMode = session.getMode();
061        return new CliSession.Action() {
062            @Override
063            public void run(CliSession session) throws Exception {
064                final PrintWriter writer = session.getWriter();
065                if (name == null) {
066                    writer.println((all ? "All" : "Available") + " commands:");
067                    for (AbstractCommand command : session.getCommands().values()) {
068                        if (all || command.getSessionModes().contains(sessionMode))
069                            writer.println(String.format("%24s - %s", command.getName(), command.getHelpSummary()));
070                    }
071                    writer.println((all ? "All" : "Available") + " functions:");
072                    for (AbstractFunction function : session.getFunctions().values()) {
073                        if (all || function.getSessionModes().contains(sessionMode))
074                            writer.println(String.format("%24s - %s", function.getName(), function.getHelpSummary()));
075                    }
076                } else {
077                    final AbstractCommand command = session.getCommands().get(name);
078                    if (command != null) {
079                        writer.println("Usage: " + command.getUsage());
080                        writer.println(command.getHelpDetail());
081                        writer.println("Supported session modes: "
082                          + command.getSessionModes().toString().replaceAll("\\[(.*)\\]", "$1"));
083                    }
084                    final AbstractFunction function = session.getFunctions().get(name);
085                    if (function != null) {
086                        writer.println("Usage: " + function.getUsage());
087                        writer.println(function.getHelpDetail());
088                        writer.println("Supported session modes: "
089                          + function.getSessionModes().toString().replaceAll("\\[(.*)\\]", "$1"));
090                    }
091                    if (command == null && function == null)
092                        writer.println("No command or function named `" + name + "' exists.");
093                }
094            }
095        };
096    }
097}
