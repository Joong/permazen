055/**
056 * {@link AtomicKVStore} based on {@link ArrayKVStore} with background compaction.
057 *
058 * <p>
059 * This implementation is designed to maximize the speed of reads and minimize the amount of memory overhead per key/value pair.
060 * It is optimized for relatively infrequent writes.
061 *
062 * <p>
063 * Instances periodically compact outstanding changes into new arrays in a background thread. A compaction is scheduled whenever:
064 *  <ul>
065 *  <li>The size of uncompacted changes exceeds the {@linkplain #setCompactLowWater compaction space low-water mark}</li>
066 *  <li>The oldest uncompacted modification is older than the {@linkplain #setCompactMaxDelay compaction maximum delay}</li>
067 *  <li>{@link #scheduleCompaction} is invoked</li>
068 *  </ul>
069 *
070 * <p>
071 * There is also a {@linkplain #setCompactHighWater high water mark for the size of uncompacted changes}: when this value
072 * is exceeded, new write attempts will block until the current compaction cycle completes. This prevents the unbounded growth
073 * of uncompacted changes when there is extremely high write volume.
074 *
075 * <p>
076 * The {@linkplain #setDirectory database directory} is a required configuration property.
077 *
078 * <p>
079 * Key and value data each must not exceed 2GB total.
080 *
081 * <p>
082 * Instances may be stopped and (re)started multiple times.
083 */
084public class AtomicArrayKVStore extends AbstractKVStore implements AtomicKVStore {

260    @Override
261    @PostConstruct
262    public void start() {
263        boolean success = false;
264        this.writeLock.lock();
265        try {
266
267            // Already started?
268            if (this.kvstore != null) {
269                success = true;
270                return;
271            }
272            this.log.info("starting " + this);
273
274            // Sanity check
275            assert this.compactFuture == null;
276            assert this.scheduledExecutorService == null;
277            assert !this.createdExecutorService;
278            assert this.generation == 0;
279            assert this.generationFile == null;
280            assert this.lockFile == null;
281            assert this.lockFileChannel == null;
282            assert this.indxFile == null;
283            assert this.keysFile == null;
284            assert this.valsFile == null;
285            assert this.modsFile == null;
286            assert this.modsFileOutput == null;
287            assert this.directoryChannel == null;
288            assert this.modsFileLength == 0;
289            assert this.modsFileSyncPoint == 0;
290            assert this.indx == null;
291            assert this.keys == null;
292            assert this.vals == null;
293            assert this.kvstore == null;
294            assert this.mods == null;
295            assert this.firstModTimestamp == 0;
296
297            // Check configuration
298            Preconditions.checkState(this.directory != null, "no directory configured");
299
300            // Create executor if needed
301            this.createdExecutorService = this.scheduledExecutorService == null;
302            if (this.createdExecutorService) {
303                this.scheduledExecutorService = Executors.newSingleThreadScheduledExecutor(new ThreadFactory() {
304                    @Override
305                    public Thread newThread(Runnable action) {
306                        final Thread thread = new Thread(action);
307                        thread.setName("Compactor for " + AtomicArrayKVStore.this);
308                        return thread;
309                    }
310                });
311            }
312
313            // Create directory if needed
314            boolean initializeFiles = false;
315            if (!this.directory.exists()) {
316                if (!this.directory.mkdirs())
317                    throw new ArrayKVException("failed to create directory `" + this.directory + "'");
318            }
319            if (!this.directory.isDirectory())
320                throw new ArrayKVException("file `" + this.directory + "' is not a directory");
321
322            // Get directory channel we can fsync()
323            this.directoryChannel = FileChannel.open(this.directory.toPath());
324
325            // Open and lock the lock file
326            this.lockFile = new File(this.directory, LOCK_FILE_NAME);
327            if (!this.lockFile.exists()) {
328                this.lockFileChannel = FileChannel.open(this.lockFile.toPath(),
329                  StandardOpenOption.WRITE, StandardOpenOption.CREATE_NEW);
330            } else {
331                this.lockFileChannel = FileChannel.open(this.lockFile.toPath(),
332                  StandardOpenOption.WRITE, StandardOpenOption.TRUNCATE_EXISTING);
333            }
334            FileLock fileLock = null;
335            try {
336                fileLock = this.lockFileChannel.tryLock();
337            } catch (OverlappingFileLockException e) {
338                // too bad
339            }
340            if (fileLock == null)
341                throw new ArrayKVException("database is already locked by another process or thread");
342
343            // If no generation file exists, initialize generation zero
344            this.generationFile = new File(this.directory, GENERATION_FILE_NAME);
345            if (!this.generationFile.exists()) {
346
347                // Create empty index, keys, and values files
348                try (
349                  final FileOutputStream indxOutput = new FileOutputStream(new File(this.directory, INDX_FILE_NAME_BASE + 0));
350                  final FileOutputStream keysOutput = new FileOutputStream(new File(this.directory, KEYS_FILE_NAME_BASE + 0));
351                  final FileOutputStream valsOutput = new FileOutputStream(new File(this.directory, VALS_FILE_NAME_BASE + 0));
352                  final ArrayKVWriter arrayWriter = new ArrayKVWriter(indxOutput, keysOutput, valsOutput)) {
353                    arrayWriter.flush();                                                        // avoid compiler warning
354                    valsOutput.getChannel().force(false);
355                    keysOutput.getChannel().force(false);
356                    indxOutput.getChannel().force(false);
357                }
358
359                // Create generation file
360                try (FileOutputStream output = new FileOutputStream(this.generationFile)) {
361                    new PrintStream(output, true).println(0);
362                    output.getChannel().force(false);
363                }
364            }
365
366            // Read current generation number
367            try {
368                this.generation = Long.parseLong(
369                  new String(Files.readAllBytes(this.generationFile.toPath()), StandardCharsets.UTF_8).trim(), 10);
370                if (this.generation < 0)
371                    throw new ArrayKVException("read negative generation number from " + this.generationFile);
372            } catch (Exception e) {
373                throw new ArrayKVException("error reading generation file", e);
374            }
375
376            // Set corresponding filenames
377            this.indxFile = new File(this.directory, INDX_FILE_NAME_BASE + this.generation);
378            this.keysFile = new File(this.directory, KEYS_FILE_NAME_BASE + this.generation);
379            this.valsFile = new File(this.directory, VALS_FILE_NAME_BASE + this.generation);
380            this.modsFile = new File(this.directory, MODS_FILE_NAME_BASE + this.generation);
381
382            // Scan directory for unexpected files
383            final List<File> expectedFiles = Arrays.asList(this.lockFile, this.generationFile,
384              this.indxFile, this.keysFile, this.valsFile, this.modsFile);
385            for (File file : this.directory.listFiles()) {
386                if (!expectedFiles.contains(file))
387                    this.log.warn("ignoring unexpected file " + file.getName() + " in my database directory");
388            }
389
390            // Create buffers that wrap the index, keys, and values files
391            try (FileInputStream input = new FileInputStream(this.indxFile)) {
392                this.indx = AtomicArrayKVStore.getBuffer(this.indxFile, input.getChannel());
393            }
394            try (FileInputStream input = new FileInputStream(this.keysFile)) {
395                this.keys = AtomicArrayKVStore.getBuffer(this.keysFile, input.getChannel());
396            }
397            try (FileInputStream input = new FileInputStream(this.valsFile)) {
398                this.vals = AtomicArrayKVStore.getBuffer(this.valsFile, input.getChannel());
399            }
400
401            // Set up underlying k/v store and uncompacted modifications
402            this.kvstore = new ArrayKVStore(this.indx, this.keys, this.vals);
403            this.mods = new MutableView(this.kvstore, null, new Writes());
404
405            // Setup modifications file
406            this.modsFileOutput = new FileOutputStream(this.modsFile, true);
407            this.modsFileLength = this.modsFileOutput.getChannel().size();
408            this.modsFileSyncPoint = this.modsFileLength;
409
410            // Read and apply pre-existing uncompacted modifications from modifications file
411            if (this.modsFileLength > 0) {
412                this.log.info("reading " + this.modsFileLength + " bytes of uncompacted modifications from " + this.modsFile);
413                try (FileInputStream input = new FileInputStream(this.modsFile)) {
414                    while (input.available() > 0) {
415                        final Writes writes;
416                        try {
417                            writes = Writes.deserialize(input);
418                        } catch (Exception e) {
419                            break;                                                      // probably a partial write
420                        }
421                        writes.applyTo(this.mods);
422                    }
423                }
424                this.firstModTimestamp = System.nanoTime() | 1;                     // avoid zero value which is special
425            }
426
427            // Schedule compaction if necessary
428            this.scheduleCompactionIfNecessary();
429
430            // Done
431            success = true;
432        } catch (IOException e) {
433            throw new ArrayKVException("startup failed", e);
434        } finally {
435            try {
436                if (!success)
437                    this.cleanup();
438            } finally {
439                this.writeLock.unlock();
440            }
441        }
442    }

444    @Override
445    @PreDestroy
446    public void stop() {
447        this.writeLock.lock();
448        try {
449
450            // Check state
451            if (this.kvstore == null)
452                return;
453            this.log.info("stopping " + this);
454
455            // Cleanup
456            this.cleanup();
457        } finally {
458            this.writeLock.unlock();
459        }
460    }

462    private void cleanup() {
463
464        // Should hold write lock now
465        assert this.lock.isWriteLockedByCurrentThread();
466
467        // Cancel compaction (if any) or wait for it to complete
468        while (this.compactFuture != null) {
469            if (!this.compactFuture.cancel(false)) {
470                try {
471                    this.compactionFinishedCondition.await();
472                } catch (InterruptedException e) {
473                    this.log.warn("thread was interrupted while waiting for compaction to complete", e);
474                }
475            }
476            this.compactFuture = null;
477        }
478
479        // Shut down executor - only if we created it
480        if (this.createdExecutorService) {
481            this.scheduledExecutorService.shutdownNow();
482            this.scheduledExecutorService = null;
483            this.createdExecutorService = false;
484        }
485
486        // Close files
487        for (Closeable closeable : new Closeable[] { this.modsFileOutput, this.directoryChannel, this.lockFileChannel }) {
488            if (closeable != null) {
489                try {
490                    closeable.close();
491                } catch (IOException e) {
492                    // ignore
493                }
494            }
495        }
496
497        // Reset state
498        this.generation = 0;
499        this.generationFile = null;
500        this.lockFile = null;
501        this.lockFileChannel = null;
502        this.indxFile = null;
503        this.keysFile = null;
504        this.valsFile = null;
505        this.modsFile = null;
506        this.modsFileOutput = null;
507        this.directoryChannel = null;
508        this.modsFileLength = 0;
509        this.modsFileSyncPoint = 0;
510        this.indx = null;
511        this.keys = null;
512        this.vals = null;
513        this.kvstore = null;
514        this.mods = null;
515        this.firstModTimestamp = 0;
516    }

616    @Override
617    public CloseableKVStore snapshot() {
618        this.readLock.lock();
619        try {
620            Preconditions.checkState(this.kvstore != null, "closed");
621
622            // Get oustanding modifications; if we are compacting, there are two sets of modifications
623            final Writes writes1;
624            if (this.mods.getKVStore() instanceof MutableView) {
625                final MutableView oldMods = (MutableView)this.mods.getKVStore();
626                assert oldMods.getKVStore() == this.kvstore;
627                writes1 = oldMods.getWrites();
628            } else
629                writes1 = null;
630            final Writes writes2 = this.mods.getWrites();
631
632            // Clone (non-empty) writes to construct snapshot
633            KVStore snapshot = this.kvstore;
634            if (writes1 != null && !writes1.isEmpty())
635                snapshot = new MutableView(snapshot, null, writes1.clone());
636            if (writes2 != null && !writes2.isEmpty())
637                snapshot = new MutableView(snapshot, null, writes2.clone());
638
639            // Done
640            return new CloseableForwardingKVStore(snapshot);
641        } finally {
642            this.readLock.unlock();
643        }
644    }

646    @Override
647    public void mutate(Mutations mutations, final boolean sync) {
648        Preconditions.checkArgument(mutations != null, "null mutations");
649        this.writeLock.lock();
650        try {
651
652            // Verify we are started
653            Preconditions.checkState(this.kvstore != null, "not started");
654
655            // Block until compaction completes if size of outstanding modifications exceeds high water mark
656            while (this.compactFuture != null && this.modsFileLength > this.compactHighWater) {
657
658                // If scheduled compaction is in the future, reschedule to be immediate
659                if (this.compactFuture.getDelay(TimeUnit.MILLISECONDS) > 0)
660                    this.scheduleCompaction(0);
661
662                // Wait for it to complete
663                try {
664                    this.compactionFinishedCondition.await();
665                } catch (InterruptedException e) {
666                    throw new ArrayKVException("thread was interrupted while waiting for compaction to complete", e);
667                }
668            }
669
670            // Get mutations as a Writes object
671            final Writes writes;
672            if (mutations instanceof Writes)
673                writes = (Writes)mutations;
674            else {
675                writes = new Writes();
676                writes.setRemoves(new KeyRanges(mutations.getRemoveRanges()));
677                for (Map.Entry<byte[], byte[]> entry : mutations.getPutPairs())
678                    writes.getPuts().put(entry.getKey(), entry.getValue());
679                for (Map.Entry<byte[], Long> entry : mutations.getAdjustPairs())
680                    writes.getAdjusts().put(entry.getKey(), entry.getValue());
681            }
682
683            // Check for the trivial case where there are zero modifications
684            if (writes.isEmpty())
685                return;
686
687            // Append mutations to uncompacted mods file
688            try {
689                writes.serialize(this.modsFileOutput);
690                this.modsFileOutput.flush();
691            } catch (IOException e) {
692                try {
693                    this.modsFileOutput.getChannel().truncate(this.modsFileLength);              // undo append
694                } catch (IOException e2) {
695                    this.log.error("error truncating log file (ignoring)", e2);
696                }
697                throw new ArrayKVException("error appending to " + this.modsFile, e);
698            }
699            final long newModsFileLength;
700            try {
701                newModsFileLength = this.modsFileOutput.getChannel().size();
702            } catch (IOException e) {
703                throw new ArrayKVException("error getting length of " + this.modsFile, e);
704            }
705            if (this.log.isDebugEnabled()) {
706                this.log.debug("appended " + (newModsFileLength - this.modsFileLength) + " bytes to "
707                  + this.modsFile + " (new length " + newModsFileLength + ")");
708            }
709            this.modsFileLength = newModsFileLength;
710
711            // Apply removes
712            for (KeyRange range : mutations.getRemoveRanges()) {
713                final byte[] min = range.getMin();
714                final byte[] max = range.getMax();
715                this.mods.removeRange(min, max);
716            }
717
718            // Apply puts
719            for (Map.Entry<byte[], byte[]> entry : mutations.getPutPairs()) {
720                final byte[] key = entry.getKey();
721                final byte[] val = entry.getValue();
722                this.mods.put(key, val);
723            }
724
725            // Apply counter adjustments
726            for (Map.Entry<byte[], Long> entry : mutations.getAdjustPairs()) {
727                final byte[] key = entry.getKey();
728                final long adjust = entry.getValue();
729                this.mods.adjustCounter(key, adjust);
730            }
731
732            // Set first uncompacted modification timestamp, if not already set
733            if (this.firstModTimestamp == 0)
734                this.firstModTimestamp = System.nanoTime() | 1;                 // avoid zero value which is special
735
736            // Schedule compaction if necessary
737            this.scheduleCompactionIfNecessary();
738
739            // If we're not syncing, we're done
740            if (!sync)
741                return;
742
743            // Update sync point, so compaction knows to also sync these mods when it copies them
744            this.modsFileSyncPoint = this.modsFileLength;
745
746            // Downgrade lock
747            this.readLock.lock();
748        } finally {
749            this.writeLock.unlock();
750        }
751
752        // Sync the mods file while holding only the read lock
753        try {
754            this.modsFileOutput.getChannel().force(false);
755        } catch (IOException e) {
756            this.log.error("error syncing log file (ignoring)", e);
757        } finally {
758            this.readLock.unlock();
759        }
760    }

764    /**
765     * Schedule a new compaction cycle, unless there is one already scheduled or running, or there are no
766     * outstanding uncompacted modifications.
767     *
768     * @return a future for the completion of the next compaction cycle, or null if there are no uncompacted modifications
769     * @throws IllegalStateException if this instance is not started
770     */
771    public Future<?> scheduleCompaction() {
772        this.writeLock.lock();
773        try {
774
775            // Sanity check
776            Preconditions.checkState(this.kvstore != null, "not started");
777
778            // Is there anything to compact?
779            if (this.modsFileLength == 0)
780                return null;
781
782            // Schedule an immediate compaction
783            return this.scheduleCompaction(0);
784        } finally {
785            this.writeLock.unlock();
786        }
787    }
788
789    private ScheduledFuture<?> scheduleCompaction(long millis) {
790
791        // Should hold write lock now
792        assert this.lock.isWriteLockedByCurrentThread();
793
794        // Compare delay to existing future, if any
795        if (this.compactFuture != null) {
796
797            // If existing compation will happen sooner, nothing to do
798            final long existingMillis = this.compactFuture.getDelay(TimeUnit.MILLISECONDS);
799            if (existingMillis <= millis)
800                return this.compactFuture;
801
802            // Try to cancel existing compaction
803            if (!this.compactFuture.cancel(false))
804                return this.compactFuture;
805        }
806
807        // Schedule new compaction
808        this.compactFuture = this.scheduledExecutorService.schedule(new Runnable() {
809            @Override
810            public void run() {
811                try {
812                    AtomicArrayKVStore.this.compact();
813                } catch (Throwable t) {
814                    AtomicArrayKVStore.this.log.error("error during compaction", t);
815                }
816            }
817        }, millis, TimeUnit.MILLISECONDS);
818
819        // Done
820        return this.compactFuture;
821    }
822
823    private void scheduleCompactionIfNecessary() {
824
825        // Should hold write lock now
826        assert this.lock.isWriteLockedByCurrentThread();
827
828        // Is there anything to do?
829        if (this.modsFileLength == 0)
830            return;
831
832        // Check space low-water mark
833        if (this.modsFileLength > this.compactLowWater) {
834            this.scheduleCompaction(0);
835            return;
836        }
837
838        // Check maximum compaction delay
839        if (this.firstModTimestamp != 0) {
840            final long firstModAgeMillis = (System.nanoTime() - this.firstModTimestamp) / 1000000L;     // convert ns -> ms
841            this.scheduleCompaction(firstModAgeMillis);
842            return;
843        }
844    }

846    @SuppressWarnings("fallthrough")
847    private void compact() throws IOException {
848        final long compactionStartTime;
849        try {
850
851            // Snapshot (and wrap) pending modifications, and mark log file position
852            final Writes writesToCompact;
853            final long previousModsFileLength;
854            final long previousModsFileSyncPoint;
855            this.writeLock.lock();
856            try {
857
858                // Mark start time and get uncompacted modifications
859                assert this.modsFileLength > 0;
860                compactionStartTime = System.nanoTime();
861                writesToCompact = this.mods.getWrites();
862
863                // It's possible the uncompacted modifications are a no-op; if so, no compaction is necessary
864                if (writesToCompact.isEmpty()) {
865                    this.modsFileOutput.getChannel().truncate(0);
866                    this.modsFileLength = 0;
867                    this.modsFileSyncPoint = 0;
868                    this.modsFileOutput.getChannel().force(false);
869                    return;
870                }
871
872                // Allow new modifications to be added by other threads while we are compacting the old modifications
873                this.mods = new MutableView(this.mods, null, new Writes());
874                previousModsFileLength = this.modsFileLength;
875                previousModsFileSyncPoint = this.modsFileSyncPoint;
876            } finally {
877                this.writeLock.unlock();
878            }
879            if (this.log.isDebugEnabled()) {
880                this.log.debug("starting compaction for generation " + this.generation + " -> " + (this.generation + 1)
881                  + " with mods file length " + previousModsFileLength);
882            }
883
884            // Create the next generation
885            final long newGeneration = this.generation + 1;
886            final File newIndxFile = new File(this.directory, INDX_FILE_NAME_BASE + newGeneration);
887            final File newKeysFile = new File(this.directory, KEYS_FILE_NAME_BASE + newGeneration);
888            final File newValsFile = new File(this.directory, VALS_FILE_NAME_BASE + newGeneration);
889            final File newModsFile = new File(this.directory, MODS_FILE_NAME_BASE + newGeneration);
890            ByteBuffer newIndx = null;
891            ByteBuffer newKeys = null;
892            ByteBuffer newVals = null;
893            FileOutputStream newModsFileOutput = null;
894            boolean success = false;
895            try {
896
897                // Merge existing compacted key/value data with uncompacted modifications
898                try (
899                  final FileOutputStream indxOutput = new FileOutputStream(newIndxFile);
900                  final FileOutputStream keysOutput = new FileOutputStream(newKeysFile);
901                  final FileOutputStream valsOutput = new FileOutputStream(newValsFile);
902                  final ArrayKVWriter arrayWriter = new ArrayKVWriter(indxOutput, keysOutput, valsOutput)) {
903
904                    // Initialize iterators
905                    final Iterator<KVPair> kvIterator = this.kvstore.getRange(null, null, false);
906                    final Iterator<KeyRange> removeIterator = writesToCompact.getRemoves().iterator();
907                    final Iterator<Map.Entry<byte[], byte[]>> putIterator = writesToCompact.getPuts().entrySet().iterator();
908                    final Iterator<Map.Entry<byte[], Long>> adjustIterator = writesToCompact.getAdjusts().entrySet().iterator();
909
910                    // Merge iterators and write the merged result to the ArrayKVWriter
911                    KVPair kv = kvIterator.hasNext() ? kvIterator.next() : null;
912                    KeyRange remove = removeIterator.hasNext() ? removeIterator.next() : null;
913                    Map.Entry<byte[], byte[]> put = putIterator.hasNext() ? putIterator.next() : null;
914                    Map.Entry<byte[], Long> adjust = adjustIterator.hasNext() ? adjustIterator.next() : null;
915                    while (true) {
916
917                        // Find the minimum key among remaining (a) key/value pairs, (b) puts, and (c) adjusts
918                        byte[] key = null;
919                        if (kv != null)
920                            key = kv.getKey();
921                        if (put != null && (key == null || ByteUtil.compare(put.getKey(), key) < 0))
922                            key = put.getKey();
923                        if (adjust != null && (key == null || ByteUtil.compare(adjust.getKey(), key) < 0))
924                            key = adjust.getKey();
925                        if (key == null)                                                        // we're done
926                            break;
927
928                        // Determine which inputs apply to the current key
929                        int inputs = 0;
930                        if (kv != null && Arrays.equals(kv.getKey(), key))
931                            inputs |= MERGE_KV;
932                        if (put != null && Arrays.equals(put.getKey(), key))
933                            inputs |= MERGE_PUT;
934                        if (adjust != null && Arrays.equals(adjust.getKey(), key))
935                            inputs |= MERGE_ADJUST;
936
937                        // Find the removal that applies to this next key, if any
938                        while (remove != null) {
939                            final byte[] removeMax = remove.getMax();
940                            if (removeMax != null && ByteUtil.compare(removeMax, key) <= 0) {
941                                remove = removeIterator.hasNext() ? removeIterator.next() : null;
942                                continue;
943                            }
944                            break;
945                        }
946                        final boolean removed = remove != null && ByteUtil.compare(remove.getMin(), key) <= 0;
947
948                        // Merge the applicable inputs
949                        switch (inputs) {
950                        case 0:                                                     // k/v store entry was removed
951                            break;
952                        case MERGE_KV:
953                            if (!removed)
954                                arrayWriter.writeKV(kv.getKey(), kv.getValue());
955                            break;
956                        case MERGE_PUT:
957                        case MERGE_PUT | MERGE_KV:
958                            arrayWriter.writeKV(put.getKey(), put.getValue());
959                            break;
960                        case MERGE_ADJUST:                                          // adjusted a non-existent value; ignore
961                            break;
962                        case MERGE_ADJUST | MERGE_KV:
963                            if (removed)                                            // adjusted a removed value; ignore
964                                break;
965                            // FALLTHROUGH
966                        case MERGE_ADJUST | MERGE_PUT:
967                        case MERGE_ADJUST | MERGE_PUT | MERGE_KV:
968                        {
969                            final byte[] encodedCount = (inputs & MERGE_PUT) != 0 ? put.getValue() : kv.getValue();
970                            final long counter;
971                            try {
972                                counter = this.kvstore.decodeCounter(encodedCount);
973                            } catch (IllegalArgumentException e) {
974                                break;
975                            }
976                            final byte[] value = this.kvstore.encodeCounter(counter + adjust.getValue());
977                            arrayWriter.writeKV(key, value);
978                            break;
979                        }
980                        default:
981                            throw new RuntimeException();
982                        }
983
984                        // Advance k/v, put, and/or adjust
985                        if ((inputs & MERGE_KV) != 0)
986                            kv = kvIterator.hasNext() ? kvIterator.next() : null;
987                        if ((inputs & MERGE_PUT) != 0)
988                            put = putIterator.hasNext() ? putIterator.next() : null;
989                        if ((inputs & MERGE_ADJUST) != 0)
990                            adjust = adjustIterator.hasNext() ? adjustIterator.next() : null;
991                    }
992
993                    // Sync file data
994                    arrayWriter.flush();
995                    valsOutput.getChannel().force(false);
996                    keysOutput.getChannel().force(false);
997                    indxOutput.getChannel().force(false);
998                }
999                assert newIndxFile.exists();
1000                assert newKeysFile.exists();
1001                assert newValsFile.exists();
1002
1003                // Create byte buffers from new files
1004                try (FileInputStream input = new FileInputStream(newIndxFile)) {
1005                    newIndx = AtomicArrayKVStore.getBuffer(newIndxFile, input.getChannel());
1006                }
1007                try (FileInputStream input = new FileInputStream(newKeysFile)) {
1008                    newKeys = AtomicArrayKVStore.getBuffer(newKeysFile, input.getChannel());
1009                }
1010                try (FileInputStream input = new FileInputStream(newValsFile)) {
1011                    newVals = AtomicArrayKVStore.getBuffer(newValsFile, input.getChannel());
1012                }
1013
1014                // Create new, empty mods file
1015                newModsFile.delete();                                           // shouldn't exist, but just in case...
1016                newModsFileOutput = new FileOutputStream(newModsFile, true);
1017                assert newModsFile.exists();
1018
1019                // Sync directory
1020                this.directoryChannel.force(false);
1021
1022                // We're done creating files
1023                success = true;
1024            } finally {
1025
1026                // Finish up
1027                this.writeLock.lock();
1028                try {
1029                    if (success) {
1030
1031                        // Initialize for wrap-up
1032                        success = false;
1033
1034                        // Size up additional mods
1035                        final long additionalModsLength = this.modsFileLength - previousModsFileLength;
1036
1037                        // Logit
1038                        if (this.log.isDebugEnabled()) {
1039                            final float duration = (System.nanoTime() - compactionStartTime) / 1000000000f;
1040                            this.log.debug("compaction for generation " + this.generation + " -> " + (this.generation + 1)
1041                              + " finishing up with " + additionalModsLength + " bytes of new modifications after "
1042                              + String.format("%.4s", duration) + " seconds");
1043                        }
1044
1045                        // Apply any changes that were made while we were unlocked and writing files
1046                        long newModsFileLength = 0;
1047                        long newModsFileSyncPoint = 0;
1048                        if (additionalModsLength > 0) {
1049                            try (
1050                              final FileChannel modsFileChannel = FileChannel.open(this.modsFile.toPath());
1051                              final FileChannel newModsFileChannel = FileChannel.open(newModsFile.toPath(),
1052                               StandardOpenOption.WRITE, StandardOpenOption.APPEND)) {
1053
1054                                // Append new data in old mods file to new mods file
1055                                while (newModsFileLength < additionalModsLength) {
1056                                    final long transferred = modsFileChannel.transferTo(
1057                                      previousModsFileLength + newModsFileLength, additionalModsLength - newModsFileLength,
1058                                      newModsFileChannel);
1059                                    assert transferred > 0;
1060                                    newModsFileLength += transferred;
1061                                }
1062
1063                                // If any of that new data was fsync()'d, we must fsync() the copy of it in the new mods file
1064                                if (this.modsFileSyncPoint > previousModsFileSyncPoint) {
1065                                    newModsFileChannel.force(false);
1066                                    newModsFileSyncPoint = newModsFileLength;
1067                                }
1068                            }
1069                        }
1070
1071                        // Atomically update new generation file contents
1072                        final AtomicUpdateFileOutputStream genOutput = new AtomicUpdateFileOutputStream(this.generationFile);
1073                        boolean genSuccess = false;
1074                        try {
1075                            new PrintStream(genOutput, true).println(newGeneration);
1076                            genOutput.getChannel().force(false);
1077                            genSuccess = true;
1078                        } finally {
1079                            if (genSuccess)
1080                                genOutput.close();
1081                            else
1082                                genOutput.cancel();
1083                        }
1084
1085                        // Declare success
1086                        success = true;
1087
1088                        // Remember old info so we can clean it up
1089                        final File oldIndxFile = this.indxFile;
1090                        final File oldKeysFile = this.keysFile;
1091                        final File oldValsFile = this.valsFile;
1092                        final File oldModsFile = this.modsFile;
1093                        final FileOutputStream oldModsFileOutput = this.modsFileOutput;
1094
1095                        // Change to the new generation
1096                        this.generation = newGeneration;
1097                        this.indx = newIndx;
1098                        this.keys = newKeys;
1099                        this.vals = newVals;
1100                        this.indxFile = newIndxFile;
1101                        this.keysFile = newKeysFile;
1102                        this.valsFile = newValsFile;
1103                        this.modsFile = newModsFile;
1104                        this.modsFileOutput = newModsFileOutput;
1105                        this.modsFileLength = newModsFileLength;
1106                        this.modsFileSyncPoint = newModsFileSyncPoint;
1107                        this.kvstore = new ArrayKVStore(this.indx, this.keys, this.vals);
1108                        this.mods = new MutableView(this.kvstore, null, this.mods.getWrites());
1109
1110                        // Sync directory prior to deleting files
1111                        try {
1112                            this.directoryChannel.force(false);
1113                        } catch (IOException e) {
1114                            this.log.error("error syncing directory " + this.directory + " (ignoring)", e);
1115                        }
1116
1117                        // Delete old files
1118                        oldIndxFile.delete();
1119                        oldKeysFile.delete();
1120                        oldValsFile.delete();
1121                        oldModsFile.delete();
1122
1123                        // Close old mods file output stream
1124                        try {
1125                            oldModsFileOutput.close();
1126                        } catch (IOException e) {
1127                            // ignore
1128                        }
1129                    }
1130                } finally {
1131                    try {
1132
1133                        // Logit
1134                        if (this.log.isDebugEnabled()) {
1135                            final float duration = (System.nanoTime() - compactionStartTime) / 1000000000f;
1136                            this.log.debug("compaction for generation " + (newGeneration - 1) + " -> " + newGeneration
1137                              + (success ? " succeeded" : " failed") + " in " + String.format("%.4s", duration) + " seconds");
1138                        }
1139
1140                        // Cleanup/undo on failure
1141                        if (!success) {
1142
1143                            // Put back the old uncompacted modifications, and merge any new mods into them
1144                            final Writes writesDuringCompaction = this.mods.getWrites();
1145                            this.mods = new MutableView(this.kvstore, null, writesToCompact);
1146                            writesDuringCompaction.applyTo(this.mods);
1147
1148                            // Delete the files we were creating
1149                            newIndxFile.delete();
1150                            newKeysFile.delete();
1151                            newValsFile.delete();
1152                            if (newModsFileOutput != null) {
1153                                newModsFile.delete();
1154                                try {
1155                                    newModsFileOutput.close();
1156                                } catch (Exception e) {
1157                                    // ignore
1158                                }
1159                            }
1160                        }
1161                    } finally {
1162                        this.writeLock.unlock();
1163                    }
1164                }
1165            }
1166        } finally {
1167            this.compactFuture = null;
1168            this.writeLock.lock();
1169            try {
1170                this.compactionFinishedCondition.signalAll();
1171            } finally {
1172                this.writeLock.unlock();
1173            }
1174        }
1175    }
