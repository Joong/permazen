096    void setState(TxState state) {
097        assert state != null;
098        synchronized (this.kvdb) {
099            assert state.compareTo(this.state) >= 0;
100            if (this.state.equals(TxState.EXECUTING) && !this.state.equals(state))
101                this.snapshot.close();
102            this.state = state;
103        }
104    }

158    /**
159     * Configure whether to reduce the consistency guarantees for this transaction from linearizable consistency
160     * to stale read consistency.
161     *
162     * <p>
163     * If so configured, this transaction guarantees a consistent, read-only view of the database as it existed
164     * at some point in the "recent past". The view is not guaranteed to be up-to-date; it's only guaranteed
165     * to be as up-to-date as is known to this node when the transaction was opened. For example, it will be
166     * at least as up to date as the most recently committed normal (linearizable) transaction on this node.
167     * However, in general (for example, if stuck in a network partition minority), the snapshot could be arbitrarily
168     * far in the past.
169     *
170     * <p>
171     * This setting may be modified freely during a transaction; it only determines behavior at {@link #commit} time.
172     * If set:
173     * <ul>
174     *  <li>The transaction will be read-only: modifications will be allowed (and reflected in subsequent reads),
175     *      but they will be discarded on {@link #commit}.</li>
176     *  <li>The {@link #commit} operation may be faster, and will not generate any additional network traffic.</li>
177     *  <li>The {@link #commit} operation will still wait for the transaction's base log entry to be committed,
178     *      which eliminates the possibility of reading uncommitted data. If the possibility of reading uncommitted data
179     *      is tolerable, then any {@link #commit} wait can be avoided simply by invoking {@link #rollback} instead.</li>
180     * </ul>
181     *
182     * @param staleReadOnly true to configure stale reads, false for normal linearizable ACID semantics
183     * @see <a href="https://aphyr.com/posts/313-strong-consistency-models">Strong consistency models</a>
184     */
185    public void setStaleReadOnly(boolean staleReadOnly) {
186        this.staleReadOnly = staleReadOnly;
187    }

195    public boolean isStaleReadOnly() {

201    /**
202     * Include a cluster configuration change when this transaction is committed.
203     *
204     * <p>
205     * The change will have been applied once this transaction is successfully committed.
206     *
207     * <p>
208     * Raft supports configuration changes that add or remove one node at a time to/from the cluster.
209     * If this method is invoked more than once in a single transaction, all but the last invocation are ignored.
210     *
211     * <p>
212     * Initially, nodes are <i>unconfigured</i>. An unconfigured node becomes configured in one of two ways:
213     * <ul>
214     *  <li>By receiving a message from a leader of some existing cluster, in which case the node joins that cluster
215     *      based on the provided configuration; or</li>
216     *  <li>By this method being invoked with {@code identity} equal to this node's identity and a non-null {@code address},
217     *      which creates a new cluster and adds this node to it.</li>
218     * </ul>
219     *
220     * <p>
221     * Therefore, this method must be used to intialize a new cluster.
222     *
223     * @param identity the identity of the node to add or remove
224     * @param address the network address of the node if adding, or null if removing
225     * @throws IllegalStateException if this method has been invoked previously on this instance
226     * @throws IllegalArgumentException if {@code identity} is null
227     */
228    public void configChange(String identity, String address) {
229        Preconditions.checkArgument(identity != null, "null identity");
230        synchronized (this.kvdb) {
231            Preconditions.checkState(this.configChange == null, "duplicate config chagne; only one is supported per transaction");
232            if (!this.state.equals(TxState.EXECUTING))
233                throw new StaleTransactionException(this);
234            this.configChange = new String[] { identity, address };
235        }
236    }
237
238    /**
239     * Get the cluster configuration change associated with this transaction, if any.
240     *
241     * <p>
242     * The returned array has length two and contains the {@code identity} and {@code address}
243     * parameters passed to {@link #configChange configChange()}.
244     *
245     * <p>
246     * The returned array is a copy; changes have no effect on this instance.
247     *
248     * @return cluster config change, or null if there is none
249     */
250    public String[] getConfigChange() {
251        synchronized (this.kvdb) {
252            return this.configChange != null ? this.configChange.clone() : null;
253        }
254    }

272    /**
273     * Set the commit timeout for this instance.
274     *
275     * <p>
276     * {@link RaftKVTransaction}s do not block while the transaction is open; the configured value is used
277     * as a timeout for the {@link #commit} operation only. If {@link #commit} takes longer than {@code timeout}
278     * milliseconds, a {@link org.jsimpledb.kv.RetryTransactionException} is thrown.
279     *
280     * <p>
281     * The default value for all transactions is configured by {@link RaftKVDatabase#setCommitTimeout}.
282     *
283     * @param timeout transaction commit timeout in milliseconds, or zero for unlimited
284     * @throws IllegalArgumentException if {@code timeout} is negative
285     */
286    @Override
287    public void setTimeout(long timeout) {
288        Preconditions.checkArgument(timeout >= 0, "timeout < 0");
289        this.timeout = (int)Math.min(timeout, Integer.MAX_VALUE);
290    }

325    @Override
326    public String toString() {
327        return this.getClass().getSimpleName()
328          + "[txId=" + this.txId
329          + ",state=" + this.state
330          + ",base=" + this.baseIndex + "t" + this.baseTerm
331          + (this.staleReadOnly ? ",staleReadOnly" : "")
332          + (this.configChange != null ? ",configChange=" + Arrays.<String>asList(this.configChange) : "")
333          + (this.state.compareTo(TxState.COMMIT_WAITING) >= 0 ? ",commit=" + this.commitIndex + "t" + this.commitTerm : "")
334          + (this.timeout != 0 ? ",timeout=" + this.timeout : "")
335          + "]";
336    }

338    @Override
339    protected void finalize() throws Throwable {
340        try {
341            synchronized (this.kvdb) {
342                if (!TxState.CLOSED.equals(this.state)) {
343                    this.log.warn(this + " leaked without commit() or rollback()");
344                    this.rollback();
345                }
346            }
347        } finally {
348            super.finalize();
349        }
350    }

352// CommitFuture

354    static class CommitFuture extends FutureTask<Void> {
355
356        private final ExceptionCallable callable;
357
358        public CommitFuture() {
359            this(new ExceptionCallable());
360        }
361
362        CommitFuture(ExceptionCallable callable) {
363            super(callable);
364            this.callable = callable;
365        }
366
367        public void succeed() {
368            Preconditions.checkState(!this.isDone(), "already done");
369            this.run();
370        }
371
372        public void fail(Exception e) {
373            Preconditions.checkState(!this.isDone(), "already done");
374            this.callable.setException(e);
375            this.run();
376        }
377    }
378
379    private static class ExceptionCallable implements Callable<Void> {
380
381        private volatile Exception e;
382
383        public void setException(Exception e) {
384            this.e = e;
385        }
386
387        @Override
388        public Void call() throws Exception {
389            if (this.e != null)
390                throw this.e;
391            return null;
392        }
393    }

395// Debug/Sanity Checking

397    void checkStateOpen(long currentTerm, long lastIndex, long raftCommitIndex) {
398        assert !this.commitFuture.isCancelled();
399        assert this.commitFuture.isDone() == this.state.compareTo(TxState.COMPLETED) >= 0;
400        assert this.baseTerm <= currentTerm;
401        switch (this.state) {
402        case EXECUTING:
403        case COMMIT_READY:
404            assert this.commitTerm == 0;
405            assert this.commitIndex == 0;
406            break;
407        case COMMIT_WAITING:
408            assert this.commitTerm >= this.baseTerm;
409            assert this.commitTerm <= currentTerm;
410            assert this.commitIndex >= this.baseIndex;                                      // equal implies a read-only tx
411            assert this.commitIndex > this.baseIndex || this.view.getWrites().isEmpty();
412            break;
413        case COMPLETED:
414            assert this.commitFuture.isDone();
415            assert this.commitTerm == 0 || this.commitTerm >= this.baseTerm;
416            assert this.commitIndex == 0 || this.commitIndex >= this.baseIndex;
417            assert this.commitTerm <= currentTerm;
418            break;
419        case CLOSED:
420        default:
421            assert false;
422            break;
423        }
424    }
