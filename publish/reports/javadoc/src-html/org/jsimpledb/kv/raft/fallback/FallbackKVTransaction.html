006package org.jsimpledb.kv.raft.fallback;
007
008import java.util.concurrent.Future;
009
010import org.jsimpledb.kv.CloseableKVStore;
011import org.jsimpledb.kv.KVDatabase;
012import org.jsimpledb.kv.KVStore;
013import org.jsimpledb.kv.KVTransaction;
014import org.jsimpledb.kv.RetryTransactionException;
015import org.jsimpledb.kv.StaleTransactionException;
016import org.jsimpledb.kv.util.ForwardingKVStore;
017
018/**
019 * A {@link KVTransaction} associated with a {@link FallbackKVDatabase}.
020 *
021 * @see FallbackKVDatabase
022 */
023public class FallbackKVTransaction extends ForwardingKVStore implements KVTransaction {
024
025    private final FallbackKVDatabase db;
026    private final KVTransaction kvt;
027    private final int migrationCount;
028
029    private boolean stale;              // protected by this.db's monitor
030
031    FallbackKVTransaction(FallbackKVDatabase db, KVTransaction kvt, int migrationCount) {
032        this.db = db;
033        this.kvt = kvt;
034        this.migrationCount = migrationCount;
035    }
036
037// ForwardingKVStore
038
039    /**
040     * Get the underlying transaction, which will be associated with either one of the fallback
041     * databases, or the standalone mode database.
042     *
043     * @return underlying transaction
044     */
045    public KVTransaction getKVTransaction() {
046        return this.kvt;
047    }
048
049    @Override
050    protected KVStore delegate() {
051        return this.kvt;
052    }
053
054// KVTransaction
055
056    @Override
057    public KVDatabase getKVDatabase() {
058        return this.db;
059    }
060
061    @Override
062    public void setTimeout(long timeout) {
063        this.kvt.setTimeout(timeout);
064    }
065
066    @Override
067    public Future<Void> watchKey(byte[] key) {
068        return this.kvt.watchKey(key);
069    }
070
071    @Override
072    public CloseableKVStore mutableSnapshot() {
073        return this.kvt.mutableSnapshot();
074    }
075
076    @Override
077    public void commit() {
078        synchronized (this.db) {
079            if (this.stale)
080                throw new StaleTransactionException(this);
081            this.stale = true;
082        }
083        this.retryIfMigrating();
084        this.kvt.commit();
085        this.retryIfMigrating();    // required to close a window where database would appear to revert to an earlier version
086    }
087
088    @Override
089    public void rollback() {
090        synchronized (this.db) {
091            this.stale = true;
092            this.kvt.rollback();
093        }
094    }
095
096    private void retryIfMigrating() {
097        synchronized (this.db) {
098            if (this.db.isMigrating() || this.migrationCount != this.db.getMigrationCount())
099                throw new RetryTransactionException(this, "fallback migration in progress");
100        }
101    }
102}
