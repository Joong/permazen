038/**
039 * Ant task for schema XML generation and/or verification.
040 *
041 * <p>
042 * This task scans the configured classpath for classes with {@link org.jsimpledb.annotation.JSimpleClass &#64;JSimpleClass}
043 * and {@link org.jsimpledb.annotation.JFieldType &#64;JFieldType} annotations and either writes the generated schema
044 * to an XML file, or verifies the schema matches an existing XML file.
045 *
046 * <p>
047 * Generation of schema XML files and the use of this task is not necessary. However, it does allow certain
048 * schema-related problems to be detected at build time instead of runtime. In particular, it can let you know
049 * if any change to your model classes requires a new JSimpleDB schema version.
050 *
051 * <p>
052 * This task can also check for conflicts between the schema in question and older schema versions that may still
053 * exist in production databases. These other schema versions are specified using nested {@code <oldschemas>}
054 * elements, which work just like {@code <fileset>}'s.
055 *
056 * <p>
057 * The following attributes are supported by this task:
058 *
059 * <div style="margin-left: 20px;">
060 * <table border="1" cellpadding="3" cellspacing="0" summary="Supported Tasks">
061 * <tr style="bgcolor:#ccffcc">
062 *  <th align="left">Attribute</th>
063 *  <th align="left">Required?</th>
064 *  <th align="left">Description</th>
065 * </tr>
066 * <tr>
067 *  <td>{@code mode}</td>
068 *  <td>No</td>
069 *  <td>
070 *      <p>
071 *      Set to {@code generate} to generate a new XML file, or {@code verify} to verify an existing XML file.
072 *      </p>
073 *
074 *      <p>
075 *      Default is {@code verify}.
076 *      </p>
077 * </td>
078 * </tr>
079 * <tr>
080 *  <td>{@code file}</td>
081 *  <td>Yes</td>
082 *  <td>
083 *      <p>
084 *      The XML file to generate or verify.
085 *      </p>
086 * </td>
087 * </tr>
088 * <tr>
089 *  <td>{@code matchNames}</td>
090 *  <td>No</td>
091 *  <td>
092 *      <p>
093 *      Whether to verify not only schema compatibility but also that the two schemas are identical, i.e.,
094 *      the same names are used for object types, fields, and composite indexes.
095 *      </p>
096 *
097 *      <p>
098 *      Two schemas that are equivalent except for names are considered compatible, because the core API uses
099 *      storage ID's, not names. However, if names change then some JSimpleDB layer operations, such as index
100 *      queries and reference path inversion, may need to be updated.
101 *      </p>
102 *
103 *      <p>
104 *      Default is {@code true}. Ignored unless {@code mode} is {@code verify}.
105 *      </p>
106 * </td>
107 * </tr>
108 * <tr>
109 *  <td>{@code failOnError}</td>
110 *  <td>No</td>
111 *  <td>
112 *      <p>
113 *      Whether to fail if verification fails when {@code mode="verify"} or when older schema
114 *      versions are specified using nested {@code <oldschemas>} elements, which work just like
115 *      {@code <fileset>}s.
116 *      </p>
117 *
118 *      <p>
119 *      Default is {@code true}.
120 *      </p>
121 * </td>
122 * </tr>
123 * <tr>
124 *  <td>{@code verifiedProperty}</td>
125 *  <td>No</td>
126 *  <td>
127 *      <p>
128 *      The name of an ant property to set to {@code true} or {@code false} depending on whether
129 *      verification succeeded or failed. Useful when {@code failOnError} is set to {@code false}
130 *      and you want to handle the failure elsewhere in the build file.
131 *      </p>
132 *
133 *      <p>
134 *      Default is to not set any property.
135 *      </p>
136 * </td>
137 * </tr>
138 * <tr>
139 *  <td>{@code classpath} or {@code classpathref}</td>
140 *  <td>Yes</td>
141 *  <td>
142 *      <p>
143 *      Specifies the search path containing classes with {@link org.jsimpledb.annotation.JSimpleClass &#64;JSimpleClass}
144 *      and {@link org.jsimpledb.annotation.JFieldType &#64;JFieldType} annotations.
145 *      </p>
146 * </td>
147 * </tr>
148 * <tr>
149 *  <td>{@code packages}</td>
150 *  <td>Yes, unless {@code classes} are specified</td>
151 *  <td>
152 *      <p>
153 *      Specifies one or more Java package names (separated by commas and/or whitespace) under which to look
154 *      for classes with {@link org.jsimpledb.annotation.JSimpleClass &#64;JSimpleClass}
155 *      or {@link org.jsimpledb.annotation.JFieldType &#64;JFieldType} annotations.
156 *
157 *      <p>
158 *      Use of this attribute requires Spring's classpath scanning classes ({@code spring-context.jar});
159 *      these must be on the {@code <taskdef>} classpath.
160 *      </p>
161 * </td>
162 * </tr>
163 * <tr>
164 *  <td>{@code classes}</td>
165 *  <td>Yes, unless {@code packages} are specified</td>
166 *  <td>
167 *      <p>
168 *      Specifies one or more Java class names (separated by commas and/or whitespace) of
169 *      classes with {@link org.jsimpledb.annotation.JSimpleClass &#64;JSimpleClass}
170 *      or {@link org.jsimpledb.annotation.JFieldType &#64;JFieldType} annotations.
171 *      </p>
172 * </td>
173 * </tr>
174 * <tr>
175 *  <td>{@code storageIdGeneratorClass}</td>
176 *  <td>No</td>
177 *  <td>
178 *      <p>
179 *      Specifies the name of an optional custom {@link StorageIdGenerator} class.
180 *      </p>
181 *
182 *      <p>
183 *      By default, a {@link DefaultStorageIdGenerator} is used.
184 *      </p>
185 * </td>
186 * </tr>
187 * </table>
188 * </div>
189 *
190 * <p>
191 * Classes are found by scanning the packages listed in the {@code "packages"} attribute.
192 * Alternatively, or in addition, specific classes may specified using the {@code "classes"} attribute.
193 *
194 * <p>
195 * To install this task into ant:
196 *
197 * <pre>
198 *  &lt;project xmlns:jsimpledb="urn:org.jsimpledb.ant" ... &gt;
199 *      ...
200 *      &lt;taskdef uri="urn:org.jsimpledb.ant" name="schema"
201 *        classname="org.jsimpledb.ant.SchemaGeneratorTask" classpathref="jsimpledb.classpath"/&gt;
202 * </pre>
203 *
204 * <p>
205 * Example of generating a schema XML file that corresponds to the specified Java model classes:
206 *
207 * <pre>
208 *  &lt;jsimpledb:schema mode="generate" classpathref="myclasses.classpath"
209 *    file="schema.xml" packages="com.example.model"/&gt;
210 * </pre>
211 *
212 * <p>
213 * Example of verifying that the schema generated from the Java model classes has not
214 * changed incompatibly (i.e., in a way that would require a new schema version):
215 *
216 * <pre>
217 *  &lt;jsimpledb:schema mode="verify" classpathref="myclasses.classpath"
218 *    file="expected-schema.xml" packages="com.example.model"/&gt;
219 * </pre>
220 *
221 * <p>
222 * Example of doing the same thing, and also verifying the generated schema is compatible with prior schema versions
223 * that may still be in use in production databases:
224 *
225 * <pre>
226 *  &lt;jsimpledb:schema mode="verify" classpathref="myclasses.classpath"
227 *    file="expected-schema.xml" packages="com.example.model"&gt;
228 *      &lt;jsimpledb:oldschemas dir="obsolete-schemas" includes="*.xml"/&gt;
229 *  &lt;/jsimpledb:schema&gt;
230 * </pre>
231 *
232 * @see org.jsimpledb.JSimpleDB
233 * @see org.jsimpledb.schema.SchemaModel
234 */
235public class SchemaGeneratorTask extends Task {
236
237    public static final String MODE_VERIFY = "verify";
238    public static final String MODE_GENERATE = "generate";
239

241    private boolean matchNames = true;
242    private boolean failOnError = true;
243    private String verifiedProperty;
244    private File file;
245    private Path classPath;
246    private String storageIdGeneratorClassName = DefaultStorageIdGenerator.class.getName();
247    private final ArrayList<OldSchemas> oldSchemasList = new ArrayList<>();
248    private final LinkedHashSet<String> classes = new LinkedHashSet<>();
249    private final LinkedHashSet<String> packages = new LinkedHashSet<>();

300    /**
301     * @throws BuildException if operation fails
302     */
303    @Override
304    public void execute() {
305
306        // Sanity check
307        if (this.file == null)
308            throw new BuildException("`file' attribute is required specifying output/verify file");
309        final boolean generate;
310        switch (this.mode) {
311        case MODE_VERIFY:
312            generate = false;
313            break;
314        case MODE_GENERATE:
315            generate = true;
316            break;
317        default:
318            throw new BuildException("`mode' attribute must be one of `" + MODE_VERIFY + "' or `" + MODE_GENERATE + "'");
319        }
320        if (this.packages == null)
321            throw new BuildException("`packages' attribute is required specifying packages to scan for Java model classes");
322        if (this.classPath == null)
323            throw new BuildException("`classpath' attribute is required specifying search path for scanned classes");
324
325        // Create directory containing file
326        if (generate && this.file.getParent() != null && !this.file.getParentFile().exists() && !this.file.getParentFile().mkdirs())
327            throw new BuildException("error creating directory `" + this.file.getParentFile() + "'");
328
329        // Set up mysterious classloader stuff
330        final AntClassLoader loader = this.getProject().createClassLoader(this.classPath);
331        final ClassLoader currentLoader = this.getClass().getClassLoader();
332        if (currentLoader != null)
333            loader.setParent(currentLoader);
334        loader.setThreadContextLoader();
335        try {
336
337            // Model and field type classes
338            final HashSet<Class<?>> modelClasses = new HashSet<>();
339            final HashSet<Class<?>> fieldTypeClasses = new HashSet<>();
340
341            // Do package scanning
342            if (!this.packages.isEmpty()) {
343
344                // Join list
345                final StringBuilder buf = new StringBuilder();
346                for (String packageName : this.packages) {
347                    if (buf.length() > 0)
348                        buf.append(' ');
349                    buf.append(packageName);
350                }
351                final String packageNames = buf.toString();
352
353                // Scan for @JSimpleClass classes
354                this.log("scanning for @JSimpleClass annotations in packages: " + packageNames);
355                for (String className : new JSimpleDBClassScanner().scanForClasses(packageNames)) {
356                    this.log("adding JSimpleDB schema class " + className);
357                    try {
358                        modelClasses.add(Class.forName(className, false, Thread.currentThread().getContextClassLoader()));
359                    } catch (ClassNotFoundException e) {
360                        throw new BuildException("failed to load class `" + className + "'", e);
361                    }
362                }
363
364                // Scan for @JFieldType classes
365                this.log("scanning for @JFieldType annotations in packages: " + packageNames);
366                for (String className : new JSimpleDBFieldTypeScanner().scanForClasses(packageNames)) {
367                    this.log("adding JSimpleDB field type class `" + className + "'");
368                    try {
369                        fieldTypeClasses.add(Class.forName(className, false, Thread.currentThread().getContextClassLoader()));
370                    } catch (Exception e) {
371                        throw new BuildException("failed to instantiate " + className, e);
372                    }
373                }
374            }
375
376            // Do specific class scanning
377            for (String className : this.classes) {
378
379                // Load class
380                final Class<?> cl;
381                try {
382                    cl = Class.forName(className, false, Thread.currentThread().getContextClassLoader());
383                } catch (ClassNotFoundException e) {
384                    throw new BuildException("failed to load class `" + className + "'", e);
385                }
386
387                // Add model classes
388                if (cl.isAnnotationPresent(JSimpleClass.class)) {
389                    this.log("adding JSimpleDB model " + cl);
390                    modelClasses.add(cl);
391                }
392
393                // Add field types
394                if (cl.isAnnotationPresent(JFieldType.class)) {
395                    this.log("adding JSimpleDB field type " + cl);
396                    fieldTypeClasses.add(cl);
397                }
398            }
399
400            // Instantiate StorageIdGenerator
401            final StorageIdGenerator storageIdGenerator;
402            try {
403                storageIdGenerator = Class.forName(this.storageIdGeneratorClassName,
404                  false, Thread.currentThread().getContextClassLoader()).asSubclass(StorageIdGenerator.class).newInstance();
405            } catch (Exception e) {
406                throw new BuildException("failed to instantiate class `" + storageIdGeneratorClassName + "'", e);
407            }
408
409            // Set up database
410            final Database db = new Database(new SimpleKVDatabase());
411
412            // Instantiate and configure field type classes
413            for (Class<?> cl : fieldTypeClasses) {
414
415                // Instantiate field types
416                this.log("instantiating " + cl + " as field type instance");
417                final FieldType<?> fieldType;
418                try {
419                    fieldType = this.asFieldTypeClass(cl).newInstance();
420                } catch (Exception e) {
421                    throw new BuildException("failed to instantiate " + cl.getName(), e);
422                }
423
424                // Add field type
425                try {
426                    db.getFieldTypeRegistry().add(fieldType);
427                } catch (Exception e) {
428                    throw new BuildException("failed to register custom field type " + cl.getName(), e);
429                }
430            }
431
432            // Set up factory
433            final JSimpleDBFactory factory = new JSimpleDBFactory();
434            factory.setDatabase(db);
435            factory.setSchemaVersion(1);
436            factory.setStorageIdGenerator(storageIdGenerator);
437            factory.setModelClasses(modelClasses);
438
439            // Build schema model
440            this.log("generating JSimpleDB schema from schema classes");
441            final SchemaModel schemaModel;
442            try {
443                schemaModel = factory.newJSimpleDB().getSchemaModel();
444            } catch (Exception e) {
445                throw new BuildException("schema generation failed: " + e, e);
446            }
447
448            // Record schema model in database
449            db.createTransaction(schemaModel, 1, true).commit();
450
451            // Verify or generate
452            boolean verified = true;
453            if (generate) {
454
455                // Write schema model to file
456                this.log("writing JSimpleDB schema to `" + this.file + "'");
457                try (BufferedOutputStream output = new BufferedOutputStream(new FileOutputStream(this.file))) {
458                    schemaModel.toXML(output, true);
459                } catch (IOException e) {
460                    throw new BuildException("error writing schema to `" + this.file + "': " + e, e);
461                }
462            } else {
463
464                // Read file
465                this.log("verifying JSimpleDB schema matches `" + this.file + "'");
466                final SchemaModel verifyModel;
467                try (BufferedInputStream input = new BufferedInputStream(new FileInputStream(this.file))) {
468                    verifyModel = SchemaModel.fromXML(input);
469                } catch (IOException e) {
470                    throw new BuildException("error reading schema from `" + this.file + "': " + e, e);
471                }
472
473                // Compare
474                final boolean matched = matchNames ? schemaModel.equals(verifyModel) : schemaModel.isCompatibleWith(verifyModel);
475                if (!matched)
476                    verified = false;
477                this.log("schema verification " + (matched ? "succeeded" : "failed"));
478                if (!matched)
479                    this.log(schemaModel.differencesFrom(verifyModel).toString());
480            }
481
482            // Check for conflicts with other schema versions
483            if (verified) {
484                int schemaVersion = 2;
485                for (OldSchemas oldSchemas : this.oldSchemasList) {
486                    for (Iterator<?> i = oldSchemas.iterator(); i.hasNext(); ) {
487                        final Resource resource = (Resource)i.next();
488                        this.log("checking schema for conflicts with " + resource);
489                        final SchemaModel otherSchema;
490                        try (BufferedInputStream input = new BufferedInputStream(resource.getInputStream())) {
491                            otherSchema = SchemaModel.fromXML(input);
492                        } catch (IOException e) {
493                            throw new BuildException("error reading schema from `" + resource + "': " + e, e);
494                        }
495                        try {
496                            db.createTransaction(otherSchema, schemaVersion++, true).commit();
497                        } catch (Exception e) {
498                            this.log("schema conflicts with " + resource + ": " + e);
499                            verified = false;
500                        }
501                    }
502                }
503            }
504
505            // Check verification results
506            if (this.verifiedProperty != null)
507                this.getProject().setProperty(this.verifiedProperty, "" + verified);
508            if (!verified && this.failOnError)
509                throw new BuildException("schema verification failed");
510        } finally {
511            loader.resetThreadContextLoader();
512            loader.cleanup();
513        }
514    }
515
516    @SuppressWarnings("unchecked")
517    private Class<? extends FieldType<?>> asFieldTypeClass(Class<?> klass) {
518        try {
519            return (Class<? extends FieldType<?>>)klass.asSubclass(FieldType.class);
520        } catch (ClassCastException e) {
521            throw new BuildException("invalid @" + JFieldType.class.getSimpleName() + " annotation on "
522              + klass + ": type is not a subclass of " + FieldType.class);
523        }
524    }
