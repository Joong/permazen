006package org.jsimpledb.spring;
007
008import com.google.common.base.Preconditions;
009
010import java.io.IOException;
011
012import org.dellroad.stuff.validation.ValidationUtil;
013import org.jsimpledb.JObject;
014import org.jsimpledb.JSimpleDB;
015import org.jsimpledb.SnapshotJTransaction;
016import org.jsimpledb.ValidationException;
017import org.jsimpledb.ValidationMode;
018import org.jsimpledb.kv.util.NavigableMapKVStore;
019import org.springframework.http.HttpInputMessage;
020import org.springframework.http.HttpOutputMessage;
021import org.springframework.http.MediaType;
022import org.springframework.http.converter.AbstractHttpMessageConverter;
023import org.springframework.http.converter.HttpMessageNotReadableException;
024
025/**
026 * Spring {@link org.springframework.http.converter.HttpMessageConverter HttpMessageConverter} capable of
027 * encoding and decoding a graph of {@link JObject}s contained in a {@link SnapshotJTransaction} that is
028 * backed by a {@link NavigableMapKVStore}.
029 *
030 * <p>
031 * The MIME type used is {@code application/x-jsimpledb-transaction}.
032 *
033 * <p>
034 * Validation of all incoming objects is supported; see {@link #setValidationGroups setValidationGroups()}.
035 *
036 * @see JObjectHttpMessageConverter
037 * @see KVStoreHttpMessageConverter
038 */
039public class SnapshotJTransactionHttpMessageConverter extends AbstractHttpMessageConverter<SnapshotJTransaction> {
040
041    /**
042     * MIME type supported by this class: {@code application/x-jsimpledb-transaction}.
043     *
044     * <p>
045     * Can be overridden in the constructor.
046     */
047    public static final MediaType MIME_TYPE = new MediaType("application", "x-jsimpledb-transaction");
048
049    private final JSimpleDB jdb;
050
051    private Class<?>[] validationGroups;
052
053    /**
054     * Constructor.
055     *
056     * @param jdb {@link JSimpleDB} instance defining the convertible types
057     */
058    public SnapshotJTransactionHttpMessageConverter(JSimpleDB jdb) {
059        super(MIME_TYPE);
060        Preconditions.checkArgument(jdb != null, "null jdb");
061        this.jdb = jdb;
062    }
063
064    /**
065     * Set validation groups used to validate all incoming objects.
066     *
067     * <p>
068     * If set to null, no validation is performed. Otherwise, all incoming objects in the transaction are validated,
069     * using the specified validation groups, or {@link javax.validation.groups.Default} if an empty is specified.
070     *
071     * <p>
072     * By default, this is null.
073     *
074     * @param groups validation group(s) to use for validation; if empty, {@link javax.validation.groups.Default} is assumed;
075     *  if null, no validation is performed
076     */
077    public void setValidationGroups(Class<?>... groups) {
078        this.validationGroups = groups;
079    }
080
081// AbstractHttpMessageConverter
082
083    @Override
084    protected Long getContentLength(SnapshotJTransaction jtx, MediaType contentType) {
085        return KVStoreHttpMessageConverter.getKVStoreContentLength(jtx.getTransaction().getKVStore());
086    }
087
088    @Override
089    protected boolean supports(Class<?> clazz) {
090        return clazz == SnapshotJTransaction.class;
091    }
092
093    @Override
094    protected SnapshotJTransaction readInternal(Class<? extends SnapshotJTransaction> clazz, HttpInputMessage input)
095      throws IOException {
096        return clazz.cast(SnapshotJTransactionHttpMessageConverter.readSnapshotTransaction(this.jdb, input, this.validationGroups));
097    }
098
099    @Override
100    protected void writeInternal(SnapshotJTransaction jtx, HttpOutputMessage output) throws IOException {
101        KVStoreHttpMessageConverter.writeKVStore(jtx.getTransaction().getKVStore(), output);
102    }
103
104// Utility methods
105
106    static SnapshotJTransaction readSnapshotTransaction(JSimpleDB jdb, HttpInputMessage input, Class<?>[] validationGroups)
107      throws IOException {
108
109        // Decode key/value store
110        final NavigableMapKVStore kvstore = new NavigableMapKVStore();
111        KVStoreHttpMessageConverter.readKVStore(kvstore, input);
112
113        // Create snapshot transaction
114        final SnapshotJTransaction jtx = jdb.createSnapshotTransaction(kvstore, true,
115          validationGroups != null ? ValidationMode.MANUAL : ValidationMode.DISABLED);
116
117        // Optionally validate
118        if (validationGroups != null) {
119            for (JObject jobj : jtx.getAll(JObject.class))
120                jobj.revalidate(validationGroups);
121            try {
122                jtx.validate();
123            } catch (ValidationException e) {
124                throw new HttpMessageNotReadableException("incoming object graph failed validation: "
125                  + ValidationUtil.describe(e.getViolations()));
126            }
127        }
128
129        // Done
130        return jtx;
131    }
132}
