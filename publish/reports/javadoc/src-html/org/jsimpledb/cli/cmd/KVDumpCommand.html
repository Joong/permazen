006package org.jsimpledb.cli.cmd;
007
008import java.io.PrintWriter;
009import java.util.Iterator;
010import java.util.Map;
011
012import org.jsimpledb.SessionMode;
013import org.jsimpledb.cli.CliSession;
014import org.jsimpledb.kv.KVPair;
015import org.jsimpledb.kv.KVTransaction;
016import org.jsimpledb.parse.ParseContext;
017import org.jsimpledb.util.ByteUtil;
018
019@Command(modes = { SessionMode.KEY_VALUE, SessionMode.CORE_API, SessionMode.JSIMPLEDB })
020public class KVDumpCommand extends AbstractCommand {
021
022    public KVDumpCommand() {
023        super("kvdump minKey? maxKey? limit:int?");
024    }
025
026    @Override
027    public String getHelpSummary() {
028        return "Dumps raw database key/value pairs to the console";
029    }
030
031    @Override
032    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
033        final String minKeyString = (String)params.get("minKey");
034        final String maxKeyString = (String)params.get("maxKey");
035        final Integer limit = (Integer)params.get("limit");
036        final byte[] minKey = minKeyString != null ? ByteUtil.parse(minKeyString) : null;
037        final byte[] maxKey = maxKeyString != null ? ByteUtil.parse(maxKeyString) : null;
038        return new CliSession.Action() {
039            @Override
040            public void run(CliSession session) throws Exception {
041                final PrintWriter writer = session.getWriter();
042                final KVTransaction kvt = session.getKVTransaction();
043                long count = 0;
044                for (Iterator<KVPair> i = kvt.getRange(minKey, maxKey, false); i.hasNext(); ) {
045                    final KVPair pair = i.next();
046                    if (limit != null && count >= limit)
047                        break;
048                    KVDumpCommand.this.print(writer, "K ", pair.getKey());
049                    KVDumpCommand.this.print(writer, "V ", pair.getValue());
050                    count++;
051                }
052                writer.println("Displayed " + count + " key/value pairs");
053            }
054        };
055    }
056
057    private void print(PrintWriter writer, String prefix, byte[] value) {
058        for (int i = 0; i < value.length; i += 32) {
059            writer.print(prefix);
060            for (int j = 0; j < 32; j++) {
061                writer.print(i + j < value.length ? String.format("%02x", value[i + j] & 0xff) : "  ");
062                if (j % 4 == 3)
063                    writer.print(' ');
064            }
065            writer.print("   ");
066            for (int j = 0; j < 32 && i + j < value.length; j++) {
067                final int ch = value[i + j];
068                writer.print(i + j < value.length ? (ch < 0x20 || ch > 0x7f ? '.' : (char)ch) : ' ');
069            }
070            writer.println();
071        }
072    }
073}
