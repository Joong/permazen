050/**
051 * JSimpleDB Java persistence layer.
052 *
053 * <p>
054 * JSimpleDB is a Java persistence solution built on three layers of abstraction:
055 *  <ul>
056 *  <li>At the bottom layer is a simple {@code byte[]} <b>key/value</b> database represented by the
057 *      {@link org.jsimpledb.kv.KVDatabase} class. Transactions are supported at this layer and are accessed
058 *      through the {@link org.jsimpledb.kv.KVTransaction} interface.
059 *      There are several available {@link org.jsimpledb.kv.KVDatabase} implementations, including "wrappers"
060 *      for several third party key/value stores.</li>
061 *  <li>On top of that sits the <b>core API</b> layer, which provides a rigourous database abstraction on top of the
062 *      key/value store. It supports simple fields of any atomic Java type, as well as list, set, and map complex fields,
063 *      tightly controlled schema versioning, simple and composite indexes, and lifecycle and change notifications.
064 *      It is not Java-specific or explicitly object-oriented. The core API is accessed through the {@link Database}
065 *      and {@link org.jsimpledb.core.Transaction} classes.</li>
066 *  <li>The top layer is a Java-centric, type safe, object-oriented persistence layer for Java applications.
067 *      It sits on top of the core API layer and provides a fully type-safe Java view of a core API {@link Transaction},
068 *      where all access is through user-supplied Java model classes. Database types and fields, and Java listener methods
069 *      are all declared using {@linkplain org.jsimpledb.annotation Java annotations}. Incremental JSR 303 validation is supported.
070 *      The {@link JSimpleDB} class represents an instance of this top layer database, and {@link JTransaction}
071 *      represents the corresonding transactions.</li>
072 *  </ul>
073 *
074 * <p>
075 * User-provided Java model classes define database fields by declaring abstract Java bean property methods.
076 * {@link JSimpleDB} generates concrete subclasses of the user-provided abstract model classes at runtime.
077 * These runtime classes implement the abstract bean property methods, as well as the {@link JObject} interface.
078 * Java model class instances are always associated with a specific {@link JTransaction}, and all of their database
079 * state derives from that the underlying key/value {@link org.jsimpledb.kv.KVTransaction}.
080 *
081 * <p>
082 * All Java model class instances have a unique {@link ObjId} which represents database identity. {@link JSimpleDB}
083 * guarantees that at most one Java model class instance instance will exist for any given {@link JTransaction} and {@link ObjId}.
084 * Instance creation, index queries, and certain other database-related tasks are initiated using a {@link JTransaction}.
085 *
086 * <p>
087 * Normal database transactions are created via {@link #createTransaction createTransaction()}. "Snapshot" transactions are
088 * purely in-memory transactions that are detached from the database and may persist indefinitely. Their purpose is to hold a
089 * snapshot of some (user-defined) portion of the database content for use outside of a regular transaction. Otherwise,
090 * they function like normal transactions, with support for index queries, listener callbacks, etc. See
091 * {@link JTransaction#createSnapshotTransaction JTransaction.createSnapshotTransaction()},
092 * {@link JTransaction#getSnapshotTransaction}, {@link JObject#copyOut JObject.copyOut()}, and
093 * {@link JObject#copyIn JObject.copyIn()}.
094 *
095 * <p>
096 * Instances of this class are usually created using a {@link JSimpleDBFactory}.
097 *
098 * @see JObject
099 * @see JTransaction
100 * @see JSimpleDBFactory
101 * @see org.jsimpledb.annotation
102 */

178    /**
179     * Primary constructor.
180     *
181     * @param database core database to use
182     * @param version schema version number of the schema derived from {@code classes},
183     *  or zero to use the highest version already recorded in the database
184     * @param storageIdGenerator generator for auto-generated storage ID's, or null to disallow auto-generation of storage ID's
185     * @param classes classes annotated with {@link JSimpleClass &#64;JSimpleClass} annotations; non-annotated classes are ignored
186     * @throws IllegalArgumentException if {@code database} or {@code classes} is null
187     * @throws IllegalArgumentException if {@code version} is not greater than zero
188     * @throws IllegalArgumentException if {@code classes} contains a null class or a class with invalid annotation(s)
189     * @throws org.jsimpledb.core.InvalidSchemaException if the schema implied by {@code classes} is invalid
190     */
191    public JSimpleDB(Database database, int version, StorageIdGenerator storageIdGenerator, Iterable<? extends Class<?>> classes) {
192
193        // Initialize
194        Preconditions.checkArgument(database != null, "null database");
195        Preconditions.checkArgument(version >= 0, "invalid negative schema version");
196        Preconditions.checkArgument(classes != null, "null classes");
197        this.db = database;
198        this.configuredVersion = version;
199        this.storageIdGenerator = storageIdGenerator;
200
201        // Inventory classes; automatically add all @JSimpleClass-annotated superclasses of @JSimpleClass-annotated classes
202        final HashSet<Class<?>> jsimpleClasses = new HashSet<>();
203        for (Class<?> type : classes) {
204
205            // Sanity check
206            Preconditions.checkArgument(type != null, "null class found in classes");
207
208            // Add type and all @JSimpleClass-annotated superclasses
209            do {
210
211                // Find annotation
212                final JSimpleClass annotation = type.getAnnotation(JSimpleClass.class);
213                if (annotation == null)
214                    continue;
215
216                // Sanity check type
217                if (type.isPrimitive() || type.isInterface() || type.isArray()) {
218                    throw new IllegalArgumentException("illegal type " + type + " for @"
219                      + JSimpleClass.class.getSimpleName() + " annotation: not a normal class");
220                }
221
222                // Add class
223                jsimpleClasses.add(type);
224            } while ((type = type.getSuperclass()) != null);
225        }
226
227        // Add Java model classes
228        for (Class<?> type : jsimpleClasses) {
229
230            // Create JClass
231            final JSimpleClass annotation = type.getAnnotation(JSimpleClass.class);
232            final String name = annotation.name().length() != 0 ? annotation.name() : type.getSimpleName();
233            if (this.log.isTraceEnabled()) {
234                this.log.trace("found @" + JSimpleClass.class.getSimpleName() + " annotation on " + type
235                  + " defining object type `" + name + "'");
236            }
237
238            // Get storage ID
239            int storageId = annotation.storageId();
240            if (storageId == 0)
241                storageId = this.getStorageIdGenerator(annotation, type).generateClassStorageId(type, name);
242
243            // Create JClass
244            JClass<?> jclass;
245            try {
246                jclass = this.createJClass(name, storageId, type);
247            } catch (IllegalArgumentException e) {
248                throw new IllegalArgumentException("invalid @" + JSimpleClass.class.getSimpleName()
249                  + " annotation on " + type + ": " + e, e);
250            }
251
252            // Add jclass
253            this.addJClass(jclass);
254            this.log.debug("added Java model class `" + jclass.name + "' with storage ID " + jclass.storageId);
255        }
256
257        // Inventory class generators
258        this.classGenerators = new ArrayList<>(this.jclasses.size() + 1);
259        for (JClass<?> jclass : this.jclasses.values())
260            this.classGenerators.add(jclass.classGenerator);
261        this.untypedClassGenerator = new ClassGenerator<UntypedJObject>(this, UntypedJObject.class);
262        this.classGenerators.add(this.untypedClassGenerator);
263
264        // Create fields
265        for (JClass<?> jclass : this.jclasses.values())
266            jclass.createFields(this);
267
268        // Create canonical field info structures
269        final HashMap<Integer, String> fieldDescriptionMap = new HashMap<>();
270        for (JClass<?> jclass : this.jclasses.values()) {
271            for (JField jfield : jclass.jfields.values()) {
272                if (jfield instanceof JComplexField) {
273                    final JComplexField complexField = (JComplexField)jfield;
274                    final JComplexFieldInfo complexFieldInfo = (JComplexFieldInfo)jfield.toJFieldInfo();
275                    for (JSimpleField subField : complexField.getSubFields()) {
276                        this.addJFieldInfo(subField, subField.toJFieldInfo(jfield.storageId), fieldDescriptionMap);
277                        final JSimpleFieldInfo subFieldInfo = (JSimpleFieldInfo)this.jfieldInfos.get(subField.storageId);
278                        complexFieldInfo.getSubFieldInfos().add(subFieldInfo);
279                    }
280                    this.addJFieldInfo(complexField, complexFieldInfo, fieldDescriptionMap);
281                } else
282                    this.addJFieldInfo(jfield, jfield.toJFieldInfo(), fieldDescriptionMap);
283            }
284        }
285
286        // Witness all simple fields to corresponding simple field info's
287        for (JClass<?> jclass : this.jclasses.values()) {
288            for (JField jfield : jclass.jfields.values()) {
289                if (jfield instanceof JSimpleField) {
290                    final JSimpleField jsimpleField = (JSimpleField)jfield;
291                    final JSimpleFieldInfo jsimpleFieldInfo = (JSimpleFieldInfo)this.jfieldInfos.get(jfield.storageId);
292                    jsimpleFieldInfo.witness(jsimpleField);
293                }
294                if (jfield instanceof JComplexField) {
295                    final JComplexField complexField = (JComplexField)jfield;
296                    final JComplexFieldInfo complexFieldInfo = (JComplexFieldInfo)this.jfieldInfos.get(jfield.storageId);
297                    for (int i = 0; i < complexField.getSubFields().size(); i++) {
298                        final JSimpleField subField = complexField.getSubFields().get(i);
299                        final JSimpleFieldInfo subFieldInfo = (JSimpleFieldInfo)this.jfieldInfos.get(subField.storageId);
300                        subFieldInfo.witness(subField);
301                    }
302                }
303            }
304        }
305
306        // Add composite indexes to class; like fields, indexes are inherited (duplicated) from superclasses
307        for (JClass<?> jclass : this.jclasses.values()) {
308            for (Class<?> type = jclass.type; type != null; type = type.getSuperclass()) {
309                final JSimpleClass annotation = type.getAnnotation(JSimpleClass.class);
310                if (annotation != null) {
311                    for (org.jsimpledb.annotation.JCompositeIndex indexAnnotation : annotation.compositeIndexes())
312                        jclass.addCompositeIndex(this, indexAnnotation);
313                }
314            }
315        }
316
317        // Create canonical info instances for indexes
318        final HashMap<Integer, String> indexDescriptionMap = new HashMap<>();
319        for (JClass<?> jclass : this.jclasses.values()) {
320            for (JCompositeIndex index : jclass.jcompositeIndexes.values()) {
321                final JCompositeIndexInfo indexInfo = index.toJCompositeIndexInfo();
322                for (JSimpleField jfield : index.jfields) {
323                    final JSimpleFieldInfo jfieldInfo = (JSimpleFieldInfo)this.jfieldInfos.get(jfield.storageId);
324                    indexInfo.getJFieldInfos().add(jfieldInfo);
325                }
326                this.addJCompositeIndexInfo(index, indexInfo, indexDescriptionMap);
327            }
328        }
329
330        // Scan for other method-level annotations
331        for (JClass<?> jclass : this.jclasses.values())
332            jclass.scanAnnotations();
333
334        // Determine which JClass's have validation requirement(s) on creation
335        for (JClass<?> jclass : this.jclasses.values())
336            jclass.calculateValidationRequirement();
337        boolean anyDefaultValidation = false;
338        for (JClass<?> jclass : this.jclasses.values()) {
339            if ((anyDefaultValidation |= jclass.requiresDefaultValidation))
340                break;
341        }
342        this.anyJClassRequiresDefaultValidation = anyDefaultValidation;
343
344        // Detect whether we have any @OnCreate, @OnDelete, and/or @OnVersionChange methods
345        boolean anyOnCreateMethods = false;
346        boolean anyOnDeleteMethods = false;
347        boolean anyOnVersionChangeMethods = false;
348        for (JClass<?> jclass : this.jclasses.values()) {
349            anyOnCreateMethods |= !jclass.onCreateMethods.isEmpty();
350            anyOnDeleteMethods |= !jclass.onDeleteMethods.isEmpty();
351            anyOnVersionChangeMethods |= !jclass.onVersionChangeMethods.isEmpty();
352        }
353        this.hasOnCreateMethods = anyOnCreateMethods;
354        this.hasOnDeleteMethods = anyOnDeleteMethods;
355        this.hasOnVersionChangeMethods = anyOnVersionChangeMethods;
356
357        // Validate schema
358        this.db.validateSchema(this.getSchemaModel());
359
360        // Eagerly load all generated Java classes so we "fail fast" if there are any loading errors
361        this.untypedClassGenerator.generateClass();
362        for (JClass<?> jclass : this.jclasses.values())
363            jclass.getClassGenerator().generateClass();
364    }

371    StorageIdGenerator getStorageIdGenerator(Annotation annotation, AnnotatedElement target) {
372        if (this.storageIdGenerator == null) {
373            throw new IllegalArgumentException("invalid @" + annotation.annotationType().getSimpleName()
374              + " annotation on " + target + ": no storage ID is given, but storage ID auto-generation is disabled"
375              + " because no " + StorageIdGenerator.class.getSimpleName() + " is configured");
376        }
377        return this.storageIdGenerator;
378    }

427    /**
428     * Get the Java model class of the given {@link JObject}.
429     *
430     * <p>
431     * If {@code jobj} is an instance of a JSimpleDB-generated subclass of a user-supplied Java model class,
432     * this returns the original Java model class. Otherwise, it returns {@code obj}'s type.
433     *
434     * @param jobj database instance
435     * @return lowest ancestor class of {@code jobj}'s class that is not a JSimpleDB-generated subclass
436     * @throws IllegalArgumentException if {@code jobj} is null
437     */
438    public static Class<?> getModelClass(JObject jobj) {
439        Preconditions.checkArgument(jobj != null, "null jobj");
440        for (Class<?> type = jobj.getClass(); type != null; type = type.getSuperclass()) {
441            if (type.getName().indexOf(GENERATED_CLASS_NAME_SUFFIX) == -1)
442                return type;
443        }
444        return null;
445    }

538    private JTransaction createTransaction(Transaction tx, ValidationMode validationMode) {
539        assert tx != null;
540        Preconditions.checkArgument(validationMode != null, "null validationMode");
541        this.actualVersion = tx.getSchema().getVersionNumber();
542        final JTransaction jtx = new JTransaction(this, tx, validationMode);
543        tx.addCallback(new CleanupCurrentCallback(jtx));
544        return jtx;
545    }

768    /**
769     * Utility method to get all of the objects directly referenced by a given object via any field.
770     *
771     * <p>
772     * Note: the returned {@link Iterable} may contain duplicates; these can be eliminated using an
773     * {@link org.jsimpledb.util.ObjIdSet} if necessary.
774     * </p>
775     *
776     * @param jobj starting object
777     * @return all objects directly referenced by {@code jobj}
778     * @throws IllegalArgumentException if {@code jobj} is null
779     */
780    public Iterable<JObject> getReferencedObjects(final JObject jobj) {
781
782        // Sanity check
783        Preconditions.checkArgument(jobj != null, "null jobj");
784        final ObjId id = jobj.getObjId();
785
786        // Visit fields
787        final ArrayList<Iterable<JObject>> iterables = new ArrayList<>();
788        for (JField jfield : this.getJClass(id).getJFieldsByStorageId().values()) {
789            jfield.visit(new JFieldSwitchAdapter<Void>() {
790
791                @Override
792                public Void caseJReferenceField(JReferenceField field) {
793                    final JObject ref = field.getValue(jobj);
794                    if (ref != null)
795                        iterables.add(Collections.singleton(ref));
796                    return null;
797                }
798
799                @Override
800                public Void caseJMapField(JMapField field) {
801                    if (field.getKeyField() instanceof JReferenceField)
802                        iterables.add(Iterables.filter(field.getValue(jobj).keySet(), JObject.class));
803                    if (field.getValueField() instanceof JReferenceField)
804                        iterables.add(Iterables.filter(field.getValue(jobj).values(), JObject.class));
805                    return null;
806                }
807
808                @Override
809                protected Void caseJCollectionField(JCollectionField field) {
810                    if (field.getElementField() instanceof JReferenceField)
811                        iterables.add(Iterables.filter(field.getValue(jobj), JObject.class));
812                    return null;
813                }
814
815                @Override
816                protected Void caseJField(JField field) {
817                    return null;
818                }
819            });
820        }
821
822        // Done
823        return Iterables.concat(iterables);
824    }

853     * @return {@link JField} instance
854     * @throws UnknownFieldException if {@code storageId} does not represent a field

856    <T extends JFieldInfo> T getJFieldInfo(int storageId, Class<T> type) {

860            throw new UnknownFieldException(storageId, "no JSimpleDB field exists with storage ID "

866            throw new UnknownFieldException(storageId, "no JSimpleDB fields exist with storage ID "
867              + storageId + " in schema version " + this.actualVersion + " (found field " + jfieldInfo + " instead)");

871    // Add new JClass, checking for storage ID conflicts
872    private void addJClass(JClass<?> jclass) {
873
874        // Check for storage ID conflict
875        final JClass<?> other = this.jclasses.get(jclass.storageId);
876        if (other != null) {
877            throw new IllegalArgumentException("illegal duplicate use of storage ID "
878              + jclass.storageId + " for both " + other + " and " + jclass);
879        }
880        this.jclasses.put(jclass.storageId, jclass);
881        assert !this.jclassesByType.containsKey(jclass.type);                   // this should never conflict, no need to check
882        this.jclassesByType.put(jclass.type, jclass);
883    }

895    // Add new info, checking for storage ID conflicts
896    private <T> void addInfo(JSchemaObject item, T info, Map<Integer, T> infoMap, Map<Integer, String> descriptionMap) {
897        final T existing = infoMap.get(item.storageId);
898        if (existing == null) {
899            infoMap.put(item.storageId, info);
900            descriptionMap.put(item.storageId, item.description);
901        } else if (!info.equals(existing)) {
902            throw new IllegalArgumentException("incompatible duplicate use of storage ID " + item.storageId
903              + " for " + descriptionMap.get(item.storageId) + " and " + item.description);
904        }
905    }

909    private class Loader extends ClassLoader {
910
911        // Set up class loader
912        public Loader() {
913            super(Thread.currentThread().getContextClassLoader());
914        }
915
916        // Find matching ClassGenerator, if any, otherwise defer to parent
917        @Override
918        protected Class<?> findClass(String name) throws ClassNotFoundException {
919            byte[] bytes = null;
920            for (ClassGenerator<?> generator : JSimpleDB.this.classGenerators) {
921                if (name.equals(generator.getClassName().replace('/', '.'))) {
922                    bytes = generator.generateBytecode();
923                    break;
924                }
925            }
926            return bytes != null ? this.defineClass(name, bytes, 0, bytes.length) : super.findClass(name);
927        }
928    }

932    private static final class CleanupCurrentCallback extends Transaction.CallbackAdapter {
933
934        private final JTransaction jtx;
935
936        CleanupCurrentCallback(JTransaction jtx) {
937            assert jtx != null;
938            this.jtx = jtx;
939        }
940
941        @Override
942        public void afterCompletion(boolean committed) {
943            final JTransaction current;
944            try {
945                current = JTransaction.getCurrent();
946            } catch (IllegalStateException e) {
947                return;
948            }
949            if (current == this.jtx)
950                JTransaction.setCurrent(null);
951        }
952
953        @Override
954        public int hashCode() {
955            return this.jtx.hashCode();
956        }
957
958        @Override
959        public boolean equals(Object obj) {
960            if (obj == null || obj.getClass() != this.getClass())
961                return false;
962            final CleanupCurrentCallback that = (CleanupCurrentCallback)obj;
963            return this.jtx.equals(that.jtx);
964        }
