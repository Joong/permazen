006package org.jsimpledb;
007
008import com.google.common.base.Converter;
009import com.google.common.reflect.TypeToken;
010
011import java.lang.reflect.Method;
012import java.util.TreeSet;
013
014import org.jsimpledb.core.DeleteAction;
015import org.jsimpledb.core.ObjId;
016import org.jsimpledb.core.ReferenceFieldType;
017import org.jsimpledb.schema.ReferenceSchemaField;
018import org.objectweb.asm.ClassWriter;
019import org.objectweb.asm.Label;
020import org.objectweb.asm.MethodVisitor;
021import org.objectweb.asm.Opcodes;
022import org.objectweb.asm.Type;
023
024/**
025 * Represents a reference field in a {@link JClass} or a reference sub-field of a complex field in a {@link JClass}.
026 */
027public class JReferenceField extends JSimpleField {
028
029    final DeleteAction onDelete;
030    final boolean cascadeDelete;
031
032    JReferenceField(JSimpleDB jdb, String name, int storageId, String description, TypeToken<?> typeToken,
033      org.jsimpledb.annotation.JField annotation, Method getter, Method setter) {
034        super(jdb, name, storageId, typeToken, new ReferenceFieldType(), true, annotation, description, getter, setter);
035        this.onDelete = annotation.onDelete();
036        this.cascadeDelete = annotation.cascadeDelete();
037    }
038
039    @Override
040    public JObject getValue(JObject jobj) {
041        return (JObject)super.getValue(jobj);
042    }
043
044    @Override
045    public <R> R visit(JFieldSwitch<R> target) {
046        return target.caseJReferenceField(this);
047    }
048
049    @Override
050    public Converter<ObjId, JObject> getConverter(JTransaction jtx) {
051        return new ReferenceConverter<JObject>(jtx, JObject.class).reverse();
052    }
053
054    /**
055     * Get the {@link DeleteAction} configured for this field.
056     *
057     * @return this field's {@link DeleteAction}
058     */
059    public DeleteAction getOnDelete() {
060        return this.onDelete;
061    }
062
063    /**
064     * Determine whether the referred-to object should be deleted when an object containing this field is deleted.
065     *
066     * @return this field's delete cascade setting
067     */
068    public boolean isCascadeDelete() {
069        return this.cascadeDelete;
070    }
071
072    @Override
073    ReferenceSchemaField toSchemaItem(JSimpleDB jdb) {
074        final ReferenceSchemaField schemaField = new ReferenceSchemaField();
075        super.initialize(jdb, schemaField);
076        schemaField.setOnDelete(this.onDelete);
077        schemaField.setCascadeDelete(this.cascadeDelete);
078        final TreeSet<Integer> objectTypes = new TreeSet<>();
079        for (JClass<?> jclass : jdb.getJClasses(this.typeToken.getRawType()))
080            objectTypes.add(jclass.storageId);
081        schemaField.setObjectTypes(objectTypes);
082        return schemaField;
083    }
084
085    @Override
086    JReferenceFieldInfo toJFieldInfo(int parentStorageId) {
087        return new JReferenceFieldInfo(this, parentStorageId);
088    }
089
090// Bytecode generation
091
092    @Override
093    void outputMethods(final ClassGenerator<?> generator, ClassWriter cw) {
094
095        // Getter
096        MethodVisitor mv = cw.visitMethod(
097          this.getter.getModifiers() & (Opcodes.ACC_PUBLIC | Opcodes.ACC_PROTECTED),
098          this.getter.getName(), Type.getMethodDescriptor(this.getter), null, generator.getExceptionNames(this.getter));
099        this.outputReadCoreValueBytecode(generator, mv);
100        mv.visitInsn(Opcodes.DUP);
101        final Label label1 = new Label();
102        mv.visitJumpInsn(Opcodes.IFNONNULL, label1);
103        mv.visitInsn(Opcodes.POP);
104        mv.visitInsn(Opcodes.ACONST_NULL);
105        mv.visitInsn(Opcodes.ARETURN);
106        mv.visitLabel(label1);
107        mv.visitFrame(Opcodes.F_SAME1, 0, new Object[0], 1, new String[] { Type.getInternalName(Object.class) });
108        mv.visitTypeInsn(Opcodes.CHECKCAST, Type.getInternalName(ObjId.class));
109        mv.visitVarInsn(Opcodes.ALOAD, 0);
110        mv.visitFieldInsn(Opcodes.GETFIELD, generator.getClassName(),
111          ClassGenerator.TX_FIELD_NAME, Type.getDescriptor(JTransaction.class));
112        mv.visitInsn(Opcodes.SWAP);
113        generator.emitInvoke(mv, ClassGenerator.GET_JOBJECT_METHOD);
114        mv.visitTypeInsn(Opcodes.CHECKCAST, Type.getInternalName(this.getter.getReturnType()));
115        mv.visitInsn(Opcodes.ARETURN);
116        mv.visitMaxs(0, 0);
117        mv.visitEnd();
118
119        // Setter
120        mv = cw.visitMethod(
121          this.setter.getModifiers() & (Opcodes.ACC_PUBLIC | Opcodes.ACC_PROTECTED),
122          this.setter.getName(), Type.getMethodDescriptor(this.setter), null, generator.getExceptionNames(this.setter));
123        mv.visitVarInsn(Opcodes.ALOAD, 1);
124        final Label label2 = new Label();
125        final Label label3 = new Label();
126        mv.visitJumpInsn(Opcodes.IFNONNULL, label2);
127        mv.visitInsn(Opcodes.ACONST_NULL);
128        mv.visitJumpInsn(Opcodes.GOTO, label3);
129        mv.visitLabel(label2);
130        mv.visitFrame(Opcodes.F_SAME, 0, new Object[0], 0, new Object[0]);
131        mv.visitVarInsn(Opcodes.ALOAD, 1);
132        generator.emitInvoke(mv, ClassGenerator.JOBJECT_GET_OBJ_ID_METHOD);
133        mv.visitLabel(label3);
134        mv.visitFrame(Opcodes.F_SAME1, 0, new Object[0], 1, new String[] { Type.getInternalName(ObjId.class) });
135        //mv.visitTypeInsn(Opcodes.CHECKCAST, Type.getInternalName(ObjId.class));
136        this.outputWriteCoreValueBytecode(generator, mv);
137        mv.visitInsn(Opcodes.RETURN);
138        mv.visitMaxs(0, 0);
139        mv.visitEnd();
140    }
141}
