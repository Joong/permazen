043    private String leader;                                                          // our leader, if known
044    private String leaderAddress;                                                   // our leader's network address
045    private String votedFor;                                                        // the candidate we voted for this term
046    private SnapshotReceive snapshotReceive;                                        // in-progress snapshot install, if any
047    private final HashMap<Long, PendingRequest> pendingRequests = new HashMap<>();  // wait for CommitResponse or log entry
048    private final HashMap<Long, PendingWrite> pendingWrites = new HashMap<>();      // wait for AppendRequest with null data
049    private final HashMap<Long, Timestamp> commitLeaderLeaseTimeoutMap              // tx's waiting for leaderLeaseTimeout's

051    private Timestamp lastLeaderMessageTime;                                        // time of most recent rec'd AppendRequest
052    private Timestamp leaderLeaseTimeout;                                           // latest rec'd leader lease timeout
053    private HashMap<String, Timestamp> probeTimestamps;                             // used only when probing majority

131// Probing mode
132
133    private int calculateProbedNodes() {
134        assert this.probeTimestamps != null;
135        int numProbed = this.raft.isClusterMember() ? 1 : 0;
136        final Timestamp now = new Timestamp();
137        for (Iterator<Timestamp> i = this.probeTimestamps.values().iterator(); i.hasNext(); ) {
138            final Timestamp timestamp = i.next();
139            if (now.offsetFrom(timestamp) >= this.raft.maxElectionTimeout) {                // timestamp is too old, discard
140                i.remove();
141                continue;
142            }
143            numProbed++;
144        }
145        return numProbed;
146    }

150    @Override
151    void setup() {
152        super.setup();
153        if (this.log.isDebugEnabled()) {
154            this.debug("entering follower role in term " + this.raft.currentTerm
155              + (this.leader != null ? "; with leader \"" + this.leader + "\" at " + this.leaderAddress : "")
156              + (this.votedFor != null ? "; having voted for \"" + this.votedFor + "\"" : ""));
157        }

160    @Override
161    void shutdown() {
162        super.shutdown();
163
164        // Cancel any in-progress snapshot install
165        if (this.snapshotReceive != null) {
166            if (this.log.isDebugEnabled())
167                this.debug("aborting snapshot install due to leaving follower role");
168            this.raft.discardFlipFloppedStateMachine();
169            this.snapshotReceive = null;
170        }
171
172        // Fail any (read-only) transactions waiting on a minimum lease timeout from deposed leader
173        for (RaftKVTransaction tx : new ArrayList<RaftKVTransaction>(this.raft.openTransactions.values())) {
174            if (tx.getState().equals(TxState.COMMIT_WAITING) && this.commitLeaderLeaseTimeoutMap.containsKey(tx.getTxId()))
175                this.raft.fail(tx, new RetryTransactionException(tx, "leader was deposed during commit"));
176        }
177
178        // Cleanup pending requests and commit writes
179        this.pendingRequests.clear();
180        for (PendingWrite pendingWrite : this.pendingWrites.values())
181            pendingWrite.cleanup();
182        this.pendingWrites.clear();
183    }

193    // Check whether the required minimum leader lease timeout has been seen, if any
194    @Override
195    boolean mayCommit(RaftKVTransaction tx) {
196
197        // Is there a required minimum leader lease timeout associated with the transaction?
198        final Timestamp commitLeaderLeaseTimeout = this.commitLeaderLeaseTimeoutMap.get(tx.getTxId());
199        if (commitLeaderLeaseTimeout == null)
200            return true;
201
202        // Do we know the leader's lease timeout yet?
203        if (this.leaderLeaseTimeout == null)
204            return false;
205
206        // Verify leader's lease timeout has extended beyond that required by the transaction
207        return this.leaderLeaseTimeout.compareTo(commitLeaderLeaseTimeout) >= 0;
208    }

210    @Override
211    void handleElectionTimeout() {
212
213        // Invalidate current leader
214        this.leader = null;
215        this.leaderAddress = null;
216
217        // Is probing enabled?
218        if (!this.raft.followerProbingEnabled) {
219            if (this.log.isDebugEnabled())
220                this.debug("follower election timeout: probing is disabled, so converting immediately to candidate");
221            this.raft.changeRole(new CandidateRole(this.raft));
222            return;
223        }
224
225        // If we are already probing, check probe results
226        if (this.probeTimestamps != null) {
227
228            // Get the number of nodes successfully probed this round, and the minimum number required (a majority)
229            final int numProbed = this.calculateProbedNodes();
230            final int numRequired = this.raft.currentConfig.size() / 2 + 1;
231
232            // Once we have successfully probed a majority we can finally become a candidate
233            if (this.log.isTraceEnabled())
234                this.trace("now we have probed " + numProbed + "/" + numRequired + " required nodes");
235            if (numProbed >= numRequired) {
236                if (this.log.isDebugEnabled())
237                    this.debug("successfully probed " + numProbed + " nodes, now converting to candidate");
238                this.raft.changeRole(new CandidateRole(this.raft));
239                return;
240            }
241        } else {
242
243            // Enter probing mode
244            if (this.log.isDebugEnabled())
245                this.debug("follower election timeout: attempting to probe a majority before becoming candidate");
246            this.probeTimestamps = new HashMap<>(this.raft.currentConfig.size() - 1);
247        }
248
249        // Send out a(nother) round of probes to all other nodes
250        final Timestamp now = new Timestamp();
251        for (String peer : this.raft.currentConfig.keySet()) {
252            if (peer.equals(this.raft.identity))
253                continue;
254            this.raft.sendMessage(new PingRequest(this.raft.clusterId, this.raft.identity, peer, this.raft.currentTerm, now));
255        }
256
257        // Restart election timeout (now it's really a ping timer)
258        this.restartElectionTimer();
259    }
260
261    /**
262     * Check whether the election timer should be running, and make it so.
263     *
264     * <p>
265     * This should be invoked:
266     * <ul>
267     *  <li>After a log entry that contains a configuration change has been added to the log</li>
268     *  <li>When a snapshot install starts</li>
269     *  <li>When a snapshot install completes</li>
270     * </ul>
271     */
272    private void updateElectionTimer() {
273        final boolean isClusterMember = this.raft.isClusterMember();
274        final boolean electionTimerRunning = this.electionTimer.isRunning();
275        if (isClusterMember && !electionTimerRunning) {
276            if (this.log.isTraceEnabled())
277                this.trace("starting up election timer because I'm now part of the current config");
278            this.restartElectionTimer();
279        } else if (!isClusterMember && electionTimerRunning) {
280            if (this.log.isTraceEnabled())
281                this.trace("stopping election timer because I'm no longer part of the current config");
282            this.electionTimer.cancel();
283        }
284    }

288    @Override
289    void checkReadyLeaderTransaction(RaftKVTransaction tx, boolean readOnly) {
290
291        // Sanity check
292        assert Thread.holdsLock(this.raft);
293        assert tx.getState().equals(TxState.COMMIT_READY);
294
295        // Did we already send a CommitRequest for this transaction?
296        final PendingRequest pendingRequest = this.pendingRequests.get(tx.getTxId());
297        if (pendingRequest != null) {
298            if (this.log.isTraceEnabled())
299                this.trace("leaving alone ready tx " + tx + " because request already sent");
300            return;
301        }
302
303        // If we are installing a snapshot, we must wait
304        if (this.snapshotReceive != null) {
305            if (this.log.isTraceEnabled())
306                this.trace("leaving alone ready tx " + tx + " because a snapshot install is in progress");
307            return;
308        }
309
310        // Handle situation where we are unconfigured and not part of any cluster yet
311        if (!this.raft.isConfigured()) {
312
313            // Get transaction mutations
314            final Writes writes = tx.getMutableView().getWrites();
315            final String[] configChange = tx.getConfigChange();
316
317            // Allow an empty read-only transaction when unconfigured
318            if (readOnly) {
319                this.raft.succeed(tx);
320                return;
321            }
322
323            // Otherwise, we can only handle an initial config change that is adding the local node
324            if (configChange == null || !configChange[0].equals(this.raft.identity) || configChange[1] == null) {
325                throw new RetryTransactionException(tx, "unconfigured system: an initial configuration change adding"
326                  + " the local node (\"" + this.raft.identity + "\") as the first member of a new cluster is required");
327            }
328
329            // Create a new cluster if needed
330            if (this.raft.clusterId == 0) {
331
332                // Pick a new, random cluster ID
333                int newClusterId;
334                do
335                    newClusterId = this.raft.random.nextInt();
336                while (newClusterId == 0);
337
338                // Join new cluster
339                this.info("creating new cluster with ID " + String.format("0x%08x", newClusterId));
340                if (!this.raft.joinCluster(newClusterId))
341                    throw new KVTransactionException(tx, "error persisting new cluster ID");
342            }
343
344            // Advance term
345            assert this.raft.currentTerm == 0;
346            if (!this.raft.advanceTerm(this.raft.currentTerm + 1))
347                throw new KVTransactionException(tx, "error advancing term");
348
349            // Append the first entry to the Raft log
350            final LogEntry logEntry;
351            try {
352                logEntry = this.raft.appendLogEntry(this.raft.currentTerm, new NewLogEntry(tx));
353            } catch (Exception e) {
354                throw new KVTransactionException(tx, "error attempting to persist transaction", e);
355            }
356            if (this.log.isDebugEnabled())
357                this.debug("added log entry " + logEntry + " for local transaction " + tx);
358            assert logEntry.getTerm() == 1;
359            assert logEntry.getIndex() == 1;
360
361            // Update transaction
362            this.advanceReadyTransaction(tx, logEntry.getTerm(), logEntry.getIndex());
363
364            // Set commit term and index from new log entry
365            this.raft.commitIndex = logEntry.getIndex();
366            this.raft.requestService(this.triggerKeyWatchesService);
367
368            // Immediately become the leader of our new single-node cluster
369            assert this.raft.isConfigured();
370            if (this.log.isDebugEnabled())
371                this.debug("appointing myself leader in newly created cluster");
372            this.raft.changeRole(new LeaderRole(this.raft));
373            return;
374        }
375
376        // If we don't have a leader yet, or leader's queue is full, we must wait
377        if (this.leader == null || this.raft.isTransmitting(this.leaderAddress)) {
378            if (this.log.isTraceEnabled()) {
379                this.trace("leaving alone ready tx " + tx + " because leader "
380                  + (this.leader == null ? "is not known yet" : "\"" + this.leader + "\" is not writable yet"));
381            }
382            return;
383        }
384
385        // Serialize reads into buffer
386        assert !readOnly || tx.getConsistency().isGuaranteesUpToDateReads();
387        final Reads reads = tx.getMutableView().getReads();
388        final long readsDataSize = reads.serializedLength();
389        if (readsDataSize != (int)readsDataSize)
390            throw new KVTransactionException(tx, "transaction read information exceeds maximum length");
391        final ByteBuffer readsData = Util.allocateByteBuffer((int)readsDataSize);
392        try (ByteBufferOutputStream output = new ByteBufferOutputStream(readsData)) {
393            reads.serialize(output);
394        } catch (IOException e) {
395            throw new RuntimeException("unexpected exception", e);
396        }
397        assert !readsData.hasRemaining();
398        readsData.flip();
399
400        // Gather writes data (i.e., mutations), but only if transaction contains mutations
401        ByteBuffer mutationData = null;
402        if (!readOnly) {
403
404            // Serialize mutations into a temporary file (but do not close or durably persist yet)
405            final Writes writes = tx.getMutableView().getWrites();
406            final File file = new File(this.raft.logDir,
407              String.format("%s%019d%s", RaftKVDatabase.TX_FILE_PREFIX, tx.getTxId(), RaftKVDatabase.TEMP_FILE_SUFFIX));
408            final FileWriter fileWriter;
409            try {
410                fileWriter = new FileWriter(file);
411            } catch (IOException e) {
412                throw new KVTransactionException(tx, "error saving transaction mutations to temporary file", e);
413            }
414            try {
415                LogEntry.writeData(fileWriter, new LogEntry.Data(writes, tx.getConfigChange()));
416                fileWriter.flush();
417            } catch (IOException e) {
418                fileWriter.getFile().delete();
419                Util.closeIfPossible(fileWriter);
420                throw new KVTransactionException(tx, "error saving transaction mutations to temporary file", e);
421            }
422
423            // Load serialized writes from file
424            final long writeLength = fileWriter.getLength();
425            try {
426                mutationData = Util.readFile(fileWriter.getFile(), writeLength);
427            } catch (IOException e) {
428                fileWriter.getFile().delete();
429                Util.closeIfPossible(fileWriter);
430                throw new KVTransactionException(tx, "error reading transaction mutations from temporary file", e);
431            }
432
433            // Record pending commit write with temporary file
434            final PendingWrite pendingWrite = new PendingWrite(tx, fileWriter);
435            this.pendingWrites.put(tx.getTxId(), pendingWrite);
436        }
437
438        // Record pending request
439        this.pendingRequests.put(tx.getTxId(), new PendingRequest(tx));
440
441        // Send commit request to leader
442        final CommitRequest msg = new CommitRequest(this.raft.clusterId, this.raft.identity, this.leader,
443          this.raft.currentTerm, tx.getTxId(), tx.getBaseTerm(), tx.getBaseIndex(), readsData, mutationData);
444        if (this.log.isTraceEnabled())
445            this.trace("sending " + msg + " to \"" + this.leader + "\" for " + tx);
446        if (!this.raft.sendMessage(msg))
447            throw new RetryTransactionException(tx, "error sending commit request to leader");
448    }

450    @Override
451    void cleanupForTransaction(RaftKVTransaction tx) {
452        this.pendingRequests.remove(tx.getTxId());
453        final PendingWrite pendingWrite = this.pendingWrites.remove(tx.getTxId());
454        if (pendingWrite != null)
455            pendingWrite.cleanup();
456        this.commitLeaderLeaseTimeoutMap.remove(tx.getTxId());
457    }

461    @Override
462    boolean mayAdvanceCurrentTerm(Message msg) {
463
464        // Deny vote if we have heard from our leader within the minimum election timeout (dissertation, section 4.2.3)
465        if (msg instanceof RequestVote
466          && this.lastLeaderMessageTime != null
467          && this.lastLeaderMessageTime.offsetFromNow() > -this.raft.minElectionTimeout)
468            return false;
469
470        // OK
471        return true;
472    }

474    @Override
475    void caseAppendRequest(AppendRequest msg) {
476
477        // Cancel probing
478        if (this.probeTimestamps != null) {
479            if (this.log.isDebugEnabled())
480                this.debug("heard from leader before we probed a majority, reverting back to normal follower");
481            this.probeTimestamps = null;
482        }
483        final Timestamp now = new Timestamp();
484
485        // Record new cluster ID if we haven't done so already
486        if (this.raft.clusterId == 0)
487            this.raft.joinCluster(msg.getClusterId());
488
489        // Record leader
490        if (!msg.getSenderId().equals(this.leader)) {
491            if (this.leader != null && !this.leader.equals(msg.getSenderId())) {
492                this.error("detected a conflicting leader in " + msg + " (previous leader was \"" + this.leader
493                  + "\") - should never happen; possible inconsistent cluster configuration (mine: " + this.raft.currentConfig
494                  + ")");
495            }
496            this.leader = msg.getSenderId();
497            this.leaderAddress = this.raft.returnAddress;
498            this.leaderLeaseTimeout = null;
499            if (this.log.isDebugEnabled())
500                this.debug("updated leader to \"" + this.leader + "\" at " + this.leaderAddress);
501            this.raft.requestService(this.checkReadyTransactionsService);     // allows COMMIT_READY transactions to be sent
502        }
503
504        // Get message info
505        final long leaderCommitIndex = msg.getLeaderCommit();
506        final long leaderPrevTerm = msg.getPrevLogTerm();
507        final long leaderPrevIndex = msg.getPrevLogIndex();
508        final long logTerm = msg.getLogEntryTerm();
509        final long logIndex = leaderPrevIndex + 1;
510
511        // Update timestamp last heard from leader
512        this.lastLeaderMessageTime = new Timestamp();
513
514        // Update leader's lease timeout
515        if (msg.getLeaderLeaseTimeout() != null
516          && (this.leaderLeaseTimeout == null || msg.getLeaderLeaseTimeout().compareTo(this.leaderLeaseTimeout) > 0)) {
517            if (this.log.isTraceEnabled())
518                this.trace("advancing leader lease timeout " + this.leaderLeaseTimeout + " -> " + msg.getLeaderLeaseTimeout());
519            this.leaderLeaseTimeout = msg.getLeaderLeaseTimeout();
520            this.raft.requestService(this.checkWaitingTransactionsService);
521        }
522
523        // If a snapshot install is in progress, cancel it
524        if (this.snapshotReceive != null) {
525            if (this.log.isDebugEnabled())
526                this.debug("rec'd " + msg + " during in-progress " + this.snapshotReceive + "; aborting snapshot install");
527            this.raft.discardFlipFloppedStateMachine();
528            this.snapshotReceive = null;
529            this.updateElectionTimer();
530        }
531
532        // Restart election timeout (if running)
533        if (this.electionTimer.isRunning())
534            this.restartElectionTimer();
535
536        // Get my last log entry's index and term
537        long lastLogTerm = this.raft.getLastLogTerm();
538        long lastLogIndex = this.raft.getLastLogIndex();
539
540        // Check whether our previous log entry term matches that of leader; if not, or it doesn't exist, request fails
541        // Note: if log entry index is prior to my last applied log entry index, Raft guarantees that term must match
542        if (leaderPrevIndex >= this.raft.lastAppliedIndex
543          && (leaderPrevIndex > lastLogIndex || leaderPrevTerm != this.raft.getLogTermAtIndex(leaderPrevIndex))) {
544            if (this.log.isDebugEnabled())
545                this.debug("rejecting " + msg + " because previous log entry doesn't match");
546            this.raft.sendMessage(new AppendResponse(this.raft.clusterId, this.raft.identity, msg.getSenderId(),
547              this.raft.currentTerm, msg.getLeaderTimestamp(), false, this.raft.lastAppliedIndex, this.raft.getLastLogIndex()));
548            return;
549        }
550
551        // Check whether the message actually contains a log entry we can append; if so, append it
552        boolean success = true;
553        if (leaderPrevIndex >= this.raft.lastAppliedIndex && !msg.isProbe()) {
554
555            // Check for a conflicting (i.e., never committed, then overwritten) log entry that we need to clear away first
556            if (logIndex <= lastLogIndex && this.raft.getLogTermAtIndex(logIndex) != msg.getLogEntryTerm()) {
557
558                // Delete conflicting log entry, and all entries that follow it, from the log
559                final int startListIndex = (int)(logIndex - this.raft.lastAppliedIndex - 1);
560                final List<LogEntry> conflictList = this.raft.raftLog.subList(startListIndex, this.raft.raftLog.size());
561                for (LogEntry logEntry : conflictList) {
562                    if (this.log.isDebugEnabled())
563                        this.debug("deleting log entry " + logEntry + " overrwritten by " + msg);
564                    if (!logEntry.getFile().delete())
565                        this.error("failed to delete log file " + logEntry.getFile());
566                }
567                try {
568                    this.raft.logDirChannel.force(true);
569                } catch (IOException e) {
570                    this.warn("errory fsync()'ing log directory " + this.raft.logDir, e);
571                }
572                conflictList.clear();
573
574                // Rebuild current config
575                this.raft.currentConfig = this.raft.buildCurrentConfig();
576
577                // Update last log entry info
578                lastLogTerm = this.raft.getLastLogTerm();
579                lastLogIndex = this.raft.getLastLogIndex();
580            }
581
582            // Append the new log entry - if we don't already have it
583            if (logIndex > lastLogIndex) {
584                assert logIndex == lastLogIndex + 1;
585                LogEntry logEntry = null;
586                do {
587
588                    // If message contains no data, we expect to get the data from the corresponding transaction
589                    final ByteBuffer mutationData = msg.getMutationData();
590                    if (mutationData == null) {
591
592                        // Find the matching pending commit write
593                        final PendingWrite pendingWrite;
594                        try {
595                            pendingWrite = Iterables.find(this.pendingWrites.values(), new Predicate<PendingWrite>() {
596                                @Override
597                                public boolean apply(PendingWrite pendingWrite) {
598                                    final RaftKVTransaction tx = pendingWrite.getTx();
599                                    return tx.getState().equals(TxState.COMMIT_WAITING)
600                                      && tx.getCommitTerm() == logTerm && tx.getCommitIndex() == logIndex;
601                                }
602                            });
603                        } catch (NoSuchElementException e) {
604                            if (this.log.isDebugEnabled()) {
605                                this.debug("rec'd " + msg + " but no read-write transaction matching commit "
606                                  + logIndex + "t" + logTerm + " found; rejecting");
607                            }
608                            break;
609                        }
610
611                        // Commit's writes are no longer pending
612                        final RaftKVTransaction tx = pendingWrite.getTx();
613                        this.pendingWrites.remove(tx.getTxId());
614
615                        // Close and durably persist the associated temporary file
616                        try {
617                            pendingWrite.getFileWriter().close();
618                        } catch (IOException e) {
619                            this.error("error closing temporary transaction file for " + tx, e);
620                            pendingWrite.cleanup();
621                            break;
622                        }
623
624                        // Append a new log entry using temporary file
625                        try {
626                            logEntry = this.raft.appendLogEntry(logTerm,
627                              new NewLogEntry(tx, pendingWrite.getFileWriter().getFile()));
628                        } catch (Exception e) {
629                            this.error("error appending new log entry for " + tx, e);
630                            pendingWrite.cleanup();
631                            break;
632                        }
633
634                        // Debug
635                        if (this.log.isDebugEnabled()) {
636                            this.debug("now waiting for commit of " + tx.getCommitIndex() + "t" + tx.getCommitTerm()
637                              + " to commit " + tx);
638                        }
639                    } else {
640
641                        // Append new log entry normally using the data from the request
642                        try {
643                            logEntry = this.raft.appendLogEntry(logTerm, new NewLogEntry(this.raft, mutationData));
644                        } catch (Exception e) {
645                            this.error("error appending new log entry", e);
646                            break;
647                        }
648                    }
649                } while (false);
650
651                // Start/stop election timer as needed
652                if (logEntry != null && logEntry.getConfigChange() != null)
653                    this.updateElectionTimer();
654
655                // Success?
656                success = logEntry != null;
657
658                // Update last log entry info
659                lastLogTerm = this.raft.getLastLogTerm();
660                lastLogIndex = this.raft.getLastLogIndex();
661            }
662        }
663
664        // Update my commit index
665        final long newCommitIndex = Math.min(Math.max(leaderCommitIndex, this.raft.commitIndex), lastLogIndex);
666        if (newCommitIndex > this.raft.commitIndex) {
667            if (this.log.isDebugEnabled())
668                this.debug("updating leader commit index from " + this.raft.commitIndex + " -> " + newCommitIndex);
669            this.raft.commitIndex = newCommitIndex;
670            this.raft.requestService(this.checkWaitingTransactionsService);
671            this.raft.requestService(this.applyCommittedLogEntriesService);
672            this.raft.requestService(this.triggerKeyWatchesService);
673        }
674
675        // Debug
676        if (this.log.isTraceEnabled()) {
677            this.trace("my updated follower state: "
678              + "term=" + this.raft.currentTerm
679              + " commitIndex=" + this.raft.commitIndex
680              + " leaderLeaseTimeout=" + this.leaderLeaseTimeout
681              + " lastApplied=" + this.raft.lastAppliedIndex + "t" + this.raft.lastAppliedTerm
682              + " log=" + this.raft.raftLog);
683        }
684
685        // Send reply
686        if (success) {
687            this.raft.sendMessage(new AppendResponse(this.raft.clusterId, this.raft.identity, msg.getSenderId(),
688              this.raft.currentTerm, msg.getLeaderTimestamp(), true, msg.isProbe() ? logIndex - 1 : logIndex,
689              this.raft.getLastLogIndex()));
690        } else {
691            this.raft.sendMessage(new AppendResponse(this.raft.clusterId, this.raft.identity, msg.getSenderId(),
692              this.raft.currentTerm, msg.getLeaderTimestamp(), false, this.raft.lastAppliedIndex, this.raft.getLastLogIndex()));
693        }
694    }

696    @Override
697    void caseCommitResponse(CommitResponse msg) {
698
699        // Find transaction
700        final RaftKVTransaction tx = this.raft.openTransactions.get(msg.getTxId());
701        if (tx == null)                                                                 // must have been rolled back locally
702            return;
703
704        // Sanity check transaction state
705        if (!tx.getState().equals(TxState.COMMIT_READY)) {
706            if (this.log.isDebugEnabled())
707                this.debug("rec'd " + msg + " for " + tx + " in state " + tx.getState() + "; ignoring");
708            return;
709        }
710        if (this.pendingRequests.remove(tx.getTxId()) == null) {
711            if (this.log.isDebugEnabled())
712                this.debug("rec'd " + msg + " for " + tx + " not expecting a response; ignoring");
713            return;
714        }
715
716        // Check result
717        if (this.log.isTraceEnabled())
718            this.trace("rec'd " + msg + " for " + tx);
719        if (msg.isSuccess()) {
720
721            // Update transaction
722            this.advanceReadyTransaction(tx, msg.getCommitTerm(), msg.getCommitIndex());
723
724            // Track leader lease timeout we must wait for, if any
725            if (msg.getCommitLeaderLeaseTimeout() != null)
726                this.commitLeaderLeaseTimeoutMap.put(tx.getTxId(), msg.getCommitLeaderLeaseTimeout());
727        } else
728            this.raft.fail(tx, new RetryTransactionException(tx, msg.getErrorMessage()));
729    }

731    @Override
732    void caseInstallSnapshot(InstallSnapshot msg) {
733
734        // Restart election timer (if running)
735        if (this.electionTimer.isRunning())
736            this.restartElectionTimer();
737
738        // Sanity check that our log is not going backwards
739        if (msg.getSnapshotIndex() < this.raft.commitIndex) {
740            this.warn("rec'd " + msg + " with retrograde index " + msg.getSnapshotIndex()
741              + " < my commit index " + this.raft.commitIndex + ", ignoring");
742            return;
743        }
744
745        // Do we have an existing install?
746        boolean startNewInstall = false;
747        if (this.snapshotReceive != null) {
748
749            // Does the message not match?
750            if (!this.snapshotReceive.matches(msg)) {
751
752                // If the message is NOT the first one in a new install, ignore it
753                if (msg.getPairIndex() != 0) {
754                    if (this.log.isDebugEnabled())
755                        this.debug("rec'd " + msg + " which doesn't match in-progress " + this.snapshotReceive + "; ignoring");
756                    return;
757                }
758
759                // The message is the first one in a new install, so discard the existing install
760                if (this.log.isDebugEnabled()) {
761                    this.debug("rec'd initial " + msg + " with in-progress " + this.snapshotReceive
762                      + "; aborting previous install");
763                }
764                startNewInstall = true;
765            }
766        } else {
767
768            // If the message is NOT the first one in a new install, ignore it
769            if (msg.getPairIndex() != 0) {
770                if (this.log.isDebugEnabled())
771                    this.debug("rec'd non-initial " + msg + " with no in-progress snapshot install; ignoring");
772                return;
773            }
774        }
775
776        // Get snapshot term and index
777        final long term = msg.getSnapshotTerm();
778        final long index = msg.getSnapshotIndex();
779
780        // Set up new install if necessary
781        if (this.snapshotReceive == null || startNewInstall) {
782            assert msg.getPairIndex() == 0;
783            if (this.raft.discardFlipFloppedStateMachine())
784                this.warn("detected left-over content in flip-flopped state machine; discarding");
785            this.updateElectionTimer();
786            this.snapshotReceive = new SnapshotReceive(PrefixKVStore.create(this.raft.kv,
787              this.raft.getFlipFloppedStateMachinePrefix()), term, index, msg.getSnapshotConfig());
788            if (this.log.isDebugEnabled()) {
789                this.debug("starting new snapshot install from \"" + msg.getSenderId()
790                  + "\" of " + index + "t" + term + " with config " + msg.getSnapshotConfig());
791            }
792        }
793        assert this.snapshotReceive.matches(msg);
794
795        // Apply next chunk of key/value pairs
796        if (this.log.isDebugEnabled())
797            this.debug("applying " + msg + " to " + this.snapshotReceive);
798        try {
799            this.snapshotReceive.applyNextChunk(msg.getData());
800        } catch (Exception e) {
801            this.error("error applying snapshot to key/value store; aborting snapshot install", e);
802            this.snapshotReceive = null;
803            this.raft.discardFlipFloppedStateMachine();
804            this.updateElectionTimer();
805            return;
806        }
807
808        // If that was the last chunk, finalize persistent state
809        if (msg.isLastChunk()) {
810            final Map<String, String> snapshotConfig = this.snapshotReceive.getSnapshotConfig();
811            if (this.log.isDebugEnabled()) {
812                this.debug("snapshot install from \"" + msg.getSenderId() + "\" of "
813                  + index + "t" + term + " with config " + snapshotConfig + " complete");
814            }
815            this.snapshotReceive = null;
816            this.raft.flipFlopStateMachine(term, index, snapshotConfig);
817            this.updateElectionTimer();
818        }
819    }

821    @Override
822    void caseRequestVote(RequestVote msg) {
823
824        // Record new cluster ID if we haven't done so already
825        if (this.raft.clusterId == 0)
826            this.raft.joinCluster(msg.getClusterId());
827
828        // Did we already vote for somebody else?
829        final String peer = msg.getSenderId();
830        if (this.votedFor != null && !this.votedFor.equals(peer)) {
831            if (this.log.isDebugEnabled())
832                this.debug("rec'd " + msg + "; rejected because we already voted for \"" + this.votedFor + "\"");
833            return;
834        }
835
836        // Verify that we are allowed to vote for this peer
837        if (msg.getLastLogTerm() < this.raft.getLastLogTerm()
838          || (msg.getLastLogTerm() == this.raft.getLastLogTerm() && msg.getLastLogIndex() < this.raft.getLastLogIndex())) {
839            if (this.log.isDebugEnabled()) {
840                this.debug("rec'd " + msg + "; rejected because their log " + msg.getLastLogIndex() + "t"
841                  + msg.getLastLogTerm() + " loses to ours " + this.raft.getLastLogIndex() + "t" + this.raft.getLastLogTerm());
842            }
843            return;
844        }
845
846        // Persist our vote for this peer (if not already persisted)
847        if (this.votedFor == null) {
848            if (this.log.isDebugEnabled())
849                this.debug("granting vote to \"" + peer + "\" in term " + this.raft.currentTerm);
850            if (!this.updateVotedFor(peer))
851                return;
852        } else {
853            if (this.log.isDebugEnabled())
854                this.debug("confirming existing vote for \"" + peer + "\" in term " + this.raft.currentTerm);
855        }
856
857        // Send reply
858        this.raft.sendMessage(new GrantVote(this.raft.clusterId, this.raft.identity, peer, this.raft.currentTerm));
859    }

869    @Override
870    void casePingResponse(PingResponse msg) {
871
872        // Are we probing?
873        if (this.probeTimestamps == null) {
874            if (this.log.isTraceEnabled())
875                this.trace("ignoring " + msg + " rec'd while not probing in " + this);
876            return;
877        }
878
879        // Update peer's ping timestamp and re-check probe status
880        this.probeTimestamps.put(msg.getSenderId(), msg.getTimestamp());
881    }

885    /**
886     * Record the peer voted for in the current term.
887     */
888    private boolean updateVotedFor(String recipient) {
889
890        // Sanity check
891        assert Thread.holdsLock(this.raft);
892        assert recipient != null;
893
894        // Update persistent store
895        final Writes writes = new Writes();
896        writes.getPuts().put(RaftKVDatabase.VOTED_FOR_KEY, this.raft.encodeString(recipient));
897        try {
898            this.raft.kv.mutate(writes, true);
899        } catch (Exception e) {
900            this.error("error persisting vote for \"" + recipient + "\"", e);
901            return false;
902        }
903
904        // Done
905        this.votedFor = recipient;
906        return true;
907    }

924    @Override
925    boolean checkState() {
926        if (!super.checkState())
927            return false;
928        assert this.leaderAddress != null || this.leader == null;
929        assert this.electionTimer.isRunning() == this.raft.isClusterMember();
930        for (Map.Entry<Long, PendingRequest> entry : this.pendingRequests.entrySet()) {
931            final long txId = entry.getKey();
932            final PendingRequest pendingRequest = entry.getValue();
933            final RaftKVTransaction tx = pendingRequest.getTx();
934            assert txId == tx.getTxId();
935            assert tx.getState().equals(TxState.COMMIT_READY);
936            assert tx.getCommitTerm() == 0;
937            assert tx.getCommitIndex() == 0;
938        }
939        for (Map.Entry<Long, PendingWrite> entry : this.pendingWrites.entrySet()) {
940            final long txId = entry.getKey();
941            final PendingWrite pendingWrite = entry.getValue();
942            final RaftKVTransaction tx = pendingWrite.getTx();
943            assert txId == tx.getTxId();
944            assert tx.getState().equals(TxState.COMMIT_READY) || tx.getState().equals(TxState.COMMIT_WAITING);
945            assert pendingWrite.getFileWriter().getFile().exists();
946        }
947        return true;
948    }

953    // but no CommitResponse has yet been received

971    // Represents a read-write transaction in COMMIT_READY or COMMIT_WAITING for which the server's AppendRequest
972    // will have null mutationData, because we will already have the data on hand waiting in a temporary file. This
973    // is a simple optimization to avoid sending the same data from leader -> follower just sent from follower -> leader.
974    private static class PendingWrite {
975
976        private final RaftKVTransaction tx;
977        private final FileWriter fileWriter;
978
979        PendingWrite(RaftKVTransaction tx, FileWriter fileWriter) {
980            this.tx = tx;
981            this.fileWriter = fileWriter;
982        }
983
984        public RaftKVTransaction getTx() {
985            return this.tx;
986        }
987
988        public FileWriter getFileWriter() {
989            return this.fileWriter;
990        }
991
992        public void cleanup() {
993            this.fileWriter.getFile().delete();
994            Util.closeIfPossible(this.fileWriter);
995        }
