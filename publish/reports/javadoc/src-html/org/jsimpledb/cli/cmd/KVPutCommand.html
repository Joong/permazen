006package org.jsimpledb.cli.cmd;
007
008import java.io.PrintWriter;
009import java.util.Map;
010
011import org.jsimpledb.SessionMode;
012import org.jsimpledb.cli.CliSession;
013import org.jsimpledb.kv.KVTransaction;
014import org.jsimpledb.parse.ParseContext;
015
016@Command(modes = { SessionMode.KEY_VALUE })
017public class KVPutCommand extends AbstractKVCommand {
018
019    public KVPutCommand() {
020        super("kvput key:bytes value:bytes");
021    }
022
023    @Override
024    public String getHelpSummary() {
025        return "Write a raw database key/value pair";
026    }
027
028    @Override
029    public String getHelpDetail() {
030        return "Writes a raw database key/value pair. The `key' and `value' parameters may be given as a hexadecimal value"
031          + " or C-style doubly-quoted string."
032          + "\n\nWARNING: this command can corrupt a JSimpleDB database.";
033    }
034
035    @Override
036    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
037        final byte[] key = (byte[])params.get("key");
038        final byte[] value = (byte[])params.get("value");
039        return new CliSession.Action() {
040            @Override
041            public void run(CliSession session) throws Exception {
042                final PrintWriter writer = session.getWriter();
043                final KVTransaction kvt = session.getKVTransaction();
044                kvt.put(key, value);
045            }
046        };
047    }
048}
