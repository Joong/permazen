006package org.jsimpledb.annotation;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.ElementType;
010import java.lang.annotation.Inherited;
011import java.lang.annotation.Retention;
012import java.lang.annotation.RetentionPolicy;
013import java.lang.annotation.Target;
014
015/**
016 * Annotates a JSimpleDB model class method that should be invoked any time the associated model object is validated.
017 *
018 * <p>
019 * This annotation does not change when an object will be enqueued for validation. It only affects the behavior once validation
020 * of an instance is actually performed.
021 *
022 * <p>
023 * When validating an object, validation via {@link OnValidate &#64;OnValidate} methods occurs after JSR 303 validation, if any.
024 *
025 * <p>
026 * An optional list of validation {@link #groups} may be specified; if so, the annotated method will only be invoked when
027 * one or more of specified validation groups (or a superclass thereof) is being validated. In particular, if {@link #groups}
028 * is non-empty and does not contain a class that extends {@link javax.validation.groups.Default}, then the method will
029 * not be invoked by automatic validation.
030 *
031 * <p>
032 * The annotated method must be an instance method (i.e., not static), return void, and take zero parameters.
033 * If validation fails, the method should throw a {@link org.jsimpledb.ValidationException}.
034 *
035 * @see org.jsimpledb.JTransaction#validate
036 * @see org.jsimpledb.JObject#revalidate
037 * @see org.jsimpledb.ValidationMode
038 */
039@Documented
040@Inherited
041@Retention(RetentionPolicy.RUNTIME)
042@Target(ElementType.METHOD)
043public @interface OnValidate {
044
045    /**
046     * Specify the validation group(s) for which the annotated method should be invoked.
047     *
048     * @return validation group(s) to use for validation; if empty, {@link javax.validation.groups.Default} is assumed
049     */
050    Class<?>[] groups() default {};
051}
