006package org.jsimpledb.kv.mvcc;
007
008import com.google.common.base.Preconditions;
009
010import javax.annotation.PostConstruct;
011import javax.annotation.PreDestroy;
012
013import org.jsimpledb.kv.AbstractKVStore;
014import org.jsimpledb.kv.CloseableKVStore;
015import org.jsimpledb.kv.KVDatabase;
016import org.jsimpledb.kv.KVPair;
017import org.jsimpledb.kv.KVPairIterator;
018import org.jsimpledb.kv.KVStore;
019import org.jsimpledb.kv.KVTransaction;
020import org.jsimpledb.kv.KeyRange;

024import org.jsimpledb.util.ByteUtil;

027/**
028 * Wrapper class that presents an {@link AtomicKVStore} view of a {@link KVDatabase},
029 * using individual transactions for each operation.
030 *
031 * <p>
032 * Warning: this class is only appropriate for use with {@link KVDatabase}s that implement some form of MVCC;
033 * {@link KVDatabase}s that use locking will likely generate conflicts if {@link #snapshot} is used concurrently
034 * with other methods.
035 *
036 * @see SnapshotKVDatabase
037 */
038public class AtomicKVDatabase extends AbstractKVStore implements AtomicKVStore {
039
040    protected final KVDatabase kvdb;
041
042    /**
043     * Constructor.
044     *
045     * @param kvdb underlying database
046     * @throws IllegalArgumentException if {@code kvdb} is null
047     */
048    public AtomicKVDatabase(KVDatabase kvdb) {
049        Preconditions.checkArgument(kvdb != null, "null kvdb");
050        this.kvdb = kvdb;
051    }
052
053// KVStore
054
055    @Override
056    public byte[] get(final byte[] key) {
057        return this.doInTransaction(new Action<byte[]>() {
058            @Override
059            public byte[] apply(KVStore kv) {
060                return kv.get(key);
061            }
062        });
063    }
064
065    @Override
066    public KVPair getAtLeast(final byte[] minKey) {
067        return this.doInTransaction(new Action<KVPair>() {
068            @Override
069            public KVPair apply(KVStore kv) {
070                return kv.getAtLeast(minKey);
071            }
072        });
073    }
074
075    @Override
076    public KVPair getAtMost(final byte[] maxKey) {
077        return this.doInTransaction(new Action<KVPair>() {
078            @Override
079            public KVPair apply(KVStore kv) {
080                return kv.getAtMost(maxKey);
081            }
082        });
083    }
084
085    @Override
086    public KVPairIterator getRange(byte[] minKey, byte[] maxKey, boolean reverse) {
087        return new KVPairIterator(this, new KeyRange(minKey != null ? minKey : ByteUtil.EMPTY, maxKey), null, reverse);
088    }
089
090    @Override
091    public void put(final byte[] key, final byte[] value) {
092        this.doInTransaction(new Action<Void>() {
093            @Override
094            public Void apply(KVStore kv) {
095                kv.put(key, value);
096                return null;
097            }
098        });
099    }
100
101    @Override
102    public void remove(final byte[] key) {
103        this.doInTransaction(new Action<Void>() {
104            @Override
105            public Void apply(KVStore kv) {
106                kv.remove(key);
107                return null;
108            }
109        });
110    }
111
112    @Override
113    public void removeRange(final byte[] minKey, final byte[] maxKey) {
114        this.doInTransaction(new Action<Void>() {
115            @Override
116            public Void apply(KVStore kv) {
117                kv.removeRange(minKey, maxKey);
118                return null;
119            }
120        });
121    }
122
123    @Override
124    public void adjustCounter(final byte[] key, final long amount) {
125        this.doInTransaction(new Action<Void>() {
126            @Override
127            public Void apply(KVStore kv) {
128                kv.adjustCounter(key, amount);
129                return null;
130            }
131        });
132    }
133
134    @Override
135    public byte[] encodeCounter(final long value) {
136        return this.doInTransaction(new Action<byte[]>() {
137            @Override
138            public byte[] apply(KVStore kv) {
139                return kv.encodeCounter(value);
140            }
141        });
142    }
143
144    @Override
145    public long decodeCounter(final byte[] bytes) {
146        return this.doInTransaction(new Action<Long>() {
147            @Override
148            public Long apply(KVStore kv) {
149                return kv.decodeCounter(bytes);
150            }
151        });
152    }
153
154// AtomicKVStore
155
156    @Override
157    @PostConstruct
158    public void start() {
159        this.kvdb.start();
160    }
161
162    @Override
163    @PreDestroy
164    public void stop() {
165        this.kvdb.stop();
166    }
167
168    @Override
169    public CloseableKVStore snapshot() {
170        final KVTransaction kvtx = this.kvdb.createTransaction();
171        boolean success = false;
172        try {
173            try {
174                kvtx.setTimeout(Integer.MAX_VALUE);
175            } catch (UnsupportedOperationException e) {
176                // ignore
177            }
178            final SnapshotKVStore kvstore = new SnapshotKVStore(kvtx);
179            success = true;
180            return kvstore;
181        } finally {
182            if (!success)
183                kvtx.rollback();
184        }
185    }
186
187    @Override
188    public void mutate(final Mutations mutations, boolean sync) {
189        Preconditions.checkArgument(mutations != null, "null mutations");
190        this.doInTransaction(new Action<Void>() {
191            @Override
192            public Void apply(KVStore kv) {
193                Writes.apply(mutations, kv);
194                return null;
195            }
196        });
197    }
198
199// Object
200
201    @Override
202    public String toString() {
203        return this.getClass().getSimpleName() + "[kvdb=" + this.kvdb + "]";
204    }
205
206// Internal methods
207
208    private <R> R doInTransaction(Action<R> action) {
209        int count = 0;
210        while (true) {
211            try {
212                final KVTransaction kvtx = this.kvdb.createTransaction();
213                boolean success = false;
214                try {
215                    final R result = action.apply(kvtx);
216                    success = true;
217                    return result;
218                } finally {
219                    if (success)
220                        kvtx.commit();
221                    else
222                        kvtx.rollback();
223                }
224            } catch (RetryTransactionException e) {
225                if (count++ < 3) {
226                    Thread.yield();
227                    continue;
228                }
229                throw e;
230            }
231        }
232    }
233
234// Action
235
236    private interface Action<R> {
237        R apply(KVStore kv);
238    }
239
240// Snapshot
241
242    private static class SnapshotKVStore extends ForwardingKVStore implements CloseableKVStore {
243
244        private final KVTransaction kvtx;
245        private final UnmodifiableKVStore delegate;
246
247        private volatile boolean closed;
248
249        SnapshotKVStore(KVTransaction kvtx) {
250            this.kvtx = kvtx;
251            this.delegate = new UnmodifiableKVStore(this.kvtx);
252        }
253
254        @Override
255        protected UnmodifiableKVStore delegate() {
256            return this.delegate;
257        }
258
259    // Object
260
261        @Override
262        protected void finalize() throws Throwable {
263            try {
264                if (!this.closed)
265                    LoggerFactory.getLogger(this.getClass()).warn(this + " leaked without invoking close()");
266                this.close();
267            } finally {
268                super.finalize();
269            }
270        }
271
272    // Closeable
273
274        @Override
275        public void close() {
276            this.closed = true;
277            this.kvtx.rollback();
278        }
279    }
280}
