006package org.jsimpledb.kv.raft;
007
008/**
009 * {@link RaftKVTransaction} supported consistency levels.
010 */
011public enum Consistency {
012
013    /**
014     * Uncommitted consistency.
015     *
016     * <p>
017     * Same as {@link #EVENTUAL}, but with the additional caveat that the view that the transaction sees may never
018     * actually get committed. This can happen, e.g., if there is a Raft leadership change during the transaction.
019     *
020     * <p>
021     * Transactions at this level may result in less network communication and/or commit faster than {@link #EVENTUAL}.
022     * Read-only transactions at this level require no network communication, commit immediately, and never result in
023     * {@link org.jsimpledb.kv.RetryTransactionException}s; in fact, read-only transactions are equivalent to invoking
024     * {@link RaftKVTransaction#rollback rollback()} instead of {@link RaftKVTransaction#commit commit()}.
025     */
026    UNCOMMITTED,
027
028    /**
029     * Eventual consistency.
030     *
031     * <p>
032     * Transactions see a consistent, committed view of the database as it existed at some point in the "recent past".
033     * The view is not guaranteed to be up-to-date; it's only guaranteed to be as up-to-date as what was known to this
034     * node when the transaction was opened. In general (for example, if stuck in a network partition minority), the view
035     * could be from arbitrarily far in the past.
036     *
037     * <p>
038     * Transactions at this level may result in less network communication and/or commit faster than {@link #LINEARIZABLE}.
039     */
040    EVENTUAL,
041
042    /**
043     * Linearizable consistency.
044     *
045     * <p>
046     * Transactions see a database that evolves step-wise through a global, linear sequence of atomic changes, where it appears as
047     * if each change occurs instantaneously at some point in time between the start and end of the corresponding transaction.
048     *
049     * <p>
050     * This is the default consistency level.
051     */
052    LINEARIZABLE;
053}
