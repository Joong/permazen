006package org.jsimpledb.core;
007
008import org.jsimpledb.kv.KVStore;
009
010/**
011 * A "snapshot" {@link Transaction} that persists indefinitely.
012 *
013 * <p>
014 * {@link SnapshotTransaction}s hold a "snapshot" of some portion of the state of a {@link Transaction}
015 * for later use. Each {@link SnapshotTransaction} contains its own set of "snapshot" objects.
016 *
017 * <p>
018 * {@link SnapshotTransaction}s can never be closed (i.e., committed or rolled-back); they persist in memory until
019 * no longer referenced. {@link Transaction.Callback}s may be registered but they will never be invoked.
020 *
021 * <p>
022 * {@link SnapshotTransaction}s can be based on an arbitrary {@link KVStore};
023 * see {@link Database#createSnapshotTransaction Database.createSnapshotTransaction()}.
024 *
025 * @see Transaction#createSnapshotTransaction Transaction.createSnapshotTransaction()
026 * @see Database#createSnapshotTransaction Database.createSnapshotTransaction()
027 * @see org.jsimpledb.SnapshotJTransaction
028 */
029public class SnapshotTransaction extends Transaction {
030
031// Constructors
032
033    SnapshotTransaction(Database db, KVStore kvstore, Schemas schemas) {
034        super(db, new SnapshotKVTransaction(kvstore), schemas);
035    }
036
037    SnapshotTransaction(Database db, KVStore kvstore, Schemas schemas, int versionNumber) {
038        super(db, new SnapshotKVTransaction(kvstore), schemas, versionNumber);
039    }
040
041    SnapshotTransaction(Database db, KVStore kvstore, Schemas schemas, Schema schema) {
042        super(db, new SnapshotKVTransaction(kvstore), schemas, schema);
043    }
044
045// Methods
046
047    /**
048     * Get the underlying {@link KVStore} that holds this snapshot transaction's state.
049     *
050     * @return underlying {@link KVStore}
051     */
052    public KVStore getKVStore() {
053        return ((SnapshotKVTransaction)this.kvt).delegate();
054    }
055
056    /**
057     * Delete all objects contained in this snapshot transaction.
058     *
059     * <p>
060     * It will contain schema meta-data but no objects.
061     * </p>
062     */
063    public void reset() {
064
065        // Sanity check
066        if (this.stale)
067            throw new StaleTransactionException(this);
068
069        // Delete all object and index keys
070        this.db.reset(this);
071    }
072
073    /**
074     * Commit this transaction.
075     *
076     * <p>
077     * {@link SnapshotTransaction}s do not support this method and will always throw {@link UnsupportedOperationException}.
078     * </p>
079     *
080     * @throws UnsupportedOperationException always
081     */
082    @Override
083    public void commit() {
084        throw new UnsupportedOperationException("snapshot transaction");
085    }
086
087    /**
088     * Roll back this transaction.
089     *
090     * <p>
091     * {@link SnapshotTransaction}s do not support this method and will always throw {@link UnsupportedOperationException}.
092     * </p>
093     *
094     * @throws UnsupportedOperationException always
095     */
096    @Override
097    public void rollback() {
098        throw new UnsupportedOperationException("snapshot transaction");
099    }
100
101    /**
102     * Register a transaction callback to be invoked when this transaction completes.
103     *
104     * <p>
105     * {@link Transaction.Callback}s registered with a {@link SnapshotTransaction} will by definition never be invoked.
106     * Therefore, this method simply discards {@code callback}.
107     * </p>
108     */
109    @Override
110    public void addCallback(Callback callback) {
111    }
112
113    /**
114     * Determine whether this transaction is still valid.
115     *
116     * <p>
117     * {@link SnapshotTransaction}s are always valid.
118     * </p>
119     *
120     * @return true always
121     */
122    @Override
123    public boolean isValid() {
124        return super.isValid();
125    }
126}
