006package org.jsimpledb.kv.util;
007
008import com.google.common.base.Preconditions;
009import com.google.common.collect.AbstractIterator;
010
011import java.io.BufferedInputStream;
012import java.io.EOFException;
013import java.io.IOException;
014import java.io.InputStream;
015import java.io.OutputStream;
016import java.util.Iterator;
017
018import org.jsimpledb.kv.KVPair;
019import org.jsimpledb.util.LongEncoder;
020import org.jsimpledb.util.UnsignedIntEncoder;
021
022/**
023 * Serializes a sequence of {@code byte[]} arrays, compressing consecutive common prefixes.
024 *
025 * <p>
026 * Keys are encoded/decode by {@link #read read()} and {@link #write write()} in one of two forms:
027 *  <ul>
028 *  <li>{@code total-length bytes...}
029 *  <li>{@code -prefix-length suffix-length suffix-bytes ...}
030 *  </ul>
031 * The first length ({@code total-length} or negative {@code prefix-length}) is encoded using {@link LongEncoder}.
032 * The {@code suffix-length}, if present, is encoded using {@link UnsignedIntEncoder}.
033 *
034 * <p>
035 * Support for encoding and decoding an entire iteration of key/value pairs is supported via
036 * {@link #readPairs readPairs()} and {@link #writePairs writePairs()}.
037 */
038public final class KeyListEncoder {
039
040    private KeyListEncoder() {
041    }
042
043    /**
044     * Write the next key, compressing its common prefix with the previous key (if any).
045     *
046     * @param out output stream
047     * @param key key to write
048     * @param prev previous key, or null for none
049     * @throws IOException if an I/O error occurs
050     * @throws IllegalArgumentException if {@code out} or {@code key} is null
051     */
052    public static void write(OutputStream out, byte[] key, byte[] prev) throws IOException {
053        Preconditions.checkArgument(out != null, "null out");
054        Preconditions.checkArgument(key != null, "null key");
055        int prefixLength = 0;
056        if (prev != null) {
057            while (prefixLength < key.length && prefixLength < prev.length && key[prefixLength] == prev[prefixLength])
058                prefixLength++;
059        }
060        if (prefixLength > 1) {
061            final int suffixLength = key.length - prefixLength;
062            LongEncoder.write(out, ~(prefixLength - 2));
063            UnsignedIntEncoder.write(out, suffixLength);
064            out.write(key, prefixLength, suffixLength);
065        } else {
066            LongEncoder.write(out, key.length);
067            out.write(key);
068        }
069    }
070
071    /**
072     * Calculate the number of bytes that would be required to write the next key via {@link #write write()}.
073     *
074     * @param key key to write
075     * @param prev previous key, or null for none
076     * @return number of bytes to be written by {@link #write write(out, key, prev)}
077     * @throws IllegalArgumentException if {@code key} is null
078     */
079    public static int writeLength(byte[] key, byte[] prev) {
080        Preconditions.checkArgument(key != null, "null key");
081        int prefixLength = 0;
082        if (prev != null) {
083            while (prefixLength < key.length && prefixLength < prev.length && key[prefixLength] == prev[prefixLength])
084                prefixLength++;
085        }
086        if (prefixLength > 1) {
087            final int suffixLength = key.length - prefixLength;
088            return LongEncoder.encodeLength(~(prefixLength - 2)) + UnsignedIntEncoder.encodeLength(suffixLength) + suffixLength;
089        } else
090            return LongEncoder.encodeLength(key.length) + key.length;
091    }
092
093    /**
094     * Read the next key.
095     *
096     * @param input input stream
097     * @param prev previous key, or null for none
098     * @return next key
099     * @throws IOException if an I/O error occurs
100     * @throws EOFException if an unexpected EOF is encountered
101     * @throws IllegalArgumentException if {@code input} is null
102     * @throws IllegalArgumentException if {@code input} contains invalid data
103     */
104    public static byte[] read(InputStream input, byte[] prev) throws IOException {
105        Preconditions.checkArgument(input != null, "null input");
106
107        // Get encoded length of prefix
108        int keyLength = KeyListEncoder.readSignedInt(input);
109        final byte[] key;
110        int prefixLength;
111
112        // Decode prefix length and read prefix, if any
113        if (keyLength < 0) {
114            if (prev == null)
115                throw new IllegalArgumentException("null `prev' given but next key has " + -keyLength + " byte shared prefix");
116            prefixLength = ~keyLength + 2;
117            final int suffixLength = UnsignedIntEncoder.read(input);
118            keyLength = prefixLength + suffixLength;
119            if (keyLength < 0)
120                throw new IllegalArgumentException("invalid prefix length " + prefixLength + " plus suffix length " + suffixLength);
121            key = new byte[keyLength];
122            System.arraycopy(prev, 0, key, 0, prefixLength);
123        } else {
124            key = new byte[keyLength];
125            prefixLength = 0;
126        }
127
128        // Read suffix
129        while (prefixLength < key.length) {
130            final int num = input.read(key, prefixLength, key.length - prefixLength);
131            if (num == -1)
132                throw new EOFException();
133            prefixLength += num;
134        }
135
136        // Done
137        return key;
138    }
139
140    /**
141     * Encode an iteration of key/value pairs.
142     *
143     * @param kvpairs key/value pair iteration
144     * @param output encoded output
145     * @throws IOException if an I/O error occurs
146     * @throws IllegalArgumentException if either parameter is null
147     */
148    public static void writePairs(Iterator<KVPair> kvpairs, OutputStream output) throws IOException {
149
150        // Sanity check
151        Preconditions.checkArgument(kvpairs != null, "null kvpairs");
152        Preconditions.checkArgument(output != null, "null output");
153
154        // Write pairs
155        byte[] prev = null;
156        while (kvpairs.hasNext()) {
157            final KVPair kv = kvpairs.next();
158            final byte[] key = kv.getKey();
159            final byte[] value = kv.getValue();
160            KeyListEncoder.write(output, key, prev);
161            KeyListEncoder.write(output, value, null);
162            prev = key;
163        }
164
165        // Write special terminator byte
166        output.write(0xff);
167    }
168
169    /**
170     * Determine the number of bytes that would be written by {@link #writePairs writePairs()}.
171     *
172     * @param kvpairs key/value pair iteration
173     * @return encoded length of this instance
174     * @throws IllegalArgumentException if {@code kvpairs} is null
175     */
176    public static long writePairsLength(Iterator<KVPair> kvpairs) {
177        long total = 1;
178        byte[] prev = null;
179        while (kvpairs.hasNext()) {
180            final KVPair kv = kvpairs.next();
181            final byte[] key = kv.getKey();
182            final byte[] value = kv.getValue();
183            total += KeyListEncoder.writeLength(key, prev);
184            total += KeyListEncoder.writeLength(value, null);
185            prev = key;
186        }
187        return total;
188    }
189
190    /**
191     * Decode an iteration of key/value pairs previously encoded by {@link #writePairs writePairs()}.
192     *
193     * <p>
194     * If an {@link IOException} occurs during iteration, the returned {@link Iterator} wraps it in a {@link RuntimeException}.
195     *
196     * <p>
197     * If invalid input is encountered during iteration, the returned {@link Iterator}
198     * will throw an {@link IllegalArgumentException}.
199     *
200     * @param input encoded input
201     * @return iteration of key/value pairs
202     * @throws IllegalArgumentException if {@code input} is null
203     */
204    public static Iterator<KVPair> readPairs(final InputStream input) {
205
206        // Sanity check
207        Preconditions.checkArgument(input != null, "null input");
208
209        // Decode
210        return new AbstractIterator<KVPair>() {
211
212            private final BufferedInputStream in = new BufferedInputStream(input, 1024);
213            private byte[] prev;
214
215            @Override
216            protected KVPair computeNext() {
217                try {
218
219                    // Check for terminator
220                    this.in.mark(1);
221                    final int b = in.read();
222                    if (b == -1)
223                        throw new EOFException("truncated input");
224                    if (b == 0xff)
225                        return endOfData();
226                    this.in.reset();
227
228                    // Read next k/v pair
229                    final byte[] key = KeyListEncoder.read(this.in, this.prev);
230                    final byte[] value = KeyListEncoder.read(this.in, null);
231                    this.prev = key;
232
233                    // Done
234                    return new KVPair(key, value);
235                } catch (IOException e) {
236                    throw new RuntimeException("I/O error during iteration", e);
237                }
238            }
239        };
240    }
241
242    private static int readSignedInt(InputStream input) throws IOException {
243        final long longValue = LongEncoder.read(input);
244        final int intValue = (int)longValue;
245        Preconditions.checkArgument(intValue == (int)longValue, "read out-of-range encoded int value %s", longValue);
246        return intValue;
247    }
248}
