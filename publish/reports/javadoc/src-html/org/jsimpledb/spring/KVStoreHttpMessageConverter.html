006package org.jsimpledb.spring;
007
008import com.google.common.base.Preconditions;
009
010import java.io.IOException;
011import java.util.Iterator;
012
013import org.jsimpledb.kv.KVPair;
014import org.jsimpledb.kv.KVStore;
015import org.jsimpledb.kv.util.KeyListEncoder;
016import org.jsimpledb.kv.util.NavigableMapKVStore;
017import org.springframework.http.HttpInputMessage;
018import org.springframework.http.HttpOutputMessage;
019import org.springframework.http.MediaType;
020import org.springframework.http.converter.AbstractHttpMessageConverter;
021import org.springframework.http.converter.HttpMessageNotReadableException;
022
023/**
024 * Spring {@link org.springframework.http.converter.HttpMessageConverter HttpMessageConverter}
025 * capable of encoding and decoding a {@link KVStore}. The {@link KeyListEncoder} class is used
026 * to compress common prefixes of consecutive keys.
027 *
028 * <p>
029 * See {@link JObjectHttpMessageConverter} for a higher level API.
030 *
031 * @see SnapshotJTransactionHttpMessageConverter
032 * @see JObjectHttpMessageConverter
033 */
034public class KVStoreHttpMessageConverter extends AbstractHttpMessageConverter<KVStore> {
035
036    /**
037     * Default MIME type supported by this instance: {@code application/x-jsimpledb-kvstore}.
038     */
039    public static final MediaType DEFAULT_MIME_TYPE = new MediaType("application", "x-jsimpledb-kvstore");
040
041    /**
042     * Constructor.
043     *
044     * <p>
045     * Configures this instance for the {@link #DEFAULT_MIME_TYPE}.
046     */
047    public KVStoreHttpMessageConverter() {
048        super(DEFAULT_MIME_TYPE);
049    }
050
051    /**
052     * Constructor.
053     *
054     * @param mimeTypes supported MIME type(s)
055     */
056    public KVStoreHttpMessageConverter(MediaType... mimeTypes) {
057        super(mimeTypes);
058    }
059
060// AbstractHttpMessageConverter
061
062    @Override
063    protected Long getContentLength(KVStore kvstore, MediaType mediaType) {
064        return KVStoreHttpMessageConverter.getKVStoreContentLength(kvstore);
065    }
066
067    @Override
068    protected boolean supports(Class<?> clazz) {
069        return clazz == KVStore.class || clazz == NavigableMapKVStore.class;
070    }
071
072    @Override
073    protected KVStore readInternal(Class<? extends KVStore> clazz, HttpInputMessage input) throws IOException {
074        final NavigableMapKVStore kvstore = new NavigableMapKVStore();
075        KVStoreHttpMessageConverter.readKVStore(kvstore, input);
076        return clazz.cast(kvstore);
077    }
078
079    @Override
080    protected void writeInternal(KVStore kvstore, HttpOutputMessage output) throws IOException {
081        KVStoreHttpMessageConverter.writeKVStore(kvstore, output);
082    }
083
084// Utility methods
085
086    /**
087     * Determine the content length of the given key/value store when encoded as payload.
088     *
089     * @param kvstore key/value store to encode
090     * @return payload content length
091     * @throws IllegalArgumentException if {@code kvstore} is null
092     */
093    public static Long getKVStoreContentLength(KVStore kvstore) {
094        Preconditions.checkArgument(kvstore != null, "null kvstore");
095        return KeyListEncoder.writePairsLength(kvstore.getRange(null, null, false));
096    }
097
098    /**
099     * Decode a key/value store HTTP payload.
100     *
101     * @param kvstore key/value store to populate from input
102     * @param input HTTP payload input
103     * @throws HttpMessageNotReadableException if {@code input} contains invalid content
104     * @throws IOException if an I/O error occurs
105     * @throws IllegalArgumentException if either parameter is null
106     */
107    public static void readKVStore(KVStore kvstore, HttpInputMessage input) throws IOException {
108        Preconditions.checkArgument(kvstore != null, "null kvstore");
109        Preconditions.checkArgument(input != null, "null input");
110        try {
111            for (Iterator<KVPair> i = KeyListEncoder.readPairs(input.getBody()); i.hasNext(); ) {
112                final KVPair kv = i.next();
113                kvstore.put(kv.getKey(), kv.getValue());
114            }
115        } catch (IllegalArgumentException e) {
116            throw new HttpMessageNotReadableException("invalid endoded key/value store", e);
117        } catch (RuntimeException e) {
118            if (e.getCause() instanceof IOException)
119                throw (IOException)e.getCause();
120            throw e;
121        }
122    }
123
124    /**
125     * Encode a key/value store HTTP payload.
126     *
127     * @param kvstore key/value store to encode
128     * @param output HTTP payload output
129     * @throws IOException if an I/O error occurs
130     * @throws IllegalArgumentException if either parameter is null
131     */
132    public static void writeKVStore(KVStore kvstore, HttpOutputMessage output) throws IOException {
133        Preconditions.checkArgument(kvstore != null, "null kvstore");
134        Preconditions.checkArgument(output != null, "null output");
135        KeyListEncoder.writePairs(kvstore.getRange(null, null, false), output.getBody());
136    }
137}
