050/**
051 * Vaadin {@link com.vaadin.data.Container} backed by {@link JSimpleDB} Java model objects.
052 *
053 * <p>
054 * Automatically creates container properties for object ID, database type, schema version, and all fields, as well as any custom
055 * properties defined by {@link org.dellroad.stuff.vaadin7.ProvidesProperty &#64;ProvidesProperty}-annotated methods in
056 * Java model classes. The properties of each {@link com.vaadin.data.Item} in the container are derived from a corresponding
057 * {@link JObject} which is usually stored in an in-memory {@link org.jsimpledb.SnapshotJTransaction} (which may contain other
058 * related objects, allowing an {@link com.vaadin.data.Item}'s properties to be derived from those related objects as well).
059 *
060 * <p>
061 * Instances are configured with a <b>type</b>, which can be any Java type. The container will then be restricted to
062 * database objects that are instances of the configured type. The type may be null, in which case there is no restriction.
063 *
064 * <p>
065 * Instances are loaded by invoking {@link #load load()} with an iteration of backing {@link JObject}s.
066 * Normally these {@link JObject}s are contained in a {@link org.jsimpledb.SnapshotJTransaction}.
067 *
068 * <p>
069 * Instances implement {@link org.dellroad.stuff.vaadin7.Connectable} and therefore must be {@link #connect connect()}'ed
070 * prior to use and {@link #disconnect disconnect()}'ed after use (usually done in the associated widget's
071 * {@link com.vaadin.ui.Component#attach attach()} and {@link com.vaadin.ui.Component#detach detach()} methods).
072 *
073 * <p>
074 * <b>Container Properties</b>
075 *
076 * <p>
077 * Instances have the following container properties:
078 * <ul>
079 *  <li>{@link #OBJECT_ID_PROPERTY}: Object {@link ObjId}</li>
080 *  <li>{@link #TYPE_PROPERTY}: Object type name (JSimpleDB type name, not Java type name, though the former
081 *      is by default the simple Java type name)</li>
082 *  <li>{@link #VERSION_PROPERTY}: Object schema version</li>
083 *  <li>{@link #REFERENCE_LABEL_PROPERTY}: Object <b>reference label</b>, which is a short description identifying the
084 *      object. Reference labels are used to provide "names" for objects that are more meaningful than object ID's
085 *      and are used as such in other {@link JSimpleDB} GUI classes, for example when displaying the object in a list.
086 *      To customize the reference label for a Java model class,
087 *      annotate a method with {@link org.dellroad.stuff.vaadin7.ProvidesProperty &#64;ProvidesProperty}{@code (}{@link
088 *      JObjectContainer#REFERENCE_LABEL_PROPERTY REFERENCE_LABEL_PROPERTY}{@code )};
089 *      otherwise, the value of this property will be the same as {@link #OBJECT_ID_PROPERTY}. Note that objects with
090 *      customized reference labels will need to be included in the snapshot transaction also if they are referenced
091 *      by an object actually in the container.
092 *      </li>
093 *  <li>A property for every {@link JSimpleDB} field that is common to all object types that sub-type
094 *      this containers's configured type. The property's ID is the field name; its value is as follows:
095 *      <ul>
096 *          <li>For simple fields, their {@linkplain org.jsimpledb.core.FieldType#toString(Object) string form}</li>
097 *          <li>For reference fields, the {@link #REFERENCE_LABEL_PROPERTY} of the referred-to object, or "Null"
098 *              if the reference is null</li>
099 *          <li>For set, list, and map fields, the first few entries in the collection</li>
100 *      </ul>
101 *  </li>
102 *  <li>A property for each {@link org.dellroad.stuff.vaadin7.ProvidesProperty &#64;ProvidesProperty}-annotated method
103 *      in the specified <b>type</b>. These properties will add to (or override) the properties listed above.
104 * </ul>
105 */

169     * Change the type restriction associated with this instance.
170     * Triggers a {@link com.vaadin.data.Container.PropertySetChangeEvent} and typically requires a reload.

172     * @param type Java type restriction, or null for none

175    public <T> void setType(Class<T> type) {
176        this.type = type;
177        this.propertyScanner = this.type != null ? new ProvidesPropertyScanner<T>(/*this.*/type) : null;
178        final ArrayList<PropertyDef<?>> propertyDefs = new ArrayList<>(this.buildPropertyDefs());
179        this.orderedPropertyNames = Collections.unmodifiableList(Lists.transform(propertyDefs,
180          new Function<PropertyDef<?>, String>() {
181            @Override
182            public String apply(PropertyDef<?> propertyDef) {
183                return propertyDef.getName();
184            }
185        }));
186        this.setProperties(propertyDefs);
187        this.fireContainerPropertySetChange();
188    }

218    /**
219     * Load this container using the supplied backing {@link JObject}s.
220     *
221     * <p>
222     * A container {@link com.vaadin.data.Item} will be created wrapping each iterated {@link JObject};
223     * {@link com.vaadin.data.Item} properties are accessible only while the containing transaction remains open.
224     *
225     * @param jobjs backing {@link JObject}s
226     */
227    @Override
228    public void load(Iterator<? extends JObject> jobjs) {
229
230        // Filter out any instances of the wrong type
231        if (JObjectContainer.this.type != null) {
232            jobjs = Iterators.filter(jobjs, new Predicate<JObject>() {
233                @Override
234                public boolean apply(JObject jobj) {
235                    return JObjectContainer.this.type.isInstance(jobj);
236                }
237            });
238        }
239
240        // Filter out nulls and duplicates
241        jobjs = this.filterNullsAndDuplicates(jobjs);
242
243        // Proceed
244        super.load(jobjs);
245    }
246
247    // This method exists solely to bind the generic type parameters
248    private <T extends JObject> Iterator<T> filterNullsAndDuplicates(final Iterator<T> jobjs) {
249        final ObjIdSet seenIds = new ObjIdSet();
250        return Iterators.filter(jobjs, new Predicate<T>() {
251            @Override
252            public boolean apply(T jobj) {
253                return jobj != null && seenIds.add(jobj.getObjId());
254            }
255        });
256    }

258    /**
259     * Update a single item in this container by updating its backing object.
260     *
261     * <p>
262     * This updates the backing object with the same object ID as {@code jobj}, if any, and then fires
263     * {@link com.vaadin.data.Property.ValueChangeEvent}s for all properties of the corresponding
264     * {@link com.vaadin.data.Item}.
265     *
266     * @param jobj updated database object
267     */
268    public void updateItem(JObject jobj) {
269        Preconditions.checkArgument(jobj != null, "null jobj");
270        final SimpleItem<JObject> item = (SimpleItem<JObject>)this.getItem(jobj.getObjId());
271        if (item != null) {
272            jobj.copyTo(item.getObject().getTransaction(), null, new CopyState());
273            item.fireValueChange();
274        }
275    }

277    /**
278     * Perform the given action within a new {@link JTransaction}.
279     *
280     * <p>
281     * The implementation in {@link JObjectContainer} performs {@code action} within a new read-only transaction.
282     *
283     * @param action the action to perform
284     */
285    protected void doInTransaction(Runnable action) {
286        final JTransaction jtx = this.jdb.createTransaction(false, ValidationMode.DISABLED);
287        jtx.getTransaction().setReadOnly(true);
288        try {
289            jtx.performAction(action);
290        } finally {
291            jtx.commit();
292        }
293    }

295// Property derivation

297    private Collection<PropertyDef<?>> buildPropertyDefs() {
298        final PropertyDefHolder pdefs = new PropertyDefHolder();
299
300        // Add properties shared by all JObjects
301        pdefs.setPropertyDef(this.refLabelPropertyDef);
302        pdefs.setPropertyDef(this.objIdPropertyDef);
303        pdefs.setPropertyDef(this.objTypePropertyDef);
304        pdefs.setPropertyDef(this.objVersionPropertyDef);
305
306        // Add properties for all fields common to all sub-types of our configured type
307        final SortedMap<Integer, JField> jfields = Util.getCommonJFields(this.jdb.getJClasses(this.type));
308        if (jfields != null) {
309            for (JField jfield : jfields.values())
310                pdefs.setPropertyDef(new ObjFieldPropertyDef(jfield.getStorageId(), jfield.getName()));
311        }
312
313        // Apply any @ProvidesProperty-annotated method properties, possibly overridding jfields
314        if (this.propertyScanner != null) {
315            for (PropertyDef<?> propertyDef : this.propertyScanner.getPropertyDefs())
316                pdefs.setPropertyDef(propertyDef);
317        }
318
319        // Done
320        return pdefs.values();
321    }

325    @Override
326    @SuppressWarnings("unchecked")
327    public <V> V getPropertyValue(JObject jobj, PropertyDef<V> propertyDef) {
328        if (propertyDef instanceof ObjPropertyDef)
329            return (V)((ObjPropertyDef<?>)propertyDef).extract(jobj);
330        if (this.propertyScanner == null)
331            throw new IllegalArgumentException("unknown property: " + propertyDef.getName());
332        return JObjectContainer.extractProperty(this.propertyScanner.getPropertyExtractor(), propertyDef, jobj);
333    }
334
335    @SuppressWarnings("unchecked")
336    private static <V> V extractProperty(PropertyExtractor<?> propertyExtractor, PropertyDef<V> propertyDef, JObject jobj) {
337        try {
338            return ((PropertyExtractor<JObject>)propertyExtractor).getPropertyValue(jobj, propertyDef);
339        } catch (DeletedObjectException e) {
340            try {
341                return propertyDef.getType().cast(new SizedLabel("<i>Unavailable</i>", ContentMode.HTML));
342            } catch (ClassCastException e2) {
343                try {
344                    return propertyDef.getType().cast("(Unavailable)");
345                } catch (ClassCastException e3) {
346                    return null;
347                }
348            }
349        }
350    }

420    /**
421     * Implements the {@link JObjectContainer#REFERENCE_LABEL_PROPERTY} property.
422     */
423    public static class RefLabelPropertyDef extends ObjPropertyDef<Component> {
424
425        public RefLabelPropertyDef() {
426            super(REFERENCE_LABEL_PROPERTY, Component.class);
427        }
428
429        @Override
430        public Component extract(JObject jobj) {
431            final ReferenceMethodInfoCache.PropertyInfo<?> propertyInfo
432              = ReferenceMethodInfoCache.getInstance().getReferenceMethodInfo(jobj.getClass());
433            if (propertyInfo == ReferenceMethodInfoCache.NOT_FOUND)
434                return new ObjIdPropertyDef().extract(jobj);
435            final Object value = JObjectContainer.extractProperty(
436              propertyInfo.getPropertyExtractor(), propertyInfo.getPropertyDef(), jobj);
437            if (value instanceof Component)
438                return (Component)value;
439            return new SizedLabel(String.valueOf(value));
440        }
441    }

443// ObjFieldPropertyDef
444
445    /**
446     * Implements a property reflecting the value of a {@link JSimpleDB} field.
447     */
448    public class ObjFieldPropertyDef extends ObjPropertyDef<Component> {
449
450        private static final int MAX_ITEMS = 3;
451
452        private final int storageId;
453
454        public ObjFieldPropertyDef(int storageId, String name) {
455            super(name, Component.class);
456            this.storageId = storageId;
457        }
458
459        @Override
460        public Component extract(final JObject jobj) {
461            final JField jfield = JObjectContainer.this.jdb.getJClass(jobj.getObjId()).getJField(this.storageId, JField.class);
462            try {
463                return jfield.visit(new JFieldSwitchAdapter<Component>() {
464
465                    @Override
466                    public Component caseJSimpleField(JSimpleField field) {
467                        return ObjFieldPropertyDef.this.handleValue(field.getValue(jobj));
468                    }
469
470                    @Override
471                    public Component caseJCounterField(JCounterField field) {
472                        return ObjFieldPropertyDef.this.handleValue(field.getValue(jobj).get());
473                    }
474
475                    @Override
476                    protected Component caseJCollectionField(JCollectionField field) {
477                        return ObjFieldPropertyDef.this.handleCollectionField(field.getValue(jobj));
478                    }
479
480                    @Override
481                    public Component caseJMapField(JMapField field) {
482                        return ObjFieldPropertyDef.this.handleMultiple(Iterables.transform(field.getValue(jobj).entrySet(),
483                          new Function<Map.Entry<?, ?>, Component>() {
484                            @Override
485                            public Component apply(Map.Entry<?, ?> entry) {
486                                final HorizontalLayout layout = new HorizontalLayout();
487                                layout.setMargin(false);
488                                layout.setSpacing(false);
489                                layout.addComponent(ObjFieldPropertyDef.this.handleValue(entry.getKey()));
490                                layout.addComponent(new SizedLabel(" \u21d2 "));        // RIGHTWARDS DOUBLE ARROW
491                                layout.addComponent(ObjFieldPropertyDef.this.handleValue(entry.getValue()));
492                                return layout;
493                            }
494                        }));
495                    }
496                });
497            } catch (UnknownFieldException e) {
498                return new SizedLabel("<i>NA</i>", ContentMode.HTML);
499            }
500        }
501
502        private Component handleCollectionField(Collection<?> col) {
503            return this.handleMultiple(Iterables.transform(col, new Function<Object, Component>() {
504                @Override
505                public Component apply(Object item) {
506                    return ObjFieldPropertyDef.this.handleValue(item);
507                }
508            }));
509        }
510
511        private Component handleMultiple(Iterable<Component> components) {
512            final HorizontalLayout layout = new HorizontalLayout();
513            layout.setMargin(false);
514            layout.setSpacing(false);
515            int count = 0;
516            for (Component component : components) {
517                if (count >= MAX_ITEMS) {
518                    layout.addComponent(new SizedLabel("..."));
519                    break;
520                }
521                if (count > 0)
522                    layout.addComponent(new SizedLabel(",&#160;", ContentMode.HTML));
523                layout.addComponent(component);
524                count++;
525            }
526            return layout;
527        }
528
529        @SuppressWarnings("unchecked")
530        private <T> Component handleValue(Object value) {
531            if (value == null)
532                return new SizedLabel("<i>Null</i>", ContentMode.HTML);
533            if (value instanceof JObject)
534                return new RefLabelPropertyDef().extract((JObject)value);
535            return new SizedLabel(String.valueOf(value));
536        }
537    }
