006package org.jsimpledb.gui;
007
008import java.util.Iterator;
009
010import org.jsimpledb.JObject;
011import org.jsimpledb.JSimpleDB;
012import org.jsimpledb.JTransaction;
013import org.jsimpledb.SnapshotJTransaction;
014import org.jsimpledb.ValidationMode;
015import org.jsimpledb.core.Transaction;
016import org.jsimpledb.kv.CloseableKVStore;
017
018/**
019 * Specialized {@link JObjectContainer} for use with key/value stores that implement
020 * {@link org.jsimpledb.kv.KVTransaction#mutableSnapshot KVTransaction.mutableSnapshot()}.
021 *
022 * <p>
023 * This container is loaded by taking a key/value store snapshot and then iterating the desired backing objects
024 * directly via {@link #iterateObjects}; no copying of objects is required.
025 *
026 * <p>
027 * Instances are (re)loaded at any time by invoking {@link #reload}. During reload, the container opens a {@link JTransaction}
028 * and then creates a {@link SnapshotJTransaction} using the {@link CloseableKVStore} returned by
029 * {@link org.jsimpledb.kv.KVTransaction#mutableSnapshot KVTransaction.mutableSnapshot()}.
030 * This {@link SnapshotJTransaction} is set as the current transaction while {@link #iterateObjects}
031 * returns the {@link JObject}s to be actually included in the container. The {@link CloseableKVStore} will
032 * remain open until the container is {@link #reload}'ed or {@link #disconnect}'ed.
033 */
034@SuppressWarnings("serial")
035public abstract class SnapshotJObjectContainer extends ReloadableJObjectContainer {
036
037    private CloseableKVStore kvstore;
038
039    /**
040     * Constructor.
041     *
042     * @param jdb {@link JSimpleDB} database
043     * @param type type restriction, or null for no restriction
044     * @throws IllegalArgumentException if {@code jdb} is null
045     */
046    protected SnapshotJObjectContainer(JSimpleDB jdb, Class<?> type) {
047        super(jdb, type);
048    }
049
050// Connectable
051
052    @Override
053    public void disconnect() {
054        super.disconnect();
055        if (this.kvstore != null) {
056            this.kvstore.close();
057            this.kvstore = null;
058        }
059    }
060
061    /**
062     * (Re)load this container.
063     *
064     * <p>
065     * This method opens a {@link JTransaction}, creates a {@link SnapshotJTransaction} using the {@link CloseableKVStore}
066     * returned by {@link org.jsimpledb.kv.KVTransaction#mutableSnapshot KVTransaction.mutableSnapshot()}, and loads
067     * the container using the objects returned by {@link #iterateObjects}.
068     */
069    public void reload() {
070
071        // Grab KVStore snapshot
072        final CloseableKVStore[] snapshotHolder = new CloseableKVStore[1];
073        this.doInTransaction(new Runnable() {
074            @Override
075            public void run() {
076                final JTransaction jtx = JTransaction.getCurrent();
077                final CloseableKVStore snapshot = jtx.getTransaction().getKVTransaction().mutableSnapshot();
078                jtx.getTransaction().addCallback(new Transaction.CallbackAdapter() {
079                    @Override
080                    public void afterCompletion(boolean committed) {
081                        if (!committed)
082                            snapshot.close();
083                    }
084                });
085                snapshotHolder[0] = snapshot;
086            }
087        });
088
089        // Replace previous KVStore snapshot (if any)
090        if (this.kvstore != null) {
091            this.kvstore.close();
092            this.kvstore = null;
093        }
094        this.kvstore = snapshotHolder[0];
095
096        // Create snapshot transaction based on the key/value store snapshot and load container
097        final SnapshotJTransaction snapshotTx = this.jdb.createSnapshotTransaction(this.kvstore, false, ValidationMode.MANUAL);
098        snapshotTx.performAction(new Runnable() {
099            @Override
100            public void run() {
101                SnapshotJObjectContainer.this.load(SnapshotJObjectContainer.this.iterateObjects());
102            }
103        });
104    }
105
106    /**
107     * Iterate the database objects that will be used to fill this container.
108     * Objects should be returned in the desired order; duplicates and null values will be ignored.
109     *
110     * <p>
111     * A {@link JTransaction} will be open in the current thread when this method is invoked.
112     *
113     * @return database objects
114     */
115    protected abstract Iterator<? extends JObject> iterateObjects();
116}
