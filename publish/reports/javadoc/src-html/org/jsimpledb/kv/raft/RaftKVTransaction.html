062    /**
063     * Constructor.
064     *
065     * @param kvdb associated database
066     * @param baseTerm term of the Raft log entry on which this transaction is based
067     * @param baseIndex index of the Raft log entry on which this transaction is based
068     * @param snapshot underlying state machine snapshot; will be closed with this instance
069     * @param view this transaction's view of the (prefixed) key/value store
070     */
071    RaftKVTransaction(RaftKVDatabase kvdb, long baseTerm, long baseIndex, CloseableKVStore snapshot, MutableView view) {
072        this.kvdb = kvdb;
073        this.baseTerm = baseTerm;
074        this.baseIndex = baseIndex;
075        this.snapshotRefs = new SnapshotRefs(snapshot);
076        this.view = view;
077    }

100    void setState(TxState state) {
101        assert state != null;
102        synchronized (this.kvdb) {
103            assert state.compareTo(this.state) >= 0;
104            if (this.state.equals(TxState.EXECUTING) && !this.state.equals(state))
105                this.snapshotRefs.unref();
106            this.state = state;
107        }
108    }

211// Configuration Stuff

213    /**
214     * Include a cluster configuration change when this transaction is committed.
215     *
216     * <p>
217     * The change will have been applied once this transaction is successfully committed.
218     *
219     * <p>
220     * Raft supports configuration changes that add or remove one node at a time to/from the cluster.
221     * If this method is invoked more than once in a single transaction, all but the last invocation are ignored.
222     *
223     * <p>
224     * Initially, nodes are <i>unconfigured</i>. An unconfigured node becomes configured in one of two ways:
225     * <ul>
226     *  <li>By receiving a message from a leader of some existing cluster, in which case the node joins that cluster
227     *      based on the provided configuration; or</li>
228     *  <li>By this method being invoked with {@code identity} equal to this node's identity and a non-null {@code address},
229     *      which creates a new cluster and adds this node to it.</li>
230     * </ul>
231     *
232     * <p>
233     * Therefore, this method must be used to intialize a new cluster.
234     *
235     * @param identity the identity of the node to add or remove
236     * @param address the network address of the node if adding, or null if removing
237     * @throws IllegalStateException if this method has been invoked previously on this instance
238     * @throws IllegalArgumentException if {@code identity} is null
239     */
240    public void configChange(String identity, String address) {
241        Preconditions.checkArgument(identity != null, "null identity");
242        synchronized (this.kvdb) {
243            Preconditions.checkState(this.configChange == null, "duplicate config chagne; only one is supported per transaction");
244            if (!this.state.equals(TxState.EXECUTING))
245                throw new StaleTransactionException(this);
246            this.configChange = new String[] { identity, address };
247        }
248    }
249
250    /**
251     * Get the cluster configuration change associated with this transaction, if any.
252     *
253     * <p>
254     * The returned array has length two and contains the {@code identity} and {@code address}
255     * parameters passed to {@link #configChange configChange()}.
256     *
257     * <p>
258     * The returned array is a copy; changes have no effect on this instance.
259     *
260     * @return cluster config change, or null if there is none
261     */
262    public String[] getConfigChange() {
263        synchronized (this.kvdb) {
264            return this.configChange != null ? this.configChange.clone() : null;
265        }
266    }

284    /**
285     * Set the commit timeout for this instance.
286     *
287     * <p>
288     * {@link RaftKVTransaction}s do not block while the transaction is open; the configured value is used
289     * as a timeout for the {@link #commit} operation only. If {@link #commit} takes longer than {@code timeout}
290     * milliseconds, a {@link org.jsimpledb.kv.RetryTransactionException} is thrown.
291     *
292     * <p>
293     * The default value for all transactions is configured by {@link RaftKVDatabase#setCommitTimeout}.
294     *
295     * @param timeout transaction commit timeout in milliseconds, or zero for unlimited
296     * @throws IllegalArgumentException if {@code timeout} is negative
297     */
298    @Override
299    public void setTimeout(long timeout) {
300        Preconditions.checkArgument(timeout >= 0, "timeout < 0");
301        this.timeout = (int)Math.min(timeout, Integer.MAX_VALUE);
302    }

319    @Override
320    public CloseableKVStore mutableSnapshot() {
321        synchronized (this.kvdb) {
322            if (!this.state.equals(TxState.EXECUTING))
323                throw new StaleTransactionException(this);
324            this.snapshotRefs.ref();
325        }
326        final Writes writes;
327        synchronized (this.view) {
328            writes = this.view.getWrites().clone();
329        }
330        final MutableView snapshotView = new MutableView(this.snapshotRefs.getKVStore(), null, writes);
331        return new CloseableForwardingKVStore(snapshotView, this.snapshotRefs.getUnrefCloseable());
332    }

357    @Override
358    public String toString() {
359        return this.getClass().getSimpleName()
360          + "[txId=" + this.txId
361          + ",state=" + this.state
362          + ",base=" + this.baseIndex + "t" + this.baseTerm
363          + ",consistency=" + this.consistency
364          + (this.readOnly ? ",readOnly" : "")
365          + (this.configChange != null ? ",configChange=" + Arrays.<String>asList(this.configChange) : "")
366          + (this.state.compareTo(TxState.COMMIT_WAITING) >= 0 ? ",commit=" + this.commitIndex + "t" + this.commitTerm : "")
367          + (this.timeout != 0 ? ",timeout=" + this.timeout : "")
368          + "]";
369    }

371    @Override
372    protected void finalize() throws Throwable {
373        try {
374            synchronized (this.kvdb) {
375                if (!TxState.CLOSED.equals(this.state)) {
376                    this.log.warn(this + " leaked without commit() or rollback()");
377                    this.rollback();
378                }
379            }
380        } finally {
381            super.finalize();
382        }
383    }

385// Debug/Sanity Checking
386
387    void checkStateOpen(long currentTerm, long lastIndex, long raftCommitIndex) {
388        assert !this.commitFuture.isCancelled();
389        assert this.commitFuture.isDone() == this.state.compareTo(TxState.COMPLETED) >= 0;
390        assert this.baseTerm <= currentTerm;
391        switch (this.state) {
392        case EXECUTING:
393        case COMMIT_READY:
394            assert this.commitTerm == 0;
395            assert this.commitIndex == 0;
396            break;
397        case COMMIT_WAITING:
398            assert this.commitTerm >= this.baseTerm;
399            assert this.commitTerm <= currentTerm;
400            assert this.commitIndex >= this.baseIndex;                                      // equal implies a read-only tx
401            assert this.commitIndex > this.baseIndex || this.view.getWrites().isEmpty();
402            break;
403        case COMPLETED:
404            assert this.commitFuture.isDone();
405            assert this.commitTerm == 0 || this.commitTerm >= this.baseTerm;
406            assert this.commitIndex == 0 || this.commitIndex >= this.baseIndex;
407            assert this.commitTerm <= currentTerm;
408            break;
409        case CLOSED:
410        default:
411            assert false;
412            break;
413        }
414    }
