006package org.jsimpledb.cli.cmd;
007
008import java.util.Map;
009
010import org.jsimpledb.SessionMode;
011import org.jsimpledb.cli.CliSession;
012import org.jsimpledb.parse.ParseContext;
013import org.jsimpledb.parse.expr.Node;
014import org.jsimpledb.parse.func.AbstractFunction;
015
016@Command(modes = { SessionMode.KEY_VALUE, SessionMode.CORE_API, SessionMode.JSIMPLEDB })
017public class RegisterFunctionCommand extends AbstractCommand {
018
019    public RegisterFunctionCommand() {
020        super("register-function class:expr");
021    }
022
023    @Override
024    public String getHelpSummary() {
025        return "Instantiates a user-supplied class subclassing AbstractFunction and registers it as an available function.";
026    }
027
028    @Override
029    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
030        final Node expr = (Node)params.get("class");
031        return new CliSession.Action() {
032            @Override
033            public void run(CliSession session) throws Exception {
034                final Object result = expr.evaluate(session).get(session);
035                if (!(result instanceof Class))
036                    throw new Exception("invalid parameter: not a " + Class.class.getName() + " instance");
037                final AbstractFunction function = session.registerFunction((Class<?>)result);
038                session.getWriter().println("Registered command `" + function.getName() + "'");
039            }
040        };
041    }
042}
