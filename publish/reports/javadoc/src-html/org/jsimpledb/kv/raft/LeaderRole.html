081    /**
082     * Get this leader's known followers.
083     *
084     * <p>
085     * The returned list is a copy; changes have no effect on this instance.
086     *
087     * @return this leader's followers
088     */
089    public List<Follower> getFollowers() {
090        final ArrayList<Follower> list;
091        synchronized (this.raft) {
092            list = new ArrayList<>(this.followerMap.values());
093        }
094        Collections.sort(list, Follower.SORT_BY_IDENTITY);
095        return list;
096    }

098    /**
099     * Get this leader's "lease timeout".
100     *
101     * <p>
102     * This is the earliest possible time at which some other, new leader could be elected in a new term.
103     * Consequently, it is the earliest possible time at which any entry that this leader is unaware of
104     * could be appended to the Raft log.
105     *
106     * <p>
107     * Normally, if followers are responding to {@link AppendRequest}s properly, this should be a value
108     * in the (near) future. This allows the leader to make the assumption, up until that point in time,
109     * that its log is fully up-to-date.
110     *
111     * <p>
112     * Until it hears from a majority of followers, a leader will not have a lease timeout established yet.
113     * In that case this method returns null.
114     *
115     * @return this leader's lease timeout, or null if none is established yet
116     */
117    public Timestamp getLeaseTimeout() {
118        synchronized (this.raft) {
119            return this.leaseTimeout;
120        }
121    }

123    /**
124     * Force this leader to step down.
125     *
126     * @throws IllegalStateException if this role is no longer active or election timer is not running
127     */
128    public void stepDown() {
129        synchronized (this.raft) {
130            Preconditions.checkState(this.raft.role == this, "role is no longer active");
131            this.debug("stepping down as leader due to invocation of stepDown()");
132            this.raft.changeRole(new FollowerRole(this.raft));
133        }
134    }

138    @Override
139    void setup() {
140        super.setup();
141        if (this.log.isDebugEnabled())
142            this.debug("entering leader role in term " + this.raft.currentTerm);
143
144        // Generate follower list
145        this.updateKnownFollowers();
146
147        // Append a "dummy" log entry with my current term. This allows us to advance the commit index when the last
148        // entry in our log is from a prior term. This is needed to avoid the problem where a transaction could end up
149        // waiting indefinitely for its log entry with a prior term number to be committed.
150        final LogEntry logEntry;
151        try {
152            logEntry = this.applyNewLogEntry(new NewLogEntry(this.raft, new LogEntry.Data(new Writes(), null)));
153        } catch (Exception e) {
154            this.error("error attempting to apply initial log entry", e);
155            return;
156        }
157        if (this.log.isDebugEnabled())
158            this.debug("added log entry " + logEntry + " to commit at the beginning of my new term");
159
160        // Start check apply timer
161        if (!this.raft.raftLog.isEmpty())
162            this.checkApplyTimer.timeoutAfter(this.raft.maxTransactionDuration);
163    }
164
165    @Override
166    void shutdown() {
167        super.shutdown();
168        for (Follower follower : this.followerMap.values())
169            follower.cleanup();
170        this.checkApplyTimer.cancel();
171    }

175    @Override
176    void outputQueueEmpty(String address) {
177
178        // Find matching follower(s) and update them if needed
179        for (Follower follower : this.followerMap.values()) {
180            if (follower.getAddress().equals(address)) {
181                if (this.log.isTraceEnabled())
182                    this.trace("updating peer \"" + follower.getIdentity() + "\" after queue empty notification");
183                this.raft.requestService(new UpdateFollowerService(follower));
184            }
185        }
186    }

188    @Override
189    void applyCommittedLogEntries() {
190        super.applyCommittedLogEntries();
191
192        // Stop check apply timer if there are none left
193        if (this.raft.raftLog.isEmpty() && this.checkApplyTimer.isRunning())
194            this.checkApplyTimer.cancel();
195    }

197    @Override
198    boolean mayApplyLogEntry(LogEntry logEntry) {
199
200        // Are we running out of memory, or keeping around too many log entries? If so, go ahead.
201        final long logEntryMemoryUsage = this.raft.getUnappliedLogMemoryUsage();
202        if (logEntryMemoryUsage > this.raft.maxUnappliedLogMemory
203          || this.raft.raftLog.size() > RaftKVDatabase.MAX_UNAPPLIED_LOG_ENTRIES) {
204            if (this.log.isTraceEnabled()) {
205                this.trace("allowing log entry " + logEntry + " to be applied because memory usage "
206                  + logEntryMemoryUsage + " > " + this.raft.maxUnappliedLogMemory + " and/or log length "
207                  + this.raft.raftLog.size() + " > " + RaftKVDatabase.MAX_UNAPPLIED_LOG_ENTRIES);
208            }
209            return true;
210        }
211
212        // Try to keep log entries around for a minimum amount of time to facilitate long-running transactions on followers
213        if (logEntry.getAge() < this.raft.maxTransactionDuration) {
214            if (this.log.isTraceEnabled()) {
215                this.trace("delaying application of " + logEntry + " because it has age "
216                  + logEntry.getAge() + "ms < " + this.raft.maxTransactionDuration + "ms");
217            }
218            return false;
219        }
220
221        // If any snapshots are in progress, we don't want to apply any log entries with index greater than the snapshot's
222        // index, because then we'd "lose" the ability to update the follower with that log entry, and as a result just have
223        // to send a snapshot again. However, we impose a limit on how long we'll wait for a slow follower.
224        for (Follower follower : this.followerMap.values()) {
225            final SnapshotTransmit snapshotTransmit = follower.getSnapshotTransmit();
226            if (snapshotTransmit == null)
227                continue;
228            if (snapshotTransmit.getSnapshotIndex() < logEntry.getIndex()
229              && snapshotTransmit.getAge() < RaftKVDatabase.MAX_SNAPSHOT_TRANSMIT_AGE) {
230                if (this.log.isTraceEnabled()) {
231                    this.trace("delaying application of " + logEntry + " because of in-progress snapshot install of "
232                      + snapshotTransmit.getSnapshotIndex() + "t" + snapshotTransmit.getSnapshotTerm()
233                      + " to " + follower);
234                }
235                return false;
236            }
237        }
238
239        // If some follower does not yet have the log entry, wait for them to get it (up to some maximum time)
240        if (logEntry.getAge() < RaftKVDatabase.MAX_SLOW_FOLLOWER_APPLY_DELAY_HEARTBEATS * this.raft.heartbeatTimeout) {
241            for (Follower follower : this.followerMap.values()) {
242                if (follower.getMatchIndex() < logEntry.getIndex()) {
243                    if (this.log.isTraceEnabled()) {
244                        this.trace("delaying application of " + logEntry + " (age " + logEntry.getAge()
245                          + " < " + (RaftKVDatabase.MAX_SLOW_FOLLOWER_APPLY_DELAY_HEARTBEATS * this.raft.heartbeatTimeout)
246                          + ") because of slow " + follower);
247                    }
248                    return false;
249                }
250            }
251        }
252
253        // OK
254        return true;
255    }

257    // We have to periodically check if we can apply log entries, because the condition is time-dependent
258    private void checkApplyEntries() {
259        this.raft.requestService(this.applyCommittedLogEntriesService);
260        if (!this.raft.raftLog.isEmpty())
261            this.checkApplyTimer.timeoutAfter(this.raft.maxTransactionDuration);
262    }

264    /**
265     * Update my {@code commitIndex} based on followers' {@code matchIndex}'s.
266     *
267     * <p>
268     * This should be invoked:
269     * <ul>
270     *  <li>After any log entry has been added to the log, if we have zero followers</li>
271     *  <li>After a log entry that contains a configuration change has been added to the log</li>
272     *  <li>After a follower's {@linkplain Follower#getMatchIndex match index} has advanced</li>
273     * </ul>
274     */
275    private void updateLeaderCommitIndex() {
276
277        // Find highest index for which a majority of cluster members have ack'd the corresponding log entry from my term
278        final int totalCount = this.raft.currentConfig.size();                          // total possible nodes
279        final int requiredCount = totalCount / 2 + 1;                                   // require a majority
280        final int startingCount = this.raft.isClusterMember() ? 1 : 0;                  // count myself, if member
281        long maxCommitIndex = this.raft.commitIndex;
282        int commitCount = -1;
283        for (long index = this.raft.commitIndex + 1; index <= this.raft.getLastLogIndex(); index++) {
284
285            // Count the number of nodes (possibly including myself) that have a copy of the log entry at index
286            int count = startingCount;
287            for (Follower follower : this.followerMap.values()) {                       // count followers who are members
288                if (follower.getMatchIndex() >= index && this.raft.isClusterMember(follower.getIdentity()))
289                    count++;
290            }
291
292            // The log entry term must match my current term (exception: unless every node has it)
293            final long term = this.raft.getLogTermAtIndex(index);
294            if (count < totalCount && term != this.raft.currentTerm)
295                continue;
296
297            // Do a majority of cluster nodes have this log entry?
298            if (count < requiredCount) {
299                if (term >= this.raft.currentTerm)                                      // there's no point in going further
300                    break;
301                continue;                                                               // a later term log entry might work
302            }
303
304            // We have a winner
305            maxCommitIndex = index;
306            commitCount = count;
307        }
308
309        // Update commit index if it advanced
310        if (maxCommitIndex > this.raft.commitIndex) {
311
312            // Update index
313            if (this.log.isDebugEnabled()) {
314                this.debug("advancing commit index from " + this.raft.commitIndex + " -> " + maxCommitIndex + " based on "
315                  + commitCount + "/" + totalCount + " nodes having received " + this.raft.getLogEntryAtIndex(maxCommitIndex));
316            }
317            this.raft.commitIndex = maxCommitIndex;
318
319            // Perform various service
320            this.raft.requestService(this.checkReadyTransactionsService);
321            this.raft.requestService(this.checkWaitingTransactionsService);
322            this.raft.requestService(this.applyCommittedLogEntriesService);
323            this.raft.requestService(this.triggerKeyWatchesService);
324
325            // Notify all (up-to-date) followers with the updated leaderCommit
326            this.updateAllSynchronizedFollowersNow();
327
328            // If we are no longer a member of the cluster, step down after the most recent config change is committed
329            if (!this.raft.isClusterMember() && this.raft.commitIndex >= this.findMostRecentConfigChange()) {
330                if (this.log.isDebugEnabled())
331                    this.log.debug("stepping down as leader of cluster (no longer a member)");
332                this.stepDown();
333            }
334        }
335    }

337    /**
338     * Update my {@code leaseTimeout} based on followers' returned {@code leaderTimeout}'s.
339     *
340     * <p>
341     * This should be invoked:
342     * <ul>
343     *  <li>After a follower has replied with an {@link AppendResponse} containing a newer
344     *      {@linkplain AppendResponse#getLeaderTimestamp leader timestamp} than before</li>
345     * </ul>
346     */
347    private void updateLeaseTimeout() {
348
349        // Only needed when we have followers
350        final int numFollowers = this.followerMap.size();
351        if (numFollowers == 0)
352            return;
353
354        // Get all cluster member leader timestamps, sorted in increasing order
355        final Timestamp[] leaderTimestamps = new Timestamp[this.raft.currentConfig.size()];
356        int index = 0;
357        if (this.raft.isClusterMember())
358            leaderTimestamps[index++] = new Timestamp();                        // this represents my own vote
359        for (Follower follower : this.followerMap.values()) {
360            if (this.raft.isClusterMember(follower.getIdentity()))
361                leaderTimestamps[index++] = follower.getLeaderTimestamp();      // note follower timestamps could be null
362        }
363        Arrays.sort(leaderTimestamps, Timestamp.NULL_FIRST_SORT);
364
365        //
366        // Calculate highest leaderTimeout shared by a majority of cluster members, based on sorted array:
367        //
368        //  # nodes    timestamps
369        //  -------    ----------
370        //     5       [ ][ ][x][x][x]        3/5 x's make a majority at index (5 - 1)/2 = 2
371        //     6       [ ][ ][x][x][x][x]     4/6 x's make a majority at index (6 - 1)/2 = 2
372        //
373        // The minimum leaderTimeout shared by a majority of nodes is at index (leaderTimestamps.length - 1) / 2.
374        // We then add the minimum election timeout, then subtract a little for clock drift.
375        //
376        final Timestamp newLeaseTimeout = leaderTimestamps[(leaderTimestamps.length + 1) / 2]
377          .offset((int)(this.raft.minElectionTimeout * (1.0f - RaftKVDatabase.MAX_CLOCK_DRIFT) - 1));
378        if (Timestamp.NULL_FIRST_SORT.compare(newLeaseTimeout, this.leaseTimeout) > 0) {
379            assert newLeaseTimeout != null;
380
381            // Update my leader lease timeout
382            if (this.log.isTraceEnabled())
383                this.trace("updating my lease timeout from " + this.leaseTimeout + " -> " + newLeaseTimeout);
384            this.leaseTimeout = newLeaseTimeout;
385
386            // Notify any followers who care
387            for (Follower follower : this.followerMap.values()) {
388                final NavigableSet<Timestamp> timeouts = follower.getCommitLeaseTimeouts().headSet(this.leaseTimeout, true);
389                if (!timeouts.isEmpty()) {
390                    follower.updateNow();                           // notify follower so it can commit waiting transaction(s)
391                    timeouts.clear();
392                }
393            }
394        }
395    }

397    /**
398     * Update our list of followers to match our current configuration.
399     *
400     * <p>
401     * This should be invoked:
402     * <ul>
403     *  <li>After a log entry that contains a configuration change has been added to the log</li>
404     *  <li>When the {@linkplain Follower#getNextIndex next index} of a follower not in the current config advances</li>
405     * </ul>
406     */
407    private void updateKnownFollowers() {
408
409        // Compare known followers with the current config and determine who needs to be be added or removed
410        final HashSet<String> adds = new HashSet<>(this.raft.currentConfig.keySet());
411        adds.removeAll(this.followerMap.keySet());
412        adds.remove(this.raft.identity);
413        final HashSet<String> dels = new HashSet<>(this.followerMap.keySet());
414        dels.removeAll(this.raft.currentConfig.keySet());
415
416        // Keep around a follower after its removal until it receives the config change that removed it
417        for (Follower follower : this.followerMap.values()) {
418
419            // Is this follower scheduled for deletion?
420            final String peer = follower.getIdentity();
421            if (!dels.contains(peer))
422                continue;
423
424            // Find the most recent log entry containing a config change in which the follower was removed
425            final String node = follower.getIdentity();
426            final long index = this.findMostRecentConfigChangeMatching(new Predicate<String[]>() {
427                @Override
428                public boolean apply(String[] configChange) {
429                    return configChange[0].equals(node) && configChange[1] == null;
430                }
431            });
432
433            // If follower has not received that log entry yet, keep on updating them until they do
434            if (follower.getMatchIndex() < index)
435                dels.remove(peer);
436        }
437
438        // Add new followers
439        for (String peer : adds) {
440            final String address = this.raft.currentConfig.get(peer);
441            final Follower follower = new Follower(peer, address, this.raft.getLastLogIndex());
442            if (this.log.isDebugEnabled())
443                this.debug("adding new follower \"" + peer + "\" at " + address);
444            follower.setUpdateTimer(new Timer(this.raft, "update timer for \"" + peer + "\"", new UpdateFollowerService(follower)));
445            this.followerMap.put(peer, follower);
446            follower.updateNow();                                               // schedule an immediate update
447        }
448
449        // Remove old followers
450        for (String peer : dels) {
451            final Follower follower = this.followerMap.remove(peer);
452            if (this.log.isDebugEnabled())
453                this.debug("removing old follower \"" + peer + "\"");
454            follower.cleanup();
455        }
456    }

458    /**
459     * Check whether a follower needs an update and send one if so.
460     *
461     * <p>
462     * This should be invoked:
463     * <ul>
464     *  <li>After a new follower has been added</li>
465     *  <li>When the output queue for a follower goes from non-empty to empty</li>
466     *  <li>After the follower's {@linkplain Follower#getUpdateTimer update timer} has expired</li>
467     *  <li>After a new log entry has been added to the log (all followers)</li>
468     *  <li>After receiving an {@link AppendResponse} that caused the follower's
469     *      {@linkplain Follower#getNextIndex next index} to change</li>
470     *  <li>After receiving the first positive {@link AppendResponse} to a probe</li>
471     *  <li>After our {@code commitIndex} has advanced (all followers)</li>
472     *  <li>After our {@code leaseTimeout} has advanced past one or more of a follower's
473     *      {@linkplain Follower#getCommitLeaseTimeouts commit lease timeouts} (with update timer reset)</li>
474     *  <li>After sending a {@link CommitResponse} with a non-null {@linkplain CommitResponse#getCommitLeaderLeaseTimeout
475     *      commit leader lease timeout} (all followers) to probe for updated leader timestamps</li>
476     *  <li>After starting, aborting, or completing a snapshot install for a follower</li>
477     * </ul>
478     */
479    private void updateFollower(Follower follower) {
480
481        // If follower has an in-progress snapshot that has become too stale, abort it
482        final String peer = follower.getIdentity();
483        SnapshotTransmit snapshotTransmit = follower.getSnapshotTransmit();
484        if (snapshotTransmit != null && snapshotTransmit.getSnapshotIndex() < this.raft.lastAppliedIndex) {
485            if (this.log.isDebugEnabled())
486                this.debug("aborting stale snapshot install for " + follower);
487            follower.cancelSnapshotTransmit();
488            follower.updateNow();
489        }
490
491        // Is follower's queue empty? If not, hold off until then
492        if (this.raft.isTransmitting(follower.getAddress())) {
493            if (this.log.isTraceEnabled())
494                this.trace("no update for \"" + peer + "\": output queue still not empty");
495            return;
496        }
497
498        // Handle any in-progress snapshot install
499        if ((snapshotTransmit = follower.getSnapshotTransmit()) != null) {
500
501            // Send the next chunk in transmission, if any
502            final long pairIndex = snapshotTransmit.getPairIndex();
503            final ByteBuffer chunk = snapshotTransmit.getNextChunk();
504            boolean synced = true;
505            if (chunk != null) {
506
507                // Send next chunk
508                final InstallSnapshot msg = new InstallSnapshot(this.raft.clusterId, this.raft.identity, peer,
509                  this.raft.currentTerm, snapshotTransmit.getSnapshotTerm(), snapshotTransmit.getSnapshotIndex(), pairIndex,
510                  pairIndex == 0 ? snapshotTransmit.getSnapshotConfig() : null, !snapshotTransmit.hasMoreChunks(), chunk);
511                if (this.raft.sendMessage(msg))
512                    return;
513                if (this.log.isDebugEnabled())
514                    this.debug("canceling snapshot install for " + follower + " due to failure to send " + msg);
515
516                // Message failed -> snapshot is fatally wounded, so cancel it
517                synced = false;
518            }
519            if (synced) {
520                if (this.log.isDebugEnabled())
521                    this.debug("completed snapshot install for out-of-date " + follower);
522            }
523
524            // Snapshot transmit is complete (or failed)
525            follower.cancelSnapshotTransmit();
526
527            // Trigger an immediate regular update
528            follower.setNextIndex(snapshotTransmit.getSnapshotIndex() + 1);
529            follower.setSynced(synced);
530            follower.updateNow();
531            this.raft.requestService(new UpdateFollowerService(follower));
532            return;
533        }
534
535        // Are we still waiting for the update timer to expire?
536        if (!follower.getUpdateTimer().pollForTimeout()) {
537            boolean waitForTimerToExpire = true;
538
539            // Don't wait for the update timer to expire if:
540            //  (a) The follower is sync'd; AND
541            //      (y) We have a new log entry that the follower doesn't have; OR
542            //      (y) We have a new leaderCommit that the follower doesn't have
543            // The effect is that we will pipeline updates to synchronized followers.
544            if (follower.isSynced()
545              && (follower.getLeaderCommit() != this.raft.commitIndex
546               || follower.getNextIndex() <= this.raft.getLastLogIndex()))
547                waitForTimerToExpire = false;
548
549            // Wait for timer to expire
550            if (waitForTimerToExpire) {
551                if (this.log.isTraceEnabled()) {
552                    this.trace("no update for \"" + follower.getIdentity() + "\": timer not expired yet, and follower is "
553                      + (follower.isSynced() ? "up to date" : "not synced"));
554                }
555                return;
556            }
557        }
558
559        // Get index of the next log entry to send to follower
560        final long nextIndex = follower.getNextIndex();
561
562        // If follower is too far behind, we must do a snapshot install
563        if (nextIndex <= this.raft.lastAppliedIndex) {
564            final MostRecentView view = new MostRecentView(this.raft, true);
565            follower.setSnapshotTransmit(new SnapshotTransmit(view.getTerm(),
566              view.getIndex(), view.getConfig(), view.getSnapshot(), view.getView()));
567            if (this.log.isDebugEnabled())
568                this.debug("started snapshot install for out-of-date " + follower);
569            this.raft.requestService(new UpdateFollowerService(follower));
570            return;
571        }
572
573        // Restart update timer here (to avoid looping if an error occurs below)
574        follower.getUpdateTimer().timeoutAfter(this.raft.heartbeatTimeout);
575
576        // Send actual data if follower is synced and there is a log entry to send; otherwise, just send a probe
577        final AppendRequest msg;
578        if (!follower.isSynced() || nextIndex > this.raft.getLastLogIndex()) {
579            msg = new AppendRequest(this.raft.clusterId, this.raft.identity, peer, this.raft.currentTerm, new Timestamp(),
580              this.leaseTimeout, this.raft.commitIndex, this.raft.getLogTermAtIndex(nextIndex - 1), nextIndex - 1);   // probe
581        } else {
582
583            // Get log entry to send
584            final LogEntry logEntry = this.raft.getLogEntryAtIndex(nextIndex);
585
586            // If the log entry correspond's to follower's transaction, don't send the data because follower already has it.
587            // But only do this optimization the first time, in case something goes wrong on the follower's end.
588            ByteBuffer mutationData = null;
589            if (!follower.getSkipDataLogEntries().remove(logEntry)) {
590                try {
591                    mutationData = logEntry.getContent();
592                } catch (IOException e) {
593                    this.error("error reading log file " + logEntry.getFile(), e);
594                    return;
595                }
596            }
597
598            // Create message
599            msg = new AppendRequest(this.raft.clusterId, this.raft.identity, peer, this.raft.currentTerm, new Timestamp(),
600              this.leaseTimeout, this.raft.commitIndex, this.raft.getLogTermAtIndex(nextIndex - 1), nextIndex - 1,
601              logEntry.getTerm(), mutationData);
602        }
603
604        // Send update
605        final boolean sent = this.raft.sendMessage(msg);
606
607        // Advance next index if a log entry was sent; we allow pipelining log entries when synchronized
608        if (sent && !msg.isProbe()) {
609            assert follower.isSynced();
610            follower.setNextIndex(Math.min(follower.getNextIndex(), this.raft.getLastLogIndex()) + 1);
611        }
612
613        // Update leaderCommit for follower
614        if (sent)
615            follower.setLeaderCommit(msg.getLeaderCommit());
616    }

618    private void updateAllSynchronizedFollowersNow() {
619        for (Follower follower : this.followerMap.values()) {
620            if (follower.isSynced())
621                follower.updateNow();
622        }
623    }

656    @Override
657    void checkReadyLeaderTransaction(RaftKVTransaction tx, boolean readOnly) {
658
659        // Sanity check
660        assert Thread.holdsLock(this.raft);
661        assert tx.getState().equals(TxState.COMMIT_READY);
662
663        // Check for conflict when LINEARIZABLE
664        if (tx.getConsistency().equals(Consistency.LINEARIZABLE)) {
665            final String error = this.checkConflicts(tx.getBaseTerm(), tx.getBaseIndex(), tx.getMutableView().getReads());
666            if (error != null) {
667                if (this.log.isDebugEnabled())
668                    this.debug("local transaction " + tx + " failed due to conflict: " + error);
669                throw new RetryTransactionException(tx, error);
670            }
671        } else {
672            if (this.log.isTraceEnabled())
673                this.trace("not checking for conflicts in " + tx.getConsistency() + " transaction " + tx);
674        }
675
676        // Handle read-only vs. read-write transaction
677        if (readOnly) {
678
679            // Set commit term and index depending on consistency level
680            switch (tx.getConsistency()) {
681            case UNCOMMITTED:
682                if (this.log.isTraceEnabled())
683                    this.trace("trivial commit for UNCOMMITTED " + tx);
684                this.raft.succeed(tx);
685                return;
686            case EVENTUAL:
687                this.advanceReadyTransaction(tx, tx.getBaseTerm(), tx.getBaseIndex());
688                return;
689            case LINEARIZABLE:
690                if (this.leaseTimeout != null && this.leaseTimeout.offsetFromNow() > 0)
691                    this.advanceReadyTransaction(tx, tx.getBaseTerm(), tx.getBaseIndex());
692                else
693                    this.advanceReadyTransaction(tx, this.raft.getLastLogTerm(), this.raft.getLastLogIndex());
694                return;
695            default:
696                throw new RuntimeException("internal error");
697            }
698        } else {
699
700            // Don't commit a new config change while there is any previous config change outstanding
701            if (tx.getConfigChange() != null && this.isConfigChangeOutstanding())
702                return;
703
704            // Commit transaction as a new log entry
705            final LogEntry logEntry;
706            try {
707                logEntry = this.applyNewLogEntry(new NewLogEntry(tx));
708            } catch (Exception e) {
709                throw new KVTransactionException(tx, "error attempting to persist transaction", e);
710            }
711            if (this.log.isDebugEnabled())
712                this.debug("added log entry " + logEntry + " for local transaction " + tx);
713
714            // Update transaction
715            this.advanceReadyTransaction(tx, logEntry.getTerm(), logEntry.getIndex());
716        }
717    }
718
719    private boolean isConfigChangeOutstanding() {
720        for (int i = (int)(this.raft.commitIndex - this.raft.lastAppliedIndex) + 1; i < this.raft.raftLog.size(); i++) {
721            if (this.raft.raftLog.get(i).getConfigChange() != null)
722                return true;
723        }
724        return false;
725    }

734    @Override
735    void caseAppendResponse(AppendResponse msg) {
736
737        // Find follower
738        final Follower follower = this.findFollower(msg);
739        if (follower == null)
740            return;
741
742        // Update follower's last rec'd leader timestamp
743        if (follower.getLeaderTimestamp() == null || msg.getLeaderTimestamp().compareTo(follower.getLeaderTimestamp()) > 0) {
744            follower.setLeaderTimestamp(msg.getLeaderTimestamp());
745            this.raft.requestService(this.updateLeaseTimeoutService);
746        }
747
748        // Ignore if a snapshot install is in progress
749        if (follower.getSnapshotTransmit() != null) {
750            if (this.log.isTraceEnabled())
751                this.trace("rec'd " + msg + " while sending snapshot install; ignoring");
752            return;
753        }
754
755        // Flag indicating we might want to update follower when done
756        boolean updateFollowerAgain = false;
757
758        // Update follower's match index
759        if (msg.getMatchIndex() > follower.getMatchIndex()) {
760            follower.setMatchIndex(msg.getMatchIndex());
761            this.raft.requestService(this.updateLeaderCommitIndexService);
762            this.raft.requestService(this.applyCommittedLogEntriesService);
763            if (!this.raft.isClusterMember(follower.getIdentity()))
764                this.raft.requestService(this.updateKnownFollowersService);
765        }
766
767        // Check result and update follower's next index
768        final boolean wasSynced = follower.isSynced();
769        final long previousNextIndex = follower.getNextIndex();
770        if (!msg.isSuccess())
771            follower.setNextIndex(Math.max(follower.getNextIndex() - 1, 1));
772        follower.setSynced(msg.isSuccess());
773        if (follower.isSynced() != wasSynced) {
774            if (this.log.isDebugEnabled()) {
775                this.debug("sync status of \"" + follower.getIdentity() + "\" changed -> "
776                  + (!follower.isSynced() ? "not " : "") + "synced");
777            }
778            updateFollowerAgain = true;
779        }
780
781        // Use follower's match index as a lower bound on follower's next index.
782        follower.setNextIndex(Math.max(follower.getNextIndex(), follower.getMatchIndex() + 1));
783
784        // Use follower's last log index as an upper bound on follower's next index.
785        follower.setNextIndex(Math.min(msg.getLastLogIndex() + 1, follower.getNextIndex()));
786
787        // Update follower again if next index has changed
788        updateFollowerAgain |= follower.getNextIndex() != previousNextIndex;
789
790        // Debug
791        if (this.log.isTraceEnabled())
792            this.trace("updated follower: " + follower + ", update again = " + updateFollowerAgain);
793
794        // Immediately update follower again (if appropriate)
795        if (updateFollowerAgain)
796            this.raft.requestService(new UpdateFollowerService(follower));
797    }

799    @Override
800    void caseCommitRequest(CommitRequest msg) {
801
802        // Find follower
803        final Follower follower = this.findFollower(msg);
804        if (follower == null)
805            return;
806
807        // Decode reads, if any, and check for conflicts
808        final ByteBuffer readsData = msg.getReadsData();
809        if (readsData != null) {
810
811            // Decode reads
812            final Reads reads;
813            try {
814                reads = Reads.deserialize(new ByteBufferInputStream(msg.getReadsData()));
815            } catch (Exception e) {
816                this.error("error decoding reads data in " + msg, e);
817                this.raft.sendMessage(new CommitResponse(this.raft.clusterId, this.raft.identity, msg.getSenderId(),
818                  this.raft.currentTerm, msg.getTxId(), "error decoding reads data: " + e));
819                return;
820            }
821
822            // Check for conflict
823            final String conflictMsg = this.checkConflicts(msg.getBaseTerm(), msg.getBaseIndex(), reads);
824            if (conflictMsg != null) {
825                if (this.log.isDebugEnabled())
826                    this.debug("commit request " + msg + " failed due to conflict: " + conflictMsg);
827                this.raft.sendMessage(new CommitResponse(this.raft.clusterId, this.raft.identity, msg.getSenderId(),
828                  this.raft.currentTerm, msg.getTxId(), conflictMsg));
829                return;
830            }
831        }
832
833        // Handle read-only vs. read-write transaction
834        if (msg.isReadOnly()) {
835
836            // Get current time
837            final Timestamp minimumLeaseTimeout = new Timestamp();
838
839            // The follower may commit as soon as it sees the transaction's BASE log entry get committed.
840            // Note, we don't need to wait for any subsequent log entries to be committed, because if they
841            // are committed they are invisible to the transaction, and if they aren't ever committed then
842            // whatever log entries replace them will necessarily have been created sometime after now.
843            final CommitResponse response;
844            if (this.leaseTimeout != null && this.leaseTimeout.compareTo(minimumLeaseTimeout) > 0) {
845
846                // No other leader could have been elected yet as of right now, so the transaction can commit immediately
847                response = new CommitResponse(this.raft.clusterId, this.raft.identity, msg.getSenderId(),
848                  this.raft.currentTerm, msg.getTxId(), msg.getBaseTerm(), msg.getBaseIndex());
849            } else {
850
851                // Remember that this follower is now going to be waiting for this particular leaseTimeout
852                follower.getCommitLeaseTimeouts().add(minimumLeaseTimeout);
853
854                // Send immediate probes to all (up-to-date) followers in an attempt to increase our leaseTimeout quickly
855                this.updateAllSynchronizedFollowersNow();
856
857                // Build response
858                response = new CommitResponse(this.raft.clusterId, this.raft.identity, msg.getSenderId(),
859                  this.raft.currentTerm, msg.getTxId(), msg.getBaseTerm(), msg.getBaseIndex(), minimumLeaseTimeout);
860            }
861
862            // Send response
863            this.raft.sendMessage(response);
864        } else {
865
866            // If the client is requesting a config change, we could check for an outstanding config change now and if so
867            // delay our response until it completes, but that's not worth the trouble. Instead, applyNewLogEntry() will
868            // throw an exception and the client will just just have to retry the transaction.
869
870            // Commit mutations as a new log entry
871            final LogEntry logEntry;
872            try {
873                logEntry = this.applyNewLogEntry(new NewLogEntry(this.raft, msg.getMutationData()));
874            } catch (Exception e) {
875                this.error("error appending new log entry for " + msg, e);
876                this.raft.sendMessage(new CommitResponse(this.raft.clusterId, this.raft.identity, msg.getSenderId(),
877                  this.raft.currentTerm, msg.getTxId(), e.getMessage() != null ? e.getMessage() : "" + e));
878                return;
879            }
880            if (this.log.isDebugEnabled())
881                this.debug("added log entry " + logEntry + " for remote " + msg);
882
883            // Follower transaction data optimization
884            follower.getSkipDataLogEntries().add(logEntry);
885
886            // Send response
887            this.raft.sendMessage(new CommitResponse(this.raft.clusterId, this.raft.identity, msg.getSenderId(),
888              this.raft.currentTerm, msg.getTxId(), logEntry.getTerm(), logEntry.getIndex()));
889        }
890    }

918    private void failDuplicateLeader(Message msg) {
919
920        // This should never happen - same term but two different leaders
921        final boolean defer = this.raft.identity.compareTo(msg.getSenderId()) <= 0;
922        this.error("detected a duplicate leader in " + msg + " - should never happen; possible inconsistent cluster"
923          + " configuration on " + msg.getSenderId() + " (mine: " + this.raft.currentConfig + "); "
924          + (defer ? "reverting to follower" : "ignoring"));
925        if (defer)
926            this.raft.changeRole(new FollowerRole(this.raft, msg.getSenderId(), this.raft.returnAddress));
927    }

958     *

965    /**
966     * Find the index of the most recent unapplied log entry having an associated config change matching the given predicate.
967     *
968     * @return most recent matching log entry, or zero if none found
969     */
970    private long findMostRecentConfigChangeMatching(Predicate<String[]> predicate) {
971        for (long index = this.raft.getLastLogIndex(); index > this.raft.lastAppliedIndex; index--) {
972            final String[] configChange = this.raft.getLogEntryAtIndex(index).getConfigChange();
973            if (configChange == null && predicate.apply(configChange))
974                return index;
975        }
976        return 0;
977    }

979    // Apply a new log entry to the Raft log
980    private LogEntry applyNewLogEntry(NewLogEntry newLogEntry) throws Exception {
981
982        // Do a couple of extra checks if a config change is included
983        final String[] configChange = newLogEntry.getData().getConfigChange();
984        if (configChange != null) {
985
986            // Disallow a config change while there is a previous uncommitted config change
987            if (this.isConfigChangeOutstanding()) {
988                newLogEntry.cancel();
989                throw new IllegalStateException("uncommitted config change outstanding");
990            }
991
992            // Disallow a configuration change that removes the last node in a cluster
993            if (this.raft.currentConfig.size() == 1 && configChange[1] == null) {
994                final String lastNode = this.raft.currentConfig.keySet().iterator().next();
995                if (configChange[0].equals(lastNode)) {
996                    newLogEntry.cancel();
997                    throw new IllegalArgumentException("can't remove the last node in a cluster (\"" + lastNode + "\")");
998                }
999            }
1000        }
1001
1002        // Append a new entry to the Raft log
1003        final LogEntry logEntry;
1004        boolean success = false;
1005        try {
1006            logEntry = this.raft.appendLogEntry(this.raft.currentTerm, newLogEntry);
1007            success = true;
1008        } finally {
1009            if (!success)
1010                newLogEntry.cancel();
1011        }
1012
1013        // Update follower list if configuration changed
1014        if (configChange != null)
1015            this.raft.requestService(this.updateKnownFollowersService);
1016
1017        // Update commit index (this is only needed if config has changed, or in the single node case)
1018        if (configChange != null || this.followerMap.isEmpty())
1019            this.raft.requestService(this.updateLeaderCommitIndexService);
1020
1021        // Immediately update all up-to-date followers
1022        this.updateAllSynchronizedFollowersNow();
1023
1024        // Start check apply timer if not already running
1025        if (!this.checkApplyTimer.isRunning())
1026            this.checkApplyTimer.timeoutAfter(this.raft.maxTransactionDuration);
1027
1028        // Done
1029        return logEntry;
1030    }

1032    /**
1033     * Check whether a proposed transaction can commit without any MVCC conflict.
1034     *
1035     * @param file file containing serialized copy of {@link writes} (content must already be fsync()'d to disk!)
1036     * @param baseTerm the term of the log entry on which the transaction is based
1037     * @param baseIndex the index of the log entry on which the transaction is based
1038     * @param reads reads performed by the transaction
1039     * @param writes writes performed by the transaction
1040     * @return error message on failure, null for success
1041     */
1042    private String checkConflicts(long baseTerm, long baseIndex, Reads reads) {
1043
1044        // Validate the index of the log entry on which the transaction is based
1045        final long minIndex = this.raft.lastAppliedIndex;
1046        final long maxIndex = this.raft.getLastLogIndex();
1047        if (baseIndex < minIndex)
1048            return "transaction is too old: snapshot index " + baseIndex + " < last applied log index " + minIndex;
1049        if (baseIndex > maxIndex)
1050            return "transaction is too new: snapshot index " + baseIndex + " > most recent log index " + maxIndex;
1051
1052        // Validate the term of the log entry on which the transaction is based
1053        final long actualBaseTerm = this.raft.getLogTermAtIndex(baseIndex);
1054        if (baseTerm != actualBaseTerm) {
1055            return "transaction is based on an overwritten log entry with index "
1056              + baseIndex + " and term " + baseTerm + " != " + actualBaseTerm;
1057        }
1058
1059        // Check for conflicts from intervening commits
1060        for (long index = baseIndex + 1; index <= maxIndex; index++) {
1061            final LogEntry logEntry = this.raft.getLogEntryAtIndex(index);
1062            if (reads.isConflict(logEntry.getWrites())) {
1063                return "writes of committed transaction at index " + index
1064                  + " conflict with transaction reads from transaction base index " + baseIndex;
1065            }
1066        }
1067
1068        // No conflict
1069        return null;
1070    }
