006package org.jsimpledb.kv.array;
007
008import com.google.common.base.Preconditions;
009import com.google.common.collect.UnmodifiableIterator;
010
011import java.nio.ByteBuffer;
012import java.util.Iterator;
013import java.util.NoSuchElementException;
014
015import org.jsimpledb.kv.AbstractKVStore;
016import org.jsimpledb.kv.KVPair;
017
018/**
019 * A simple read-only {@link KVStore} based on a sorted array of key/value pairs.
020 *
021 * <p>
022 * Instances query three {@link ByteBuffer}s, one for the array index, one for the key data, and one for the value data.
023 * Data for these {@link ByteBuffer}s is created using {@link ArrayKVWriter}.
024 *
025 * <p>
026 * Instances are optimized for minimal memory overhead and queries using keys sharing a prefix with the previously
027 * queried key. Key data is prefix-compressed.
028 *
029 * <p>
030 * Key and value data must each not exceed 2GB.
031 */
032public class ArrayKVStore extends AbstractKVStore {
033
034    private final ByteBuffer indx;
035    private final ByteBuffer keys;
036    private final ByteBuffer vals;
037    private final int size;
038
039    private final ThreadLocal<ArrayKVFinder> finderThreadLocal = new ThreadLocal<ArrayKVFinder>() {
040        @Override
041        protected ArrayKVFinder initialValue() {
042            return new ArrayKVFinder(ArrayKVStore.this.indx, ArrayKVStore.this.keys, ArrayKVStore.this.vals);
043        }
044    };
045
046    /**
047     * Constructor.
048     *
049     * @param indx buffer containing index data written by a {@link ArrayKVWriter}
050     * @param keys buffer containing key data written by a {@link ArrayKVWriter}
051     * @param vals buffer containing value data written by a {@link ArrayKVWriter}
052     * @throws IllegalArgumentException if any parameter is null
053     * @throws IllegalArgumentException if {@code indx} size is not a correct multiple
054     */
055    public ArrayKVStore(ByteBuffer indx, ByteBuffer keys, ByteBuffer vals) {
056        Preconditions.checkArgument(indx != null, "null indx");
057        Preconditions.checkArgument(keys != null, "null keys");
058        Preconditions.checkArgument(vals != null, "null vals");
059        Preconditions.checkArgument(indx.capacity() % 8 == 0, "index size is not a multiple of 8");
060        this.indx = indx;
061        this.keys = keys;
062        this.vals = vals;
063        this.size = this.indx.capacity() / 8;
064    }
065
066    @Override
067    public byte[] get(byte[] key) {
068        final ArrayKVFinder finder = this.finderThreadLocal.get();
069        final int index = finder.find(key);
070        if (index < 0)
071            return null;
072        return finder.readValue(index);
073    }
074
075    @Override
076    public KVPair getAtLeast(byte[] minKey) {
077        final ArrayKVFinder finder = this.finderThreadLocal.get();
078        int index = finder.find(minKey);
079        if (index < 0) {
080            index = ~index;
081            if (index == this.size)
082                return null;
083        }
084        return finder.readKV(index);
085    }
086
087    @Override
088    public KVPair getAtMost(byte[] maxKey) {
089        final ArrayKVFinder finder = this.finderThreadLocal.get();
090        int index = finder.find(maxKey);
091        if (index < 0)
092            index = ~index;
093        if (index == 0)
094            return null;
095        return finder.readKV(index - 1);
096    }
097
098    @Override
099    public Iterator<KVPair> getRange(byte[] minKey, byte[] maxKey, final boolean reverse) {
100
101        // Find min index
102        final ArrayKVFinder finder = this.finderThreadLocal.get();
103        int index;
104        if (minKey == null || minKey.length == 0)
105            index = 0;
106        else {
107            index = finder.find(minKey);
108            if (index < 0)
109                index = ~index;
110        }
111        final int minIndex = index;
112
113        // Find max index
114        if (maxKey == null)
115            index = this.size;
116        else {
117            index = finder.find(maxKey);
118            if (index < 0)
119                index = ~index;
120        }
121        final int maxIndex = index;
122
123        // Return iterator over array indexes
124        return new UnmodifiableIterator<KVPair>() {
125
126            private int index = reverse ? maxIndex : minIndex;
127
128            @Override
129            public boolean hasNext() {
130                return reverse ? this.index > minIndex : this.index < maxIndex;
131            }
132
133            @Override
134            public KVPair next() {
135                if (!this.hasNext())
136                    throw new NoSuchElementException();
137                if (reverse)
138                    this.index--;
139                final KVPair kv = ArrayKVStore.this.finderThreadLocal.get().readKV(this.index);
140                if (!reverse)
141                    this.index++;
142                return kv;
143            }
144        };
145    }
146
147    @Override
148    public void put(byte[] key, byte[] value) {
149        throw new UnsupportedOperationException();
150    }
151
152    @Override
153    public void remove(byte[] key) {
154        throw new UnsupportedOperationException();
155    }
156
157    @Override
158    public void removeRange(byte[] minKey, byte[] maxKey) {
159        throw new UnsupportedOperationException();
160    }
161}
