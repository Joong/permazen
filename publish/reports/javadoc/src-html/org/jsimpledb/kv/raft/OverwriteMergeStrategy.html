006package org.jsimpledb.kv.raft;
007
008import java.util.Iterator;
009
010import org.jsimpledb.kv.KVPair;
011import org.jsimpledb.kv.KVTransaction;
012
013/**
014 * {@link MergeStrategy} that completely overwrites the destination database with the content of the source database.
015 */
016public class OverwriteMergeStrategy implements MergeStrategy {
017
018    @Override
019    public void merge(KVTransaction src, KVTransaction dst) {
020        dst.removeRange(null, null);
021        final Iterator<KVPair> i = src.getRange(null, null, false);
022        while (i.hasNext()) {
023            final KVPair pair = i.next();
024            dst.put(pair.getKey(), pair.getValue());
025        }
026        Util.closeIfPossible(i);
027    }
028
029    @Override
030    public String toString() {
031        return "overwrite merge strategy";
032    }
033}
