007
008import com.google.common.base.Preconditions;
009
010import java.io.BufferedOutputStream;
011import java.io.Closeable;
012import java.io.IOException;
013import java.io.OutputStream;
014import java.util.Arrays;
015import java.util.Iterator;
016import java.util.Map;
017
018import org.jsimpledb.kv.KVPair;
019import org.jsimpledb.kv.KVStore;
020import org.jsimpledb.kv.KeyRange;
021import org.jsimpledb.kv.mvcc.Mutations;
022import org.jsimpledb.util.ByteUtil;

024/**
025 * Writes {@link ArrayKVStore} index, key, and value data, given a sorted sequence of key/value pairs.
026 *
027 * <p>
028 * Key and value data must not exceed 2GB (each separately).
029 */
030public class ArrayKVWriter implements Closeable {
031
032    private static final int BUFFER_SIZE = 1024 * 1024;
033
034    // Flags used by writeMerged()
035    private static final int MERGE_KV = 0x01;
036    private static final int MERGE_PUT = 0x02;
037    private static final int MERGE_ADJUST = 0x04;
038
039    private final BufferedOutputStream indxOutput;
040    private final BufferedOutputStream keysOutput;
041    private final BufferedOutputStream valsOutput;
042
043    private int keysLength;
044    private int valsLength;
045
046    private int nextIndex;
047    private byte[] prevKey;
048    private byte[] baseKey;
049    private int baseKeyOffset;
050    private boolean closed;
051
052    /**
053     * Constructor.
054     *
055     * @param indxOutput index file output
056     * @param keysOutput key data file output
057     * @param valsOutput value data file output
058     */
059    public ArrayKVWriter(OutputStream indxOutput, OutputStream keysOutput, OutputStream valsOutput) {
060        Preconditions.checkArgument(indxOutput != null, "null indxOutput");
061        Preconditions.checkArgument(keysOutput != null, "null keysOutput");
062        Preconditions.checkArgument(valsOutput != null, "null valsOutput");
063        this.indxOutput = new BufferedOutputStream(indxOutput, BUFFER_SIZE);
064        this.keysOutput = new BufferedOutputStream(keysOutput, BUFFER_SIZE);
065        this.valsOutput = new BufferedOutputStream(valsOutput, BUFFER_SIZE);
066    }
067
068    /**
069     * Get the number of bytes written so far to the index file.
070     *
071     * @return length of the index file
072     */
073    public int getIndxLength() {
074        return this.nextIndex * 8;
075    }
076
077    /**
078     * Get the number of bytes written so far to the key data file.
079     *
080     * @return length of the key data file
081     */
082    public int getKeysLength() {
083        return this.keysLength;
084    }
085
086    /**
087     * Get the number of bytes written so far to the value data file.
088     *
089     * @return length of the value data file
090     */
091    public int getValsLength() {
092        return this.valsLength;
093    }
094
095    /**
096     * Write out the next key/value pair.
097     *
098     * @param key key
099     * @param val value
100     * @throws IllegalArgumentException if {@code key} is out of order (i.e., not strictly greater then the previous key)
101     * @throws IllegalArgumentException if {@code key} or {@code val} is null
102     * @throws IllegalStateException if either the key or data file would grow larger than 2<sup>31</sup>-1 bytes
103     * @throws IOException if an I/O error occurrs
104     */
105    public void writeKV(byte[] key, byte[] val) throws IOException {
106
107        // Sanity checks
108        Preconditions.checkArgument(key != null, "null key");
109        Preconditions.checkArgument(val != null, "null value");
110        Preconditions.checkArgument(this.prevKey == null || ByteUtil.compare(key, this.prevKey) > 0, "key <= previous key");
111        Preconditions.checkState((this.nextIndex * 8) + 8 > 0, "too much index data");
112        Preconditions.checkState(this.keysLength == 0 || this.keysLength + key.length > 0, "too much key data");
113        Preconditions.checkState(this.valsLength == 0 || this.valsLength + val.length > 0, "too much value data");
114
115        // Write key index entry and data
116        if ((this.nextIndex & 0x1f) == 0) {
117
118            // Write a base key index entry every 32 entries
119            this.writeIndxValue(this.keysLength);
120            this.baseKeyOffset = this.keysLength;
121            this.baseKey = key;
122
123            // Write key data
124            this.keysOutput.write(key);
125            this.keysLength += key.length;
126        } else {
127
128            // Match maximal prefix of most recent base key, up to 255 bytes
129            final int maxPrefixLength = Math.min(Math.min(this.baseKey.length, key.length), 0xff);
130            int prefixLength = 0;
131            while (prefixLength < maxPrefixLength && key[prefixLength] == this.baseKey[prefixLength])
132                prefixLength++;
133            final int suffixLength = key.length - prefixLength;
134            assert suffixLength > 0;
135
136            // Write encoded { base key prefix length, offset to key suffix }
137            final int suffixRelativeOffset = this.keysLength - this.baseKeyOffset;
138            Preconditions.checkState((suffixRelativeOffset & 0xff000000) == 0, "key(s) too long");
139            this.writeIndxValue(prefixLength << 24 | suffixRelativeOffset);
140
141            // Write key data - suffix only
142            this.keysOutput.write(key, prefixLength, suffixLength);
143            this.keysLength += suffixLength;
144        }
145
146        // Write value index entry
147        this.writeIndxValue(this.valsLength);
148
149        // Write value data
150        this.valsOutput.write(val);
151        this.valsLength += val.length;
152
153        // Update state
154        this.prevKey = key;
155        this.nextIndex++;
156    }
157
158    /**
159     * Merge the given key/value pair iteration with the specified mutations and write out the merged combination.
160     * What's written is the result of the key/value iteration with the mutations applied.
161     *
162     * <p>
163     * The {@link KVStore} parameter will only be used to invoke {@link KVStore#encodeCounter KVStore.encodeCounter()}
164     * and {@link KVStore#decodeCounter KVStore.decodeCounter()} in order to handle counter adjustments.
165     *
166     * @param kvstore callback for encoding and decoding counter values
167     * @param kvIterator key/value pairs - must be sorted in order
168     * @param mutations mutations to apply
169     * @throws IllegalArgumentException if any parameter is null
170     * @throws IllegalArgumentException if {@code kvs} iterates out of order
171     * @throws IllegalStateException if either the key or data file would grow larger than 2<sup>31</sup>-1 bytes
172     * @throws IOException if an I/O error occurrs
173     */
174    @SuppressWarnings("fallthrough")
175    public void writeMerged(KVStore kvstore, Iterator<KVPair> kvIterator, Mutations mutations) throws IOException {
176
177        // Sanity checks
178        Preconditions.checkArgument(kvstore != null, "null kvstore");
179        Preconditions.checkArgument(kvIterator != null, "null kvIterator");
180        Preconditions.checkArgument(mutations != null, "null mutations");
181
182        // Initialize iterators
183        final Iterator<? extends KeyRange> removeIterator = mutations.getRemoveRanges().iterator();
184        final Iterator<? extends Map.Entry<byte[], byte[]>> putIterator = mutations.getPutPairs().iterator();
185        final Iterator<? extends Map.Entry<byte[], Long>> adjustIterator = mutations.getAdjustPairs().iterator();
186
187        // Merge iterators and write the merged result to the ArrayKVWriter
188        KVPair kv = kvIterator.hasNext() ? kvIterator.next() : null;
189        KeyRange remove = removeIterator.hasNext() ? removeIterator.next() : null;
190        Map.Entry<byte[], byte[]> put = putIterator.hasNext() ? putIterator.next() : null;
191        Map.Entry<byte[], Long> adjust = adjustIterator.hasNext() ? adjustIterator.next() : null;
192        while (true) {
193
194            // Find the minimum key among remaining (a) key/value pairs, (b) puts, and (c) adjusts
195            byte[] key = null;
196            if (kv != null)
197                key = kv.getKey();
198            if (put != null && (key == null || ByteUtil.compare(put.getKey(), key) < 0))
199                key = put.getKey();
200            if (adjust != null && (key == null || ByteUtil.compare(adjust.getKey(), key) < 0))
201                key = adjust.getKey();
202            if (key == null)                                                        // we're done
203                break;
204
205            // Determine which inputs apply to the current key
206            int inputs = 0;
207            if (kv != null && Arrays.equals(kv.getKey(), key))
208                inputs |= MERGE_KV;
209            if (put != null && Arrays.equals(put.getKey(), key))
210                inputs |= MERGE_PUT;
211            if (adjust != null && Arrays.equals(adjust.getKey(), key))
212                inputs |= MERGE_ADJUST;
213
214            // Find the removal that applies to this next key, if any
215            while (remove != null) {
216                final byte[] removeMax = remove.getMax();
217                if (removeMax != null && ByteUtil.compare(removeMax, key) <= 0) {
218                    remove = removeIterator.hasNext() ? removeIterator.next() : null;
219                    continue;
220                }
221                break;
222            }
223            final boolean removed = remove != null && ByteUtil.compare(remove.getMin(), key) <= 0;
224
225            // Merge the applicable inputs
226            switch (inputs) {
227            case 0:                                                     // k/v store entry was removed
228                break;
229            case MERGE_KV:
230                if (!removed)
231                    this.writeKV(kv.getKey(), kv.getValue());
232                break;
233            case MERGE_PUT:
234            case MERGE_PUT | MERGE_KV:
235                this.writeKV(put.getKey(), put.getValue());
236                break;
237            case MERGE_ADJUST:                                          // adjusted a non-existent value; ignore
238                break;
239            case MERGE_ADJUST | MERGE_KV:
240                if (removed)                                            // adjusted a removed value; ignore
241                    break;
242                // FALLTHROUGH
243            case MERGE_ADJUST | MERGE_PUT:
244            case MERGE_ADJUST | MERGE_PUT | MERGE_KV:
245            {
246                final byte[] encodedCount = (inputs & MERGE_PUT) != 0 ? put.getValue() : kv.getValue();
247                final long counter;
248                try {
249                    counter = kvstore.decodeCounter(encodedCount);
250                } catch (IllegalArgumentException e) {
251                    break;
252                }
253                final byte[] value = kvstore.encodeCounter(counter + adjust.getValue());
254                this.writeKV(key, value);
255                break;
256            }
257            default:
258                throw new RuntimeException();
259            }
260
261            // Advance k/v, put, and/or adjust
262            if ((inputs & MERGE_KV) != 0)
263                kv = kvIterator.hasNext() ? kvIterator.next() : null;
264            if ((inputs & MERGE_PUT) != 0)
265                put = putIterator.hasNext() ? putIterator.next() : null;
266            if ((inputs & MERGE_ADJUST) != 0)
267                adjust = adjustIterator.hasNext() ? adjustIterator.next() : null;
268        }
269    }
270
271    private void writeIndxValue(int offset) throws IOException {
272        this.indxOutput.write(offset >> 24);
273        this.indxOutput.write(offset >> 16);
274        this.indxOutput.write(offset >> 8);
275        this.indxOutput.write(offset);
276    }
277
278    /**
279     * Flush all three outputs.
280     *
281     * @throws IOException if an I/O error occurrs
282     */
283    public void flush() throws IOException {
284        this.indxOutput.flush();
285        this.keysOutput.flush();
286        this.valsOutput.flush();
287    }
288
289    /**
290     * Close all three outputs.
291     *
292     * @throws IOException if an I/O error occurrs
293     */
294    @Override
295    public void close() throws IOException {
296        if (this.closed)
297            return;
298        this.closed = true;
299        this.indxOutput.close();
300        this.keysOutput.close();
301        this.valsOutput.close();
302    }
