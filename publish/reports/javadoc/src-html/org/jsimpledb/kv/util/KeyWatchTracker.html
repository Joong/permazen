006package org.jsimpledb.kv.util;
007
008import com.google.common.base.Function;
009import com.google.common.base.Preconditions;
010import com.google.common.collect.Iterables;
011import com.google.common.util.concurrent.AbstractFuture;
012import com.google.common.util.concurrent.ListenableFuture;
013
014import java.util.ArrayList;
015import java.util.HashSet;
016import java.util.Map;
017import java.util.NavigableMap;
018import java.util.TreeMap;
019
020import org.jsimpledb.kv.KeyRange;
021import org.jsimpledb.kv.mvcc.Mutations;
022import org.jsimpledb.util.ByteUtil;
023
024/**
025 * Utility class used to track key watches.
026 *
027 * <p>
028 * For space efficiency, this class does not track original values. Therefore, spurious notifications can result.
029 *
030 * <p>
031 * Instances are thread safe.
032 *
033 * @see org.jsimpledb.kv.KVTransaction#watchKey
034 */
035public class KeyWatchTracker {
036
037    private final TreeMap<byte[], KeyInfo> keyInfos = new TreeMap<>(ByteUtil.COMPARATOR);
038
039    /**
040     * Register a new watch.
041     *
042     * <p>
043     * If the returned {@link java.util.concurrent.Future} is {@link java.util.concurrent.Future#cancel cancel()}'ed,
044     * the watch is automatically unregistered.
045     *
046     * @param key the key to watch
047     * @return a {@link ListenableFuture} that returns {@code key} when the value associated with {@code key} is modified
048     * @throws IllegalArgumentException if {@code key} is null
049     */
050    public synchronized ListenableFuture<Void> register(byte[] key) {
051        Preconditions.checkArgument(key != null, "null key");
052        KeyInfo keyInfo = this.keyInfos.get(key);
053        if (keyInfo == null) {
054            key = key.clone();                              // avoid external mutation of key contents
055            keyInfo = new KeyInfo(key);
056            this.keyInfos.put(key, keyInfo);
057        }
058        return keyInfo.add();
059    }
060
061    /**
062     * Count the number of keys being watched.
063     *
064     * <p>
065     * Note that the same key can be watched more than once, so this only counts keys being watched, not total watches.
066     *
067     * @return number of keys being watched
068     */
069    public synchronized int getNumKeysWatched() {
070        return this.keyInfos.size();
071    }
072
073    /**
074     * Trigger all watches associated with the given key.
075     *
076     * @param key the key that has been modified
077     * @return true if any watches were triggered, otherwise false
078     * @throws IllegalArgumentException if {@code key} is null
079     */
080    public boolean trigger(byte[] key) {
081        Preconditions.checkArgument(key != null, "null key");
082        final KeyInfo keyInfo;
083        synchronized (this) {
084            keyInfo = KeyWatchTracker.this.keyInfos.remove(key);
085        }
086        if (keyInfo != null) {
087            keyInfo.succeed();
088            return true;
089        }
090        return false;
091    }
092
093    /**
094     * Trigger all watches associated with the given keys.
095     *
096     * @param keys keys that have been modified
097     * @return true if any watches were triggered, otherwise false
098     * @throws IllegalArgumentException if {@code keys} is null
099     */
100    public boolean trigger(Iterable<byte[]> keys) {
101        Preconditions.checkArgument(keys != null, "null keys");
102        final ArrayList<KeyInfo> triggerList = new ArrayList<>();
103        synchronized (this) {
104            for (byte[] key : keys) {
105                final KeyInfo keyInfo = KeyWatchTracker.this.keyInfos.remove(key);
106                if (keyInfo != null)
107                    triggerList.add(keyInfo);
108            }
109        }
110        for (KeyInfo keyInfo : triggerList)
111            keyInfo.succeed();
112        return !triggerList.isEmpty();
113    }
114
115    /**
116     * Trigger all watches associated with keys in the given range.
117     *
118     * @param range range of keys that have been modified
119     * @return true if any watches were triggered, otherwise false
120     * @throws IllegalArgumentException if {@code range} is null
121     */
122    public boolean trigger(KeyRange range) {
123        Preconditions.checkArgument(range != null, "null range");
124        final ArrayList<KeyInfo> triggerList = new ArrayList<>();
125        synchronized (this) {
126            final NavigableMap<byte[], KeyInfo> subMap = range.getMax() != null ?
127              this.keyInfos.subMap(range.getMin(), true, range.getMax(), false) : this.keyInfos.tailMap(range.getMin(), true);
128            triggerList.addAll(subMap.values());
129            subMap.clear();
130        }
131        for (KeyInfo keyInfo : triggerList)
132            keyInfo.succeed();
133        return !triggerList.isEmpty();
134    }
135
136    /**
137     * Trigger all watches associated with the given mutations.
138     *
139     * @param mutations mutations
140     * @return true if any watches were triggered, otherwise false
141     * @throws IllegalArgumentException if {@code mutations} is null
142     */
143    public boolean trigger(Mutations mutations) {
144        Preconditions.checkArgument(mutations != null, "null mutations");
145        boolean result = false;
146        for (KeyRange range : mutations.getRemoveRanges())
147            result |= this.trigger(range);
148        final EntryKeyFunction keyFunction = new EntryKeyFunction();
149        result |= this.trigger(this.applyEntryKeyFunction(mutations.getPutPairs(), keyFunction));
150        result |= this.trigger(this.applyEntryKeyFunction(mutations.getAdjustPairs(), keyFunction));
151        return result;
152    }
153
154    // This method exists solely to bind the generic type parameters
155    private <E extends Map.Entry<byte[], ?>> Iterable<byte[]> applyEntryKeyFunction(Iterable<E> i, EntryKeyFunction keyFunction) {
156        return Iterables.<E, byte[]>transform(i, keyFunction);
157    }
158
159    /**
160     * Trigger all watches.
161     */
162    public synchronized void triggerAll() {
163        for (KeyInfo keyInfo : this.keyInfos.values())
164            keyInfo.succeed();
165        this.keyInfos.clear();
166    }
167
168    /**
169     * Discard all outstanding key watches and fail them with the given exception.
170     *
171     * @param e failing exception
172     */
173    public synchronized void failAll(Exception e) {
174        for (KeyInfo keyInfo : this.keyInfos.values())
175            keyInfo.fail(e);
176        this.keyInfos.clear();
177    }
178
179// KeyInfo
180
181    private class KeyInfo {
182
183        private final byte[] key;
184        private final HashSet<KeyFuture> futures = new HashSet<>(1);
185
186        KeyInfo(byte[] key) {
187            assert key != null;
188            this.key = key;
189        }
190
191        KeyFuture add() {
192            assert Thread.holdsLock(KeyWatchTracker.this);
193            final KeyFuture future = new KeyFuture(this);
194            this.futures.add(future);
195            return future;
196        }
197
198        void cancel(KeyFuture future) {
199            synchronized (KeyWatchTracker.this) {
200                if (this.futures.remove(future) && this.futures.isEmpty())
201                    KeyWatchTracker.this.keyInfos.remove(this.key);
202            }
203        }
204
205        void succeed() {
206            for (KeyFuture future : this.futures)
207                future.set(null);
208        }
209
210        void fail(Exception e) {
211            for (KeyFuture future : this.futures)
212                future.setException(e);
213        }
214    }
215
216// KeyFuture
217
218    private static class KeyFuture extends AbstractFuture<Void> {
219
220        private final KeyInfo keyInfo;
221
222        KeyFuture(KeyInfo keyInfo) {
223            assert keyInfo != null;
224            this.keyInfo = keyInfo;
225        }
226
227        @Override
228        protected boolean set(Void value) {
229            return super.set(value);
230        }
231
232        @Override
233        protected boolean setException(Throwable t) {
234            return super.setException(t);
235        }
236
237        @Override
238        public boolean cancel(boolean mayInterruptIfRunning) {
239            this.keyInfo.cancel(this);
240            return super.cancel(mayInterruptIfRunning);
241        }
242    }
243
244// EntryKeyFunction
245
246    private static class EntryKeyFunction implements Function<Map.Entry<byte[], ?>, byte[]> {
247
248        @Override
249        public byte[] apply(Map.Entry<byte[], ?> entry) {
250            return entry.getKey();
251        }
252    }
253}
