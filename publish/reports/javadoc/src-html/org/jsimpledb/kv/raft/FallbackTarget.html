017/**
018 * Represents one of the underlying {@link RaftKVDatabase}s managed by a {@link FallbackKVDatabase}.
019 *
020 * <p>
021 * Instances must be configured with the {@link RaftKVDatabase} and several parameters used to determine

032     *
033     * @see #setTransactionTimeout setTransactionTimeout()
034     */

036

038     * Default check interval ({@value #DEFAULT_CHECK_INTERVAL}ms).
039     *
040     * @see #setCheckInterval setCheckInterval()
041     */

044    /**
045     * Default minimum available type ({@value #DEFAULT_MIN_AVAILABLE_TIME}ms).

047     * @see #setMinAvailableTime setMinAvailableTime()

234    /**
235     * Perform an availability assessment of the underlying {@link RaftKVDatabase} associated with this instance.
236     *
237     * <p>
238     * This method may block for as long as necessary to determine availability, but it should not block
239     * indefinitely. If a {@link KVTransaction} is opened, {@link KVTransaction#setTimeout setTimeout()}
240     * should be used to set a time limit.
241     *
242     * <p>
243     * This method will not be invoked concurrently by two different threads.
244     *
245     * <p>
246     * If this method throws an unchecked exception, the database will be assumed to be unavailable.
247     *
248     * <p>
249     * The implementation in {@link FallbackTarget} determines availability by attempting to commit a read-only,
250     * {@link Consistency#LINEARIZABLE} transaction within the configured maximum timeout.
251     *
252     * @return true if database is available, false otherwise
253     */
254    protected boolean checkAvailability() {
255
256        // Setup
257        if (this.log.isTraceEnabled())
258            this.log.trace("checking availability of " + this.raft + " with timeout of " + this.transactionTimeout + "ms");
259        final long startTimeNanos = System.nanoTime();
260        boolean success = false;
261
262        // Perform transaction
263        final KVTransaction tx = this.raft.createTransaction();
264        try {
265            tx.setTimeout(this.transactionTimeout);
266            tx.getAtLeast(ByteUtil.EMPTY);
267            tx.commit();
268            success = true;
269        } finally {
270            if (!success)
271                tx.rollback();
272        }
273
274        // Check timeout
275        final int duration = (int)((System.nanoTime() - startTimeNanos) / 1000000L);
276        final boolean timedOut = duration > this.transactionTimeout;
277        if (this.log.isTraceEnabled()) {
278            this.log.trace("availability transaction on " + this.raft + " completed in "
279              + duration + "ms (" + (timedOut ? "failed" : "successful") + ")");
280        }
281        if (timedOut && this.log.isDebugEnabled()) {
282            this.log.debug("availability transaction on " + this.raft + " completed in "
283              + duration + " > " + this.transactionTimeout + " ms, returning unavailable");
284        }
285
286        // Done
287        return !timedOut;
288    }

292    public FallbackTarget clone() {
293        try {
294            return (FallbackTarget)super.clone();
295        } catch (CloneNotSupportedException e) {
296            throw new RuntimeException(e);
297        }
298    }
