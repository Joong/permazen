006package org.jsimpledb.cli.cmd;
007
008import java.io.PrintWriter;
009import java.text.SimpleDateFormat;
010import java.util.Date;
011import java.util.List;
012import java.util.Map;
013
014import org.jsimpledb.SessionMode;
015import org.jsimpledb.cli.CliSession;
016import org.jsimpledb.kv.raft.fallback.FallbackKVDatabase;
017import org.jsimpledb.kv.raft.fallback.FallbackTarget;
018import org.jsimpledb.parse.ParseContext;
019
020@Command(modes = { SessionMode.KEY_VALUE, SessionMode.CORE_API, SessionMode.JSIMPLEDB })
021public class RaftFallbackStatusCommand extends AbstractCommand {
022
023    public RaftFallbackStatusCommand() {
024        super("raft-fallback-status");
025    }
026
027    @Override
028    public String getHelpSummary() {
029        return "Displays the status of the Raft fallback database";
030    }
031
032    @Override
033    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
034        return new CliSession.Action() {
035            @Override
036            public void run(CliSession session) throws Exception {
037                if (!(session.getKVDatabase() instanceof FallbackKVDatabase))
038                    throw new Exception("key/value store is not Raft fallback");
039                RaftFallbackStatusCommand.this.displayStatus(session.getWriter(), (FallbackKVDatabase)session.getKVDatabase());
040            }
041        };
042    }
043
044    private void displayStatus(PrintWriter writer, FallbackKVDatabase db) {
045
046    // TODO show configuration here...
047
048        final List<FallbackTarget> targets = db.getFallbackTargets();
049        final Date lastStandaloneActiveTime = db.getLastStandaloneActiveTime();
050
051        // Show configuration
052        writer.println();
053        writer.println("Configuration");
054        writer.println("=============");
055        writer.println();
056        writer.println(String.format("%15s: %s", "Standalone KV", db.getStandaloneTarget()));
057        writer.println(String.format("%15s: %s", "State file", db.getStateFile()));
058        writer.println();
059        writer.println("Raft Targets");
060        writer.println("============");
061        writer.println();
062        writer.println(String.format("  %5s %-10s %-10s %-9s %-11s %-20.20s %-20.20s %s",
063          "Index", "Check Int.", "Tx Timeout", "Min Avail", "Min Unavail", "Merge", "Rejoin", "Description"));
064        writer.println(String.format("  %5s %-10s %-10s %-9s %-11s %-20.20s %-20.20s %s",
065          "-----", "----------", "----------", "---------", "-----------", "-----", "------", "-----------"));
066        for (int i = 0; i < targets.size(); i++) {
067            final FallbackTarget target = targets.get(i);
068            writer.println(String.format("  %5s %10s %10s %9s %11s %-20.20s %-20.20s %.38s",
069              i, target.getCheckInterval() + "ms", target.getTransactionTimeout() + "ms",
070              target.getMinAvailableTime() + "ms", target.getMinUnavailableTime() + "ms",
071              this.objString(target.getUnavailableMergeStrategy()), this.objString(target.getRejoinMergeStrategy()),
072              target));
073        }
074
075        // Show status
076        writer.println();
077        writer.println("Fallback Status");
078        writer.println("===============");
079        writer.println();
080        writer.println(String.format("  %5s %-6s %-9s %-20s %s", "Index", "Active", "Available", "Last Change", "Last Active"));
081        writer.println(String.format("  %5s %-6s %-9s %-20s %s", "-----", "------", "---------", "-----------", "-----------"));
082        for (int i = -1; i < targets.size(); i++) {
083            final FallbackTarget target = i >= 0 ? targets.get(i) : null;
084            final boolean active = i == db.getCurrentTargetIndex();
085            final boolean available = target == null || target.isAvailable();
086            writer.println(String.format("  %3d   %3s    %6s    %-20s %s",
087              i, active ? "*" : "", available ? "Yes" : "No ",
088              this.date(target != null ? target.getLastChangeTime() : null),
089              active ? "Now" : this.date(target != null ? target.getLastActiveTime() : null)));
090        }
091        writer.println();
092    }
093
094    private String date(Date date) {
095        if (date == null)
096            return "N/A";
097        return new SimpleDateFormat("yyyy-MM-dd HH:mm:ss").format(date);
098    }
099
100    // Strip package name from default toString() output: "foo.bar.Jam@1234" -> "Jam@1234"
101    private String objString(Object obj) {
102        return String.valueOf(obj).replaceAll("^(\\p{javaJavaIdentifierStart}\\p{javaJavaIdentifierPart}*\\.)*"
103          + "(\\p{javaJavaIdentifierStart}\\p{javaJavaIdentifierPart}*@)", "$2");
104    }
105}
