006package org.jsimpledb.cli;
007
008import java.io.BufferedInputStream;
009import java.io.File;
010import java.io.FileDescriptor;
011import java.io.FileInputStream;
012import java.io.InputStream;
013import java.util.ArrayDeque;
014import java.util.ArrayList;
015import java.util.Arrays;
016import java.util.LinkedHashSet;
017import java.util.List;
018
019import org.jsimpledb.JSimpleDB;
020import org.jsimpledb.SessionMode;
021import org.jsimpledb.cli.cmd.Command;
022import org.jsimpledb.core.Database;
023import org.jsimpledb.parse.ParseUtil;
024import org.jsimpledb.parse.func.Function;
025import org.jsimpledb.schema.SchemaModel;
026import org.jsimpledb.spring.AnnotatedClassScanner;
027import org.jsimpledb.util.AbstractMain;
028
029/**
030 * CLI main entry point.
031 */
032public class Main extends AbstractMain {
033
034    public static final String HISTORY_FILE = ".jsimpledb_history";
035
036    private File schemaFile;
037    private SessionMode mode = SessionMode.JSIMPLEDB;
038    private final LinkedHashSet<Class<?>> commandClasses = new LinkedHashSet<>();
039    private final LinkedHashSet<Class<?>> functionClasses = new LinkedHashSet<>();
040    private final ArrayList<String> oneShotCommands = new ArrayList<>();
041
042    @Override
043    protected boolean parseOption(String option, ArrayDeque<String> params) {
044        if (option.equals("--schema-file")) {
045            if (params.isEmpty())
046                this.usageError();
047            this.schemaFile = new File(params.removeFirst());
048            this.allowAutoDemo = false;
049        } else if (option.equals("--command") || option.equals("-c")) {
050            if (params.isEmpty())
051                this.usageError();
052            this.oneShotCommands.add(params.removeFirst());
053        } else if (option.equals("--core-mode"))
054            this.mode = SessionMode.CORE_API;
055        else if (option.equals("--kv-mode"))
056            this.mode = SessionMode.KEY_VALUE;
057        else if (option.equals("--cmd-pkg")) {
058            if (params.isEmpty())
059                this.usageError();
060            this.scanCommandClasses(params.removeFirst());
061        } else if (option.equals("--func-pkg")) {
062            if (params.isEmpty())
063                this.usageError();
064            this.scanFunctionClasses(params.removeFirst());
065        } else
066            return false;
067        return true;
068    }
069
070    private void scanCommandClasses(String pkgname) {
071        for (String className : new AnnotatedClassScanner(Command.class).scanForClasses(pkgname.split("[\\s,]")))
072            this.commandClasses.add(this.loadClass(className));
073    }
074
075    private void scanFunctionClasses(String pkgname) {
076        for (String className : new AnnotatedClassScanner(Function.class).scanForClasses(pkgname.split("[\\s,]")))
077            this.functionClasses.add(this.loadClass(className));
078    }
079
080    @Override
081    public int run(String[] args) throws Exception {
082
083        // Parse command line
084        final ArrayDeque<String> params = new ArrayDeque<String>(Arrays.asList(args));
085        final int result = this.parseOptions(params);
086        if (result != -1)
087            return result;
088        switch (params.size()) {
089        case 0:
090            break;
091        default:
092            this.usageError();
093            return 1;
094        }
095
096        // Read schema file from `--schema-file' (if any)
097        SchemaModel schemaModel = null;
098        if (this.schemaFile != null) {
099            try {
100                final InputStream input = new BufferedInputStream(new FileInputStream(this.schemaFile));
101                try {
102                    schemaModel = SchemaModel.fromXML(input);
103                } finally {
104                    input.close();
105                }
106            } catch (Exception e) {
107                System.err.println(this.getName() + ": can't load schema from `" + this.schemaFile + "': " + e.getMessage());
108                if (this.verbose)
109                    e.printStackTrace(System.err);
110                return 1;
111            }
112        }
113
114        // Set up Database
115        final Database db = this.startupKVDatabase();
116
117        // Load JSimpleDB layer, if specified
118        final JSimpleDB jdb = this.schemaClasses != null ? this.getJSimpleDBFactory(db).newJSimpleDB() : null;
119
120        // Sanity check consistent schema model if both --schema-file and --model-pkg were specified
121        if (jdb != null) {
122            if (schemaModel != null) {
123                if (!schemaModel.equals(jdb.getSchemaModel())) {
124                    System.err.println(this.getName() + ": schema from `" + this.schemaFile + "' conflicts with schema generated"
125                      + " from scanned classes");
126                    System.err.println(schemaModel.differencesFrom(jdb.getSchemaModel()));
127                    return 1;
128                }
129            } else
130                schemaModel = jdb.getSchemaModel();
131        }
132
133        // Downgrade to Core API mode from JSimpleDB mode if no Java model classes provided
134        if (jdb == null && this.mode.equals(SessionMode.JSIMPLEDB)) {
135            System.err.println(this.getName() + ": entering core API mode because no Java model classes were specified");
136            this.mode = SessionMode.CORE_API;
137        }
138
139        // Set up console
140        final Console console;
141        switch (this.mode) {
142        case KEY_VALUE:
143            console = new Console(db.getKVDatabase(), new FileInputStream(FileDescriptor.in), System.out);
144            break;
145        case CORE_API:
146            console = new Console(db, new FileInputStream(FileDescriptor.in), System.out);
147            break;
148        case JSIMPLEDB:
149            console = new Console(jdb, new FileInputStream(FileDescriptor.in), System.out);
150            break;
151        default:
152            console = null;
153            assert false;
154            break;
155        }
156        console.setHistoryFile(new File(new File(System.getProperty("user.home")), HISTORY_FILE));
157
158        // Set up CLI session
159        final CliSession session = console.getSession();
160        session.setDatabaseDescription(this.getDatabaseDescription());
161        session.setReadOnly(this.readOnly);
162        session.setVerbose(this.verbose);
163        session.setSchemaModel(schemaModel);
164        session.setSchemaVersion(this.schemaVersion);
165        session.setAllowNewSchema(this.allowNewSchema);
166        session.registerStandardFunctions();
167        session.registerStandardCommands();
168        try {
169            for (Class<?> cl : this.commandClasses) {
170                final Command annotation = cl.getAnnotation(Command.class);
171                if (annotation != null && Arrays.asList(annotation.modes()).contains(session.getMode()))
172                    session.registerCommand(cl);
173            }
174            for (Class<?> cl : this.functionClasses) {
175                final Function annotation = cl.getAnnotation(Function.class);
176                if (annotation != null && Arrays.asList(annotation.modes()).contains(session.getMode()))
177                    session.registerFunction(cl);
178            }
179        } catch (IllegalArgumentException e) {
180            System.err.println(this.getName() + ": " + e.getMessage());
181            if (this.verbose)
182                e.printStackTrace(System.err);
183            return 1;
184        }
185
186        // Handle one-shot command mode
187        if (!this.oneShotCommands.isEmpty()) {
188            for (String text : this.oneShotCommands) {
189
190                // Parse command(s)
191                final List<CliSession.Action> actions = console.parseCommand(text);
192                if (actions == null) {
193                    session.getWriter().println("Error: failed to parse command `" + ParseUtil.truncate(text, 40) + "'");
194                    return 1;
195                }
196
197                // Execute command(s)
198                for (CliSession.Action action : console.parseCommand(text)) {
199                    if (!session.performCliSessionAction(action))
200                        return 1;
201                }
202            }
203            return 0;
204        }
205
206        // Run console
207        console.run();
208
209        // Shut down KV database
210        this.shutdownKVDatabase();
211
212        // Done
213        return 0;
214    }
215
216    @Override
217    protected String getName() {
218        return "jsimpledb";
219    }
220
221    @Override
222    protected void usageMessage() {
223        System.err.println("Usage:");
224        System.err.println("  " + this.getName() + " [options]");
225        System.err.println("Options:");
226        this.outputFlags(new String[][] {
227          { "--schema-file file",       "Load core database schema from XML file" },
228          { "--cmd-pkg package",        "Register @Command-annotated classes found under the specified Java package" },
229          { "--func-pkg package",       "Register @Function-annotated classes found under the specified Java package" },
230          { "--core-mode",              "Force core API mode (default if neither Java model classes nor schema are provided)" },
231          { "--kv-mode",                "Force key/value mode" },
232          { "--command, -c command",    "Execute the given command and then exit (may be repeated)" },
233        });
234        System.err.println("Exactly one key/value database type must be configured; optional flags must"
235          + " follow the main flag for the key/value database.");
236        System.err.println("Raft requires its own separate key/value store, which should be specified after the Raft options.");
237        System.err.println("For Raft fallback, specify Raft followed by the Raft key/value store, then Raft fallback, followed"
238          + " by the fallback key/value store.");
239    }
240
241    public static void main(String[] args) throws Exception {
242        new Main().doMain(args);
243    }
244}
