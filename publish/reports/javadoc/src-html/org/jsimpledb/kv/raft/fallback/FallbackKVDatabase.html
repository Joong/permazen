040/**
041 * A partition-tolerant {@link KVDatabase} that automatically migrates between a clustered {@link RaftKVDatabase}
042 * and a local, non-clustered "standalone mode" {@link KVDatabase}, based on availability of the Raft cluster.
043 *
044 * <p>
045 * A {@link RaftKVDatabase} requires that the local node be part of a cluster majority, otherwise, transactions
046 * cannot commit (even read-only ones) and application progress halts. This class adds partition tolerance to
047 * a {@link RaftKVDatabase}, by maintaining a separate private "standalone mode" {@link KVDatabase} that can be used
048 * in lieu of the normal {@link RaftKVDatabase} when the Raft cluster is unavailable.
049 *
050 * <p>
051 * Instances transparently and automatically switch over to standalone mode {@link KVDatabase} when they determine
052 * that the {@link RaftKVDatabase} is unavailable.
053 *
054 * <p>
055 * Of course, this sacrifices consistency. To address that, a configurable {@link MergeStrategy} is used to migrate the data
056 * When switching between normal mode and standalone mode. The {@link MergeStrategy} is given read-only access to the
057 * database being switched away from, and read-write access to the database being switched to; when switching away from
058 * the {@link RaftKVDatabase}, {@link Consistency#EVENTUAL_COMMITTED} is used to eliminate the requirement for
059 * communication with the rest of the cluster.
060 *
061 * <p>
062 * Although more exotic, instances support migrating between multiple {@link RaftKVDatabase}s in a prioritized list.
063 * For example, the local node may be part of two independent {@link RaftKVDatabase} clusters: a higher priority one
064 * containing every node, and a lower priority one containing only nodes in the same data center as the local node.
065 * In any case, the "standalone mode" database (which is not a clustered database) always has the lowest priority.
066 *
067 * <p>
068 * Raft cluster availability is determined by {@link FallbackTarget#checkAvailability}; subclasses may override the default
069 * implementation if desired.
070 */
071public class FallbackKVDatabase implements KVDatabase {

227    @Override
228    public synchronized void start() {
229
230        // Already started?
231        if (this.started)
232            return;
233        this.startCount++;
234
235        // Sanity check
236        Preconditions.checkState(this.stateFile != null, "no state file configured");
237        Preconditions.checkState(this.targets != null, "no targets configured");
238        try {
239
240            // Logging
241            if (this.log.isDebugEnabled())
242                this.log.info("starting up " + this);
243
244            // Create executor
245            this.executor = Executors.newScheduledThreadPool(this.targets.size(), new ExecutorThreadFactory());
246
247            // Start underlying databases
248            this.standaloneKV.start();
249            for (FallbackTarget target : this.targets)
250                target.getRaftKVDatabase().start();
251
252            // Start periodic availability checks
253            for (FallbackTarget target : this.targets) {
254                target.future = this.executor.scheduleWithFixedDelay(
255                  new AvailabilityCheckTask(target), 0, target.getCheckInterval(), TimeUnit.MILLISECONDS);
256            }
257
258            // Initialize target runtime state
259            for (FallbackTarget target : this.targets) {
260                target.available = true;
261                target.lastChangeTimestamp = null;
262            }
263            this.currentTargetIndex = this.targets.size() - 1;
264            this.migrationCount = 0;
265
266            // Read state file, if present
267            if (this.stateFile.exists()) {
268                try {
269                    this.readStateFile();
270                } catch (IOException e) {
271                    throw new RuntimeException("error reading persistent state file " + this.stateFile, e);
272                }
273            }
274
275            // Set up periodic migration checks
276            this.migrationCheckFuture = this.executor.scheduleWithFixedDelay(
277              new MigrationCheckTask(), MIGRATION_CHECK_INTERVAL, MIGRATION_CHECK_INTERVAL, TimeUnit.MILLISECONDS);
278
279            // Done
280            this.started = true;
281        } finally {
282            if (!this.started)
283                this.cleanup();
284        }
285    }
286
287    @Override
288    public synchronized void stop() {
289
290        // Already stopped?
291        if (!this.started)
292            return;
293        this.cleanup();
294    }

296    private void cleanup() {
297
298        // Sanity check
299        assert Thread.holdsLock(this);
300
301        // Logging
302        if (this.log.isDebugEnabled())
303            this.log.info("shutting down " + this);
304
305        // Wait for migration to complete
306        if (this.migrating) {
307            if (this.log.isDebugEnabled())
308                this.log.info("waiting for migration to finish to shut down " + this);
309            do {
310                try {
311                    this.wait();
312                } catch (InterruptedException e) {
313                    this.log.warn("interrupted during " + this + " shutdown while waiting for migration to finish (ignoring)", e);
314                }
315                if (!this.started)              // we lost a race with another thread invoking stop()
316                    return;
317            } while (this.migrating);
318            if (this.log.isDebugEnabled())
319                this.log.info("migration finished, continuing with shut down of " + this);
320        }
321
322        // Reset target runtime state
323        for (FallbackTarget target : this.targets) {
324            target.available = false;
325            target.lastChangeTimestamp = null;
326        }
327
328        // Stop periodic checks
329        for (FallbackTarget target : this.targets) {
330            if (target.future != null) {
331                target.future.cancel(true);
332                target.future = null;
333            }
334        }
335        if (this.migrationCheckFuture != null) {
336            this.migrationCheckFuture.cancel(true);
337            this.migrationCheckFuture = null;
338        }
339
340        // Shut down thread pool
341        if (this.executor != null) {
342            this.executor.shutdownNow();
343            this.executor = null;
344        }
345
346        // Stop databases
347        for (FallbackTarget target : this.targets) {
348            try {
349                target.getRaftKVDatabase().stop();
350            } catch (Exception e) {
351                this.log.warn("error stopping database target " + target + " (ignoring)", e);
352            }
353        }
354        try {
355            this.standaloneKV.stop();
356        } catch (Exception e) {
357            this.log.warn("error stopping fallback database " + this.standaloneKV + " (ignoring)", e);
358        }
359
360        // Done
361        this.started = false;
362    }

369    @Override
370    public synchronized FallbackKVTransaction createTransaction() {
371
372        // Sanity check
373        Preconditions.checkState(this.started, "not started");
374
375        // Create inner transaction from current database
376        final KVDatabase currentKV = this.currentTargetIndex == -1 ?
377          this.standaloneKV : this.targets.get(this.currentTargetIndex).getRaftKVDatabase();
378        KVTransaction tx = currentKV.createTransaction();
379
380        // Wrap it
381        return new FallbackKVTransaction(this, tx, this.migrationCount);
382    }

405    // Perform availability check on the specified target
406    private void performCheck(FallbackTarget target, final int startCount) {
407
408        // Check for shutdown race condition
409        synchronized (this) {
410            if (!this.started || startCount != this.startCount)
411                return;
412        }
413
414        // Logging
415        if (this.log.isTraceEnabled())
416            this.log.trace("performing availability check for " + target);
417
418        // Perform check
419        boolean available = false;
420        try {
421            available = target.checkAvailability();
422        } catch (Exception e) {
423            if (this.log.isDebugEnabled())
424                this.log.debug("checkAvailable() for " + target + " threw exception", e);
425        }
426
427        // Handle result
428        synchronized (this) {
429
430            // Check for shutdown race condition
431            if (!this.started || startCount != this.startCount)
432                return;
433
434            // Prevent timestamp roll-over
435            if (target.lastChangeTimestamp != null && target.lastChangeTimestamp.isRolloverDanger())
436                target.lastChangeTimestamp = null;
437
438            // Any state change?
439            if (available == target.available)
440                return;
441
442            // Update availability and schedule an immediate migration check
443            this.log.info(target + " has become " + (available ? "" : "un") + "available");
444            target.available = available;
445            target.lastChangeTimestamp = new Timestamp();
446            this.executor.submit(new MigrationCheckTask());
447        }
448    }

450    // Perform migration if necessary
451    private void checkMigration(final int startCount) {
452        final int currIndex;
453        int bestIndex;
454        final FallbackTarget currTarget;
455        final FallbackTarget bestTarget;
456        synchronized (this) {
457
458            // Check for shutdown race condition
459            if (!this.started || startCount != this.startCount)
460                return;
461
462            // Logging
463            if (this.log.isTraceEnabled())
464                this.log.trace("performing migration check");
465
466            // Allow only one migration at a time
467            if (this.migrating)
468                return;
469
470            // Get the highest priority (i.e., best choice) database that is currently available
471            bestIndex = this.targets.size() - 1;
472            while (bestIndex >= 0) {
473                final FallbackTarget target = this.targets.get(bestIndex);
474
475                // Enforce hysteresis: don't change state unless sufficient time has past since target's last state change
476                final boolean previousAvailable = bestIndex >= this.currentTargetIndex;
477                final boolean currentAvailable = target.available;
478                final int timeSinceChange = target.lastChangeTimestamp != null ?
479                  -target.lastChangeTimestamp.offsetFromNow() : Integer.MAX_VALUE;
480                final boolean hysteresisAvailable;
481                if (currentAvailable)
482                    hysteresisAvailable = previousAvailable || timeSinceChange >= target.getMinAvailableTime();
483                else
484                    hysteresisAvailable = previousAvailable && timeSinceChange < target.getMinUnavailableTime();
485                if (this.log.isTraceEnabled()) {
486                    this.log.trace(target + " availability: previous=" + previousAvailable + ", current=" + currentAvailable
487                      + ", hysteresis=" + hysteresisAvailable);
488                }
489
490                // If this target is available, use it
491                if (hysteresisAvailable)
492                    break;
493
494                // Try next best one
495                bestIndex--;
496            }
497
498            // Already there?
499            currIndex = this.currentTargetIndex;
500            if (currIndex == bestIndex)
501                return;
502
503            // Get targets
504            currTarget = currIndex != -1 ? this.targets.get(currIndex) : null;
505            bestTarget = bestIndex != -1 ? this.targets.get(bestIndex) : null;
506
507            // Start migration
508            this.migrating = true;
509        }
510        try {
511            final String desc = "migration from "
512              + (currIndex != -1 ? "fallback target #" + currIndex : "standalone database")
513              + " to "
514              + (bestIndex != -1 ? "fallback target #" + bestIndex : "standalone database");
515            try {
516
517                // Gather info
518                final KVDatabase currKV = currTarget != null ? currTarget.getRaftKVDatabase() : this.standaloneKV;
519                final KVDatabase bestKV = bestTarget != null ? bestTarget.getRaftKVDatabase() : this.standaloneKV;
520                final Date lastActiveTime = bestTarget != null ? bestTarget.lastActiveTime : this.lastStandaloneActiveTime;
521                final MergeStrategy mergeStrategy = bestIndex < currIndex ?
522                  currTarget.getUnavailableMergeStrategy() : bestTarget.getRejoinMergeStrategy();
523
524                // Logit
525                this.log.info("starting fallback " + desc + " using " + mergeStrategy);
526
527                // Create source transaction. Note the combination of read-only and EVENTUAL_COMMITTED is important, because this
528                // guarantees that the transaction will generate no network traffic (and not require any majority) on commit().
529                final KVTransaction src;
530                if (currKV instanceof RaftKVDatabase) {
531                    src = ((RaftKVDatabase)currKV).createTransaction(Consistency.EVENTUAL_COMMITTED);
532                    ((RaftKVTransaction)src).setReadOnly(true);
533                } else
534                    src = currKV.createTransaction();
535                boolean srcCommitted = false;
536                try {
537
538                    // Create destination transaction
539                    final KVTransaction dst = bestKV.createTransaction();
540                    boolean dstCommitted = false;
541                    try {
542
543                        // Get timestamp
544                        final Date currentTime = new Date();
545
546                        // Perform merge
547                        mergeStrategy.merge(src, dst, lastActiveTime);
548
549                        // Commit transactions
550                        src.commit();
551                        srcCommitted = true;
552                        dst.commit();
553                        dstCommitted = true;
554
555                        // Redirect new transactions
556                        this.log.info(desc + " succeeded");
557                        synchronized (this) {
558                            if (currTarget != null)
559                                currTarget.lastActiveTime = currentTime;
560                            else
561                                this.lastStandaloneActiveTime = currentTime;
562                            this.currentTargetIndex = bestIndex;
563                            this.migrationCount++;
564                        }
565
566                    } finally {
567                        if (!dstCommitted)
568                            dst.rollback();
569                    }
570                } finally {
571                    if (!srcCommitted)
572                        src.rollback();
573                }
574            } catch (RetryTransactionException e) {
575                this.log.info(desc + " failed (will try again later): " + e);
576            } catch (Throwable t) {
577                this.log.error(desc + " failed", t);
578            }
579        } finally {
580            synchronized (this) {
581                this.migrating = false;
582                this.notifyAll();
583            }
584        }
585
586        // Update state file
587        try {
588            this.writeStateFile();
589        } catch (IOException e) {
590            this.log.error("error writing state to state file " + this.stateFile, e);
591        }
592    }

594    private void readStateFile() throws IOException {
595
596        // Sanity check
597        assert Thread.holdsLock(this);
598
599        // Read data
600        final int targetIndex;
601        final long standaloneActiveTime;
602        final long[] lastActiveTimes;
603        try (DataInputStream input = new DataInputStream(new BufferedInputStream(new FileInputStream(this.stateFile)))) {
604            final int cookie = input.readInt();
605            if (cookie != STATE_FILE_COOKIE)
606                throw new IOException("invalid state file " + this.stateFile + " (incorrect header)");
607            final int formatVersion = input.readInt();
608            switch (formatVersion) {
609            case CURRENT_FORMAT_VERSION:
610                break;
611            default:
612                throw new IOException("invalid state file " + this.stateFile + " format version (expecting "
613                  + CURRENT_FORMAT_VERSION + ", found " + formatVersion + ")");
614            }
615            final int numTargets = input.readInt();
616            if (numTargets != this.targets.size()) {
617                this.log.warn("state file " + this.stateFile + " lists " + numTargets + " != " + this.targets.size()
618                  + ", assuming configuration change and ignoring file");
619                return;
620            }
621            targetIndex = input.readInt();
622            if (targetIndex < -1 || targetIndex >= this.targets.size())
623                throw new IOException("invalid state file " + this.stateFile + " target index " + targetIndex);
624            standaloneActiveTime = input.readLong();
625            lastActiveTimes = new long[numTargets];
626            for (int i = 0; i < numTargets; i++)
627                lastActiveTimes[i] = input.readLong();
628        }
629
630        // Apply data
631        this.currentTargetIndex = targetIndex;
632        for (int i = 0; i < this.targets.size(); i++) {
633            final FallbackTarget target = this.targets.get(i);
634            target.lastActiveTime = lastActiveTimes[i] != 0 ? new Date(lastActiveTimes[i]) : null;
635        }
636        this.lastStandaloneActiveTime = standaloneActiveTime != 0 ? new Date(standaloneActiveTime) : null;
637    }
638
639    private void writeStateFile() throws IOException {
640
641        // Sanity check
642        assert Thread.holdsLock(this);
643
644        // Write data
645        try (DataOutputStream output = new DataOutputStream(
646          new BufferedOutputStream(new AtomicUpdateFileOutputStream(this.stateFile)))) {
647            output.writeInt(STATE_FILE_COOKIE);
648            output.writeInt(CURRENT_FORMAT_VERSION);
649            output.writeInt(this.targets.size());
650            output.writeInt(this.currentTargetIndex);
651            output.writeLong(this.lastStandaloneActiveTime != null ? this.lastStandaloneActiveTime.getTime() : 0);
652            for (int i = 0; i < this.targets.size(); i++) {
653                final FallbackTarget target = this.targets.get(i);
654                output.writeLong(target.lastActiveTime != null ? target.lastActiveTime.getTime() : 0);
655            }
656        }
657    }

673// AvailabilityCheckTask
674
675    private class AvailabilityCheckTask implements Runnable {
676
677        private final FallbackTarget target;
678        private final int startCount;
679
680        AvailabilityCheckTask(FallbackTarget target) {
681            assert Thread.holdsLock(FallbackKVDatabase.this);
682            this.target = target;
683            this.startCount = FallbackKVDatabase.this.startCount;
684        }
685
686        @Override
687        public void run() {
688            try {
689                FallbackKVDatabase.this.performCheck(this.target, this.startCount);
690            } catch (Throwable t) {
691                FallbackKVDatabase.this.log.error("exception from " + this.target + " availability check", t);
692            }
693        }
694    }

698    private class MigrationCheckTask implements Runnable {
699
700        private final int startCount;
701
702        MigrationCheckTask() {
703            assert Thread.holdsLock(FallbackKVDatabase.this);
704            this.startCount = FallbackKVDatabase.this.startCount;
705        }
706
707        @Override
708        public void run() {
709            try {
710                FallbackKVDatabase.this.checkMigration(this.startCount);
711            } catch (Throwable t) {
712                FallbackKVDatabase.this.log.error("exception from migration check", t);
713            }
714        }
