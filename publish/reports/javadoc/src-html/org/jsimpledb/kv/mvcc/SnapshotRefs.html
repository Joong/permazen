006package org.jsimpledb.kv.mvcc;
007
008import com.google.common.base.Preconditions;
009
010import java.io.Closeable;
011
012import org.jsimpledb.kv.CloseableKVStore;
013import org.jsimpledb.kv.KVStore;
014import org.slf4j.LoggerFactory;
015
016/**
017 * Holds a count of references to a {@link CloseableKVStore} and actually {@link CloseableKVStore#close close()}'s it
018 * when the reference count reaches zero.
019 *
020 * <p>
021 * Instances are thread safe.
022 */
023public class SnapshotRefs {
024
025    private final CloseableKVStore snapshot;
026    private int refs;
027
028    /**
029     * Constructor.
030     *
031     * <p>
032     * Initially the reference count is set to one.
033     *
034     * @param snapshot database snapshot
035     */
036    public SnapshotRefs(CloseableKVStore snapshot) {
037        Preconditions.checkArgument(snapshot != null, "null snapshot");
038        this.snapshot = snapshot;
039        this.refs = 1;
040        synchronized (this) { }
041    }
042
043    public synchronized KVStore getKVStore() {
044        Preconditions.checkState(this.refs > 0, "no longer referenced");
045        return this.snapshot;
046    }
047
048    public synchronized void ref() {
049        Preconditions.checkState(this.refs > 0, "no longer referenced");
050        this.refs++;
051    }
052
053    public synchronized void unref() {
054        Preconditions.checkState(this.refs > 0, "no longer referenced");
055        if (--this.refs == 0)
056            this.snapshot.close();
057    }
058
059    /**
060     * Create an empty {@link Closeable} that, when {@link Closeable#close close()}'d, decrements the reference count.
061     *
062     * @return unreferencing {@link Closeable}
063     */
064    public Closeable getUnrefCloseable() {
065        return new Closeable() {
066            private boolean closed;
067            @Override
068            public synchronized void close() {
069                if (this.closed)
070                    return;
071                SnapshotRefs.this.unref();
072                this.closed = true;
073            }
074        };
075    }
076
077    @Override
078    protected void finalize() throws Throwable {
079        try {
080            if (this.refs > 0) {
081                LoggerFactory.getLogger(this.getClass()).warn(this + " leaked with " + this.refs + " remaining references");
082                this.refs = 0;
083                this.snapshot.close();
084            }
085        } finally {
086            super.finalize();
087        }
088    }
089}
