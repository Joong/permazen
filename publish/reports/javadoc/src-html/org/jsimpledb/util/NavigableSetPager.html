006package org.jsimpledb.util;
007
008import com.google.common.base.Preconditions;
009import com.google.common.collect.Lists;
010
011import java.util.ArrayList;
012import java.util.List;
013import java.util.NavigableSet;
014
015/**
016 * Allows efficient paging through a {@link NavigableSet}.
017 *
018 * <p>
019 * Instances are configured with a page size and a view order (ascending or descending).
020 *
021 * <p>
022 * Instances maintain a cursor position that anchors one "page" of consecutive items within the set. Each time
023 * {@link #readCurrentPage} is invoked, the set is reacquired from {@link #getNavigableSet} and the
024 * contents of the current page are read from it.
025 *
026 * <p>
027 * For navigation, {@link #nextPage} and {@link #prevPage} move to the next or previous page,
028 * {@link #firstPage} and {@link #lastPage} jump to the first or last page.
029 * After movement, {@link #hasNextPage} and {@link #hasPrevPage} indicate whether adjacent pages exist.
030 *
031 * @param <E> element type
032 */
033public abstract class NavigableSetPager<E> {
034
035    /**
036     * Default page size ({@value #DEFAULT_PAGE_SIZE}).
037     *
038     * @see #setPageSize setPageSize()
039     */
040    public static final int DEFAULT_PAGE_SIZE = 100;
041
042    private int pageSize = DEFAULT_PAGE_SIZE;
043    private boolean descending;                                 // query ordering
044    private Bound anchor;                                       // current page min (forward) or max (backward)
045    private Bound limit;                                        // marks the end of the current page, if known
046    private boolean backwards;                                  // most recent move was backward (i.e., prev() instead of next())
047    private boolean dataBefore;                                 // there is more data behind current page
048    private boolean dataAfter;                                  // there is more data ahead of current page
049
050// Configuration
051
052    /**
053     * Set the page size.
054     *
055     * @param pageSize maximum number of rows in a page
056     * @throws IllegalArgumentException if {@code pageSize} is zero or less
057     */
058    public void setPageSize(int pageSize) {
059        Preconditions.checkArgument(pageSize > 0, "pageSize <= 0");
060        this.pageSize = pageSize;
061    }
062
063    /**
064     * Get view ordering. Default is ascending.
065     *
066     * @return false if container order is ascending, true if descending
067     */
068    public boolean isDescending() {
069        return this.descending;
070    }
071
072    /**
073     * Set view ordering.
074     *
075     * @param descending true for descending view, false for ascending view
076     */
077    public void setDescending(boolean descending) {
078        if (this.descending != descending) {
079            this.descending = descending;
080            this.limit = null;
081        }
082    }
083
084// Paging
085
086    /**
087     * Indicates that there are more results after the current page.
088     *
089     * @return true if there are more results in the forward direction, false if we are on the last page
090     */
091    public boolean hasNextPage() {
092        return this.backwards ? this.dataBefore : this.dataAfter;
093    }
094
095    /**
096     * Indicates that there are more results before the current page.
097     *
098     * @return true if there are more results in the reverse direction, false if we are on the first page
099     */
100    public boolean hasPrevPage() {
101        return this.backwards ? this.dataAfter : this.dataBefore;
102    }
103
104    /**
105     * Advance forward to the next page of results, if any.
106     *
107     * <p>
108     * This will advance to the next higher page of results if configured for an ascending view,
109     * or the next lower page of results if configured for a descending view.
110     *
111     * <p>
112     * After invoking this method, {@link #readCurrentPage} must be invoked for it to take effect.
113     * Duplicate invocations of this method without an intervening call to {@link #readCurrentPage}
114     * will have no effect and return false.
115     *
116     * @return true if successful, false if we were already on the last page
117     */
118    public boolean nextPage() {
119        return this.step(false);
120    }
121
122    /**
123     * Advance backward to the previous page of results, if any.
124     *
125     * <p>
126     * This will regress to be the next lower page of results if configured for an ascending view,
127     * or the next higher page of results if configured for a descending view.
128     *
129     * <p>
130     * After invoking this method, {@link #readCurrentPage} must be invoked for it to take effect.
131     * Duplicate invocations of this method without an intervening call to {@link #readCurrentPage}
132     * will have no effect and return false.
133     *
134     * @return true if successful, false if we were already on the first page
135     */
136    public boolean prevPage() {
137        return this.step(true);
138    }
139
140    private boolean step(boolean backwards) {
141        if (!(backwards ^ this.backwards)) {
142            if (this.limit == null)
143                return false;
144            this.anchor = this.limit;
145        } else {
146            if (this.anchor == null)
147                return false;
148            this.backwards = backwards;
149        }
150        this.limit = null;
151        return true;
152    }
153
154    /**
155     * Jump to the first page.
156     */
157    public void firstPage() {
158        this.jump(false);
159    }
160
161    /**
162     * Jump to the last page.
163     */
164    public void lastPage() {
165        this.jump(true);
166    }
167
168    private void jump(boolean last) {
169        this.anchor = null;
170        this.limit = null;
171        this.backwards = last;
172        this.dataBefore = last;
173        this.dataAfter = !last;
174    }
175
176    /**
177     * Set the current cursor position, i.e., page anchor.
178     *
179     * <p>
180     * The next page returned by {@link #readCurrentPage} will start at {@code cursor},
181     * inclusive for an ascending ordering or exclusive for a descending ordering.
182     *
183     * <p>
184     * The {@code cursor} must not be null unless the underlying {@link NavigableSet} supports null values.
185     *
186     * @param cursor new cursor position for the current page
187     */
188    public void setCursor(E cursor) {
189        this.anchor = new Bound(cursor);
190        this.limit = null;
191        this.backwards = false;
192        this.dataBefore = true;
193        this.dataAfter = true;
194    }
195
196    /**
197     * Read the contents of the current page.
198     *
199     * <p>
200     * The list is read starting from the current cursor position (inclusive for an ascending ordering,
201     * exclusive for a descending ordering).
202     *
203     * <p>
204     * If configured for an ascending view, the list will have ascending ordering;
205     * if configured for a descending view, the list will have descending ordering.
206     *
207     * @return list of items in the current page
208     */
209    public List<E> readCurrentPage() {
210
211        // Get set
212        NavigableSet<E> set = this.getNavigableSet();
213
214        // Set overall ordering
215        if (this.descending)
216            set = set.descendingSet();
217
218        // Set page anchor point, which determines the first object on the page
219        NavigableSet<E> anchorSet;
220        if (this.anchor != null) {
221            anchorSet = this.backwards ?
222              set.headSet(this.anchor.getValue(), false) :                      // anchor point is maximum
223              set.tailSet(this.anchor.getValue(), true);                        // anchor point is minimum
224        } else
225            anchorSet = set;
226
227        // Iterate away from the anchor point in the same direction we're "moving"
228        if (this.backwards)
229            anchorSet = anchorSet.descendingSet();
230
231        // Gather up to one page's worth of data and remember the limit we reach, if any, to become the next anchor point
232        this.limit = null;
233        List<E> page = new ArrayList<>(this.pageSize);
234        for (E item : anchorSet) {
235            if (this.backwards)
236                page.add(item);
237            if (page.size() >= this.pageSize) {
238                this.limit = new Bound(item);
239                break;
240            }
241            if (!this.backwards)
242                page.add(item);
243        }
244
245        // If we ran out of items, bounce off the end and load a full page going in the opposite direction
246        if (this.anchor != null && this.limit == null) {
247            this.anchor = null;
248            this.backwards = !this.backwards;
249            return this.readCurrentPage();
250        }
251
252        // Determine if there is any data before the anchor point
253        if (this.anchor != null) {
254            this.dataBefore = this.backwards ?
255              !set.tailSet(this.anchor.getValue(), true).isEmpty() :
256              !set.headSet(this.anchor.getValue(), false).isEmpty();
257        } else
258            this.dataBefore = false;
259
260        // Determine if there is any data after the page limit
261        if (this.limit != null) {
262            this.dataAfter = this.backwards ?
263              !set.headSet(this.limit.getValue(), false).isEmpty() :
264              true;                             // we already know there's one because we read it and it became this.limit
265        } else
266            this.dataAfter = false;
267
268        // Un-do the effect of previous reversal
269        if (this.backwards)
270            page = Lists.reverse(page);
271
272        // Done
273        return page;
274    }
275
276// Subclass methods
277
278    /**
279     * Get the {@link NavigableSet} through which to page.
280     *
281     * <p>
282     * This method is invoked anew each time {@link #readCurrentPage} is invoked.
283     *
284     * @return the entire query domain in ascending order
285     */
286    protected abstract NavigableSet<E> getNavigableSet();
287
288// Bound
289
290    private class Bound {
291
292        private final E value;
293
294        Bound(E value) {
295            this.value = value;
296        }
297
298        public E getValue() {
299            return this.value;
300        }
301    }
302}
