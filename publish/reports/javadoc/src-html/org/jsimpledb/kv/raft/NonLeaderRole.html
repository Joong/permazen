006package org.jsimpledb.kv.raft;
007
008import com.google.common.base.Preconditions;
009
010import org.jsimpledb.kv.raft.msg.AppendResponse;
011import org.jsimpledb.kv.raft.msg.CommitRequest;
012
013/**
014 * Support superclass for the {@linkplain FollowerRole follower} and {@linkplain CandidateRole candidate} roles,
015 * both of which have an election timer.
016 */
017public abstract class NonLeaderRole extends Role {
018
019    final Timer electionTimer = new Timer(this.raft, "election timer", new Service(this, "election timeout") {
020        @Override
021        public void run() {
022            NonLeaderRole.this.checkElectionTimeout();
023        }
024    });
025    private final boolean startElectionTimer;
026
027// Constructors
028
029    NonLeaderRole(RaftKVDatabase raft, boolean startElectionTimer) {
030        super(raft);
031        this.startElectionTimer = startElectionTimer;
032    }
033
034// Status & Debugging
035
036    /**
037     * Get the election timer deadline, if currently running.
038     *
039     * <p>
040     * For a follower that is not a member of its cluster, this will return null because no election timer is running.
041     * For all other cases, this will return the time at which the election timer expires.
042     *
043     * @return current election timer expiration deadline, or null if not running
044     */
045    public Timestamp getElectionTimeout() {
046        synchronized (this.raft) {
047            return this.electionTimer.getDeadline();
048        }
049    }
050
051    /**
052     * Force an immediate election timeout.
053     *
054     * @throws IllegalStateException if this role is no longer active or election timer is not running
055     */
056    public void startElection() {
057        synchronized (this.raft) {
058            Preconditions.checkState(this.raft.role == this, "role is no longer active");
059            Preconditions.checkState(this.electionTimer.isRunning(), "election timer is not running");
060            this.debug("triggering immediate election timeout due to invocation of startElection()");
061            this.electionTimer.timeoutNow();
062        }
063    }
064
065// Lifecycle
066
067    @Override
068    void setup() {
069        super.setup();
070        if (this.startElectionTimer)
071            this.restartElectionTimer();
072    }
073
074    @Override
075    void shutdown() {
076        super.shutdown();
077        this.electionTimer.cancel();
078    }
079
080// Service
081
082    // Check for an election timeout
083    private void checkElectionTimeout() {
084        if (this.electionTimer.pollForTimeout()) {
085            if (this.log.isDebugEnabled())
086                this.debug("election timeout while in " + this);
087            this.handleElectionTimeout();
088        }
089    }
090
091    void restartElectionTimer() {
092
093        // Sanity check
094        assert Thread.holdsLock(this.raft);
095
096        // Generate a randomized election timeout delay
097        final int range = this.raft.maxElectionTimeout - this.raft.minElectionTimeout;
098        final int randomizedPart = Math.round(this.raft.random.nextFloat() * range);
099
100        // Restart timer
101        this.electionTimer.timeoutAfter(this.raft.minElectionTimeout + randomizedPart);
102    }
103
104    abstract void handleElectionTimeout();
105
106// MessageSwitch
107
108    @Override
109    void caseAppendResponse(AppendResponse msg) {
110        this.failUnexpectedMessage(msg);
111    }
112
113    @Override
114    void caseCommitRequest(CommitRequest msg) {
115        this.failUnexpectedMessage(msg);
116    }
117}
