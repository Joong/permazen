006package org.jsimpledb.kv;
007
008import java.util.concurrent.Future;
009
010/**
011 * {@link KVDatabase} transaction API.
012 *
013 * <p>
014 * Provides a transactional view of a {@link KVStore}.
015 *
016 * <p>
017 * Instances may throw {@link KVTransactionException} during any operation if the transaction cannot be continued.
018 * In particular, {@link StaleTransactionException} is thrown by a transaction that is no longer open, and
019 * {@link RetryTransactionException} is thrown when the transaction should be retried due to a transient
020 * problem (such as a write conflict with another transaction).
021 *
022 * <p>
023 * When {@link RetryTransactionException} is thrown by {@link #commit}, the transaction may have actually been committed.
024 * Therefore, transactions should be written to be idempotent.
025 *
026 * <p>
027 * No matter what state it is in, instances must support invoking {@link #rollback} at any time.
028 *
029 * <p>
030 * If an instance throws a {@link KVTransactionException}, the transaction should be implicitly rolled back.
031 * Any subsequent operation other than {@link #rollback} should throw {@link StaleTransactionException}.
032 *
033 * <p>
034 * Implementations must throw {@link StaleTransactionException} if {@link #commit} or {@link #rollback} has already
035 * been invoked, or if the {@link KVTransaction} instance is no longer usable for some other reason. In particular,
036 * implementations should throw {@link TransactionTimeoutException} if an operation is attempted on a transaction
037 * that has been held open past some maximum allowed time limit.
038 *
039 * <p>
040 * Implementations are responsible for ensuring modifications to {@code byte[]} arrays after method
041 * invocations do no harm. This usually means {@code byte[]} array parameters and return values must be copied.
042 *
043 * <p>
044 * Implementations are not required to support accessing keys that start with {@code 0xff},
045 * and if not may throw {@link IllegalArgumentException} if such keys are accessed.
046 */
047public interface KVTransaction extends KVStore {
048
049    /**
050     * Get the {@link KVDatabase} with which this instance is associated.
051     *
052     * @return associated database
053     */
054    KVDatabase getKVDatabase();
055
056    /**
057     * Change the timeout for this transaction from its default value (optional operation).
058     *
059     * @param timeout transaction timeout in milliseconds, or zero for unlimited
060     * @throws UnsupportedOperationException if this transaction does not support timeouts
061     * @throws IllegalArgumentException if {@code timeout} is negative
062     * @throws StaleTransactionException if this transaction is no longer usable
063     */
064    void setTimeout(long timeout);
065
066    /**
067     * Watch a key to monitor for changes in its value.
068     *
069     * <p>
070     * When this method is invoked, {@code key}'s current value (if any) as read by this transaction is remembered. The returned
071     * {@link Future} completes if and when a different value for {@code key} is subsequently committed by some transaction,
072     * including possibly this one.
073     *
074     * <p>
075     * Key watches outlive the transaction in which they are created, persisting until they complete or are
076     * {@link Future#cancel cancel()}'ed. When a {@link KVDatabase} is {@link KVDatabase#stop}'ed, all outstanding
077     * key watches are implicitly {@link Future#cancel cancel()}'ed.
078     *
079     * <p><b>Caveats</b></p>
080     *
081     * <p>
082     * Key watches are not without overhead; applications should avoid overuse. For example, consider creating a
083     * single key that is used to consolidate modifications to a set of keys.
084     *
085     * <p>
086     * Conceptually, detection of changes behaves as if by a background thread that periodically creates a new transaction
087     * and reads the key's value (the actual implementation will likely be more efficient). This means a change that is
088     * quickly reverted could be missed, and that multiple changes could occur before notification. In addition, spurious
089     * notifications may occur, where the key's value has not changed.
090     *
091     * <p>
092     * A key watch is only guaranteed to be valid if the transaction in which it was created successfully commits.
093     *
094     * <p>
095     * Key watch support is optional; instances that don't support key watches throw {@link UnsupportedOperationException}.
096     *
097     * @param key the key to watch
098     * @return a {@link Future} that returns {@code key} when the value associated with {@code key} is modified
099     * @throws StaleTransactionException if this transaction is no longer usable
100     * @throws RetryTransactionException if this transaction must be retried and is no longer usable
101     * @throws KVDatabaseException if an unexpected error occurs
102     * @throws UnsupportedOperationException if this instance does not support key watches
103     * @throws IllegalArgumentException if {@code key} starts with {@code 0xff} and such keys are not supported
104     * @throws IllegalArgumentException if {@code key} is null
105     * @see org.jsimpledb.JTransaction#getKey(org.jsimpledb.JObject, String) JTransaction.getKey()
106     */
107    Future<Void> watchKey(byte[] key);
108
109    /**
110     * Commit this transaction.
111     *
112     * <p>
113     * Note that if this method throws a {@link RetryTransactionException},
114     * the transaction was either successfully committed or rolled back. In either case,
115     * this instance is no longer usable.
116     * </p>
117     *
118     * @throws StaleTransactionException if this transaction is no longer usable
119     * @throws RetryTransactionException if this transaction must be retried and is no longer usable
120     */
121    void commit();
122
123    /**
124     * Cancel this transaction.
125     *
126     * <p>
127     * After this method returns, this instance is no longer usable.
128     *
129     * <p>
130     * This method may be invoked at any time, even after a previous invocation of
131     * {@link #commit} or {@link #rollback}, in which case the invocation will be ignored.
132     */
133    void rollback();
134
135    /**
136     * Create a mutable copy of the database content represented by this transaction.
137     *
138     * <p>
139     * The returned {@link KVStore} should be mutable, and be completely independent of this transaction
140     * (subsequent changes to this transaction do not affect it, and vice-versa).
141     *
142     * <p>
143     * Note that as with any other information extracted from a {@link KVTransaction}, the returned content
144     * should not be considered valid until this transaction has been successfully committed.
145     *
146     * <p>
147     * The returned {@link KVStore} should be promply {@link CloseableKVStore#close close()}'d when no longer
148     * needed to release any underlying resources. In particular, the caller must ensure that the {@link KVStore}
149     * is {@link CloseableKVStore#close close()}'d even if this transaction's commit fails. This may require
150     * adding a transaction synchronization callback, etc.
151     *
152     * <p>
153     * This is an optional method; only some underlying key/value store technologies can efficiently support it.
154     * Implementations may throw {@link UnsupportedOperationException} if not supported.
155     *
156     * @return independent, mutable copy of this transaction's entire database content
157     * @throws UnsupportedOperationException if this method is not supported
158     * @throws StaleTransactionException if this transaction is no longer usable
159     * @throws RetryTransactionException if this transaction must be retried and is no longer usable
160     */
161    CloseableKVStore mutableSnapshot();
162}
