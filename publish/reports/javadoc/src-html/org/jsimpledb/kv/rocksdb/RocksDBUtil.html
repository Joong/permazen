006package org.jsimpledb.kv.rocksdb;
007
008import com.google.common.base.Preconditions;
009
010import java.lang.reflect.Method;
011
012import org.rocksdb.RocksObject;
013
014/**
015 * Utility methods for use with RocksDB.
016 */
017public final class RocksDBUtil {
018
019    private static Method rocksObjectInitializedMethod;
020
021    private RocksDBUtil() {
022    }
023
024    /**
025     * Determine whether the given {@link RocksObject} is still valid, i.e., has not been deposed.
026     *
027     * @param obj object to check
028     * @return true if {@code obj} is still valid
029     * @throws IllegalArgumentException if {@code obj} is null
030     */
031    public static boolean isValid(RocksObject obj) {
032        Preconditions.checkArgument(obj != null, "null obj");
033        try {
034            if (RocksDBUtil.rocksObjectInitializedMethod == null) {
035                RocksDBUtil.rocksObjectInitializedMethod = RocksObject.class.getDeclaredMethod("isInitialized");
036                RocksDBUtil.rocksObjectInitializedMethod.setAccessible(true);
037            }
038            return (Boolean)RocksDBUtil.rocksObjectInitializedMethod.invoke(obj);
039        } catch (Exception e) {
040            throw new RuntimeException("internal error", e);
041        }
042    }
043}
