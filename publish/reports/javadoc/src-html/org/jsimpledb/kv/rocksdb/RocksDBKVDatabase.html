006package org.jsimpledb.kv.rocksdb;
007
008import org.jsimpledb.kv.mvcc.SnapshotKVDatabase;
009import org.jsimpledb.kv.mvcc.SnapshotVersion;
010
011/**
012 * {@link org.jsimpledb.kv.KVDatabase} implementation based on a {@link RocksDBAtomicKVStore}, providing concurrent transactions
013 * and linearizable ACID semantics.
014 */
015public class RocksDBKVDatabase extends SnapshotKVDatabase {
016
017// Properties
018
019    /**
020     * Configure the underlying {@link RocksDBAtomicKVStore} used by this instance. Required property.
021     *
022     * @param kvstore underlying key/value store
023     * @throws IllegalStateException if this instance is already {@link #start}ed
024     */
025    public void setKVStore(RocksDBAtomicKVStore kvstore) {
026        super.setKVStore(kvstore);
027    }
028
029// KVDatabase
030
031    @Override
032    public synchronized RocksDBKVTransaction createTransaction() {
033        return (RocksDBKVTransaction)super.createTransaction();
034    }
035
036// SnapshotKVDatabase
037
038    @Override
039    protected RocksDBKVTransaction createSnapshotKVTransaction(SnapshotVersion versionInfo) {
040        return new RocksDBKVTransaction(this, versionInfo);
041    }
042}
