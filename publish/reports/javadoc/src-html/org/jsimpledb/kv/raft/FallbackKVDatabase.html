024/**
025 * A partition-tolerant {@link KVDatabase} that automatically migrates between a clustered {@link RaftKVDatabase}
026 * and a local, non-clustered "emergency standalone mode" {@link KVDatabase}, based on availability of the Raft cluster.
027 *
028 * <p>
029 * A {@link RaftKVDatabase} requires that the local node be part of a cluster majority, otherwise, transactions
030 * cannot commit (even read-only ones) and application progress halts. This class adds partition tolerance to
031 * a {@link RaftKVDatabase}, by having a local {@link KVDatabase} that can be used when the cluster is unavailable.
032 *
033 * <p>
034 * Instances transparently and automatically switch over to the "standalone mode" {@link KVDatabase} when they determine
035 * that the {@link RaftKVDatabase} is unavailable. If/when availability is been restored, the local node rejoins the
036 * {@link RaftKVDatabase} cluster. In both cases, a configurable {@link MergeStrategy} is used to migrate the data.
037 * In the fallback operation, the most up-to-date committed version of the Raft database is migrated; reading this
038 * information does not require communication with the rest of the cluster and therefore is possible during a partition.
039 *
040 * <p>
041 * Although more exotic, instances support migrating between multiple {@link RaftKVDatabase}s in a prioritized list.
042 * For example, the local node may be part of two independent {@link RaftKVDatabase} clusters: a higher priority one
043 * containing every node, and a lower priority one containing only nodes in the same data center as the local node.
044 * In any case, the "standalone mode" database (which is not a clustered database) always has the lowest priority.
045 *
046 * <p>
047 * Raft cluster availability is determined by {@link FallbackTarget#checkAvailability}; subclasses may override the default
048 * implementation if desired.
049 */
050public class FallbackKVDatabase implements KVDatabase {
051
052    private static final int MIGRATION_CHECK_INTERVAL = 1000;                   // every 1 second
053
054    protected final Logger log = LoggerFactory.getLogger(this.getClass());
055
056    // Configured state
057    private KVDatabase standaloneKV;
058    private ArrayList<FallbackTarget> targets;
059
060    // Runtime state
061    private boolean migrating;
062    private int migrationCount;
063    private ScheduledExecutorService executor;
064    private ScheduledFuture<?> migrationCheckFuture;
065    private int startCount;
066    private boolean started;
067    private int currentTargetIndex;
068
069// Methods
070
071    /**
072     * Configure the local "standalone mode" {@link KVDatabase} to be used when all {@link FallbackTarget}s are unavailable.
073     *
074     * @param standaloneKV "standalone mode" database
075     * @throws IllegalArgumentException if {@code standaloneKV} is null
076     * @throws IllegalStateException if this instance is already started
077     */
078    public synchronized void setStandaloneTarget(KVDatabase standaloneKV) {
079        Preconditions.checkArgument(standaloneKV != null, "null standaloneKV");
080        Preconditions.checkState(!this.started, "already started");
081        this.standaloneKV = standaloneKV;
082    }
083
084    /**
085     * Configure the underlying {@link FallbackTarget}(s).
086     *
087     * <p>
088     * If multiple {@link FallbackTarget}s are provided, they should be ordered in order of increasing preference.
089     *
090     * @param targets targets in order of increasing preference
091     * @throws IllegalArgumentException if {@code targets} is null
092     * @throws IllegalArgumentException if {@code targets} is empty
093     * @throws IllegalArgumentException if any target is null
094     * @throws IllegalArgumentException if any target does not have a {@link RaftKVDatabase} configured
095     * @throws IllegalStateException if this instance is already started
096     */
097    public synchronized void setFallbackTargets(List<? extends FallbackTarget> targets) {
098        Preconditions.checkArgument(targets != null, "null targets");
099        Preconditions.checkArgument(!targets.isEmpty(), "empty targets");
100        Preconditions.checkState(!this.started, "already started");
101        this.targets = new ArrayList<>(targets.size());
102        for (FallbackTarget target : targets) {
103            Preconditions.checkArgument(target != null, "null target");
104            target = target.clone();
105            Preconditions.checkArgument(target.getRaftKVDatabase() != null, "target with no database configured");
106        }
107    }
108
109// KVDatabase
110
111    @Override
112    public synchronized void start() {
113
114        // Already started?
115        if (this.started)
116            return;
117        this.startCount++;
118
119        // Sanity check
120        Preconditions.checkState(this.targets != null, "no targets configured");
121        try {
122
123            // Create executor
124            this.executor = Executors.newScheduledThreadPool(this.targets.size(), new ExecutorThreadFactory());
125
126            // Start underlying databases
127            for (FallbackTarget target : this.targets)
128                target.getRaftKVDatabase().start();
129
130            // Start periodic checks
131            final int currentStartCount = this.startCount;
132            for (FallbackTarget target : this.targets) {
133                target.future = this.executor.scheduleWithFixedDelay(
134                  new AvailabilityCheckTask(target), 0, target.getCheckInterval(), TimeUnit.MILLISECONDS);
135            }
136
137            // Initialize target runtime state
138            for (FallbackTarget target : this.targets) {
139                target.available = true;
140                target.lastChangeTimestamp = null;
141            }
142            this.currentTargetIndex = this.targets.size() - 1;
143            this.migrationCount = 0;
144
145            // Set up periodic migration checks
146            this.migrationCheckFuture = this.executor.scheduleWithFixedDelay(
147              new MigrationCheckTask(), MIGRATION_CHECK_INTERVAL, MIGRATION_CHECK_INTERVAL, TimeUnit.MILLISECONDS);
148
149            // Done
150            this.started = true;
151        } finally {
152            if (!this.started)
153                this.cleanup();
154        }
155    }
156
157    @Override
158    public synchronized void stop() {
159
160        // Already stopped?
161        if (!this.started)
162            return;
163        this.cleanup();
164    }
165
166    private void cleanup() {
167
168        // Sanity check
169        assert Thread.holdsLock(this);
170
171        // Wait for migration to complete
172        if (this.migrating) {
173            if (this.log.isDebugEnabled())
174                this.log.info("waiting for migration to finish to shut down " + this);
175            do {
176                try {
177                    this.wait();
178                } catch (InterruptedException e) {
179                    this.log.warn("interrupted during " + this + " shutdown while waiting for migration to finish (ignoring)", e);
180                }
181                if (!this.started)              // we lost a race with another thread invoking stop()
182                    return;
183            } while (this.migrating);
184            if (this.log.isDebugEnabled())
185                this.log.info("migration finished, continuing with shut down of " + this);
186        }
187
188        // Reset target runtime state
189        for (FallbackTarget target : this.targets) {
190            target.available = false;
191            target.lastChangeTimestamp = null;
192        }
193
194        // Stop periodic checks
195        for (FallbackTarget target : this.targets) {
196            if (target.future != null) {
197                target.future.cancel(true);
198                target.future = null;
199            }
200        }
201        if (this.migrationCheckFuture != null) {
202            this.migrationCheckFuture.cancel(true);
203            this.migrationCheckFuture = null;
204        }
205
206        // Shut down thread pool
207        if (this.executor != null) {
208            this.executor.shutdownNow();
209            this.executor = null;
210        }
211
212        // Stop databases
213        for (FallbackTarget target : this.targets) {
214            try {
215                target.getRaftKVDatabase().stop();
216            } catch (Exception e) {
217                this.log.warn("error stopping database target " + target + " (ignoring)", e);
218            }
219        }
220
221        // Done
222        this.started = false;
223    }
224
225    @Override
226    public synchronized KVTransaction createTransaction() {
227
228        // Sanity check
229        Preconditions.checkState(this.started, "not started");
230
231        // Create inner transaction from current database
232        final KVDatabase currentKV = this.currentTargetIndex == -1 ?
233          this.standaloneKV : this.targets.get(this.currentTargetIndex).getRaftKVDatabase();
234        KVTransaction tx = currentKV.createTransaction();
235
236        // Wrap it
237        return new FallbackKVTransaction(this, tx, this.migrationCount);
238    }

247        return this.migrationCount;

250// Internal methods
251
252    // Perform availability check on the specified target
253    private void performCheck(FallbackTarget target, final int startCount) {
254
255        // Check for shutdown race condition
256        synchronized (this) {
257            if (!this.started || startCount != this.startCount)
258                return;
259        }
260
261        // Perform check
262        boolean available = false;
263        try {
264            available = target.checkAvailability();
265        } catch (Exception e) {
266            if (this.log.isDebugEnabled())
267                this.log.debug("checkAvailable() for " + target + " threw exception", e);
268        }
269
270        // Handle result
271        synchronized (this) {
272
273            // Check for shutdown race condition
274            if (!this.started || startCount != this.startCount)
275                return;
276
277            // Any state change?
278            if (available == target.available)
279                return;
280
281            // Enforce hysteresis
282            if (target.lastChangeTimestamp != null) {
283                final int minDelay = target.available ? target.getMinAvailableTime() : target.getMinUnavailableTime();
284                if (target.lastChangeTimestamp.offsetFromNow() + minDelay > 0) {
285                    if (this.log.isTraceEnabled()) {
286                        this.log.trace(target + " is " + (target.available ? "" : "un") + "available, but minimum delay of "
287                          + minDelay + "ms has not yet elapsed");
288                    }
289                    return;
290                }
291            }
292
293            // Change state and schedule migration
294            this.log.info(target + " has become " + (available ? "" : "un") + "available");
295            target.available = available;
296            target.lastChangeTimestamp = new Timestamp();
297            this.executor.submit(new MigrationCheckTask());
298        }
299    }
300
301    // Perform migration if necessary
302    private void checkMigration(final int startCount) {
303        final int currIndex;
304        int bestIndex;
305        final FallbackTarget currTarget;
306        final FallbackTarget bestTarget;
307        synchronized (this) {
308
309            // Check for shutdown race condition
310            if (!this.started || startCount != this.startCount)
311                return;
312
313            // Allow only one migration at a time
314            if (this.migrating)
315                return;
316
317            // Get the highest priority (i.e., best choice) database that is currently available
318            bestIndex = this.targets.size() - 1;
319            while (bestIndex >= 0 && !this.targets.get(bestIndex).available)
320                bestIndex--;
321
322            // Already there?
323            currIndex = this.currentTargetIndex;
324            if (currIndex == bestIndex)
325                return;
326
327            // Get targets
328            currTarget = currIndex != -1 ? this.targets.get(currIndex) : null;
329            bestTarget = bestIndex != -1 ? this.targets.get(bestIndex) : null;
330
331            // Start migration
332            this.migrating = true;
333        }
334        final String desc = "migration from " + currTarget + " (index " + currIndex + ") to "
335          + bestTarget + " (index " + bestIndex + ")";
336        Throwable error = null;
337        try {
338
339            // Gather info
340            final KVDatabase currKV = currTarget != null ? currTarget.getRaftKVDatabase() : this.standaloneKV;
341            final KVDatabase bestKV = bestTarget != null ? bestTarget.getRaftKVDatabase() : this.standaloneKV;
342            final MergeStrategy mergeStrategy = bestIndex < currIndex ?
343              currTarget.getUnavailableMergeStrategy() : bestTarget.getRejoinMergeStrategy();
344
345            // Logit
346            if (this.log.isDebugEnabled())
347                this.log.debug("starting " + desc + " using " + mergeStrategy);
348
349            // Create source transaction
350            final KVTransaction src;
351            if (currKV instanceof RaftKVDatabase)
352                src = ((RaftKVDatabase)currKV).createTransaction(true);    // using committed and EVENTUAL avoids network traffic
353            else
354                src = currKV.createTransaction();
355            boolean srcCommitted = false;
356            try {
357
358                // Special configuration for Raft source transactions to avoid network traffic
359                if (src instanceof RaftKVTransaction) {
360                    final RaftKVTransaction tx = (RaftKVTransaction)src;
361                    tx.setConsistency(Consistency.EVENTUAL);
362                    tx.setReadOnly(true);
363                }
364
365                // Create destination transaction
366                final KVTransaction dst = bestKV.createTransaction();
367                boolean dstCommitted = false;
368                try {
369
370                    // Perform merge
371                    mergeStrategy.merge(src, dst);
372
373                    // Commit transactions
374                    src.commit();
375                    srcCommitted = true;
376                    dst.commit();
377                    dstCommitted = true;
378                } finally {
379                    if (!dstCommitted)
380                        dst.rollback();
381                }
382            } catch (Error e) {
383                error = e;
384                throw e;
385            } catch (RuntimeException e) {
386                error = e;
387                throw e;
388            } catch (Throwable t) {
389                error = t;
390                throw new RuntimeException(t);
391            } finally {
392                if (!srcCommitted)
393                    src.rollback();
394            }
395        } finally {
396
397            // Finish up
398            synchronized (this) {
399                if (error == null) {
400                    this.currentTargetIndex = bestIndex;
401                    this.migrationCount++;
402                }
403                this.migrating = false;
404                this.notifyAll();
405            }
406
407            // Logit
408            if (error != null)
409                this.log.error(desc + " failed", error);
410            else if (this.log.isDebugEnabled())
411                this.log.debug(desc + " succeeded");
412        }
413    }

417    private class ExecutorThreadFactory implements ThreadFactory {
418
419        private final AtomicInteger id = new AtomicInteger();
420
421        @Override
422        public Thread newThread(Runnable action) {
423            final Thread thread = new Thread(action);
424            thread.setName("Executor#" + this.id.incrementAndGet() + " for " + FallbackKVDatabase.this);
425            return thread;
426        }
427    }

429// AvailabilityCheckTask
430
431    private class AvailabilityCheckTask implements Runnable {
432
433        private final FallbackTarget target;
434        private final int startCount;
435
436        AvailabilityCheckTask(FallbackTarget target) {
437            assert Thread.holdsLock(FallbackKVDatabase.this);
438            this.target = target;
439            this.startCount = FallbackKVDatabase.this.startCount;
440        }
441
442        @Override
443        public void run() {
444            try {
445                FallbackKVDatabase.this.performCheck(this.target, this.startCount);
446            } catch (Throwable t) {
447                FallbackKVDatabase.this.log.error("exception from " + this.target + " availability check", t);
448            }
449        }
450    }

452// MigrationCheckTask
453
454    private class MigrationCheckTask implements Runnable {
455
456        private final int startCount;
457
458        MigrationCheckTask() {
459            assert Thread.holdsLock(FallbackKVDatabase.this);
460            this.startCount = FallbackKVDatabase.this.startCount;
461        }
462
463        @Override
464        public void run() {
465            try {
466                FallbackKVDatabase.this.checkMigration(this.startCount);
467            } catch (Throwable t) {
468                FallbackKVDatabase.this.log.error("exception from migration check", t);
469            }
470        }
471    }
