006package org.jsimpledb.kv.raft.fallback;
007
008import java.util.Date;
009import java.util.Iterator;
010
011import org.jsimpledb.kv.KVPair;
012import org.jsimpledb.kv.KVTransaction;
013
014/**
015 * {@link MergeStrategy} that completely overwrites the destination database with the content of the source database.
016 */
017public class OverwriteMergeStrategy implements MergeStrategy {
018
019    @Override
020    public void merge(KVTransaction src, KVTransaction dst, Date lastActiveTime) {
021        dst.removeRange(null, null);
022        final Iterator<KVPair> i = src.getRange(null, null, false);
023        while (i.hasNext()) {
024            final KVPair pair = i.next();
025            dst.put(pair.getKey(), pair.getValue());
026        }
027        try {
028            ((AutoCloseable)i).close();
029        } catch (Exception e) {
030            // ignore
031        }
032    }
033
034    @Override
035    public String toString() {
036        return "Overwrite";
037    }
038}
