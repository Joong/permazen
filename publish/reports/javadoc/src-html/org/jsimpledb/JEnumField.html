006package org.jsimpledb;
007
008import com.google.common.base.Converter;
009import com.google.common.reflect.TypeToken;
010
011import java.lang.reflect.Method;
012
013import org.dellroad.stuff.java.EnumUtil;
014import org.jsimpledb.core.EnumFieldType;
015import org.jsimpledb.core.EnumValue;
016import org.jsimpledb.schema.EnumSchemaField;
017import org.jsimpledb.schema.SimpleSchemaField;
018import org.objectweb.asm.ClassWriter;
019import org.objectweb.asm.FieldVisitor;
020import org.objectweb.asm.MethodVisitor;
021import org.objectweb.asm.Opcodes;
022import org.objectweb.asm.Type;
023
024/**
025 * Represents an {@link Enum} field in a {@link JClass}.
026 */
027public class JEnumField extends JSimpleField {
028
029    final EnumConverter<?> converter;
030
031    @SuppressWarnings({ "unchecked", "rawtypes" })
032    JEnumField(JSimpleDB jdb, String name, int storageId, Class<? extends Enum<?>> enumType,
033      org.jsimpledb.annotation.JField annotation, String description, Method getter, Method setter) {
034        super(jdb, name, storageId, TypeToken.of(enumType.asSubclass(Enum.class)),
035          new EnumFieldType((Class)enumType), annotation.indexed(), annotation, description, getter, setter);
036        this.converter = EnumConverter.createEnumConverter(enumType);
037    }
038
039    @Override
040    public <R> R visit(JFieldSwitch<R> target) {
041        return target.caseJEnumField(this);
042    }
043
044    @Override
045    @SuppressWarnings("unchecked")
046    public TypeToken<? extends Enum<?>> getTypeToken() {
047        return (TypeToken<? extends Enum<?>>)this.typeToken;
048    }
049
050    @Override
051    public Converter<EnumValue, ? extends Enum<?>> getConverter(JTransaction jtx) {
052        return this.converter.reverse();
053    }
054
055    @Override
056    EnumSchemaField toSchemaItem(JSimpleDB jdb) {
057        final EnumSchemaField schemaField = new EnumSchemaField();
058        this.initialize(jdb, schemaField);
059        return schemaField;
060    }
061
062    @SuppressWarnings("unchecked")
063    void initialize(JSimpleDB jdb, SimpleSchemaField schemaField0) {
064        super.initialize(jdb, schemaField0);
065        final EnumSchemaField schemaField = (EnumSchemaField)schemaField0;
066        schemaField.getIdentifiers().clear();
067        for (Enum<?> value : EnumUtil.getValues((Class<Enum<?>>)this.getTypeToken().getRawType()))
068            schemaField.getIdentifiers().add(value.name());
069    }
070
071    @Override
072    JEnumFieldInfo toJFieldInfo(int parentStorageId) {
073        return new JEnumFieldInfo(this, parentStorageId);
074    }
075
076// Bytecode generation
077
078    @Override
079    void outputFields(ClassGenerator<?> generator, ClassWriter cw) {
080        final FieldVisitor valueField = cw.visitField(Opcodes.ACC_PRIVATE | Opcodes.ACC_STATIC | Opcodes.ACC_FINAL,
081          ClassGenerator.ENUM_CONVERTER_FIELD_PREFIX + this.storageId, Type.getDescriptor(EnumConverter.class), null, null);
082        valueField.visitEnd();
083    }
084
085    @Override
086    boolean hasClassInitializerBytecode() {
087        return true;
088    }
089
090    @Override
091    void outputClassInitializerBytecode(ClassGenerator<?> generator, MethodVisitor mv) {
092        mv.visitLdcInsn(Type.getType(this.typeToken.getRawType()));
093        generator.emitInvoke(mv, ClassGenerator.ENUM_CONVERTER_CREATE_METHOD);
094        mv.visitFieldInsn(Opcodes.PUTSTATIC, generator.getClassName(),
095          ClassGenerator.ENUM_CONVERTER_FIELD_PREFIX + this.storageId, Type.getDescriptor(EnumConverter.class));
096    }
097
098    @Override
099    void outputMethods(final ClassGenerator<?> generator, ClassWriter cw) {
100
101        // Getter
102        MethodVisitor mv = cw.visitMethod(
103          this.getter.getModifiers() & (Opcodes.ACC_PUBLIC | Opcodes.ACC_PROTECTED),
104          this.getter.getName(), Type.getMethodDescriptor(this.getter), null, generator.getExceptionNames(this.getter));
105        mv.visitFieldInsn(Opcodes.GETSTATIC, generator.getClassName(),
106          ClassGenerator.ENUM_CONVERTER_FIELD_PREFIX + this.storageId, Type.getDescriptor(EnumConverter.class));
107        generator.emitInvoke(mv, ClassGenerator.CONVERTER_REVERSE_METHOD);
108        this.outputReadCoreValueBytecode(generator, mv);
109        generator.emitInvoke(mv, ClassGenerator.CONVERTER_CONVERT_METHOD);
110        mv.visitTypeInsn(Opcodes.CHECKCAST, Type.getInternalName(this.getter.getReturnType()));
111        mv.visitInsn(Opcodes.ARETURN);
112        mv.visitMaxs(0, 0);
113        mv.visitEnd();
114
115        // Setter
116        mv = cw.visitMethod(
117          this.setter.getModifiers() & (Opcodes.ACC_PUBLIC | Opcodes.ACC_PROTECTED),
118          this.setter.getName(), Type.getMethodDescriptor(this.setter), null, generator.getExceptionNames(this.setter));
119        mv.visitFieldInsn(Opcodes.GETSTATIC, generator.getClassName(),
120          ClassGenerator.ENUM_CONVERTER_FIELD_PREFIX + this.storageId, Type.getDescriptor(EnumConverter.class));
121        mv.visitVarInsn(Opcodes.ALOAD, 1);
122        generator.emitInvoke(mv, ClassGenerator.CONVERTER_CONVERT_METHOD);
123        this.outputWriteCoreValueBytecode(generator, mv);
124        mv.visitInsn(Opcodes.RETURN);
125        mv.visitMaxs(0, 0);
126        mv.visitEnd();
127    }
128}
