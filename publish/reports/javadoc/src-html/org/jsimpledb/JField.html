006package org.jsimpledb;
007
008import java.lang.reflect.Method;
009
010import org.jsimpledb.core.ObjId;
011import org.jsimpledb.schema.SchemaField;
012import org.objectweb.asm.ClassWriter;
013import org.objectweb.asm.FieldVisitor;
014import org.objectweb.asm.Label;
015import org.objectweb.asm.MethodVisitor;
016import org.objectweb.asm.Opcodes;
017import org.objectweb.asm.Type;
018
019/**
020 * Represents a field in a Java model object.
021 */
022public abstract class JField extends JSchemaObject {
023
024    final Method getter;
025
026    JSchemaObject parent;
027    boolean requiresValidation;
028
029    JField(JSimpleDB jdb, String name, int storageId, String description, Method getter) {
030        super(jdb, name, storageId, description);
031        this.getter = getter;
032    }
033
034    void calculateRequiresValidation() {
035        this.requiresValidation = this.getter != null && Util.requiresValidation(this.getter);
036    }
037
038    @Override
039    abstract SchemaField toSchemaItem(JSimpleDB jdb);
040
041    /**
042     * Get the getter method associated with this field.
043     *
044     * @return field property getter method, or null if this field is a sub-field of a complex field
045     */
046    public Method getGetter() {
047        return this.getter;
048    }
049
050    /**
051     * Get the Java value of this field in the given Java model object.
052     * Does not alter the schema version of the object.
053     *
054     * @param jobj object containing this field
055     * @throws org.jsimpledb.core.DeletedObjectException if {@code jobj} does not exist in its associated {@link JTransaction}
056     * @throws org.jsimpledb.kv.StaleTransactionException if the {@link JTransaction} associated with {@code jobj}
057     *  is no longer usable
058     * @return this field's value in {@code jobj}
059     * @throws IllegalArgumentException if {@code jobj} is null
060     */
061    public abstract Object getValue(JObject jobj);
062
063    /**
064     * Apply visitor pattern.
065     *
066     * @param target target to invoke
067     * @param <R> visit return type
068     * @return value from the method of {@code target} corresponding to this instance's type
069     * @throws NullPointerException if {@code target} is null
070     */
071    public abstract <R> R visit(JFieldSwitch<R> target);
072
073    /**
074     * Create a {@link JFieldInfo} instance that corresponds to this instance.
075     */
076    abstract JFieldInfo toJFieldInfo();
077
078// Bytecode generation
079
080    boolean hasClassInitializerBytecode() {
081        return false;
082    }
083
084    void outputClassInitializerBytecode(ClassGenerator<?> generator, MethodVisitor mv) {
085        throw new UnsupportedOperationException();
086    }
087
088    void outputFields(ClassGenerator<?> generator, ClassWriter cw) {
089    }
090
091    abstract void outputMethods(ClassGenerator<?> generator, ClassWriter cw);
092
093    void outputCachedValueField(ClassGenerator<?> generator, ClassWriter cw) {
094        final FieldVisitor valueField = cw.visitField(Opcodes.ACC_PRIVATE,
095          this.getCachedValueFieldName(), Type.getDescriptor(this.getCachedValueFieldType()), null, null);
096        valueField.visitEnd();
097    }
098
099    void outputCachedValueGetterMethod(ClassGenerator<?> generator, ClassWriter cw, Method fieldReaderMethod) {
100        final MethodVisitor mv = generator.startMethod(cw, this.getter);
101        mv.visitVarInsn(Opcodes.ALOAD, 0);
102        mv.visitFieldInsn(Opcodes.GETFIELD, generator.getClassName(),
103          this.getCachedValueFieldName(), Type.getDescriptor(this.getCachedValueFieldType()));
104        mv.visitInsn(Opcodes.DUP);
105        final Label valueReady = new Label();
106        mv.visitJumpInsn(Opcodes.IFNONNULL, valueReady);
107        mv.visitInsn(Opcodes.POP);
108        mv.visitVarInsn(Opcodes.ALOAD, 0);
109        mv.visitFieldInsn(Opcodes.GETFIELD, generator.getClassName(),
110          ClassGenerator.TX_FIELD_NAME, Type.getDescriptor(JTransaction.class));
111        mv.visitVarInsn(Opcodes.ALOAD, 0);
112        mv.visitFieldInsn(Opcodes.GETFIELD, generator.getClassName(),
113          ClassGenerator.ID_FIELD_NAME, Type.getDescriptor(ObjId.class));
114        mv.visitLdcInsn(this.storageId);
115        mv.visitInsn(Opcodes.ICONST_1);                                                             // i.e., true
116        mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, Type.getInternalName(JTransaction.class),
117          fieldReaderMethod.getName(), Type.getMethodDescriptor(fieldReaderMethod), false);
118        mv.visitTypeInsn(Opcodes.CHECKCAST, Type.getInternalName(this.getCachedValueFieldType()));
119        mv.visitInsn(Opcodes.DUP);
120        mv.visitVarInsn(Opcodes.ALOAD, 0);
121        mv.visitInsn(Opcodes.SWAP);
122        mv.visitFieldInsn(Opcodes.PUTFIELD, generator.getClassName(),
123          this.getCachedValueFieldName(), Type.getDescriptor(this.getCachedValueFieldType()));
124        mv.visitLabel(valueReady);
125        mv.visitFrame(Opcodes.F_SAME1, 0, new Object[0], 1, new String[] { Type.getInternalName(this.getCachedValueFieldType()) });
126        mv.visitInsn(Opcodes.ARETURN);
127        mv.visitMaxs(0, 0);
128        mv.visitEnd();
129    }
130
131    private String getCachedValueFieldName() {
132        return ClassGenerator.JFIELD_FIELD_PREFIX + this.storageId;
133    }
134
135    private Class<?> getCachedValueFieldType() {
136        return this.getter.getReturnType();
137    }
138}
