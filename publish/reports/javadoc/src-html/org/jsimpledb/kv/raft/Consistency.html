006package org.jsimpledb.kv.raft;
007
008/**
009 * {@link RaftKVTransaction} supported consistency levels.
010 *
011 * <p>
012 * The differences only apply to read-only transactions; read-write transactions (i.e., transactions containing
013 * any mutations or cluster configuration changes) always behave with {@link #LINEARIZABLE} consistency.
014 *
015 * @see RaftKVTransaction#setConsistency RaftKVTransaction.setConsistency()
016 * @see <a href="https://aphyr.com/posts/313-strong-consistency-models">Strong consistency models</a>
017 */
018public enum Consistency {
019
020    /**
021     * Uncommitted eventual consistency.
022     *
023     * <p>
024     * This level is only meaningful for read-only transactions; read-write transactions are always {@link #LINEARIZABLE}.
025     *
026     * <p>
027     * Transactions see a consistent, view of the database that is either already committed or likely to be committed soon.
028     * This is the same as {@link #EVENTUAL_COMMITTED}, but with the additional (unlikely) possibility that the view that the
029     * transaction sees may never actually get committed; this can only happen if there is a Raft leadership change
030     * during the transaction.
031     *
032     * <p>
033     * Transactions at this level require no network communication, commit immediately, and never result in
034     * {@link org.jsimpledb.kv.RetryTransactionException}s. Committing a transaction at this level is actually
035     * equivalent to invoking {@link RaftKVTransaction#rollback rollback()} instead of {@link RaftKVTransaction#commit commit()}.
036     *
037     * <p>
038     * In Raft terms, transactions are based on the latest uncommitted log entry, the commit operation does not wait
039     * for that log entry to be committed, and up-to-date reads are not guaranteed.
040     */
041    UNCOMMITTED(false, false, false),
042
043    /**
044     * Committed eventual consistency.
045     *
046     * <p>
047     * This level is only meaningful for read-only transactions; read-write transactions are always {@link #LINEARIZABLE}.
048     *
049     * <p>
050     * Transactions see a consistent, committed view of the database as it existed at some point in the "recent past".
051     * The view is not guaranteed to be up-to-date; it's only guaranteed to be as up-to-date as what was known to this
052     * node when the transaction was opened. In general (for example, if stuck in a network partition minority), the view
053     * could be from arbitrarily far in the past.
054     *
055     * <p>
056     * Unlike {@link #EVENTUAL}, transactions at this level require no network communication, commit immediately, and never
057     * result in {@link org.jsimpledb.kv.RetryTransactionException}s; however, they see a potentiall older view. Compared
058     * to {@link #UNCOMMITTED}, transactions at this level see a view that is potentially older, in exchange for the guarantee
059     * that it has definitely been committed.
060     *
061     * <p>
062     * This consistency level is useful when it's important to allow reading the database in any situation, e.g., even when
063     * the local node is in a minority partition of the cluster. Of course, the data will only be as up-to-date as is known
064     * locally.
065     *
066     * <p>
067     * In Raft terms, transactions are based on the latest committed log entry, and up-to-date reads are not guaranteed.
068     */
069    EVENTUAL_COMMITTED(true, false, false),
070
071    /**
072     * Eventual consistency.
073     *
074     * <p>
075     * This level is only meaningful for read-only transactions; read-write transactions are always {@link #LINEARIZABLE}.
076     *
077     * <p>
078     * Transactions see a consistent, committed view of the database as it existed at some point in the "recent past".
079     * The view is not guaranteed to be up-to-date; it's only guaranteed to be as up-to-date as what was known to this
080     * node when the transaction was opened. In general (for example, if stuck in a network partition minority), the view
081     * could be from arbitrarily far in the past.
082     *
083     * <p>
084     * No network traffic is generated by {@link RaftKVTransaction#commit commit()}, and transactions can never throw
085     * {@link org.jsimpledb.kv.RetryTransactionException}s due to read/write conflicts (i.e., stale read checks).
086     * A {@link RaftKVTransaction#commit commit()} blocks until the log entry on which the transaction is based is committed;
087     * if no concurrent write transactions are occurring, this will happen immediately.
088     *
089     * <p>
090     * In Raft terms, transactions are based on the latest uncommitted log entry, the commit operation waits
091     * for that log entry to be committed, and up-to-date reads are not guaranteed.
092     */
093    EVENTUAL(false, true, false),
094
095    /**
096     * Linearizable consistency.
097     *
098     * <p>
099     * Transactions see a database that evolves step-wise through a global, linear sequence of atomic changes, where
100     * each change appears to have occurred instantaneously at some point in time between the start and end of the
101     * corresponding transaction.
102     *
103     * <p>
104     * This is the default consistency level.
105     *
106     * <p>
107     * In Raft terms, transactions are based on the latest uncommitted log entry, the commit operation waits
108     * for that log entry to be committed, and up-to-date reads are guaranteed.
109     */
110    LINEARIZABLE(false, true, true);
111
112    private final boolean basedOnCommittedLogEntry;
113    private final boolean waitsForLogEntryToBeCommitted;
114    private final boolean guaranteesUpToDateReads;
115
116    private Consistency(boolean basedOnCommittedLogEntry,
117      boolean waitsForLogEntryToBeCommitted, boolean guaranteesUpToDateReads) {
118        this.basedOnCommittedLogEntry = basedOnCommittedLogEntry;
119        this.waitsForLogEntryToBeCommitted = waitsForLogEntryToBeCommitted;
120        this.guaranteesUpToDateReads = guaranteesUpToDateReads;
121    }
122
123    /**
124     * Determines the log entry on which transactions at this level are based.
125     *
126     * <p>
127     * When false, transactions are based on this node's most recent log entry, whether committed or not.
128     * When true, transactions are based on this node's most recent committed log entry. The former provides
129     * a more up-to-date view, but {@link RaftKVTransaction#commit commit()} can block (and could possibly
130     * fail with a {@link org.jsimpledb.kv.RetryTransactionException}) when {@link #isWaitsForLogEntryToBeCommitted} is true.
131     * The latter, when combined with {@link #isGuaranteesUpToDateReads} being false, allows read-only transactions
132     * to commit immediately with no network traffic, even when in a minority partition.
133     *
134     * @return true if transactions at this level are always based on committed log entries
135     */
136    public boolean isBasedOnCommittedLogEntry() {
137        return this.basedOnCommittedLogEntry;
138    }
139
140    /**
141     * Determines whether transactions at this level block in {@link RaftKVTransaction#commit commit()} until
142     * the log entry on which they are based is committed.
143     *
144     * <p>
145     * This setting has no effect if {@link #isBasedOnCommittedLogEntry} returns true, because in that case
146     * the log entry on which the transaction is based is already committed when transaction starts.
147     *
148     * @return true if transactions at this level wait for their base log entry to be committed
149     */
150    public boolean isWaitsForLogEntryToBeCommitted() {
151        return this.waitsForLogEntryToBeCommitted;
152    }
153
154    /**
155     * Determines whether transactions at this level guarantee reading the most up-to-date information.
156     *
157     * <p>
158     * This setting has no effect on transactions that contain mutations; they always read the most up-to-date information.
159     *
160     * <p>
161     * If this returns false, read-only transactions may read out-of-date information ("eventual consistency").
162     *
163     * @return true if transactions at this level guarantee up-to-date reads
164     */
165    public boolean isGuaranteesUpToDateReads() {
166        return this.guaranteesUpToDateReads;
167    }
168}
