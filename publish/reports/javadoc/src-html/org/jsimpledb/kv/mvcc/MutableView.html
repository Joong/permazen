026/**
027 * Provides a mutable view of an underlying, read-only {@link KVStore}.
028 *
029 * <p>
030 * Instances intercept all operations to the underlying {@link KVStore}, recording mutations in a {@link Writes} instance
031 * instead of applying them to the {@link KVStore}. Instances then provide a view of the mutated {@link KVStore} based those
032 * mutations. Mutations that overwrite previous mutations are consolidated.
033 * </p>
034 *
035 * <p>
036 * Unlike writes, reads are passed through to the underlying {@link KVStore}, except where they intersect a previous write.
037 * Reads may also be optionally recorded.
038 * </p>
039 *
040 * <p>
041 * In all cases, then underlying {@link KVStore} is never modified.
042 * </p>
043 *
044 * <p>
045 * Instances are also capable of performing certain MVCC-related calculations, such as whether two transactions may be re-ordered.
046 * </p>
047 *
048 * <p>
049 * Instances are thread safe; however, directly accessing the associated {@link Reads} or {@link Writes} is not thread safe
050 * without first locking this instance.
051 * </p>
052 */
053public class MutableView extends AbstractKVStore implements SizeEstimating {
054
055    private final KVStore kv;
056    private final Writes writes;
057    private Reads reads;
058
059// Constructors
060
061    /**
062     * Constructor.
063     *
064     * @param kv underlying {@link KVStore}
065     * @throws IllegalArgumentException if {@code kv} is null
066     */
067    public MutableView(KVStore kv) {
068        this(kv, new Reads(), new Writes());
069    }
070
071    /**
072     * Constructor using caller-provided {@link Reads} (optional} and {@link Writes}.
073     *
074     * @param kv underlying {@link KVStore}
075     * @param reads recorded reads, or null for none
076     * @param writes recorded writes
077     * @throws IllegalArgumentException if {@code kv} is null
078     * @throws IllegalArgumentException if {@code writes} is null
079     */
080    public MutableView(KVStore kv, Reads reads, Writes writes) {
081        Preconditions.checkArgument(kv != null, "null kv");
082        Preconditions.checkArgument(writes != null, "null writes");
083        this.kv = kv;
084        this.reads = reads;
085        this.writes = writes;
086        synchronized (this) { }                                     // because this.reads is not final
087    }

091    /**
092     * Get the {@link Reads} associated with this instance.
093     *
094     * <p>
095     * This includes all keys explicitly or implicitly read by calls to
096     * {@link #get get()}, {@link #getAtLeast getAtLeast()}, {@link #getAtMost getAtMost()}, and {@link #getRange getRange()}.
097     *
098     * @return reads recorded, or null if this instance is not configured to record reads
099     */

101        return this.reads;
102    }

105     * Get the {@link Writes} associated with this instance.
106     *

108     */
109    public synchronized Writes getWrites() {
110        return this.writes;
111    }

113    /**
114     * Disable read tracking and discard the {@link Reads} associated with this instance.
115     *
116     * <p>
117     * Can be used to save some memory when read tracking information is no longer needed.
118     */
119    public synchronized void disableReadTracking() {
120        this.reads = null;
121    }

123// KVStore

125    @Override
126    public synchronized byte[] get(byte[] key) {
127
128        // Check puts
129        byte[] value = this.writes.getPuts().get(key);
130        if (value != null)
131            return this.applyCounterAdjustment(key, value);
132
133        // Check removes
134        if (this.writes.getRemoves().contains(key))
135            return null;
136
137        // Read from underlying k/v store
138        value = this.kv.get(key);
139
140        // Record the read
141        this.recordReads(key, ByteUtil.getNextKey(key));
142
143        // Apply counter adjustments
144        if (value != null)                                          // we can ignore adjustments of missing values
145            value = this.applyCounterAdjustment(key, value);
146
147        // Done
148        return value;
149    }
150
151    @Override
152    public Iterator<KVPair> getRange(byte[] minKey, byte[] maxKey, boolean reverse) {
153
154        // Build iterator
155        return new RangeIterator(minKey, maxKey, reverse);
156    }
157
158    @Override
159    public synchronized void put(byte[] key, byte[] value) {
160
161        // Sanity check
162        Preconditions.checkArgument(key != null, "null key");
163        Preconditions.checkArgument(value != null, "null value");
164
165        // Overwrite any counter adjustment
166        this.writes.getAdjusts().remove(key);
167
168        // Record the put
169        this.writes.getPuts().put(key.clone(), value.clone());
170    }
171
172    @Override
173    public synchronized void remove(byte[] key) {
174
175        // Sanity check
176        Preconditions.checkArgument(key != null, "null key");
177
178        // Overwrite any counter adjustment
179        this.writes.getAdjusts().remove(key);
180
181        // Overwrite any put
182        this.writes.getPuts().remove(key);
183
184        // Record the remove
185        this.writes.setRemoves(this.writes.getRemoves().add(new KeyRange(key)));
186    }
187
188    @Override
189    public synchronized void removeRange(byte[] minKey, byte[] maxKey) {
190
191        // Realize minKey
192        if (minKey == null)
193            minKey = ByteUtil.EMPTY;
194
195        // Overwrite any puts and counter adjustments
196        if (maxKey != null) {
197            this.writes.getPuts().subMap(minKey, maxKey).clear();
198            this.writes.getAdjusts().subMap(minKey, maxKey).clear();
199        } else {
200            this.writes.getPuts().tailMap(minKey).clear();
201            this.writes.getAdjusts().tailMap(minKey).clear();
202        }
203
204        // Record the remove
205        this.writes.setRemoves(this.writes.getRemoves().add(new KeyRange(minKey, maxKey)));
206    }

208    @Override
209    public byte[] encodeCounter(long value) {
210        return this.kv.encodeCounter(value);
211    }
212
213    @Override
214    public long decodeCounter(byte[] bytes) {
215        return this.kv.decodeCounter(bytes);
216    }

218    @Override
219    public synchronized void adjustCounter(byte[] key, long amount) {
220
221        // Check puts
222        final byte[] putValue = this.writes.getPuts().get(key);
223        if (putValue != null) {
224            final long value;
225            try {
226                value = this.kv.decodeCounter(putValue);
227            } catch (IllegalArgumentException e) {
228                return;                                 // previously put value was not decodable, so ignore this adjustment
229            }
230            this.writes.getPuts().put(key, this.kv.encodeCounter(value + amount));
231            return;
232        }
233
234        // Check removes
235        if (this.writes.getRemoves().contains(key))
236            return;
237
238        // Calculate new, cumulative adjustment
239        final Long oldAdjust = this.writes.getAdjusts().get(key);
240        final long adjust = (oldAdjust != null ? oldAdjust : 0) + amount;
241
242        // Record/update adjustment
243        if (adjust != 0)
244            this.writes.getAdjusts().put(key, adjust);
245        else
246            this.writes.getAdjusts().remove(key);
247    }

261        estimator
262          .addObjectOverhead()
263          .addReferenceField()                              // kv
264          .addField(this.reads)                             // reads
265          .addField(this.writes);                           // writes
266    }

278// Internal methods

280    // Apply accumulated counter adjustments to the value, if any
281    private synchronized byte[] applyCounterAdjustment(byte[] key, byte[] value) {
282
283        // Is there an adjustment of this key?
284        assert key != null;
285        final Long adjust = this.writes.getAdjusts().get(key);
286        if (adjust == null)
287            return value;
288
289        // Decode value we just read as a counter
290        final long counterValue;
291        try {
292            counterValue = this.kv.decodeCounter(value);
293        } catch (IllegalArgumentException e) {
294            this.writes.getAdjusts().remove(key);       // previous adjustment was bogus because value was not decodable
295            return value;
296        }
297
298        // Adjust counter value by accumulated adjustment value and re-encode
299        return this.kv.encodeCounter(counterValue + adjust);
300    }
301
302    // Record that keys were read in the range [minKey, maxKey)
303    private synchronized void recordReads(byte[] minKey, byte[] maxKey) {
304
305        // Not tracking reads?
306        if (this.reads == null)
307            return;
308
309        // Realize minKey
310        if (minKey == null)
311            minKey = ByteUtil.EMPTY;
312
313        // Already tracked?
314        final KeyRange readRange = new KeyRange(minKey, maxKey);
315        if (this.reads.getReads().contains(readRange))
316            return;
317
318        // Subtract out the part of the read range that did not really go through to k/v store due to puts or removes
319        KeyRanges readRanges = new KeyRanges(readRange);
320        final Set<byte[]> putKeys = (maxKey != null ?
321          this.writes.getPuts().subMap(minKey, maxKey) : this.writes.getPuts().tailMap(minKey)).keySet();
322        for (byte[] key : putKeys)
323            readRanges = readRanges.remove(new KeyRange(key));
324        readRanges = readRanges.intersection(this.writes.getRemoves().inverse());
325
326        // Record reads
327        if (!readRanges.isEmpty())
328            this.reads.setReads(this.reads.getReads().union(readRanges));
329    }

331// RangeIterator

333    private class RangeIterator implements Iterator<KVPair>, Closeable {
334
335        private final boolean reverse;          // iteration direction
336        private final byte[] limit;             // limit of iteration; exclusive if forward, inclusive if reverse
337
338        private byte[] cursor;                  // current position; inclusive if forward, exclusive if reverse
339        private KVPair next;                    // the next k/v pair queued up, or null if not found yet
340        private byte[] removeKey;               // key to remove if remove() is invoked
341        private boolean finished;
342
343        // Position in underlying k/v store
344        private Iterator<KVPair> kviter;        // k/v store iterator, if any left
345        private KVPair kvnext;                  // next kvstore pair, if already retrieved
346
347        // Position in puts
348        private KVPair putnext;                 // next put pair, if already retrieved
349        private boolean putdone;                // no more pairs left in puts
350
351        RangeIterator(byte[] minKey, byte[] maxKey, boolean reverse) {
352
353            // Realize minKey
354            if (minKey == null)
355                minKey = ByteUtil.EMPTY;
356
357            // Initialize cursors
358            this.kviter = MutableView.this.kv.getRange(minKey, maxKey, reverse);
359            this.cursor = reverse ? maxKey : minKey;
360            this.limit = reverse ? minKey : maxKey;
361            this.reverse = reverse;
362        }
363
364        @Override
365        public synchronized boolean hasNext() {
366            return this.next != null || this.findNext();
367        }
368
369        @Override
370        public synchronized KVPair next() {
371            if (this.next == null && !this.findNext())
372                throw new NoSuchElementException();
373            final KVPair pair = this.next;
374            assert pair != null;
375            this.removeKey = pair.getKey();
376            this.next = null;
377            return pair;
378        }
379
380        @Override
381        public synchronized void remove() {
382            Preconditions.checkState(this.removeKey != null);
383            MutableView.this.remove(this.removeKey);
384            this.removeKey = null;
385        }
386
387        private synchronized boolean findNext() {
388
389            // Invariants & checks
390            assert this.next == null;
391            assert this.cursor != null || this.reverse;
392            assert this.limit != null || !this.reverse;
393            assert this.kviter != null || this.kvnext == null;
394
395            // Exhausted?
396            if (this.finished)
397                return false;
398
399            // Find the next underlying k/v pair, if we don't already have it
400            final KeyRanges removes;
401            synchronized (MutableView.this) {
402                removes = MutableView.this.writes.getRemoves();
403            }
404            if (this.kviter != null && this.kvnext == null) {
405                while (true) {
406
407                    // Get next k/v pair in underlying key/value store, if any
408                    if (!this.kviter.hasNext()) {
409                        this.closeKVStoreIterator();
410                        break;
411                    }
412                    this.kvnext = this.kviter.next();
413                    assert this.kvnext != null;
414                    assert !this.isPastLimit(this.kvnext.getKey());
415                    assert this.isPast(this.kvnext.getKey(), this.cursor) :
416                      "key " + ByteUtil.toString(this.kvnext.getKey()) + " is not past cursor " + ByteUtil.toString(this.cursor);
417
418                    // If k/v pair has been removed, skip past the matching remove range
419                    final KeyRange[] ranges = removes.findKey(this.kvnext.getKey());
420                    if (ranges[0] == ranges[1] && ranges[0] != null) {
421                        final KeyRange removeRange = ranges[0];
422
423                        // Find the end of the remove range (if any)
424                        final byte[] removeRangeEnd = this.reverse ? removeRange.getMin() : removeRange.getMax();
425                        if (removeRangeEnd == null
426                         || this.isPastLimit(removeRangeEnd)
427                         || (this.reverse && Arrays.equals(removeRangeEnd, this.limit))) {
428                            this.closeKVStoreIterator();
429                            break;
430                        }
431
432                        // Skip over it and restart iterator
433                        this.closeKVStoreIterator();
434                        final byte[] iterMin;
435                        final byte[] iterMax;
436                        if (this.reverse) {
437                            iterMin = this.limit;
438                            iterMax = removeRangeEnd;
439                        } else {
440                            iterMin = removeRangeEnd;
441                            iterMax = this.limit;
442                        }
443                        this.kviter = MutableView.this.kv.getRange(iterMin, iterMax, this.reverse);
444                        continue;
445                    }
446
447                    // Got one
448                    break;
449                }
450            }
451
452            // Find next put pair, if we don't already have it
453            if (!this.putdone && this.putnext == null) {
454                Map.Entry<byte[], byte[]> putEntry;
455                synchronized (MutableView.this) {
456                    if (this.reverse) {
457                        putEntry = this.cursor != null ?
458                          MutableView.this.writes.getPuts().lowerEntry(this.cursor) :
459                          MutableView.this.writes.getPuts().lastEntry();
460                    } else {
461                        putEntry = MutableView.this.writes.getPuts().ceilingEntry(this.cursor);
462                    }
463                }
464                if (putEntry == null || this.isPastLimit(putEntry.getKey())) {
465                    this.putnext = null;
466                    this.putdone = true;
467                } else
468                    this.putnext = new KVPair(putEntry);
469            }
470
471            // Figure out which pair appears first (k/v or put); if there's a tie, the put wins
472            if (this.kvnext == null && this.putnext == null)
473                this.next = null;
474            else if (this.kvnext == null) {
475                this.next = this.putnext;
476                this.putnext = null;
477            } else if (this.putnext == null) {
478                this.next = this.kvnext;
479                this.kvnext = null;
480            } else {
481                int diff = ByteUtil.compare(this.putnext.getKey(), this.kvnext.getKey());
482                if (reverse)
483                    diff = -diff;
484                if (diff <= 0) {
485                    this.next = this.putnext;
486                    this.putnext = null;
487                    if (diff == 0)
488                        this.kvnext = null;                     // the kvstore key was overridden by the put key
489                } else {
490                    this.next = this.kvnext;
491                    this.kvnext = null;
492                }
493            }
494
495            // Record that we read from everything we just scanned over in the underlying KVStore
496            final byte[] skipMin;
497            final byte[] skipMax;
498            if (this.reverse) {
499                skipMin = this.next != null ? this.next.getKey() : ByteUtil.EMPTY;
500                skipMax = this.cursor;
501            } else {
502                skipMin = this.cursor;
503                skipMax = this.next != null ? ByteUtil.getNextKey(this.next.getKey()) : null;
504            }
505            if (skipMax == null || ByteUtil.compare(skipMin, skipMax) < 0)
506                MutableView.this.recordReads(skipMin, skipMax);
507
508            // Finished?
509            if (this.next == null) {
510                this.finished = true;
511                return false;
512            }
513
514            // Apply any counter adjustment to the retrieved value, if appropriate
515            final byte[] adjustedValue = MutableView.this.applyCounterAdjustment(this.next.getKey(), this.next.getValue());
516            if (adjustedValue != this.next.getValue())
517                this.next = new KVPair(this.next.getKey(), adjustedValue);
518
519            // Update cursor
520            this.cursor = this.reverse ? this.next.getKey() : ByteUtil.getNextKey(this.next.getKey());
521
522            // Done
523            return true;
524        }
525
526        private boolean isPastLimit(byte[] key) {
527            return this.isPast(key, this.limit);
528        }
529
530        private boolean isPast(byte[] key, byte[] mark) {
531            return this.reverse ?
532              mark == null || ByteUtil.compare(key, mark) < 0 :
533              mark != null && ByteUtil.compare(key, mark) >= 0;
534        }
535
536        private void closeKVStoreIterator() {
537            assert Thread.holdsLock(this);
538            if (this.kviter != null) {
539                try {
540                    ((AutoCloseable)this.kviter).close();
541                } catch (Exception e) {
542                    // ignore;
543                }
544                this.kviter = null;
545            }
546            this.kvnext = null;
547        }
548
549    // Closeable
550
551        @Override
552        public synchronized void close() {
553            this.closeKVStoreIterator();
554            this.putdone = true;
555        }
556    }
