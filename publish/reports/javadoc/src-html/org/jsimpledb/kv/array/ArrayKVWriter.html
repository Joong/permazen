006package org.jsimpledb.kv.array;
007
008import com.google.common.base.Preconditions;
009
010import java.io.BufferedOutputStream;
011import java.io.Closeable;
012import java.io.IOException;
013import java.io.OutputStream;
014
015import org.jsimpledb.util.ByteUtil;
016
017/**
018 * Writes {@link ArrayKVStore} index, key, and value data, given a sorted sequence of key/value pairs.
019 *
020 * <p>
021 * Key and value data must not exceed 2GB.
022 */
023public class ArrayKVWriter implements Closeable {
024
025    private static final int BUFFER_SIZE = 1024 * 1024;
026
027    private final BufferedOutputStream indxOutput;
028    private final BufferedOutputStream keysOutput;
029    private final BufferedOutputStream valsOutput;
030
031    private int keysLength;
032    private int valsLength;
033
034    private int nextIndex;
035    private byte[] prevKey;
036    private byte[] baseKey;
037    private int baseKeyOffset;
038    private boolean closed;
039
040    /**
041     * Constructor.
042     *
043     * @param indxOutput index file output
044     * @param keysOutput key data file output
045     * @param valsOutput value data file output
046     */
047    public ArrayKVWriter(OutputStream indxOutput, OutputStream keysOutput, OutputStream valsOutput) throws IOException {
048        Preconditions.checkArgument(indxOutput != null, "null indxOutput");
049        Preconditions.checkArgument(keysOutput != null, "null keysOutput");
050        Preconditions.checkArgument(valsOutput != null, "null valsOutput");
051        this.indxOutput = new BufferedOutputStream(indxOutput, BUFFER_SIZE);
052        this.keysOutput = new BufferedOutputStream(keysOutput, BUFFER_SIZE);
053        this.valsOutput = new BufferedOutputStream(valsOutput, BUFFER_SIZE);
054    }
055
056    /**
057     * Get the number of bytes written so far to the index file.
058     */
059    public int getIndxLength() {
060        return this.nextIndex * 8;
061    }
062
063    /**
064     * Get the number of bytes written so far to the key data file.
065     */
066    public int getKeysLength() {
067        return this.keysLength;
068    }
069
070    /**
071     * Get the number of bytes written so far to the value data file.
072     */
073    public int getValsLength() {
074        return this.valsLength;
075    }
076
077    /**
078     * Write out the next key/value pair.
079     *
080     * @param key key
081     * @param val value
082     * @throws IllegalArgumentException if {@code key} is out of order (i.e., not strictly greater then the previous key)
083     * @throws IllegalArgumentException if {@code key} or {@code val} is null
084     * @throws IllegalStateException if either the key or data file would grow larger than 2<sup>31</sup>-1 bytes
085     */
086    public void writeKV(byte[] key, byte[] val) throws IOException {
087
088        // Sanity checks
089        Preconditions.checkArgument(key != null, "null key");
090        Preconditions.checkArgument(val != null, "null value");
091        Preconditions.checkArgument(this.prevKey == null || ByteUtil.compare(key, this.prevKey) > 0, "key <= previous key");
092        Preconditions.checkState((this.nextIndex * 8) + 8 > 0, "too much index data");
093        Preconditions.checkState(this.keysLength == 0 || this.keysLength + key.length > 0, "too much key data");
094        Preconditions.checkState(this.valsLength == 0 || this.valsLength + val.length > 0, "too much value data");
095
096        // Write key index entry and data
097        if ((this.nextIndex & 0x1f) == 0) {
098
099            // Write a base key index entry every 32 entries
100            this.writeIndxValue(this.keysLength);
101            this.baseKeyOffset = this.keysLength;
102            this.baseKey = key;
103
104            // Write key data
105            this.keysOutput.write(key);
106            this.keysLength += key.length;
107        } else {
108
109            // Match maximal prefix of most recent base key, up to 255 bytes
110            final int maxPrefixLength = Math.min(Math.min(this.baseKey.length, key.length), 0xff);
111            int prefixLength = 0;
112            while (prefixLength < maxPrefixLength && key[prefixLength] == this.baseKey[prefixLength])
113                prefixLength++;
114            final int suffixLength = key.length - prefixLength;
115            assert suffixLength > 0;
116
117            // Write encoded { base key prefix length, offset to key suffix }
118            final int suffixRelativeOffset = this.keysLength - this.baseKeyOffset;
119            Preconditions.checkState((suffixRelativeOffset & 0xff000000) == 0, "key(s) too long");
120            this.writeIndxValue(prefixLength << 24 | suffixRelativeOffset);
121
122            // Write key data - suffix only
123            this.keysOutput.write(key, prefixLength, suffixLength);
124            this.keysLength += suffixLength;
125        }
126
127        // Write value index entry
128        this.writeIndxValue(this.valsLength);
129
130        // Write value data
131        this.valsOutput.write(val);
132        this.valsLength += val.length;
133
134        // Update state
135        this.prevKey = key;
136        this.nextIndex++;
137    }
138
139    private void writeIndxValue(int offset) throws IOException {
140        this.indxOutput.write(offset >> 24);
141        this.indxOutput.write(offset >> 16);
142        this.indxOutput.write(offset >> 8);
143        this.indxOutput.write(offset);
144    }
145
146    /**
147     * Flush all three outputs.
148     */
149    public void flush() throws IOException {
150        this.indxOutput.flush();
151        this.keysOutput.flush();
152        this.valsOutput.flush();
153    }
154
155    /**
156     * Close all three outputs.
157     */
158    @Override
159    public void close() throws IOException {
160        if (this.closed)
161            return;
162        this.closed = true;
163        this.indxOutput.close();
164        this.keysOutput.close();
165        this.valsOutput.close();
166    }
167}
