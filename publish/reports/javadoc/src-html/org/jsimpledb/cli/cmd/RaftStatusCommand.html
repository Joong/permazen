006package org.jsimpledb.cli.cmd;
007
008import java.io.PrintWriter;
009import java.util.List;
010import java.util.Map;
011
012import org.jsimpledb.SessionMode;
013import org.jsimpledb.cli.CliSession;
014import org.jsimpledb.kv.raft.CandidateRole;
015import org.jsimpledb.kv.raft.Follower;
016import org.jsimpledb.kv.raft.FollowerRole;
017import org.jsimpledb.kv.raft.LeaderRole;
018import org.jsimpledb.kv.raft.LogEntry;
019import org.jsimpledb.kv.raft.RaftKVDatabase;
020import org.jsimpledb.kv.raft.RaftKVTransaction;
021import org.jsimpledb.kv.raft.Role;
022import org.jsimpledb.kv.raft.Timestamp;
023import org.jsimpledb.kv.raft.TxState;
024import org.jsimpledb.parse.ParseContext;
025
026@Command(modes = { SessionMode.KEY_VALUE, SessionMode.CORE_API, SessionMode.JSIMPLEDB })
027public class RaftStatusCommand extends AbstractRaftCommand {
028
029    public RaftStatusCommand() {
030        super("raft-status");
031    }
032
033    @Override
034    public String getHelpSummary() {
035        return "Displays the Raft cluster state of the local node";
036    }
037
038    @Override
039    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
040        return new RaftAction() {
041            @Override
042            protected void run(CliSession session, RaftKVDatabase db) throws Exception {
043                RaftStatusCommand.this.displayStatus(session.getWriter(), db);
044            }
045        };
046    }
047
048    private void displayStatus(PrintWriter writer, RaftKVDatabase db) {
049
050        // Configuration info
051        writer.println();
052        writer.println("Configuration");
053        writer.println("=============");
054        writer.println();
055        writer.println(String.format("%-24s: %s", "Log directory", db.getLogDirectory()));
056        writer.println(String.format("%-24s: %d.%03d sec", "Min election timeout",
057          db.getMinElectionTimeout() / 1000, db.getMinElectionTimeout() % 1000));
058        writer.println(String.format("%-24s: %d.%03d sec", "Max election timeout",
059          db.getMaxElectionTimeout() / 1000, db.getMaxElectionTimeout() % 1000));
060        writer.println(String.format("%-24s: %d.%03d sec", "Heartbeat timeout",
061          db.getHeartbeatTimeout() / 1000, db.getHeartbeatTimeout() % 1000));
062        writer.println(String.format("%-24s: %d.%03d sec", "Commit timeout",
063          db.getCommitTimeout() / 1000, db.getCommitTimeout() % 1000));
064        writer.println(String.format("%-24s: %d.%03d sec", "Max transaction duration",
065          db.getMaxTransactionDuration() / 1000, db.getMaxTransactionDuration() % 1000));
066
067        // Cluster info
068        writer.println();
069        writer.println("Cluster Info");
070        writer.println("============");
071        writer.println();
072        writer.println(String.format("%-24s: \"%s\"", "Cluster identity", db.getIdentity()));
073        writer.println(String.format("%-24s: %s", "Cluster ID",
074          db.getClusterId() != 0 ? String.format("0x%08x", db.getClusterId()) : "Unconfigured"));
075        writer.println(String.format("%-24s: %s", "Node is cluster member", db.isClusterMember() ? "Yes" : "No"));
076        final Map<String, String> config = db.getCurrentConfig();
077        if (config.isEmpty())
078            writer.println(String.format("%-24s: %s", "Cluster configuration", "Unconfigured"));
079        else {
080            writer.println();
081            writer.println(String.format("Cluster configuration:"));
082            writer.println();
083            writer.println(String.format("%1s %-16s %s", "", "Identity", "Address"));
084            writer.println(String.format("%1s %-16s %s", "", "--------", "-------"));
085            for (Map.Entry<String, String> entry : db.getCurrentConfig().entrySet()) {
086                final String identity = entry.getKey();
087                final String address = entry.getValue();
088                writer.println(String.format("%1s %-16s %s",
089                  identity.equals(db.getIdentity()) ? "*" : "", "\"" + identity + "\"", address));
090            }
091        }
092
093        // General Raft status
094        writer.println();
095        writer.println("Raft State");
096        writer.println("==========");
097        writer.println();
098        writer.println(String.format("%-24s: %dt%d", "Last applied log entry", db.getLastAppliedIndex(), db.getLastAppliedTerm()));
099        writer.println(String.format("%-24s: %d", "Commit Index", db.getCommitIndex()));
100        writer.println(String.format("%-24s: %d", "Current term", db.getCurrentTerm()));
101        final Role role = db.getCurrentRole();
102        writer.println(String.format("%-24s: %s", "Current Role",
103          role instanceof LeaderRole ? "Leader" :
104          role instanceof FollowerRole ? "Follower" :
105          role instanceof CandidateRole ? "Candidate" : "?" + role));
106        writer.println(String.format("%-24s: %d", "Unapplied memory usage", db.getUnappliedLogMemoryUsage()));
107        final List<LogEntry> log = db.getUnappliedLog();
108        writer.println(String.format("%-24s: %d", "Unapplied log entries", log.size()));
109        if (!log.isEmpty()) {
110            writer.println();
111            writer.println(String.format("%-10s %-6s %-10s %-8s %s", "Entry", "Commit", "Size", "Age", "Config Change"));
112            writer.println(String.format("%-10s %-6s %-10s %-8s %s", "-----", "------", "----", "---", "-------------"));
113            for (LogEntry entry : log) {
114                writer.println(String.format("%-10s %-6s %-10d %-8s %s", entry.getIndex() + "t" + entry.getTerm(),
115                  entry.getIndex() <= db.getCommitIndex() ? "Yes" : "No",
116                  entry.getFileSize(), String.format("%d.%03ds", entry.getAge() / 1000, entry.getAge() % 1000),
117                  this.describe(entry.getConfigChange())));
118            }
119        }
120
121        // Role-specific info
122        if (role instanceof LeaderRole) {
123            final LeaderRole leader = (LeaderRole)role;
124
125            writer.println();
126            writer.println("Leader Info");
127            writer.println("===========");
128            writer.println();
129            writer.println(String.format("%-24s: %s", "Lease Timeout", leader.getLeaseTimeout() != null ?
130              String.format("%+dms", leader.getLeaseTimeout().offsetFromNow()) : "Not established"));
131            final List<Follower> followers = leader.getFollowers();
132            writer.println(String.format("%-24s: %d", "Followers", followers.size()));
133            if (!followers.isEmpty()) {
134                writer.println();
135                writer.println(String.format("  %-16s %-8s %-6s %-6s %-6s %s",
136                  "Identity", "Status", "Match", "Next", "Commit", "Timestamp"));
137                writer.println(String.format("  %-16s %-8s %-6s %-6s %-6s %s",
138                  "--------", "------", "-----", "----", "------", "---------"));
139                for (Follower follower : leader.getFollowers()) {
140                    writer.println(String.format("  %-16s %-8s %-6s %-6s %-6s %s", follower.getIdentity(),
141                      follower.isReceivingSnapshot() ? "Snapshot" : follower.isSynced() ? "Synced" : "No Sync",
142                      follower.getMatchIndex(), follower.getNextIndex(), follower.getLeaderCommit(),
143                      follower.getLeaderTimestamp() != null ?
144                       String.format("%+dms", follower.getLeaderTimestamp().offsetFromNow()) : "None"));
145                }
146            }
147        } else if (role instanceof FollowerRole) {
148            final FollowerRole follower = (FollowerRole)role;
149
150            writer.println();
151            writer.println("Follower Info");
152            writer.println("=============");
153            writer.println();
154            writer.println(String.format("%-24s: %s", "Leader Identity",
155              follower.getLeaderIdentity() != null ? "\"" + follower.getLeaderIdentity() + "\"" : "Unknown"));
156            writer.println(String.format("%-24s: %s", "Leader Address",
157              follower.getLeaderAddress() != null ? follower.getLeaderAddress() : "Unknown"));
158            writer.println(String.format("%-24s: %s", "Voted For",
159              follower.getVotedFor() != null ? "\"" + follower.getVotedFor() + "\"" : "Nobody"));
160            writer.println(String.format("%-24s: %s", "Installing snapshot", follower.isInstallingSnapshot() ? "Yes" : "No"));
161            final Timestamp electionTimeout = follower.getElectionTimeout();
162            writer.println(String.format("%-24s: %s", "Election timer running",
163              electionTimeout != null ? "Yes; expires in " + electionTimeout.offsetFromNow() + "ms" : "No"));
164            final int probed = follower.getNodesProbed();
165            writer.println(String.format("%-24s: %s", "Election nodes probed",
166              probed != -1 ? String.format("%d / %d", follower.getNodesProbed(), config.size()) : "Not probing"));
167        } else if (role instanceof CandidateRole) {
168            final CandidateRole candidate = (CandidateRole)role;
169
170            writer.println();
171            writer.println("Candidate Info");
172            writer.println("==============");
173            writer.println();
174            writer.println(String.format("%-24s: %d", "Votes Required", candidate.getVotesRequired()));
175            writer.println(String.format("%-24s: %d", "Votes Received", candidate.getVotesReceived()));
176        }
177
178        // Transactions
179        writer.println();
180        writer.println("Open Transactions");
181        writer.println("=================");
182        writer.println();
183        writer.println(String.format("%1s %-6s %-10s %-5s %-12s %-8s %-8s %s",
184          "", "ID", "State", "Type", "Consistency", "Base", "Commit", "Config Change"));
185        writer.println(String.format("%1s %-6s %-10s %-5s %-12s %-8s %-8s %s",
186          "", "--", "-----", "----", "-----------", "----", "------", "-------------"));
187        for (RaftKVTransaction tx2 : db.getOpenTransactions()) {
188            writer.println(String.format("  %-6d %-10s %-5s %-12s %-8s %-8s %s", tx2.getTxId(),
189              tx2.getState(), tx2.isReadOnly() ? "R/O" : "R/W", tx2.getConsistency(), tx2.getBaseIndex() + "t" + tx2.getBaseTerm(),
190              tx2.getState().compareTo(TxState.COMMIT_WAITING) >= 0 ? tx2.getCommitIndex() + "t" + tx2.getCommitTerm() : "",
191              this.describe(tx2.getConfigChange())));
192        }
193        writer.println();
194    }
195
196    // Describe a config change
197    private String describe(String[] change) {
198        return change != null ?
199          (change[1] != null ? String.format("+\"%s\"@%s", change[0], change[1]) : "-\"" + change[0] + "\"") : "";
200    }
201}
