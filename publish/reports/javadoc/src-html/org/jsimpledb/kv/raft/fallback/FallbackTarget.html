020/**
021 * Represents one of the underlying {@link RaftKVDatabase}s managed by a {@link FallbackKVDatabase}.
022 *
023 * <p>
024 * Instances must at least be configured with the associated {@link RaftKVDatabase}; all other properties have defaults.
025 *
026 * Two {@link MergeStrategy}s are configured to handle switching into
027 * (see {@link #setUnavailableMergeStrategy setUnavailableMergeStrategy()}) and out of
028 * (see {@link #setRejoinMergeStrategy setRejoinMergeStrategy()}) standalone mode.
029 *
030 * <p>
031 * Other parameters configure how cluster availability is determined and enforce hysteresis; see
032 * {@link #setTransactionTimeout setTransactionTimeout()}, {@link #setCheckInterval setCheckInterval()},
033 * {@link #setMinAvailableTime setMinAvailableTime()}, and {@link #setMinUnavailableTime setMinUnavailableTime()}.
034 */

274// Subclass Methods

276    /**
277     * Perform an availability assessment of the underlying {@link RaftKVDatabase} associated with this instance.
278     *
279     * <p>
280     * This method may block for as long as necessary to determine availability, but it should not block
281     * indefinitely. If a {@link KVTransaction} is opened, {@link KVTransaction#setTimeout setTimeout()}
282     * should be used to set a time limit.
283     *
284     * <p>
285     * This method will not be invoked concurrently by two different threads.
286     *
287     * <p>
288     * If this method throws an unchecked exception, the database will be assumed to be unavailable.
289     *
290     * <p>
291     * The implementation in {@link FallbackTarget} determines availability by attempting to commit a read-only,
292     * {@link org.jsimpledb.kv.raft.Consistency#LINEARIZABLE} transaction within the configured maximum timeout.
293     *
294     * @return true if database is available, false otherwise
295     */
296    protected boolean checkAvailability() {
297
298        // Setup
299        if (this.log.isTraceEnabled())
300            this.log.trace("checking availability of " + this.raft + " with timeout of " + this.transactionTimeout + "ms");
301        final long startTimeNanos = System.nanoTime();
302        boolean success = false;
303
304        // Perform transaction
305        final KVTransaction tx = this.raft.createTransaction();
306        try {
307            tx.setTimeout(this.transactionTimeout);
308            tx.getAtLeast(ByteUtil.EMPTY);
309            tx.commit();
310            success = true;
311        } finally {
312            if (!success)
313                tx.rollback();
314        }
315
316        // Check timeout
317        final int duration = (int)((System.nanoTime() - startTimeNanos) / 1000000L);
318        final boolean timedOut = duration > this.transactionTimeout;
319        if (this.log.isTraceEnabled()) {
320            this.log.trace("availability transaction on " + this.raft + " completed in "
321              + duration + "ms (" + (timedOut ? "failed" : "successful") + ")");
322        }
323        if (timedOut && this.log.isDebugEnabled()) {
324            this.log.debug("availability transaction on " + this.raft + " completed in "
325              + duration + " > " + this.transactionTimeout + " ms, returning unavailable");
326        }
327
328        // Done
329        return !timedOut;
330    }
