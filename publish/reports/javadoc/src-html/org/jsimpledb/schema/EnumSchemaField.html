006package org.jsimpledb.schema;
007
008import com.google.common.collect.Iterators;
009import com.google.common.collect.PeekingIterator;
010
011import java.util.ArrayList;
012import java.util.List;
013import java.util.TreeMap;
014
015import javax.xml.stream.XMLStreamException;
016import javax.xml.stream.XMLStreamReader;
017import javax.xml.stream.XMLStreamWriter;
018
019import org.dellroad.stuff.string.StringEncoder;
020import org.jsimpledb.core.EnumFieldType;
021import org.jsimpledb.core.InvalidSchemaException;
022import org.jsimpledb.util.Diffs;
023
024/**
025 * An enum field in a {@link SchemaObjectType}.
026 */
027public class EnumSchemaField extends SimpleSchemaField {
028
029    private /*final*/ List<String> idents = new ArrayList<>();
030
031    /**
032     * Get the ordered list of identifiers.
033     *
034     * @return enum identifier list
035     */
036    public List<String> getIdentifiers() {
037        return this.idents;
038    }
039
040    @Override
041    void validate() {
042        super.validate();
043        try {
044            EnumFieldType.validateIdentifiers(this.idents);
045        } catch (IllegalArgumentException e) {
046            throw new InvalidSchemaException("invalid " + this + ": " + e.getMessage(), e);
047        }
048    }
049
050    @Override
051    public <R> R visit(SchemaFieldSwitch<R> target) {
052        return target.caseEnumSchemaField(this);
053    }
054
055    // For enum types, we don't care if the type names are different; we only care if the identifier sets are different.
056    // This allows enum types to change Java packages without creating an incompatible schema.
057    @Override
058    boolean isCompatibleWithInternal(AbstractSchemaItem that0) {
059        final EnumSchemaField that = (EnumSchemaField)that0;
060        if (this.isIndexed() != that.isIndexed())
061            return false;
062        if (!this.idents.equals(that.idents))
063            return false;
064        return true;
065    }
066
067    @Override
068    void readSubElements(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
069        while (this.expect(reader, true, IDENTIFIER_TAG))
070            this.idents.add(reader.getElementText());
071    }
072
073    @Override
074    void writeXML(XMLStreamWriter writer, boolean includeName) throws XMLStreamException {
075        writer.writeStartElement(ENUM_FIELD_TAG.getNamespaceURI(), ENUM_FIELD_TAG.getLocalPart());
076        this.writeAttributes(writer, includeName);
077        for (String ident : this.idents) {
078            writer.writeStartElement(IDENTIFIER_TAG.getNamespaceURI(), IDENTIFIER_TAG.getLocalPart());
079            writer.writeCharacters(StringEncoder.encode(ident, false));
080            writer.writeEndElement();
081        }
082        writer.writeEndElement();
083    }
084
085// DiffGenerating
086
087    @Override
088    public Diffs differencesFrom(SimpleSchemaField other) {
089        final Diffs diffs = new Diffs(super.differencesFrom(other));
090        if (!(other instanceof EnumSchemaField)) {
091            diffs.add("change type from " + other.getClass().getSimpleName() + " to " + this.getClass().getSimpleName());
092            return diffs;
093        }
094        final EnumSchemaField that = (EnumSchemaField)other;
095        if (!this.idents.equals(that.idents)) {
096            final Diffs enumDiffs = new Diffs();
097            final TreeMap<String, Integer> thisOrdinals = new TreeMap<>();
098            final TreeMap<String, Integer> thatOrdinals = new TreeMap<>();
099            for (int i = 0; i < this.idents.size(); i++)
100                thisOrdinals.put(this.idents.get(i), i);
101            for (int i = 0; i < that.idents.size(); i++)
102                thatOrdinals.put(that.idents.get(i), i);
103            final PeekingIterator<String> thisIterator = Iterators.peekingIterator(thisOrdinals.keySet().iterator());
104            final PeekingIterator<String> thatIterator = Iterators.peekingIterator(thatOrdinals.keySet().iterator());
105            while (thisIterator.hasNext() || thatIterator.hasNext()) {
106                final String thisName = thisIterator.hasNext() ? thisIterator.peek() : null;
107                final String thatName = thatIterator.hasNext() ? thatIterator.peek() : null;
108                assert thisName != null || thatName != null;
109                final int diff = thisName == null ? 1 : thatName == null ? -1 : thisName.compareTo(thatName);
110                if (diff < 0)
111                    enumDiffs.add("added `" + thisName + "' (ordinal " + thisOrdinals.get(thisName) + ")");
112                else
113                    thatIterator.next();
114                if (diff > 0)
115                    enumDiffs.add("removed `" + thatName + "' (ordinal " + thatOrdinals.get(thatName) + ")");
116                else
117                    thisIterator.next();
118            }
119            diffs.add("changed enum identifier list", enumDiffs);
120        }
121        return diffs;
122    }
123
124// Object
125
126    @Override
127    public boolean equals(Object obj) {
128        if (obj == this)
129            return true;
130        if (!super.equals(obj))
131            return false;
132        final EnumSchemaField that = (EnumSchemaField)obj;
133        return this.idents.equals(that.idents);
134    }
135
136    @Override
137    public int hashCode() {
138        return super.hashCode() ^ this.idents.hashCode();
139    }
140
141// Cloneable
142
143    @Override
144    public EnumSchemaField clone() {
145        final EnumSchemaField clone = (EnumSchemaField)super.clone();
146        clone.idents = new ArrayList<>(this.idents);
147        return clone;
148    }
149}
