006package org.jsimpledb.core;
007
008import com.google.common.base.Preconditions;
009import com.google.common.reflect.TypeToken;
010
011import java.util.Comparator;
012
013import org.jsimpledb.parse.ParseContext;
014import org.jsimpledb.parse.ParseUtil;
015import org.jsimpledb.util.ByteReader;
016import org.jsimpledb.util.ByteWriter;
017
018/**
019 * Defines the encoding, ordering, and range of possible values for a {@link SimpleField}.
020 *
021 * <p>
022 * A {@link FieldType} maps between instances of its supported Java type and the self-delimited {@code byte[]} encoding of
023 * those instances used in the database. The {@code byte[]} encoding also implicitly defines the database sort order
024 * (via unsigned lexicographical ordering). This ordering is also reflected via {@link #compare compare()}.
025 * </p>
026 *
027 * <p>
028 * A {@link FieldType} also defines a mapping between Java instances and {@link String} values.
029 * There are two separate {@link String} forms, a regular form and a self-delimiting form.
030 * </p>
031 *
032 * <p>
033 * {@link FieldType}s have these requirements and properties:
034 * <ul>
035 *  <li>They have a unique {@linkplain #getName name}; typically the same as their {@linkplain #getTypeToken supported type}.</li>
036 *  <li>All possible values can be represented in Java as an instance of the associated Java type (possibly including null).</li>
037 *  <li>Instances {@linkplain #compare totally order} their Java values. If the associated Java type implements {@link Comparable},
038 *      then the two orderings do not necessarily have to match, but they should if possible.</li>
039 *  <li>All possible values can be encoded/decoded into a self-delimiting binary string (i.e., {@code byte[]} array)
040 *      without losing information, and these binary strings, when sorted lexicographically using unsigned comparison,
041 *      sort consistently with the total ordering of the corresponding Java values defined by {@link #compare compare()}.</li>
042 *  <li>All possible values can be encoded/decoded to/from {@link String}s without losing information,
043 *      with both a {@linkplain #toString(Object) regular string form} for non-null values and a
044 *      {@linkplain #toParseableString self-delimiting string form} for any value (these may be the same).</li>
045 *  <li>{@code null} may or may not be a supported value; if so, it must be handled by {@link #compare compare()} (typically
046 *      null values sort last) and have binary and string encodings just like any other value.</li>
047 *  <li>There is a {@linkplain #getDefaultValue default value}. For types that support null, the default value must be null.</li>
048 *  <li>An optional {@linkplain #getEncodingSignature encoding signature} protects against incompatible encodings
049 *      when a {@link FieldType}'s binary or string encoding changes without changing the {@linkplain #getName name}.</li>
050 * </ul>
051 *
052 * <p>
053 * Two {@link FieldType} instances should be equal according to {@link #equals equals()} if only if they behave identically
054 * with respect to all of the above.
055 * </p>
056 *
057 * <p>
058 * A {@link FieldTypeRegistry} contains a registry of {@link FieldType}s indexed by name.
059 * </p>
060 *
061 * @param <T> The associated Java type
062 * @see FieldTypeRegistry
063 */
064public abstract class FieldType<T> implements Comparator<T> {
065
066    /**
067     * The regular expression that {@link FieldType} names must match. This pattern is the same as is required
068     * for Java identifiers, except that the following additional characters are allowed after the first character:
069     * dot (`.') and dash (`-'), and lastly up to 255 pairs of square brackets (`[]') to indicate an array type.
070     */
071    public static final String NAME_PATTERN = "\\p{javaJavaIdentifierStart}[-.\\p{javaJavaIdentifierPart}]*(\\[\\]){0,255}";
072
073    /**
074     * Type name for reference types.
075     */
076    public static final String REFERENCE_TYPE_NAME = "reference";
077
078    final String name;
079    final TypeToken<T> typeToken;
080    final long signature;
081    final T defaultValueObject;
082
083    byte[] defaultValue;
084
085// Constructors
086
087    /**
088     * Constructor.
089     *
090     * @param name the name of this type
091     * @param typeToken Java type for the field's values
092     * @param signature binary encoding signature
093     * @param defaultValue default value for this type
094     * @throws IllegalArgumentException if any parameter is null
095     * @throws IllegalArgumentException if {@code name} is invalid
096     */
097    protected FieldType(String name, TypeToken<T> typeToken, long signature, T defaultValue) {
098        Preconditions.checkArgument(name != null, "null name");
099        Preconditions.checkArgument(name.matches(FieldType.NAME_PATTERN), "invalid type name `" + name + "'");
100        Preconditions.checkArgument(typeToken != null, "null typeToken");
101        this.name = name;
102        this.typeToken = typeToken;
103        this.signature = signature;
104        this.defaultValueObject = defaultValue;
105    }
106
107    /**
108     * Constructor taking a {@link Class} object.
109     * The {@linkplain #getName name} of this instance will be the {@linkplain Class#getName name} of the given class.
110     *
111     * @param type Java type for the field's values
112     * @param signature binary encoding signature
113     * @param defaultValue default value for this type
114     * @throws NullPointerException if {@code type} is null
115     */
116    protected FieldType(Class<T> type, long signature, T defaultValue) {
117        this(type.getName(), TypeToken.of(type), signature, defaultValue);
118    }
119
120// Public methods
121
122    /**
123     * Get the name of this type. {@link FieldType} names must be unique in the registry.
124     *
125     * @return this type's name
126     */
127    public String getName() {
128        return this.name;
129    }
130
131    /**
132     * Get the Java type corresponding to this type's values.
133     *
134     * @return this type's Java type
135     */
136    public TypeToken<T> getTypeToken() {
137        return this.typeToken;
138    }
139
140    /**
141     * Get the binary encoding signature of this type.
142     *
143     * <p>
144     * The binary encoding signature is analogous to the {@code serialVersionUID} used by Java serialization.
145     * It represents a specific binary and/or {@link String} encoding for Java values. In the case that a {@link FieldType}
146     * implementation changes its binary encoding, but not it's name, it <b>must</b> use a new, different binary encoding
147     * signature to eliminate the possibility of mixing incompatible encodings in software vs. persistent storage.
148     * Typically a value of zero is used until if/when such a change occurs.
149     * </p>
150     *
151     * <p>
152     * Note that another option when encodings change is simply to change the {@linkplain #getName name} of the type
153     * (encoding signatures are scoped to a single {@link FieldType} name).
154     * </p>
155     *
156     * @return binary encoding signature
157     */
158    public long getEncodingSignature() {
159        return this.signature;
160    }
161
162    /**
163     * Read a value from the given input.
164     *
165     * @param input byte input
166     * @return field value (possibly null)
167     * @throws IllegalArgumentException if invalid input is encountered
168     * @throws IndexOutOfBoundsException if input is truncated
169     */
170    public abstract T read(ByteReader input);
171
172    /**
173     * Write a value to the given output.
174     *
175     * @param writer byte output
176     * @param value value to write (possibly null)
177     * @throws IllegalArgumentException if {@code value} is null and this type does not support null
178     */
179    public abstract void write(ByteWriter writer, T value);
180
181    /**
182     * Get the default value for this field type encoded as a {@code byte[]} array.
183     *
184     * @return encoded default value
185     */
186    public final byte[] getDefaultValue() {
187        if (this.defaultValue == null) {
188            final ByteWriter writer = new ByteWriter();
189            try {
190                this.write(writer, this.defaultValueObject);
191            } catch (IllegalArgumentException e) {
192                throw new UnsupportedOperationException(this + " does not have a default value");
193            }
194            this.defaultValue = writer.getBytes();
195        }
196        return this.defaultValue.clone();
197    }
198
199    /**
200     * Get the default value for this field type.
201     *
202     * @return default value
203     */
204    public final T getDefaultValueObject() {
205        return this.defaultValueObject;
206    }
207
208    /**
209     * Read and discard a value from the given input.
210     *
211     * @param reader byte input
212     * @throws IllegalArgumentException if invalid input is encountered
213     * @throws IndexOutOfBoundsException if input is truncated
214     */
215    public abstract void skip(ByteReader reader);
216
217    /**
218     * Encode a non-null value as a {@link String} for later decoding by {@link #fromString fromString()}.
219     *
220     * <p>
221     * Each of the characters in the returned {@link String} must be one of the valid XML characters
222     * (tab, newline, carriage return, <code>&#92;u0020 - &#92;ud7ff</code>, and <code>&#92;ue000 - &#92;fffdf</code>).
223     * </p>
224     *
225     * <p>
226     * The implementation in {@link FieldType} checks that {@code value} is not null, then delegates to {@link #toParseableString}.
227     * Subclasses that override this method should also override {@link #fromString fromString()}.
228     * </p>
229     *
230     * @param value actual value, never null
231     * @return string encoding of {@code value} acceptable to {@link #fromString fromString()}
232     * @throws IllegalArgumentException if {@code value} is null
233     * @see <a href="http://www.w3.org/TR/REC-xml/#charsets">The XML 1.0 Specification</a>
234     */
235    public String toString(T value) {
236        Preconditions.checkArgument(value != null, "null value");
237        return this.toParseableString(value);
238    }
239
240    /**
241     * Parse a non-null value previously encoded by {@link #toString(Object) toString(T)}.
242     *
243     * <p>
244     * The implementation in {@link FieldType} creates a new {@link ParseContext} based on {@code string},
245     * delegates to {@link #toParseableString} to parse it, and verifies that all of {@code string} was consumed
246     * during the parse. Subclasses that override this method should also override {@link #toString(Object) toString(T)}.
247     * </p>
248     *
249     * @param string non-null value previously encoded as a {@link String} by {@link #toString(Object) toString(T)}
250     * @return actual value
251     * @throws IllegalArgumentException if the input is invalid
252     */
253    public T fromString(String string) {
254        final ParseContext ctx = new ParseContext(string);
255        final T value = this.fromParseableString(ctx);
256        if (!ctx.isEOF()) {
257            throw new IllegalArgumentException("found trailing garbage starting with `"
258              + ParseUtil.truncate(ctx.getInput(), 20) + "'");
259        }
260        return value;
261    }
262
263    /**
264     * Encode a possibly null value as a {@link String} for later decoding by {@link #fromParseableString fromParseableString()}.
265     * The string value must be <i>self-delimiting</i>, i.e., decodable even when followed by arbitrary additional characters,
266     * and must not start with whitespace or closing square bracket ({@code "]"}).
267     *
268     * <p>
269     * In addition, each of the characters in the returned {@link String} must be one of the valid XML characters
270     * (tab, newline, carriage return, <code>&#92;u0020 - &#92;ud7ff</code>, and <code>&#92;ue000 - &#92;fffdf</code>).
271     * </p>
272     *
273     * @param value actual value (possibly null)
274     * @return string encoding of {@code value} acceptable to {@link #fromParseableString fromParseableString()}
275     * @throws IllegalArgumentException if {@code value} is null and this type does not support null
276     * @see <a href="http://www.w3.org/TR/REC-xml/#charsets">The XML 1.0 Specification</a>
277     */
278    public abstract String toParseableString(T value);
279
280    /**
281     * Parse a value previously encoded by {@link #toParseableString toParseableString()} as a self-delimited {@link String}
282     * and positioned at the start of the given parsing context.
283     *
284     * @param context parse context starting with a string previously encoded via {@link #toParseableString toParseableString()}
285     * @return actual value (possibly null)
286     * @throws IllegalArgumentException if the input is invalid
287     */
288    public abstract T fromParseableString(ParseContext context);
289
290    /**
291     * Verify the given object is a valid instance of this {@link FieldType}'s Java type and cast it to that type.
292     *
293     * <p>
294     * Note that this method must throw {@link IllegalArgumentException}, not {@link ClassCastException}
295     * or {@link NullPointerException}, if {@code obj} does not have the correct type, or is an illegal null value.
296     * </p>
297     *
298     * <p>
299     * This method is allowed to perform widening conversions of the object that lose no information, e.g.,
300     * from {@link Integer} to {@link Long}.
301     * </p>
302     *
303     * <p>
304     * The implementation in {@link FieldType} simply casts the value using this instance's raw Java type.
305     * Subclasses should override this method to implement any other restrictions, e.g., disallowing null values.
306     * </p>
307     *
308     * @param obj object to validate
309     * @return {@code obj} cast to this field's type
310     * @throws IllegalArgumentException if {@code obj} in not of type T
311     * @throws IllegalArgumentException if {@code obj} is null and this type does not support null values
312     * @throws IllegalArgumentException if {@code obj} is in any other way not supported by this {@link FieldType}
313     */
314    @SuppressWarnings("unchecked")
315    public T validate(Object obj) {
316        try {
317            return (T)this.typeToken.getRawType().cast(obj);
318        } catch (ClassCastException e) {
319            throw new IllegalArgumentException(this + " does not support values of type " + obj.getClass().getName());
320        }
321    }
322
323    /**
324     * Order two field values.
325     *
326     * <p>
327     * This method must provide a total ordering of all supported Java values that is consistent with the database ordering,
328     * i.e., the unsigned lexicographical ordering of the corresponding {@code byte[]} encoded field values.
329     *
330     * <p>
331     * If null is a supported Java value, then the returned {@link Comparator} must accept null parameters without
332     * throwing an exception (note, this is a stronger requirement than the {@link Comparator} interface normally requires).
333     *
334     * <p>
335     * Note: by convention, null values usually sort last.
336     *
337     * @throws IllegalArgumentException if {@code value1} or {@code value2} is null and this type does not support null
338     */
339    @Override
340    public abstract int compare(T value1, T value2);
341
342    /**
343     * Determine whether any of this field type's encoded values start with a {@code 0x00} byte.
344     * Certain optimizations are possible when this is not the case. It is safe for this method to always return true.
345     *
346     * <p>
347     * Note: changing the return value of this method usually means changing the binary encoding, resulting in
348     * an incompatible type.
349     * </p>
350     *
351     * <p>
352     * The implementation in {@link FieldType} returns {@code true}.
353     * </p>
354     *
355     * @return true if an encoded value starting with {@code 0x00} exists
356     */
357    public boolean hasPrefix0x00() {
358        return true;
359    }
360
361    /**
362     * Determine whether any of this field type's encoded values start with a {@code 0xff} byte.
363     * Certain optimizations are possible when this is not the case. It is safe for this method to always return true.
364     *
365     * <p>
366     * Note: changing the return value of this method usually means changing the binary encoding, resulting in
367     * an incompatible type.
368     * </p>
369     *
370     * <p>
371     * The implementation in {@link FieldType} returns {@code true}.
372     * </p>
373     *
374     * @return true if an encoded value starting with {@code 0xff} exists
375     */
376    public boolean hasPrefix0xff() {
377        return true;
378    }
379
380    /**
381     * Convenience method for generic type binding.
382     *
383     * @param obj object to validate
384     * @throws IllegalArgumentException if {@code obj} in not of type T
385     * @throws IllegalArgumentException if {@code obj} is null and this type does not support null values
386     * @throws IllegalArgumentException if {@code obj} is in any other way not supported by this {@link FieldType}
387     */
388    void validateAndWrite(ByteWriter writer, Object obj) {
389        this.write(writer, this.validate(obj));
390    }
391
392// Object
393
394    @Override
395    public String toString() {
396        return "field type `" + this.name + "'";
397    }
398
399    @Override
400    public int hashCode() {
401        return this.name.hashCode() ^ this.typeToken.hashCode();
402    }
403
404    @Override
405    public boolean equals(Object obj) {
406        if (obj == this)
407            return true;
408        if (obj == null || obj.getClass() != this.getClass())
409            return false;
410        final FieldType<?> that = (FieldType<?>)obj;
411        return this.name.equals(that.name) && this.typeToken.equals(that.typeToken);
412    }
413}
